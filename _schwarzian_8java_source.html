283 public static double[] final_three(double[] uzians) {
284 int m=uzians.length;
285 int n=m+3;
286 // return originals and add the last 3
287 double[] fl=new double[n];
288 for (int j=0;j<m;j++) {
289 fl[j]=uzians[j];
290 }
291 if (n==3) { // 3-flower
292 fl[m]=fl[m+1]=fl[m+2]=oosq3;
293 return fl;
294 }
295 if (n==4) { // 4-flower
296 fl[m]=fl[m+2]=2.0/(3.0*fl[0]);
297 fl[2]=fl[0];
298 return fl;
299 }
300 if (n==5) { // 5-flower
301 for (int k=0;k<3;k++) {
302 fl[m+k]=
303 (fl[k]-oosq3)/(3*fl[k]*fl[k+1]-1.0);
304 }
305 return fl;
306 }
307 if (n==6) { // 6-flower
308 for (int k=0;k<3;k++) {
309 double num=fl[0+k]*fl[1+k];
310 double denom=3.0*fl[0+k]*fl[1+k]*fl[2+k]-fl[0+k]-fl[2+k];
311 fl[m+k]=num/denom;
312 }
313 return fl;
314 }
315 if (n==7) { // 7-flower
316 for (int k=0;k<3;k++) {
317 double num=3.0*(3.0*fl[0+k]*fl[1+k]*fl[2+k]-fl[0+k]-fl[2+k])+oosq3;
318 double denom=3.0*(3.0*fl[0+k]*fl[1+k]*fl[2+k]*fl[3+k]-
319 fl[0+k]*fl[1+k]-fl[0+k]*fl[3+k]-fl[2+k]*fl[3+k])+1.0;
320 fl[m+k]=num/denom;
321 }
322 return fl;
323 }
324 if (n==8) { // 8-flower
325 for (int k=0;k<3;k++) {
326 double num=3.0*(3.0*fl[0+k]*fl[1+k]*fl[2+k]*fl[3+k]-
327 fl[0+k]*fl[1+k]-fl[0+k]*fl[3+k]-fl[2+k]*fl[3+k])+2.0;
328 double denom=27.0*fl[0+k]*fl[1+k]*fl[2+k]*fl[3+k]*fl[4+k]-
329 9.0*(fl[0+k]*fl[1+k]*fl[2+k]+fl[0+k]*fl[1+k]*fl[4+k]-fl[2+k]*fl[3+k]*fl[4+k])-
330 3.0*(fl[0+k]+fl[2+k]-fl[4+k]);
331 fl[m+k]=num/denom;
332 }
333 return fl;
334 }
335
336 else {
337 throw new DataException("No formulas next schwarzian for degree > 8");
338 }

351 public static double rad_to_schwarzian(double[] rad,int hes) {
352 CircleSimple[] sC=new CircleSimple[4];
353 for (int i=0;i<4;i++) {
354 sC[i]=new CircleSimple();
355 sC[i].rad=rad[i];
356 }
357
358 // get the four centers in the appropriate geometry
359 Complex []Z=new Complex[4];
360
361 // find centers for face f <v,w,a>
362 double[] ivd= {1.0,1.0,1.0}; // tangency packings only
363 int ans=CommonMath.placeOneFace(sC[0],sC[1],sC[2],ivd,hes);
364 if (ans<0) {
365 throw new DataException(
366 "Problem in 'rad_to_schwarzian' placeOneFace");
367 }
368 Z[0]=new Complex(sC[0].center);
369 Z[1]=new Complex(sC[1].center);
370 Z[2]=new Complex(sC[2].center);
371
372 // find center for b in <w,v,b>
373 sC[3]=CommonMath.comp_any_center(sC[1].center,
374 sC[0].center,sC[1].rad,sC[0].rad,
375 rad[3],ivd[0],ivd[1],ivd[2], hes);
376 sC[3].rad=rad[3];
377 Z[3]=new Complex(sC[3].center);
378
379 // compute the face tangency points, then face mobius, i.e.,
380 // the mobius maps FROM the base equilateral to the face
381 DualTri dtri=new DualTri(Z[0],Z[1],Z[2],hes); // <v,w,u>
382 if (dtri.TangPts==null)
383 throw new DataException("'rad_to_schwarzian' failed to get 'dri'");
384 Complex []tanPts=new Complex[3];
385 for (int j=0;j<3;j++)
386 tanPts[j]=new Complex(dtri.TangPts[j]);
387 Mobius fbase=Mobius.mob_xyzXYZ(
388 CPBase.omega3[0],CPBase.omega3[1],CPBase.omega3[2],
389 tanPts[0],tanPts[1],tanPts[2],0,hes);
390
391 dtri=new DualTri(Z[1],Z[0],Z[3],hes); // <w,v,b>
392 if (dtri.TangPts==null)
393 throw new DataException("'rad_to_schwarzian' failed with second 'dri'");
394 for (int j=0;j<3;j++)
395 tanPts[j]=new Complex(dtri.TangPts[j]);
396 Mobius gbase=Mobius.mob_xyzXYZ(
397 CPBase.omega3[0],CPBase.omega3[1],CPBase.omega3[2],
398 tanPts[0],tanPts[1],tanPts[2],0,hes);
399
400 Mobius dMob = Schwarzian.getIntrinsicSch(fbase,gbase,0,0);
401 if (Math.abs(dMob.c.y)>.001)
402 throw new DataException("error: Schwarzian is not real");
403 return dMob.c.x;
404 }

425 public static Mobius getIntrinsicSch(Mobius bm_f,Mobius bm_g,
426 int indx_f,int indx_g) {
427
428 // Let F be the base equilateral, edge 0 centered on 1,
429 // let G be the contiguous equilateral across that edge.
430 // To get the edge Mobius derivative, we must pre-compose
431 // bm_f and bm_g, which map F to f and g, to get mu_f, mu_g.
432 // For mu_f, pre-rotate so the first edge of F maps
433 // to the indx_f edge of f; i.e., rotation by omega3[indx_f].
434 // For mu_g we want to map G to g. We pre-compose bm_g
435 // by translation by -2, rotation by omega3[indx_g], then
436 // rotation by pi.
437 Mobius pre_f=new Mobius(CPBase.omega3[indx_f],
438 new Complex(0.0),new Complex(0.0),new Complex(1.0));
439 Mobius mu_f=(Mobius)bm_f.rmultby(pre_f);
440
441 Complex wi=new Complex(CPBase.omega3[indx_g]).times(-1.0);
442 Mobius pre_g=new Mobius(wi,wi.times(-2.0),
443 new Complex(0.0),new Complex(1.0));
444 Mobius mu_g=(Mobius)bm_g.rmultby(pre_g); // mob_g.det();
445 Mobius edgeMob=(Mobius)mu_g.inverse().rmultby(mu_f);
446 edgeMob.normalize();
447
448 // check c essentially real? trace essentially 2?
449 if (Math.abs(edgeMob.c.y)>.00001)
450 throw new MobException("c entry should be real");
451 if (edgeMob.a.add(edgeMob.d).abs()-2.0>.0001)
452 throw new MobException("trace should be 2.0");
453 double s=edgeMob.c.x;
454
455 boolean debug=false;
456 if (debug) { // debug=true;
457 deBugging.DebugHelp.mob4matlab("bm_f",bm_f);
458 deBugging.DebugHelp.mob4matlab("bm_g",bm_g);
459 deBugging.DebugHelp.mob4matlab("pre_f",pre_f);
460 deBugging.DebugHelp.mob4matlab("mob_f",mu_f);
461 deBugging.DebugHelp.mob4matlab("pre_g",pre_g);
462 deBugging.DebugHelp.mob4matlab("mob_g",mu_g);
463 deBugging.DebugHelp.mob4matlab("edgeMob",edgeMob);
464 debug=false;
465 }
466
467 // return clean Mobius
468 Mobius outmob=new Mobius();
469 outmob.a=new Complex(1.0+s);
470 outmob.b=new Complex(-1.0*s);
471 outmob.c=new Complex(s);
472 outmob.d=new Complex(1.0-s);
473
474 return outmob;
475 }

672 public static SchwarzData getSchData(TriAspect domf,TriAspect tmpg,
673 TriAspect rngF,TriAspect tmpG) {
674 boolean debug=false;
675
676 // duplicates, to avoid messing up originals
677 TriAspect domg=new TriAspect(tmpg);
678 TriAspect rngG=new TriAspect(tmpG);
679 SchwarzData schData=null;
680 try {
681 schData=comp_Sch_Deriv(domf,domg,rngF,rngG);
682 } catch (Exception ex) {
683 throw new CombException("failed to get Sch_Deriv");
684 }
685
686 // unit outward normal eta; depends on geom
687 int indx_f=domf.nghb_Tri(domg);
688 Complex cv=domf.getCenter(indx_f);
689 Complex cw=domf.getCenter((indx_f+1)%3);
690 if (domf.hes>0) { // sph
691 CircleSimple sC1=SphericalMath.s_to_e_data(cv,
692 domf.getRadius(indx_f));
693 cv=sC1.center;
694 CircleSimple sC2=SphericalMath.s_to_e_data(cw,
695 domf.getRadius((indx_f+1)%3));
696 cw=sC2.center;
697 if (sC1.flag==-1 || sC2.flag==-1)
698 throw new MiscException("A disc contains "+
699 "infinity: Schwarz not ready "+
700 "for this.");
701 }
702 if (domf.hes<0) { // hyp
703 CircleSimple sC=HyperbolicMath.h_to_e_data(cv,
704 domf.getRadius(indx_f));
705 cv=sC.center;
706 sC=HyperbolicMath.h_to_e_data(cw,
707 domf.getRadius((indx_f+1)%3));
708 cw=sC.center;
709 }
710
711 // get outward unit normal
712 Complex eta=cw.minus(cv).times(new Complex(0,-1));
713 eta=eta.divide(eta.abs());
714
715 Complex tmp=schData.Schw_Deriv.divide(eta.conj());
716 schData.Schw_coeff=tmp.x;

776 public static SchwarzData comp_Sch_Deriv(TriAspect domf,TriAspect domg,
777 TriAspect rngF,TriAspect rngG) {
778 boolean debug=false;
779 File tmpdir=null;
780 BufferedWriter fbw=null;
781
782 // recompute baseMob's
783 Mobius dmf=new Mobius(domf.setBaseMobius());
784 Mobius dmg=new Mobius(domg.setBaseMobius());
785 Mobius rmF=new Mobius(rngF.setBaseMobius());
786 Mobius rmG=new Mobius(rngG.setBaseMobius());
787 if (dmf==null || dmg==null || rmF==null || rmG==null)
788 return null;
789
790
791 if (debug) { // debug=true;
792 tmpdir=new File(System.getProperty("java.io.tmpdir"));
793 // for debugging, comparing to matlab output
794 fbw=CPFileManager.openWriteFP(tmpdir,
795 "InitialBaseMobs.mlab",false);
796 try {
797 fbw.write("dmf="+dmf.toMatlabString()+"\n");
798 fbw.write("dmg="+dmg.toMatlabString()+"\n");
799 fbw.write("rmF="+rmF.toMatlabString()+"\n");
800 fbw.write("rmG="+rmG.toMatlabString()+"\n");
801 } catch(Exception ex) {
802 CirclePack.cpb.errMsg("failed initial output in 'schwarzian.java'");
803 }
804 }
805
806 // Find shared edge index for g, align g to f if necessary
807 int indx_g=domg.nghb_Tri(domf);
808 HalfEdge g_he=domg.baseEdge;
809 if (indx_g==1)
810 g_he=g_he.next;
811 else if (indx_g==2)
812 g_he=g_he.next.next;
813 Mobius tmob;
814 if ((tmob=domg.alignMe(domf, g_he, 1))!=null)
815 dmg=(Mobius)dmg.lmultby(tmob);
816
817 // Find shared edge index for G, align G to F if necessary
818 int indx_G=rngG.nghb_Tri(rngF);
819 HalfEdge G_he=rngG.baseEdge;
820 if (indx_G==1)
821 G_he=G_he.next;
822 else if (indx_G==2)
823 G_he=G_he.next.next;
824 if ((tmob=(Mobius)rngG.alignMe(rngF, G_he, 1))!=null)
825 rmG=(Mobius)rmG.lmultby(tmob);
826
827 // Find shared edge indices for f/F
828 int indx_f=domf.nghb_Tri(domg);
829 int indx_F=rngF.nghb_Tri(rngG);
830
831 // get v/w to start 'schData'
832 HalfEdge f_he=domf.baseEdge;
833 if (indx_f==1)
834 f_he=f_he.next;
835 else if (indx_f==2)
836 f_he=f_he.next.next;
837
838 // get started: assume vert indices agree in domain/range
839 SchwarzData schData=new SchwarzData(
840 f_he.origin.vertIndx,f_he.next.origin.vertIndx);
841
842 // must pre-rotate 'baseMob's so shared edge
843 // corresponds to base edge through 1
844 dmf.a=dmf.a.times(CPBase.omega3[indx_f]);
845 dmf.c=dmf.c.times(CPBase.omega3[indx_f]);
846 dmf.normalize();
847
848 // get dmf derivative at 1
849 Complex dmfd1=new Complex(dmf.c).add(dmf.d);
850 schData.dmf_deriv=new Complex(1.0).divide(dmfd1.times(dmfd1));
851
852 rmF.a=rmF.a.times(CPBase.omega3[indx_F]);
853 rmF.c=rmF.c.times(CPBase.omega3[indx_F]);
854 rmF.normalize();
855
856 dmg.a=dmg.a.times(CPBase.omega3[indx_g]);
857 dmg.c=dmg.c.times(CPBase.omega3[indx_g]);
858 dmg.normalize();
859
860 rmG.a=rmG.a.times(CPBase.omega3[indx_G]);
861 rmG.c=rmG.c.times(CPBase.omega3[indx_G]);
862 rmG.normalize();
863
864 Mobius mf=(Mobius)rmF.rmultby(dmf.inverse());
865 mf.normalize();
866 Mobius mg=(Mobius)rmG.rmultby(dmg.inverse());
867 mg.normalize();
868 Mobius mgimf=(Mobius)mg.inverse().rmultby(mf);
869 mgimf.normalize();
870 schData.Schw_Deriv=mgimf.c;
871
872 if (debug && tmpdir!=null) {
873 try {
874 // baseMob's, adjusted for indices
875 fbw.write("\nAdjusted baseMob's:\n");
876 fbw.write("dmf="+dmf.toMatlabString()+"\n");
877 fbw.write("dmg="+dmg.toMatlabString()+"\n");
878 fbw.write("rmF="+rmF.toMatlabString()+"\n");
879 fbw.write("rmG="+rmG.toMatlabString()+"\n");
880 fbw.write("\n");
881
882 // maps f->F and g->G
883 fbw.write("mf="+mf.toMatlabString()+"\n");
884 fbw.write("mg="+mg.toMatlabString()+"\n");
885 fbw.write("\n");
886
887 fbw.flush();
888 fbw.close();
889 } catch(Exception ex) {
890 CirclePack.cpb.errMsg("failed initial output in 'schwarzian.java'");
891 }
892 }
893
894 return schData;
895 }

905 public static Double getIntrinsicSch(TriAspect tri_f,TriAspect tri_g) {
906 Mobius dmf=new Mobius(tri_f.setBaseMobius());
907 Mobius dmg=new Mobius(tri_g.setBaseMobius());
908 if (dmf==null || dmg==null)
909 return null;
910 boolean debug=false; // debug=true;
911
912 // Find shared edges
913 int indx_f=tri_f.nghb_Tri(tri_g);
914 int indx_g=tri_g.nghb_Tri(tri_f);
915
916 // modify dmf; pre-rotate base by omega[indx_f]
917 if (indx_f>0) {
918
919 dmf.a=dmf.a.times(CPBase.omega3[indx_f]);
920 dmf.c=dmf.c.times(CPBase.omega3[indx_f]);
921 }
922
923 // modify dmg; pre-rotate, then pre-compose
924 // by rotation about 1 (so map is from equilateral
925 // across edge through 1).
926 if (indx_g==0)
927 dmg=(Mobius)dmg.rmultby(gFix0);
928 else if (indx_g==1)
929 dmg=(Mobius)dmg.rmultby(gFix1);
930 else if (indx_g==2)
931 dmg=(Mobius)dmg.rmultby(gFix2);
932
933 if (debug) { // debug=true;
934 // draw rgb dots, red should be shared edge
935 tri_f.deBugHelp(dmf,true);
936 tri_g.deBugHelp(dmg,false);
937 }
938
939 // our definition uses dmg^{-1}.dmf
940 Mobius dom_sch=(Mobius)dmg.inverse().rmultby(dmf);
941 dom_sch.normalize();
942 Complex c=dom_sch.c;
943 if (Math.abs(c.y)>.001)
944 throw new MobException("intrinsic schwarzian has complex part");
945 return (Double)c.x;
946 }
