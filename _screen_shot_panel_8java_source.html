102 public boolean storeCPImage(PackData p) {
103 String num=String.format("%03d",imageCount);
104 String prefix=dirField.getText().trim();
105 if (prefix.startsWith("~")) {
106 try {
107 prefix=new String(CPFileManager.HomeDirectory.getCanonicalPath()+
108 prefix.substring(1));
109 } catch (Exception ex) {
110 }
111 }
112 String tmpName=new String(prefix+File.separator+nameField.getText().trim()
113 +"-"+num+"."+CPBase.IMG);
114 File locFile = new File(tmpName);
115 BufferedImage bI;
116 try {
117
118 // create any directories necessary
119 File canonFile=null;
120 try {
121 canonFile = locFile.getCanonicalFile();
122 String parent = canonFile.getParent();
123 if (parent!=null) {
124 File parentFile=new File(parent);
125 if (!parentFile.exists())
126 if (!parentFile.mkdirs())
127 throw new InOutException("Couldn't create directory '"+parent+"'");
128 }
129 } catch (IOException e) {
130 return false;
131 }
132 if (p!=null) // capture 'packImage' for this packing
133 bI=p.cpDrawing.packImage;
134 else { // capture Map frame 'pairPanel' of two packings
135 BufferedImage domPI=PackControl.mapPairFrame.getDomainCPS().packImage;
136 BufferedImage rangePI=PackControl.mapPairFrame.getRangeCPS().packImage;
137 int pairWidth=domPI.getWidth()+9+rangePI.getWidth(); // border: each end 2, mid 3
138 int pairHeight=domPI.getHeight()+6; // border: 2 top/bottom
139
140 // put in new BufferedImage
141 bI=new BufferedImage(pairWidth,pairHeight,BufferedImage.TYPE_INT_RGB);
142 Graphics2D g2=(Graphics2D)bI.getGraphics();
143
144 // domain
145 g2.drawImage(domPI,3,3,domPI.getWidth(),domPI.getHeight(),this);
146 // range
147 g2.drawImage(rangePI,domPI.getWidth()+6,3,
148 rangePI.getWidth(),rangePI.getHeight(),null);
149 }
150
151 ImageIO.write(bI,CPBase.IMG,locFile);
152 } catch (Exception ex) {
153 CirclePack.cpb.myErrorMsg("Screen dump to '"+tmpName+"' has failed.");
154 return false;
155 }
156 CirclePack.cpb.msg("Saved screendump to '"+tmpName+"'");
157 ThumbNail tnail=null;
158 try {
159 tnail=new ThumbNail(locFile,thumbMenu);
160
161 } catch (InOutException ine) {
162 CirclePack.cpb.myErrorMsg(ine.getMessage());
163 return false;
164 }
165 thumbNails.add(tnail);
166 thumbPanel.add(tnail);
167 scrollPane.revalidate();
168 imageCount++;
169 return true;
170 }
