138 public int close_psfile(CPdrawing cpd) {
139 if (!isOpen()) // writer to 'bodyFile' is not open
140 return 0;
141 try {
142 fp.flush();
143 fp.close(); // close 'bodyFile' (temp storage of main postscript)
144 } catch (Exception ex) {
145 throw new InOutException("problem closing postscript: "+ex.getMessage());
146 }
147 fp=null;
148 pF=null; // close 'PostFactory'
149 CommandStrParser.pF=null;
150
151 try {
152 BufferedWriter finalPS=new BufferedWriter(new FileWriter(psUltimateFile));
153
154 // preamble text
155 preAmble(finalPS,cpd);
156
157 // main body
158 File bodyFile = new File( System.getProperty("java.io.tmpdir"),
159 new String(File.separator+id+"_"+postFilename));
160 BufferedReader bodyFP=new BufferedReader(new FileReader(bodyFile));
161 int c;
162 while ((c=bodyFP.read())>=0)
163 finalPS.write(c);
164 bodyFP.close();
165
166 // trailer to close
167 if (PackControl.postManager.textBuffer==null ||
168 PackControl.postManager.textBuffer.length()==0) // no bottom text
169 finalPS.write("\ngrestore\nend\nshowpage\n"); // page wrapup
170 else {
171 finalPS.write("\ngrestore\n");
172 finalPS.write(PackControl.postManager.textBuffer.toString());
173 finalPS.write("\nend\nshowpage\n");
174 }
175 finalPS.flush();
176 finalPS.close();
177 } catch (Exception ex) {
178 throw new InOutException("closing postscript file: "+ex.getMessage());
179 }
180 postFilename=null;
181 textBuffer=null;
182 textLineCount=0;
183 return 1;
184 }

250 public int post_size_settings(BufferedWriter bw,CPdrawing cpDrawing,int mode)
251 throws IOException {
252 double size=(double)CPBase.DEFAULT_PS_PAGE_SIZE;
253 if((mode & 1)==1) { // BoundingBox
254 int bblx=(int)(72*(4.25-size/2))-10;
255 int bbly=(int)(72*(5.5-size/2))-10;
256 if (textBuffer!=null) // need to lower bottom for text
257 bbly -= textLineCount*18;
258 int bbrx=(int)(72*(4.25+size/2))+10;
259 int bbry=(int)(72*(5.5+size/2))+10;
260 bw.write("%%BoundingBox: "+bblx+" "+bbly+" "+bbrx+" "+bbry+"\n");
261 }
262 if ((mode & 2)==2) { // basic size info
263 bw.write("% ---------- pack size settings\n");
264 double factor= size/(cpDrawing.XWidth);
265 /* -------- scaling/translation for screen */
266 bw.write(" "+factor+" "+(double)((1.01)*factor)+" sc \n "+
267 (double)(-(cpDrawing.realBox.lz.x+cpDrawing.realBox.rz.x)/2.0)+" "+
268 (double)(-(cpDrawing.realBox.lz.y+cpDrawing.realBox.rz.y)/2.0)+" tr\n");
269 /* -------- define 'ourlinewidth' at 1 point */
270 bw.write(" /ourlinewidth\n { 72 div "+factor+" div "+
271 ps_linewidth+" mul setlinewidth} def\n");
272 /* -------- pre-defined thickness factors */
273 bw.write(" /onetk\n {1 ourlinewidth} def\n");
274 bw.write(" /twotk\n {2 ourlinewidth} def\n");
275 bw.write(" /threetk\n {3 ourlinewidth} def\n");
276 bw.write(" /fourtk\n {4 ourlinewidth} def\n");
277 bw.write(" /fivetk\n {5 ourlinewidth} def\n");
278 bw.write(" /sixtk\n {6 ourlinewidth} def\n");
279 bw.write(" /seventk\n {7 ourlinewidth} def\n");
280 bw.write(" /eighttk\n {8 ourlinewidth} def\n");
281 bw.write(" /ninetk\n {9 ourlinewidth} def\n");
282 bw.write(" /tentk\n {10 ourlinewidth} def\n");
283 bw.write(" "+(double)ps_linewidth+" ourlinewidth\n 0 sg\n");
284 bw.write("/Times-Roman ff "+(double)(.15/factor)+
285 " scf sf\n% ------------\n"); // scale font
286 }
287 if ((mode & 4)==4) /* clip window */
288 {
289 /* ----- now scaled based on screens size and desired output size */
290 double lx=cpDrawing.realBox.lz.x;
291 double ly=cpDrawing.realBox.lz.y;
292 double rx=cpDrawing.realBox.rz.x;
293 double ry=cpDrawing.realBox.rz.y;
294 bw.write("n\n"+lx+" "+ry+" m\n"+lx+" "+ly+" l\n"+rx+" "+ly+" l\n"+
295 rx+" "+ry+" l\ncp\n%gs s gr\nclip\nn\n"); // clip box
296 }
297 return 1;
298 }
