120 public static int[] markGenerations(PackDCEL pdcel,int[] seedstop) {
121 int gen[] = new int[pdcel.vertCount+1];
122 NodeLink v_curr=new NodeLink();
123 NodeLink v_next=new NodeLink();

125 // list/mark generation 1
126 for (int v=1;v<=pdcel.vertCount;v++) {
127 if (seedstop[v]>0) {
128 v_next.add(v);
129 gen[v]=1;
130 gen[0]++;
131 }
132 }
133
134 // no seeds given, find a first seed
135 if (v_next.size()==0) {
136 int base=0;
137 if (pdcel.alpha!=null) {
138 int alp=pdcel.alpha.origin.vertIndx;
139 if (seedstop[alp]<0) { // try for a nghb
140 int[] petals=pdcel.alpha.origin.getPetals();
141 for (int j=0;j<petals.length;j++) {
142 if (seedstop[petals[j]]==0)
143 base=petals[j];
144 }
145 }
146 if (seedstop[alp]==0)
147 base=alp;
148 }
149 if (base==0) { // else get any non-stop vertex
150 for (int v=1;(v<=pdcel.vertCount && base==0);v++)
151 if (seedstop[v]==0)
152 base=v;
153 }
154 if (base==0)
155 throw new CombException("no vertex to act as seed");
156
157 // list/mark this seed
158 v_next.add(base);
159 gen[base]=1;
160 gen[0]++;
161 }
162
163 int gcount=1;
164 while (v_next.size()>0) {
165 v_curr=v_next;
166 v_next=new NodeLink();
167 gcount++; // keep track of generation in 'gen'
168 Iterator<Integer> vis=v_curr.iterator();
169 while (vis.hasNext()) {
170 int v=vis.next();
171 int[] petals=pdcel.vertices[v].getPetals();
172 for (int j=0;j<petals.length;j++) {
173 int w=petals[j];
174 if (seedstop[w]<0)
175 gen[w]=-gcount;
176 else if (seedstop[w]==0 && gen[w]==0) {
177 gen[w]=gcount;
178 gen[0]++;
179 v_next.add(w);
180 }
181 }
182 }
183 }
184 return gen;

850 public static int enfold_raw(PackDCEL pdcel,int v) {
851 if (!pdcel.vertices[v].isBdry() || pdcel.countPetals(v)<=2)
852 throw new CombException("can't enfold v = "+v+" with just 2 nghbs");
853
854 // key players
855 Vertex vert=pdcel.vertices[v];
856 HalfEdge out_edge=vert.halfedge;
857 HalfEdge in_edge=out_edge.twin.next.twin;
858 HalfEdge bdryedge=out_edge.twin;
859 HalfEdge twin_prev=bdryedge.prev;
860 HalfEdge twin_next=in_edge.twin.next;
861 DcelFace ideal=bdryedge.face;
862
863 // check if bdry component has just two edges:
864 // toss 'oldedge', reconnect, and toss the red chain.
865 if (bdryedge.next.next==bdryedge) {
866
867 Vertex wert=bdryedge.origin;
868 if (wert.halfedge==in_edge)
869 wert.halfedge=bdryedge;

871 // save connections so we can orphan 'in_edge'
872 HalfEdge oldnxt=in_edge.next;
873 HalfEdge oldprev=in_edge.prev;
874
875 bdryedge.next=oldnxt;
876 oldnxt.prev=bdryedge;
877
878 bdryedge.prev=oldprev;
879 oldprev.next=bdryedge;
880
881 bdryedge.face=in_edge.face;

883 vert.bdryFlag=wert.bdryFlag=0;
884 vert.redFlag=wert.redFlag=false;
885 vert.aim=wert.aim=2.0*Math.PI;
886 pdcel.redChain=null;
887 pdcel.triData=null;
888 return 2;
889 }
890 // check if the bdry has just three edges: create
891 // a new common face, toss red chain
892 if (bdryedge.next.next.next==bdryedge) {
893 DcelFace dface=new DcelFace();
894 Vertex wert=bdryedge.origin;
895 Vertex uert=bdryedge.prev.origin;
896 bdryedge.face=dface;
897 bdryedge.face.edge=bdryedge;
898 bdryedge.next.face=dface;
899 bdryedge.next.next.face=dface;

901 out_edge.next.origin.redFlag=false;

903 vert.bdryFlag=wert.bdryFlag=uert.bdryFlag=0;
904 vert.redFlag=wert.redFlag=uert.redFlag=false;
905 pdcel.redChain=null;
906 pdcel.triData=null;
907 return 3;
908 }
909
910 // new edge and twin
911 HalfEdge new_edge=new HalfEdge(in_edge.origin);
912 HalfEdge new_twin=new HalfEdge(out_edge.twin.origin);
913
914 new_edge.twin=new_twin;
915 new_twin.twin=new_edge;
916
917 new_edge.face=null; // don't create new face
918 new_twin.face=ideal;
919 if (ideal.edge==out_edge.twin || ideal.edge==in_edge.twin)
920 ideal.edge=new_twin;
921 out_edge.twin.face=null;
922 in_edge.twin.face=null;
923
924 // new red
925 RedEdge new_red=new RedEdge(new_edge);
926 new_edge.myRedEdge=new_red;
927
928 // relink things
929 in_edge.twin.next=new_edge;
930 new_edge.prev=in_edge.twin;
931
932 new_edge.next=out_edge.twin;
933 out_edge.twin.prev=new_edge;
934
935 new_twin.prev=twin_prev;
936 twin_prev.next=new_twin;
937
938 new_twin.next=twin_next;
939 twin_next.prev=new_twin;
940
941 if (pdcel.redChain==out_edge.myRedEdge || pdcel.redChain==in_edge.myRedEdge)
942 pdcel.redChain=out_edge.myRedEdge.nextRed;
943
944 // simple red chain case
945 if (in_edge.myRedEdge.nextRed==out_edge.myRedEdge) {
946 vert.redFlag=false;
947
948 new_red.prevRed=in_edge.myRedEdge.prevRed;
949 in_edge.myRedEdge.prevRed.nextRed=new_red;
950
951 new_red.nextRed=out_edge.myRedEdge.nextRed;
952 out_edge.myRedEdge.nextRed.prevRed=new_red;
953
954 in_edge.myRedEdge=null;
955 out_edge.myRedEdge=null;
956
957 vert.redFlag=false;
958 }
959 // otherwise there are one or more intervening red chains
960 else {
961 RedEdge red=new RedEdge(out_edge.twin);
962
963 new_red.prevRed=in_edge.myRedEdge.prevRed;
964 in_edge.myRedEdge.prevRed.nextRed=new_red;
965
966 new_red.nextRed=red;
967 red.prevRed=new_red;
968
969 red.nextRed=in_edge.myRedEdge.nextRed;
970 in_edge.myRedEdge.nextRed.prevRed=red;
971
972 red.twinRed=out_edge.myRedEdge;
973 out_edge.myRedEdge.twinRed=red;
974
975 in_edge.myRedEdge=null;
976 }
977
978 vert.bdryFlag=0;
979 vert.aim=2.0*Math.PI;
980 new_edge.origin.halfedge=new_edge;
981 pdcel.triData=null;
982 return pdcel.countPetals(vert.vertIndx);

1002 HalfEdge edge,boolean multi_bary) {

1005 pdcel.redChain=null;
1006 ideal=true;

1008
1009 // make room
1010 int node=pdcel.vertCount+1; // new index

1012 pdcel.alloc_vert_space(node+10,true);

1015 if (hlink==null || hlink.size()<=2)
1016 return 0;

1019 Vertex newV=new Vertex(node); // this is the barycenter

1071 newV.bdryFlag=0;

1073 pdcel.vertices[node]=newV;
1074 pdcel.vertCount=node;

1076 pdcel.triData=null;
