754 public int brptExists(int j,int type) {
755 if(branchPts==null || branchPts.size()<=1)
756 return -1;
757 Iterator<GenBrModPt> bps=branchPts.iterator();
758 bps.next(); // flush first null entry
759 while (bps.hasNext()) {
760 GenBrModPt gbp=bps.next();
761 if (gbp!=null && gbp.myType==type) {
762 if (type==GenBrModPt.SINGULAR) {
763 combinatorics.komplex.DcelFace face=extenderPD.packDCEL.faces[j];
764 HalfEdge he=face.edge;
765 do {
766 if (he==gbp.myEdge) {
767 return gbp.branchID;
768 }
769 he=he.next;
770 } while(he!=face.edge);
771 return -1;
772 }
773 else if (type==GenBrModPt.CHAPERONE) {
774 if (gbp.myEdge.origin.vertIndx==j)
775 return gbp.branchID;
776 return -1;
777 }
778 else if (type==GenBrModPt.QUADFACE) {
779 return -1; // not yet implemented
780 }
781 }
782 }
783 return -1;
784 }

937 public double []getClickData(PackData refPack, Complex pt) {
938 if (pt==null)
939 return null;
940 double []ans=null;
941
942 // first check if pt is inside a circle
943 int v=NodeLink.grab_one_vert(refPack,"z "+pt.x+" "+pt.y);
944
945 // not in circle, try interstice
946 if (v<=0) {
947 // barycentric coords relative to angles,
948 // see 'HyperbolicMath.ideal_bary'
949 BaryPoint bpt=BaryLink.grab_one_barypoint(refPack,
950 new String("-i "+pt.x+" "+pt.y));
951 if (bpt!=null) {
952 ans=new double[4];
953 ans[0]=1.0;
954 ans[1]=(double)bpt.face;
955 ans[2]=bpt.b0;
956 ans[3]=bpt.b1;
957 return ans;
958 }
959 return null; // nope
960 }
961
962 // OK, now we are in a circle.
963 if (refPack.isBdry(v)) { // can't yet branch at boundary circle
964 CirclePack.cpb.errMsg("Can't yet branch at boundary vertex v="+v);
965 throw new DataException(v+" is a bdry vertex");
966 }
967
968 // Key variables are R, r, and shadow.
969 int num=refPack.countFaces(v);
970 int[] flower=refPack.getFlower(v);
971 double R=refPack.getRadius(v);
972 Complex cent=refPack.getCenter(v);
973 Complex cent2pt=pt.minus(cent);
974 double r=R-cent2pt.abs(); // ideally, radius of smaller sister when packed
975 double c2pt_arg=cent2pt.arg();
976 if (r<0) // error: not in the right circle
977 throw new DataException("hum...?, selected point is not in circle "+v);
978
979 // close to center of v? use traditional branch point
980 // TODO: need to experiment to find best 'dcutoff'
981 double dcutoff=.025;
982 if ((R-r)<dcutoff*R) {
983 ans=new double[2];
984 ans[0]=3.0; // traditional
985 ans[1]=(double)v;
986 return ans;
987 }
988
989 // else prepare return data for circle case
990 ans=new double[6];
991 ans[0]=2.0; // circle
992 ans[1]=(double)v; // circle index
993
994 // closed list the arguments of edges from v to petals v_j
995
996 // TODO: If nearOne>=0, pt is very close to a tangency point,
997 // may want to use singular branching or some special routine.
998// int nearOne=-1;
999 double []petalAngs=new double[num+1]; // closed cycle of angles to petals
1000 for (int j=0;j<num;j++) {
1001 Complex c2pet=refPack.getCenter(flower[j]).minus(cent);
1002 petalAngs[j]=(c2pet.arg()+m2pi)%m2pi;
1003// c2pet=c2pet.times(R/c2pet.abs());
1004// if (c2pet.minus(cent2pt).abs()<.025*R) {
1005// nearOne=flower[j];
1006// }
1007 }
1008 petalAngs[num]=petalAngs[0]; // close up
1009
1010 // "click" data for a point on the circle is a double c = J + F,
1011 // where the integer part J is the index of the petal
1012 // containing the point, and the F is the cclw fraction of
1013 // angle the point occupies in that petal.
1014 // We get the upstream and downstream max and min data we can use
1015 ClickModValue centClick=
1016 petalClicks(c2pt_arg,petalAngs); // click data for pt itself
1017 ClickModValue maxUpClick=
1018 petalClicks(c2pt_arg-CPBase.piby2,petalAngs); // pi/2 clw
1019 ClickModValue maxDownClick=
1020 petalClicks(c2pt_arg+CPBase.piby2,petalAngs); // pi/2 cclw
1021
1022 // move one petal upstream, one petal downstream
1023 ClickModValue minUpClick=
1024 new ClickModValue((centClick.petal-1.0+num)%num+
1025 centClick.fraction);
1026 ClickModValue minDownClick=
1027 new ClickModValue((centClick.petal+1.0)%num+
1028 centClick.fraction);
1029
1030 // get the 'shadow' cast by pt; this is half the angular
1031 // arc subtended by a circle through pt, perpendicular to
1032 // 'cent2pt' and hitting the circle of radius R at right
1033 // angles (i.e., hyp geodesic, vis-a-vis this disc).
1034 // 'shadow' is in [0,pi/2].
1035 double shadow=HyperbolicMath.shadow_angle((R-r)/R); // geodesic half-length
1036
1037 // TODO: CAUTION, not at all sure about this (4/2016)
1038 // Use transform T to map real click data for 'pt' and 'shadow'
1039 // monotonically to virtual click data used to find jumps and overlaps.
1040 // On the upstream side: T: [0,pi/2] --> [minright, maxright].
1041 // On the downstream side: T: [0,pi/2] --> [minleft, maxleft].
1042 ClickModValue vertualUp=
1043 clickTransform(shadow,minUpClick,maxUpClick,num);
1044 ClickModValue vertualDown=
1045 clickTransform(shadow,minDownClick,maxDownClick,num);
1046
1047 // jump and overlap depend on upstream or downstream
1048 int upj=(int)((vertualUp.petal+1)%num);
1049 ans[2]=flower[upj];
1050 ans[4]=1.0-vertualUp.fraction;
1051
1052 int downj=(int)(vertualDown.petal+1)%num;
1053 ans[3]=flower[downj];
1054 ans[5]=vertualDown.fraction;
1055
1056 return ans;
1057 }

1076

1080 double result=((double)fcount+fracdiff)*comp;

1122 public double pseudoAngle(double myang,double []angs) {
1123 double pi2=2.0*Math.PI;
1124
1125 // make sure myang is in [0,2pi].
1126 while (myang<0)
1127 myang +=pi2;
1128 while (myang>pi2)
1129 myang -=pi2;
1130
1131 int k=angs.length;
1132 double accumAng=0.0;
1133 int tick=0;
1134 double incr=0.0;
1135 while (tick<k && myang>accumAng) {
1136 incr=angs[tick];
1137 if (myang<=(accumAng+incr))
1138 return ((double)tick+(myang-accumAng)/incr)*Math.PI;
1139 tick++;
1140 accumAng +=incr;
1141 }
1142
1143 return ((double)k)*Math.PI; // myang should ~ full anglesum
1144 }
