53 public static Path2D.Double readpath(File dir,String filename,boolean script_flag) {
54 Path2D.Double gpath=null;
55 BufferedReader fp=null;
56
57 // look in script
58 if (script_flag) {
59 fp=CPFileManager.openReadFP(null,filename,script_flag);
60 if (fp==null)
61 return null;
62 }
63
64 // given directory
65 else if (dir!=null) {
66 fp=CPFileManager.openReadFP(dir,filename,script_flag);
67 }
68
69 // check default locations
70 if (fp==null && (fp=CPFileManager.openReadFP(CPFileManager.PackingDirectory,filename,script_flag))==null &&
71 (fp=CPFileManager.openReadFP(CPFileManager.CurrentDirectory,filename,script_flag))==null) {
72 return null;
73 }
74
75 String line;
76
77 // Must find 'PATH:' before reading further
78 PathReadState state=PathReadState.LOOK_FOR_PATH;
79 try {
80 while(state==PathReadState.LOOK_FOR_PATH
81 && (line=StringUtil.ourNextLine(fp))!=null) {
82 StringTokenizer tok = new StringTokenizer(line);
83 while(tok.hasMoreTokens()) {
84 String mainTok = tok.nextToken();
85 if(mainTok.startsWith("PATH")) {
86 gpath=new Path2D.Double();
87 state=PathReadState.START_SEG;
88 }
89 }
90 } // end of while for 'PATH'
91
92 // now expect next non-empty line to have two doubles
93 while (state==PathReadState.START_SEG
94 && ((line=StringUtil.ourNextLine(fp))!=null)) {
95 try {
96 StringTokenizer tok = new StringTokenizer(line);
97 double x=Double.parseDouble((String)tok.nextToken());
98 double y=Double.parseDouble((String)tok.nextToken());
99 gpath.moveTo(x,y);
100 state=PathReadState.READ_XY;
101 } catch (Exception ex) {
102 CirclePack.cpb.myErrorMsg("Exception in reading path from '"+filename+"'.");
103 return null;
104 }
105
106 while(state==PathReadState.READ_XY
107 && (line=StringUtil.ourNextLine(fp))!=null) {
108 StringTokenizer tok = new StringTokenizer(line);
109 String xstr=null;
110 // expect two doubles per line
111 try {
112 xstr=(String)tok.nextToken();
113 double x=Double.parseDouble(xstr);
114 double y=Double.parseDouble((String)tok.nextToken());
115 gpath.lineTo(x,y);
116 } catch (NumberFormatException nfe) {
117 if (xstr.contains("BREAK")) { // indicates a new component of path
118 gpath.closePath();
119 state=PathReadState.START_SEG;
120 }
121 else state=PathReadState.CLOSE;
122 }
123 }
124 }
125 if (state==PathReadState.CLOSE) {
126 gpath.closePath();
127 return gpath;
128 }
129 else return null;
130 } catch (Exception ex) {
131 CirclePack.cpb.myErrorMsg("Exception in reading path from '"+filename+"'.");
132 return null;
133 }

144 throws IOException {
145 if (fp==null) return 0;

149 Vector<Vector<Complex>> polyGamma = PathUtil.gpPolygon(gpath,
