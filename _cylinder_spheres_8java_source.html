560 public double[] computeRz(int p,int q) {
561 double R;
562 double z1;
563
564 // set R between min/max possible, but not less than 0.5
565 R=(p+q)/(3*Math.PI);
566 if (R<0.5)
567 R=0.5;
568 if (p<q) {
569 int hold=p;
570 p=q;
571 q=hold;
572 }
573
574 // special situations
575 if (p==q) {
576 double[] ans=new double[4];
577 ans[0]=Math.sqrt(3.0/(8.0*(1.0-Math.cos(2.0*Math.PI/(2.0*p)))));
578 ans[1]=-0.5;
579 ans[2]=0;
580 ans[3]=0;
581 return ans;
582 }
583 if (q==0) {
584 double[] ans=new double[4];
585 ans[0]=Math.sqrt(1.0/(2.0*(1.0-Math.cos(2.0*Math.PI/p))));
586 ans[1]=0;
587 ans[2]=0;
588 ans[3]=0;
589 return ans;
590 }
591
592 z1=what_is_z1(R,p,q,-1.0); // get started
593 double t1=what_is_t(R,z1);
594 Complex s1=computeS2(R,z1);
595 double t_error=Math.abs(p*t1+q*s1.x-R*CPBase.pi2);
596 double z_error=Math.abs(p*z1+q*s1.y);
597 int safety=1000;
598 while ((t_error+z_error)>.000001 && safety>0) {
599 safety=safety-1;
600 R=newtees(R,z1,p,q);
601 z1=what_is_z1(R,p,q,z1);
602 t1=what_is_t(R,z1);
603 Complex s2=computeS2(R,z1);
604 t_error=Math.abs(p*t1+q*s2.x-R*CPBase.pi2);
605 z_error=Math.abs(p*z1+q*s2.y);
606 }
607 double[] ans=new double[4];
608 if (safety==0)
609 CirclePack.cpb.errMsg("safetied out in 'computeRz'");
610 ans[0]=R;
611 ans[1]=z1;
612 ans[2]=t_error;
613 ans[3]=z_error;
614 return ans;
615 }

626 public Complex computeS2(double R,double z1) {
627 z1=-1*Math.abs(z1); // z1 should lie in (-ZR,0].
628 double ZR=what_is_Z_R(R);
629 if (z1<-1.0*ZR)
630 z1=-1.0*ZR;
631 double min_z=0;
632 double max_z=ZR;
633 double z2=min_z;
634 double best=S2_distance(R,z1,z2);
635 int safety=1000;
636 // TODO: improve using Newton method
637 while (Math.abs(best-1.0)>0.0000000000001 && safety>0) {
638 safety=safety--;
639 z2=(min_z+max_z)/2.0;
640 best=S2_distance(R,z1,z2);
641 if (best<1)
642 min_z=z2;
643 else if (best>1)
644 max_z=z2;
645 }
646 if (safety==0) {
647 if (Math.abs(best-1.0)>.001) {
648 CirclePack.cpb.errMsg("safety error in computeS2");
649 return null;
650 }
651 }
652 double t2=R*Math.acos(1+(z2*z2-1)/(2*R*R));
653 return new Complex(t2,z2);
654 }

664 public double newtees(double RR,double zz1,int p,int q) {
665 double R=RR;
666 double z1=zz1;
667 double t1=what_is_t(R,z1);
668 Complex s2=computeS2(R,z1);
669 double value=p*t1+q*s2.x-2.0*Math.PI*R; // want this to be zero.
670 int safety=1000;
671 while (Math.abs(value)>.000000001 && safety>0) {
672 safety--;
673 double dt1=der_tz(R,z1);
674 double dt2=der_tz(R,s2.y);
675 double dvdR=p*dt1+q*dt2-2*Math.PI; // derivative or value wrt R
676 R=R-value/dvdR;
677 t1=what_is_t(R,z1);
678 s2=computeS2(R,z1);
679 value=p*t1+q*s2.x-2.0*Math.PI*R;
680 }
681 if (safety==0)
682 CirclePack.cpb.errMsg("safetied out in 'newtees'");
683 return R;
684 }

694 public Point3D faceBarycenter(double R,HalfEdge edge) {
695 Complex a=edge.origin.center;
696 Complex b=edge.next.origin.center;
697 Complex c=edge.next.next.origin.center;
698 Point3D a3D=set3D(R,a);
699 Point3D b3D=set3D(R,b);
700 Point3D c3D=set3D(R,c);
701 double cx=(a3D.x+b3D.x+c3D.x)/3.0;
702 double cy=(a3D.y+b3D.y+c3D.y)/3.0;
703 double cz=(a3D.z+b3D.z+b3D.z)/3.0;
704 return new Point3D(cx,cy,cz);
705 }

715 public double S2_distance(double R,double z1,double x) {
716 // (a) Find c1:
717 double c1=1.0+(z1*z1-1.0)/(2.0*R*R);
718 double s1=Math.sqrt(1.0-c1*c1);
719 // (b) Find c2 as a function of x
720 double c2=1.0+(x*x-1)/(2*R*R);
721 double s2=Math.sqrt(1.0-c2*c2);
722 double d=(x-z1)*(x-z1)-(2*R*R)*(c1*c2+s1*s2-1.0);
723 return Math.sqrt(d);
724 }

733 public double der_tz(double R,double z) {
734 double y=(1.0-z*z);
735 return -2.0*R*z/(Math.sqrt(y)*Math.sqrt(4.0*R*R-y));
736 }

746 public double what_is_z1(double R,int p,int q,double z1) {
747 if (q==0)
748 return 0;
749
750 double max_z1=0;
751 double ZR=what_is_Z_R(R);
752 double min_z1=-ZR;
753 if (z1<-.9)
754 z1=min_z1;
755 double z_out=z1;
756 Complex s2=computeS2(R,z1);
757 double p1q2=p*z_out+q*s2.y;
758 int safety=1000;
759 while (Math.abs(p1q2)>.00000001) {
