91 public int set_sp_Mobius() {
92 if (mateIndex<0)
93 return 0; // this is a "border" side
94 Complex A=startEdge.getCenter();
95 Complex a=startEdge.twinRed.nextRed.getCenter();
96 Complex B=endEdge.nextRed.getCenter();
97 Complex b=endEdge.twinRed.getCenter();
98
99 mob=new Mobius(); // start with identity
100 mobErr=0.0;
101
102 /* Now have everything for mobius transformation. (The mobius
103 * for 'pairedEdge' is set separately (should be inverse to this)). */
104 if (hes<0) { // hyp
105 if (a.abs()>=Mobius.MOD1 && b.abs()>=Mobius.MOD1
106 && A.abs()>=Mobius.MOD1 && B.abs()>=Mobius.MOD1) {
107 /* We have another real degree of freedom, so we go to a face
108 * midway along this paired side and match its centers.*/
109
110 int count=0;
111 RedEdge etrace=startEdge;
112 while (etrace!=endEdge) {
113 count++;
114 etrace=etrace.nextRed;
115 }
116
117 if (count<2) // side is too short
118 return 0;
119 // move to center edge-pairing and use ends
120 etrace=startEdge;
121 for (int i=0;i<=(int)(count/2);i++)
122 etrace=etrace.nextRed;
123 Complex C=etrace.getCenter();
124 Complex c=etrace.twinRed.nextRed.getCenter();
125
126 if (c.abs()>=Mobius.MOD1 || C.abs()>=Mobius.MOD1)
127 return 0; // didn't get matching centers or one/both these
128 // centers were too close to unit circle.
129
130 mob=Mobius.trans_abAB(a,b,A,B,c,C);
131 mobErr=mob.error;
132 }
133 else if (a.abs()>=Mobius.MOD1 || b.abs()>=Mobius.MOD1
134 || A.abs()>=Mobius.MOD1 || B.abs()>=Mobius.MOD1)
135 return 0; /* fixup: don't have routine to handle some of points
136 * "on" unit circle and others off. Also,
137 * may have too much error for either trans_abAB
138 * or auto_abAB. */
139 else mob=Mobius.auto_abAB(a,b,A,B);
140 mobErr=mob.error;
141 }
142 else if (hes==0) { // eucl
143 mob=Mobius.mob_abAB(a,b,A,B);
144 mobErr=mob.error;
145 }
146 // TODO: sph not yet implemented, but may be set elsewhere,
147 // e.g., under projection to the sphere.
148 else return 0;
149 return 1;
150 }
