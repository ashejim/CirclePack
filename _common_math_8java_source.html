56 public static int normAlphaGamma(PackData p,int a,int g) {
57 Mobius mob=new Mobius();
58 Vertex alpvert=p.packDCEL.vertices[a];
59 if (alpvert.isBdry())

90 public static int placeOneFace(CircleSimple sC0,CircleSimple sC1,
91 CircleSimple sC2,
92 double[] invDist,int hes) {
93
94 // first is always at origin
95 sC0.center=new Complex(0.0);
96
97 if (hes<0) { // hyp case
98 double x0=sC0.rad;
99 if (x0<=0) { // horocycle? can't handle this
100 CirclePack.cpb.errMsg("Can't place hyp triple: first is horocycle");
101 return 0;
102 }
103
104 // next circle on positive axis
105 double x1=sC1.rad;
106 if (x1<=0) { // horocycle?
107 double R=HyperbolicMath.h_horo_rad(x1,invDist[2]);
108 sC1.center=new Complex(1.0);
109 sC1.rad=-R; // negative of euclidean radius
110 }
111 else {
112 double h=HyperbolicMath.h_ivd_length(x0, x1, invDist[2]);
113 double e=Math.exp(h);
114 sC1.center=new Complex((e-1)/(e+1));
115 }
116 }
117 else if (hes>0) { // sphere case
118 // next out pos x-axis
119 double sdist=SphericalMath.s_ivd_length(sC0.rad,sC1.rad,invDist[2]);
120 sC1.center = new Complex(0.0,sdist);
121 }
122 else { // eucl case
123 // next on positive x-axis
124 double r=sC0.rad;
125 double r2=sC1.rad;
126 sC1.center=new Complex(Math.sqrt(r*r+r2*r2+2*r*r2*invDist[2]));
127 }
128
129 CircleSimple sC=new CircleSimple();
130 sC=CommonMath.comp_any_center(sC0.center, sC1.center,sC0.rad,sC1.rad,
131 sC2.rad, invDist[0],invDist[1],invDist[2],hes);
132 sC2.center=sC.center;
133 return sC.flag; // in case there's an error
134 }

445 // TODO: improve notion of "generalized tangency" to
446 // get conformal invariant definition and
447 // tangency in (essentially) tangent case.
448 // Note: sph case can go wrong if one or both circles
449 // contain infinity.
450 /* As of 6/2021:
451 * Work with eucl circles c1=(z1,r1), c2=(z2,r2).
452 * The GT point will be located on the ray L from
453 * z1 through z2, so we work with real number
454 * distances on L.
455 *
456 * First get inversive distance between c1/c2.
457 * invD=(|z1-z2|^2-(r1^2+r2^2))/(2*r1*r2).
458 * (this may or may not correspond with "intended"
459 * inversive distance).
460 *
461 * We identify the c1/c2 configuration with a
462 * normalized configuration C-/C+ that is a
463 * mobius image: namely, based on inD alone,
464 * can compute radius R and center circles
465 * C-, C+ of radius R on the negative/positive
466 * real axis (resp.) the left edge of C- hits -1
467 * and the right edge of C+ hits +1.
468 *
469 * Here's R:
470 * R=(-2+sqrt(2+2*invD))/(invD-1)
471 *
472 * Triples of points to be identified are {A,B,C}
473 * on L and {a,b,c} in the normalized situation.
474 *
475 * For {A,B,C}, let x = |z1-z2|. Then A = x-r2,
476 * B=r1, C=x+r2. (These are points along L)
477 * A is the distance along L from z1 to the first
478 * intersection point of L and c2, B is intersection
479 * point of L and c1, and C is the further intersection
480 * point of L and c2. (Note that B and C are positive,
481 * but A may be negative in case of deep overlap.)
482 *
483 * The corresponding points {a,b,c} of the real line
484 * are: a is the left edge of C+, b is the right edge
485 * of C-, and c is +1. Thus, a = 1-2R, b = -1+2R, and
486 * c = 1.
487 *
488 * If T is the Mobius map: T{a,b,c} --> {A,B,C}, then
489 * by obvious symmetry about the y-axis in the
490 * normalized situation, T(0) is what we are after.
491 * Return the point along L which is distance T(0)
492 * from z1 in the direction of z2.
493 *
494 * @param cs1 CircleSimple
495 * @param cs2 CircleSimple
496 * @param hes int
497 * @return new Complex
498 */
499 public static Complex genTangPoint(CircleSimple cs1,
500 CircleSimple cs2,int hes) {
501
502 // convert to eucl circles
503 if (hes<0) {
504 cs1=HyperbolicMath.h_to_e_data(cs1);
505 cs2=HyperbolicMath.h_to_e_data(cs2);
506 }
507 else if (hes>0) {
508 cs1=SphericalMath.s_to_e_data(cs1);
509 cs2=SphericalMath.s_to_e_data(cs2);
510 }
511
512 double x=cs1.center.minus(cs2.center).abs();
513 double t=cs1.rad; // prepare for tangency
514
515 double invD=(x*x-(cs1.rad*cs1.rad+cs2.rad*cs2.rad))/(2*cs1.rad*cs2.rad);
516
517 // if not essentially tangent, find real value t
518 if (Math.abs(invD-1.0)>.001) {
519
520 // points on L
521 Complex A=new Complex(x-cs2.rad);
522 Complex B=new Complex(cs1.rad);
523 Complex C=new Complex(x+cs2.rad);
524
525 // radius and points for normalized config
526 double R=(-2.0+Math.sqrt(2.0+2.0*invD))/(invD-1.0);
527 Complex a=new Complex(1.0-2.0*R);
528 Complex b=new Complex(-1+2.0*R);
529 Complex c=new Complex(1.0);
530
531 Mobius mob=Mobius.mob_xyzXYZ(a,b,c,A,B,C,0,0);
532 t=mob.apply(new Complex(0.0)).x;
533 }
534
535 // have t, find point distance t along L from cs1.center
536 Complex pt=cs2.center.minus(cs1.center).times(t/x).add(cs1.center);
537 if (hes>0)
538 return SphericalMath.proj_pt_to_sph(pt);
539 return pt;
540 }
