109 public int compute() {
110
111 // initialize
112 int[] hits=new int[N+1]; // mark when computed
113
114 // want to track constraints to see how to
115 // adjust for branching: recursion for j<=N-3;
116 // straints[j+1]=sqrt3*uzian[j]*straints[j]-straints[j-1],
117 double[] straints=new double[N+1];
118 straints[0]=0.0;
119 straints[1]=1.0;
120 straints[2]=sqrt3*uzian[1];
121
122 t[1]=0.0;
123 radius[1]=radius[N-1]=recipSqRad[1]=recipSqRad[N-1]=1.0;
124 hits[1]=1; // c_1 is in place to start
125 branchDeg=0; // increment for each branching event
126
127 // compute data for c_2
128 if (N>3) {
129 double[] tdata=Sit2(uzian[1]);
130 t[2]=t[1]+tdata[0];
131 radius[2]=1/(tdata[1]*tdata[1]);
132 recipSqRad[2]=tdata[1];
133 hits[2]=1;
134 }
135
136 // compute as generic up to and including N-2;
137 // watch for dspmt very large, negative, or 0, which
138 // means there's branching.
139 double[] tdata=null;
140 double dspmt;
141 for (int j=3;j<=(N-2);j++) {
142 if (recipSqRad[j-1]==0) { // get a halfplane.
143 if (j==N-2) {
144 CirclePack.cpb.errMsg("Can't complete a flower "+
145 "since c_{n-2} is a halfplane");
146 return 0;
147 }
148 double r=1/(recipSqRad[j-2]*recipSqRad[j-2]); // same rad as previous
149 t[j]=t[j-2]-2.0*uzian[j]*sqrt3*r; // but to left of c(j-1)
150 recipSqRad[j]=recipSqRad[j-2];
151 straints[j]=straints[j-2];
152 hits[j]=1;
153 }
154 else if (recipSqRad[j-1]<0) { // branching event
155 tdata=Sit4(uzian[j-1],recipSqRad[j-2],recipSqRad[j-1]);
156 t[j]=t[j-1]+tdata[0];
157 recipSqRad[j]=tdata[1];
158 radius[j]=1/(tdata[1]*tdata[1]);
159 straints[j]=(-1.0)*(sqrt3*uzian[j-1]*straints[j-1]-straints[j-2]);
160 branchDeg++;
161 hits[j]=1;
162 }
163 else {
164 tdata=Sit3(uzian[j-1],recipSqRad[j-2],recipSqRad[j-1]);
165 dspmt=tdata[0];
166 recipSqRad[j]=tdata[1];
167 t[j]=t[j-1]+dspmt;
168 radius[j]=1/(tdata[1]*tdata[1]);
169 straints[j]=sqrt3*uzian[j-1]*straints[j-1]-straints[j-2];
170 hits[j]=1;
171 }
172 }
173
174 // now, compute the last three uzians
175 if (N==3) {
176 uzian[N-2]=uzian[N-1]=uzian[N]=oosq3;
177 return 1;
178 }
179
180 // Now mandate c(n-1)= 1, compute last data
181 radius[N-1]=1.0;
182 double[] lastdata=compLast(t[N-2],radius[N-3],radius[N-2]);
183 t[N-1]=lastdata[0];
184 uzian[N-2]=lastdata[1];
185 uzian[N-1]=lastdata[2];
186 uzian[N]=lastdata[3];
187
188// debugging
189// alternate computation of u_{n-2}
190// double unm2=(1+straints[N-3])/(sqrt3*straints[N-2]);
191// System.out.println("u_{n-2} by 'compLast' is "+uzian[N-2]+
192// "; and with constraints is "+unm2);
193
194 if (uzian[N]<0)
195 throw new DataException("Flower illigitimate as s_{N}>1.");
196
197 if (branchDeg==0)
198 return 1;
199 else
200 return -branchDeg;
201 }

322 public static ArrayList<Double> constraints(double[] uz,Boolean allpos) {
323 allpos=true;
324 int N=uz.length-1;
325 if (N<4)
326 return null;
327
328 ArrayList<Double> cons=new ArrayList<Double>(0);
329 cons.add(0,Double.valueOf(0.0));
330 cons.add(1,Double.valueOf(1.0));
331 double Cj=1.0;
332 double Cjm1=0.0;
333 // recursively define C_{j+1} for j=1,...,(n-3)
334 // keep going even if a negative is encountered
335 for (int j=1;j<=(N-3);j++) {
336 double cjp1=sqrt3*uz[j]*Cj-Cjm1;
337 if (cjp1<=0.0) {
338 allpos=false;
339 cjp1 *=-1.0;
340 }
341 cons.add(Double.valueOf(cjp1));
342 Cjm1=Cj;
343 Cj=cjp1;
344 }
345 cons.add(N-1,Double.valueOf(1.0));
346 return cons;
347 }
