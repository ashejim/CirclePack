42 public static int processQuery(PackData p,
43 Vector<String> queryStr,boolean forMsg) {
44 StringBuilder strbld=new StringBuilder();
45 for (int j=0;j<queryStr.size();j++)
46 strbld.append((String)queryStr.get(j)+" ");
47
48
49 // remove the '?'
50 strbld.deleteCharAt(0);
51 while (strbld.length()>0 && strbld.charAt(0)==' ')
52 strbld.deleteCharAt(0);
53 if (strbld.length()==0)
54 return 0;
55
56 // split off 'query': up to first " "
57 int k=strbld.indexOf(" ");
58 String result=null;
59 if (k>0) {
60 String query=strbld.substring(0,k);
61 strbld.delete(0,k);
62
63 // break rest into usual flag segments (though generally no flags)
64 Vector<Vector<String>> flagSegs=StringUtil.flagSeg(strbld.toString());
65 result=queryParse(p,query,flagSegs,forMsg);
66 }
67 else
68 result=queryParse(p,strbld.toString(),(Vector<Vector<String>>)null,forMsg);
69
70 if (result==null)
71 return 0;
72 CirclePack.cpb.msg(result);
73 return 1;
74 }
