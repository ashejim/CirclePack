283 Iterator<HalfEdge> sis=spokes.iterator();

329 v2parent[n]=n2o;

331 }

336 for (int n=1;n<=vertCount;n++) {
337 int v=p_Indices[n];
338 radii[n]=p.getRadius(v);
339 centers[n]=p.getCenter(v);
340 }

354 public PackData convertTo() {
355
356 if (vertCount < 4 || intVertCount < 1 || flowerCount == 0) {
357 CirclePack.cpb
358 .errMsg("Conversion of PackLite failed: missing data");
359 return null;
360 }
361
362 int[] indx=new int[vertCount+1];
363 for (int ii=1;ii<=vertCount;ii++)
364 indx[ii]=ii;
365
366 if (parent.nodeCount==vertCount)
367 for (int ii=1;ii<=vertCount;ii++)
368 indx[ii]=v2parent[ii];
369
370 // build the vertex bouquet
371 int[][] bouquet=new int[vertCount+1][];
372 int tick=0;
373 for (int vv = 1; vv <= vertCount; vv++) {
374 int v=indx[vv];
375 int num=flowerHeads[tick++];
376 int[] flower=new int[num+1];
377 for (int j=0;j<=num;j++)
378 flower[j]=indx[flowerHeads[tick++]];
379 bouquet[v]=flower;
380 }
381
382 PackDCEL pdcel=CombDCEL.getRawDCEL(bouquet);
383 PackData p = new PackData((CPdrawing) null);
384 pdcel.fixDCEL(p);
385 p.hes = parent.hes;
386 p.setAlpha(indx[1]);
387 p.setGamma(indx[intVertCount+1]);
388
389 // set up vertexMap:
390 p.vertexMap = new VertexMap();
391 for (int v=1;v<=vertCount;v++)
392 p.vertexMap.add(new EdgeSimple(v,indx[v]));
393
394 // some default stuff
395 double rad = 0.025;
396 if (hes < 0)
397 rad = 1.0 - Math.exp(-1.0);
398 for (int v = 1; v <= vertCount; v++) {
399 p.setRadius(indx[v],rad);
400 p.setCenter(indx[v],new Complex(0.0));
401 }
402
403 // set vertexMap (whether we converted back to original or not)
404 for (int v = 1; v <= vertCount; v++) {
405 p.vertexMap.add(new EdgeSimple(v, indx[v]));
406 }
407
408 // store radii
409 if (radii != null) {
410 for (int v = 1; v <= vertCount; v++)
411 p.setRadius(indx[v],radii[v]);
412 }
413
414 // store centers
415 if (centers != null) {
416 for (int v = 1; v <= vertCount; v++)
417 p.setCenter(indx[v],new Complex(centers[v]));
418 }
419
420 // nondefault aims
421 p.set_aim_default();
422 if (aimCount > 0) {
423 for (int i = 0; i < aimCount; i++)
424 p.setAim(indx[aimIndices[i]],aims[i]);
425 }
426
427 if (invDistCount > 0 && invDistLink != null) {
428 tick = 0;
429 Iterator<EdgeSimple> iL = invDistLink.iterator();
430 while (tick < invDistCount && iL.hasNext()) {
431 EdgeSimple es=iL.next();
432 es.v=indx[es.v];
433 es.w=indx[es.w];
434 HalfEdge edge=p.packDCEL.findHalfEdge(es);
435 if (edge==null)
436 throw new DataException("failed to find edge "+es);
437 edge.setInvDist(invDistances[tick++]);
438 }
439 }
440
441 return p;
442 }
