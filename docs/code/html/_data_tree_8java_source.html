198 public static String setTopologyStr(PackData p) {
199 if (!p.status)
200 return new String("pack is empty");
201 if (p.genus==0) { // genus zero
202 if (p.getBdryCompCount()==0) // sphere
203 return new String("sphere");
204 if (p.getBdryCompCount()==1) // disc
205 return new String("topological disc");
206 if (p.getBdryCompCount()==2) // annulus
207 return new String("topological annulus");
208 else // n-connected, planar
209 return new String("planar, "+p.getBdryCompCount()+"-connected");
210 }
211 if (p.getBdryCompCount()==0) // n-torus
212 return new String("topological "+p.genus+"-torus");
213 else return new String("genus "+p.genus+", bordered");
214 }
