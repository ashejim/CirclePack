121 public boolean putVariable(PackData p,String vkey,
122 Vector<Vector<String>> flagSegs) {
123 Vector<String> items=flagSegs.get(0);
124 String theStuff=null;
125
126 String firstStr=items.get(0);
127 char c=firstStr.charAt(0);
128
129 // this is form '{..cmd..}', so do a 'valueExecute',
130 if (c=='{') {
131 theStuff=StringUtil.
132 getBracesString(StringUtil.reconstitute(flagSegs));
133 CallPacket cP=CommandStrParser.valueExecute(p,theStuff);
134 if (cP==null || cP.error)
135 return false;
136
137 // expect double value or else int value
138 try {
139 if (cP.double_vec!=null)
140 varModel.put(vkey,cP.double_vec.get(0).toString());
141 else if (cP.int_vec!=null)
142 varModel.put(vkey,cP.int_vec.get(0).toString());
143 return true;
144 } catch (Exception ex) {
145 return false;
146 }
147 }
148
149 // math comp case
150 if (c=='$') {
151 theStuff=StringUtil.getMathString(StringUtil.reconstitute(flagSegs));
152 try {
153 varModel.put(vkey, theStuff);
154 return true;
155 } catch (Exception ex) {
156 return false;
157 }
158 }
159
160 // routine case
161 if (c!='?' && c!='&') {
162 theStuff=StringUtil.reconstitute(flagSegs);
163 }
164
165 // otherwise, need to find 'query' and remaining flagSegs
166 else {
167 String query=null;
168 // have to worry about incidental white space after it
169 if (firstStr.length()>1) {
170 query=firstStr.substring(1);
171 items.remove(0);
172 }
173 else { // remove white space
174 try {
175 items.remove(0); // toss '?' or '&'
176 query=items.remove(0);
177 } catch (Exception ex) {
178 throw new ParserException("query seems to be messed up");
179 }
180 }
181 if (items.size()==0)
182 flagSegs.remove(0); // used all of first segment
183
184 // now, get theStuff
185 if (c=='?')
186 theStuff=QueryParser.queryParse(p,query,flagSegs,false);
187 else
188 theStuff=QueryParser.curValueParse(p,query,flagSegs);
189 }
190
191 // finally, store 'theStuff' under 'vkey'
192 try {
193 varModel.put(vkey,theStuff); // in case it's displacing something
194 varModel.fireTableDataChanged();
195 } catch (Exception ex) {
196 CirclePack.cpb.errMsg("failed to store variable");
197 return false;
198 }
199 return true;

554 public static String []isSpecVariable(String strbld) {
555 if (strbld==null || strbld.charAt(0)!='[')
556 return null;
557 int leftbs=1;
558 int rightbs=0;
559 int j=1;
560 int length=strbld.length();
561 while (rightbs<leftbs && j<length) {
562 if (strbld.charAt(j)==']')
563 rightbs++;
564 if (strbld.charAt(j)=='[')
565 leftbs++;
566 j++;
567 }
568
569 // malformed? -- unmatched [
570 if (j==length && rightbs<leftbs)
571 return null;
572
573 String inbrackets=strbld.substring(1,j-1).trim();
574 String []results=new String[3];
575
576 // is this a slider?
577 if (inbrackets.substring(0,6).toUpperCase().startsWith("SLIDER")) {
578 results[0]=new String("SLIDER");
579 String afterR=inbrackets.substring(6,inbrackets.length()).trim();
580 if (afterR.length()>0)
581 results[1]=new String(" "+afterR); // add a space
582 else
583 results[1]="";
584 results[2]=strbld.substring(j,strbld.length()).trim();
585 return results;
586 }
587
588 // TODO: in future, may want other specifications in brackets
589 return null;
590 }
