158 /* Added arguments are positional: if omitted, they take default values
159 as given in the header. Namely,
160
161 passNo = PASS_FIRST | PASS_MIDDLE | PASS_LAST
162 cutp = null
163 state = null
164
165 This tells riffle to first allocate memory, initialize superstep
166 variables (PASS_FIRST), do specified number of iterations
167 (PASS_MIDDLE), and clean up (PASS_LAST). The two null pointers tell
168 riffle not to store any of the state variables from this call to
169 riffle, so that superstepping will have to start over the next time
170 riffle is called. (The aim was to be able to use in a parallel
171 and/or interruptable environment.) So, the default values for these
172 last three parameters should make modified routines behave exactly like the
173 original routines.
174 */

180 public int startRiffle() throws PackingException { // initiate packing computation
181
182 if (status!=LOADED) throw new PackingException();
183
184 maxBadCuts = 0;
185 minBadCuts = 0;
186 sumBadCuts = 0;
187 cntBadCuts = 0;
188
189 // set up parameters
190 ttoler = 3*aimnum*RP_TOLER; // adjust tolerance
191 key = 1; // initial superstep type
192 m = 1; // Type 1 multiplier
193 sct = 1; // Type 1 count
194 fct = 2; // Type 2 minimum count
195
196 // do one iteration to get started
197 accumErr2 = 0;
198 try {
199 for (int j=0;j<aimnum;j++) {
200 int v = index[j];
201 double faim = p.packDCEL.vertices[v].aim; // get target sum
202 double r = getTriRadius(v); // get present label
203
204 // compute anglesum (using local data)
205 double fbest=compTriCurv(v,r);
206
207 // use the model to predict the next value
208 int N = 2*vNum[v];
209 double del = Math.sin(faim/N);
210 double bet = Math.sin(fbest/N);
211 double r2 = r*bet*(1-del)/(del*(1-bet));
212 // store as new radius label
213 if (r2<0)
214 throw new PackingException();
215 setTriRadius(v,r2);
216 p.packDCEL.vertices[v].curv = fbest; // store new angle sum
217 fbest -= faim;
218 accumErr2 += fbest*fbest; // accum abs error
219 }
220 accumErr2 = Math.sqrt(accumErr2);
221
222 } catch (Exception ex) {
223 status=FAILURE;
224 throw new PackingException();
225 }
226 status=RIFFLE;
227 totalPasses=localPasses=1;
228 return RIFFLE;
229 }

248 public int continueRiffle(int passL) throws PackingException {
249 double fbest;
250 double faim;
251 double c1;
252
253 if (status!=RIFFLE) throw new PackingException();
254 localPasses=0;
255 passLimit=passL;
256
257 // ============ Begin Main Loop ============
258 // Iterate until L2 error is below tolerance or pass limit reached
259 while ((accumErr2 >ttoler && localPasses<passLimit)) {
260
261 // Snapshot current radii before this pass (for superstep computation)
262 for (int i=1;i<=p.nodeCount;i++)
263 R1[i] = getTriRadius(i);
264
265 int numBadCuts = 0;
266 double factor=0.0;
267 do { // Retry loop: ensure error ratio factor < 1.0 (i.e., progress)
268 c1 = 0.0;
269 for (int j=0;j<aimnum;j++) {
270
271 int v = index[j]; // index of adjustable vertex
272 faim = p.packDCEL.vertices[v].aim; // target angle sum α(v)
273 double ra = getTriRadius(v); // current radius r(v)
274
275 // Compute current angle sum θ(v) from surrounding radii
276 fbest=compTriCurv(v,ra);
277
278 // === Uniform Neighbor Model (UNM) prediction ===
279 // Model: if v had 2N equal neighbors, each face angle
280 // would be θ/(2N). The radius producing target α/(2N) is:
281 // r_new = r · sin(θ/2N)(1−sin(α/2N)) / [sin(α/2N)(1−sin(θ/2N))]
282 int N = 2*vNum[v];
283 double del = Math.sin(faim/N); // sin(α/2N)
284 double bet = Math.sin(fbest/N); // sin(θ/2N)
285 double r2 = ra*bet*(1-del)/(del*(1-bet));
286 // store as new radius label
287 if (r2<0)
288 throw new PackingException();
289 setTriRadius(v,r2);
290 p.packDCEL.vertices[v].curv = fbest; // store new angle sum
291 fbest -= faim;
292 c1 += fbest*fbest; // accum abs error
293 }
294 c1 = Math.sqrt(c1);
295
296 factor = c1/accumErr2;
297 if (factor >= 1.0) {
298 accumErr2 = c1;
299 key = 1;
300 numBadCuts++;
301 }
302 if(numBadCuts > MAX_ALLOWABLE_BAD_CUTS)
303 throw new PackingException(
304 "exceeded MAX_ALLOWABLE_BAD_CUTS");
305 } while(factor >= 1.0);
306
307 if(cntBadCuts == 0) {
308 maxBadCuts = numBadCuts;
309 minBadCuts = numBadCuts;
310 sumBadCuts = numBadCuts;
311 }
312 else {
313 maxBadCuts = (numBadCuts > maxBadCuts) ? numBadCuts : maxBadCuts;
314 minBadCuts = (numBadCuts < minBadCuts) ? numBadCuts : minBadCuts;
315 sumBadCuts += numBadCuts;
316 }
317 cntBadCuts++;
318
319 // ========== Superstep Acceleration ==========
320 // After one UNM pass, radii moved from R1→R2 with error
321 // reduction factor = c1/accumErr2. We extrapolate further
322 // along direction (R2−R1) by factor λ to speed convergence.
323
324 // Snapshot post-UNM radii
325 for (int i=1;i<=p.nodeCount;i++)
326 R2[i] = getTriRadius(i);
327
328 // Find maximum safe λ: need R2+λ(R2−R1) > 0 for all vertices.
329 // For vertices where radius decreased (R2−R1 < 0),
330 // λ must be < R2/(R1−R2) to keep radius positive.
331 double lmax = 10000;
332 double fact0=0.0;
333 for (int j=0;j<aimnum;j++) {
334 int v = index[j];
335 double rat = R2[v] - R1[v]; // direction of change
336 double tr=0.0;
337 if (rat < 0)
338 lmax = (lmax < (tr= (-R2[v]/rat))) ? lmax : tr;
339 }
340 lmax = lmax/2; // safety factor of 2
341
342 // Compute λ using one of two strategies
343 double lambda=0.0;
344 if (key==1) { // Type 1: growing multiplier m
345 lambda = m*factor;
346 double mmax = 0.75/(1-factor); // cap on multiplier
347 double mm=0.0;
348 m = (mmax < (mm=(1+0.8/(sct+1))*m)) ? mmax : mm;
349 }
350 else { // Type 2: geometric extrapolation
351 fact0=0.0;
352 double ftol=0.0;
353 if (sct>fct && Math.abs(factor-fact0)<ftol) {
354 lambda = factor/(1-factor); // geometric series sum
355 sct = -1;
356 }
357 else
358 lambda = factor; // fallback
359 }
360 lambda = (lambda>lmax) ? lmax : lambda; // clamp to safe max
361
362 // Apply superstep: R_new = R2 + λ·(R2 − R1)
363 for (int j=0;j<aimnum;j++) {
364 int v = index[j];
365 double nwr=R2[v]+lambda*(R2[v]-R1[v]);
366 if(nwr<0)
367 throw new PackingException("negative rad at "+v);
368 setTriRadius(v,nwr);
369 }
370 sct++;
371 fact0 = factor;
372
373 // ========== Verify Superstep ==========
374 // Recompute angle sums with extrapolated radii and do
375 // one more UNM update to check improvement.
376 accumErr2 = 0;
377 for (int j=0;j<aimnum;j++) {
378 int v = index[j];
379
380 faim = p.packDCEL.vertices[v].aim;
381 double rc = getTriRadius(v);
382
383 // Recompute angle sum after superstep
384 fbest=compTriCurv(v,rc);
385
386 // UNM prediction from post-superstep state
387 int N = 2*vNum[v];
388
389 double del = Math.sin(faim/N);
390 double bet = Math.sin(fbest/N);
391
392 double r2 = rc*bet*(1-del)/(del*(1-bet));
393 if (r2<0)
394 throw new PackingException();
395 setTriRadius(v,r2);
396 p.packDCEL.vertices[v].curv = fbest;
397 fbest -= faim;
398 accumErr2 += fbest*fbest;
399 }
400 accumErr2 = Math.sqrt(accumErr2);
401
402 // ========== Evaluate Superstep Outcome ==========
403 // Compare actual error reduction to the predicted reduction
404 // factor^λ (linear convergence extrapolation).
405 double pred = Math.exp(lambda*Math.log(factor)); // predicted: factor^λ
406 double act = accumErr2/c1; // actual ratio
407 if (act<1) { // error decreased: superstep helped
408 if (act>pred) { // but not as much as predicted: reset multiplier
409 m = 1;
410 sct = 0;
411 if (key==1) key = 2; // switch superstep strategy
412 }
413 // else: accept result and keep growing multiplier
414 }
415 else { // error increased: roll back to pre-superstep state
416 m = 1;
417 sct =0;
418 for (int i=1;i<=p.nodeCount;i++)
419 setTriRadius(i,R2[i]); // restore R2 (post-UNM, pre-superstep)
420 accumErr2 = c1;
421 if (key==2) key = 1; // switch strategy for next attempt
422 }
423
424 // show activity
425 if ((localPasses % 10)==0)
426 repack_activity_msg();
427
428 localPasses++;
429 } // end of main while loop
430
431 totalPasses+=localPasses;
432 return RIFFLE;
433 }

453 double accum=0.0;
454 int N=5; // iterations in each radius comp
455
456 // find vertices to work on and set cutoff value
457 int []inDex =new int[p.nodeCount+1];
458 aimnum=0;

520 public double e_RadCalc(int v,double r, double aim,int N) {
521 double lower=0.5;
522 double upper=0.5;
523 double factor=0.5;
524 double upcurv;
525 double lowcurv;
526
527 // compute initial curvature
528 double curv=compTriCurv(v,r);
529 double bestcurv=lowcurv=upcurv=curv;
530
531 // may hit upper/lower bounds on radius
532 if (bestcurv>(aim+RP_OKERR)) {
533 upper=r/factor;
534 upcurv=compTriCurv(v,upper);
535 if (upcurv>aim) { // return max possible
536 return upper;
537 }
538 }
539 else if (bestcurv<(aim-RP_OKERR)) {
540 lower=r*factor;
541 lowcurv=compTriCurv(v,lower);
542 if (lowcurv<aim) { // return min possible
543 return lower;
544 }
545 }
546
547 // iterative secand method
548 for (int n=1;n<=N;n++) {
549 if (bestcurv>(aim+RP_OKERR)) {
550 lower=r;
551 lowcurv=bestcurv;
552 r += (aim-bestcurv)*(upper-r)/(upcurv-bestcurv);
553 }
554 else if (bestcurv<(aim-RP_OKERR)) {
555 upper=r;
556 upcurv=bestcurv;
557 r -= (bestcurv-aim)*(lower-r)/(lowcurv-bestcurv);
558 }
559 else
560 return r;
561
562 bestcurv=compTriCurv(v,r);
563 }
564
565 return r;
566 }

728 public static double[] euclPerron(PackData p, int direction, int passes) {
729 double[] results = new double[4];
730 results[0] = results[1] = results[2] = results[3] = FAILURE;
731 double dirSign = -1.0;
732 if (direction >= 0)
733 dirSign = 1.0;
734 boolean both = false;
735 if (direction == 2 || direction == -2 || direction == 0)
736 both = true;
737 int count = 0;
738
739 // whom to repack?
740 int aimNum = 0;
741 int[] inDex = new int[p.nodeCount + 1];
742 for (int k = 1; k <= p.nodeCount; k++) {
743 if (p.getAim(k) > 0) {
744 inDex[aimNum] = k;
745 aimNum++;
746 }
747 }
748 if (aimNum == 0) { // nothing to repack
749 return results;
750 }
751
752 // =============== First stage ======================
753 // radii going up if direction>=0 or else down:
754 // compute discrepency from being super/subpacking
755 double discrepency = 0.0;
756 double fbest = 0.0;
757 for (int j = 0; j < aimNum; j++) {
758 int v = inDex[j];
759 double vaim = p.getAim(v);
760 double r = p.getRadius(v);
761 fbest = 0.0;
762 int[] flower = p.packDCEL.vertices[v].getFlower(true);
763 int num=flower.length-1;
764 double r2 = p.getRadius(flower[0]);
765 double m2 = r2 / (r + r2);
766 for (int n = 1; n <= num; n++) {
767 double m1 = m2;
768 r2 = p.getRadius(flower[n]);
769 m2 = r2 / (r + r2);
770 fbest += Math.acos(1 - 2 * m1 * m2);
771 }
772 p.setCurv(v, fbest);
773 double diff2 = dirSign * (fbest - vaim);
774 if (diff2 > 0.0)
775 discrepency += diff2 * diff2;
776 }
777
778 // Perron until we have a super/subpacking (or pass out)
779 int safety = 0;
780 while (safety < passes && discrepency > RP_TOLER) {
781 safety++;
782 discrepency = 0.0;
783 for (int j = 0; j < aimNum; j++) {
784 int v = inDex[j];
785 double vaim = p.getAim(v);
786 double r = p.getRadius(v);
787
788 // update anglesum for initial r (nghb's may have changed)
789 fbest = 0.0;
790 int[] flower = p.packDCEL.vertices[v].getFlower(true);
791 int num=flower.length-1;
792 double r2 = p.getRadius(flower[0]);
793 double m2 = r2 / (r + r2);
794 for (int n = 1; n <= num; n++) {
795 double m1 = m2;
796 r2 = p.getRadius(flower[n]);
797 m2 = r2 / (r + r2);
798 fbest += Math.acos(1 - 2 * m1 * m2);
799 }
800 p.setCurv(v, fbest);
801
802 // sub/superpacking at v? want to adjust radius
803 double diff2 = dirSign * (fbest - vaim);
804 if (diff2 > RP_TOLER) { // uniform neighbor computation
805 int N = 2 * num;
806 double del = Math.sin(vaim / N);
807 double bet = Math.sin(fbest / N);
808 r = r * bet * (1 - del) / (del * (1 - bet));
809 p.setRadius(v, r);
810 count++;
811
812 // update the anglesum for new r
813 fbest = 0.0;
814 r2 = p.getRadius(flower[0]);
815 m2 = r2 / (r + r2);
816 for (int n = 1; n <= num; n++) {
817 double m1 = m2;
818 r2 = p.getRadius(flower[n]);
819 m2 = r2 / (r + r2);
820 fbest += Math.acos(1 - 2 * m1 * m2);
821 }
822 p.setCurv(v, fbest);
823
824 // get squared discrepency
825 diff2 = dirSign * (fbest - vaim);
826 if (diff2 < -(100 * RP_TOLER)) {
827 throw new PackingException("uniform neighbor computation overshoots at v=" + v);
828 }
829 }
830 if (diff2 > 0.0)
831 discrepency += diff2 * diff2;
832 }
833 } // end of while loop
834 if (dirSign >= 0) // upward
835 results[1] = Math.sqrt(discrepency);
836 else
837 results[2] = Math.sqrt(discrepency);
838
839 // failed?
840 if (safety == passes) {
841 if (dirSign > 0)
842 CirclePack.cpb.errMsg("failed to get a superpacking in euclPerron");
843 else
844 CirclePack.cpb.errMsg("failed to get a subpacking in euclPerron");
845 }
846
847 // to prepare for stage 2 or return, update discrepency and accum
848 double accum = 0.0;
849 discrepency = 0.0;
850 for (int j = 0; j < aimNum; j++) {
851 int v = inDex[j];
852 double vaim = p.getAim(v);
853 double r = p.getRadius(v);
854 fbest = 0.0;
855 int[] flower = p.packDCEL.vertices[v].getFlower(true);

998 public static double uniFactor(int num,double asum,double aim) {
999 int N = 2*num;
1000 // assume all petals have radius 1.0
1001 double del = Math.sin(aim/N);
1002 double bet = Math.sin(asum/N);
1003 // current rad r=(1/bet)-1.0, desired rad R=(1/del)-1.0,
1004 // so return R/r
