43 extensionAbbrev="FK";
44 toolTip="'FeedBack' allows radii (eucl only) adjustments "+
45 "based on other radii. Matrix DM (dependency matrix, nonnegative " +
46 "entries) adjusts radii via vector mutlt NewR = DM*CurR, where " +
47 "CurR is vector of current radii ratios (vis-a-vis eucl max radii) "+
48 "and NewR is the vector of new radii ratios.";
49 registerXType();

51 // convert packData to euclidean, record its maximal packing
52 try {
53 maxEucl=p.copyPackTo();
54 cpCommand(maxEucl,"max_pack 10000");
55 cpCommand(maxEucl,"geom_to_e");
56 cpCommand(extenderPD,"geom_to_e");
57 running=true;
58 } catch (Exception ex) {
59 errorMsg("FeedBack: error in preparing 'maxPack'");
60 running=false;
61 }
62 // count boundary vertices
63 bdryCount=0;
64 for (int v=1;v<=extenderPD.nodeCount;v++)
65 if (extenderPD.isBdry(v))
66 bdryCount++;
67 if (bdryCount==0) {
68 errorMsg("FeedBack: packing has no boundary");
69 running=false;
70 }
71
72 // yes, seems okay
73 if (running) {
74 extenderPD.packExtensions.add(this);
75 depMatrix=new Vector<FeedPacket>(1);
76 }

79 public int cmdParser(String cmd,Vector<Vector<String>> flagSegs) {
80 int count=0;
81 Vector<String> items=null;
82
83 // ----------- reset ------------
84 if (cmd.startsWith("reset")) {
85 depMatrix=new Vector<FeedPacket>(1);
86 return 1;
87 }
88
89 // -----------
90 else if (cmd.startsWith("radMult")) {
91 double x=1.0;
92 NodeLink sourceVerts=null;
93 NodeLink targetVerts=null;
94 try {
95 items=flagSegs.get(0);
96 // not a flag, should be a double
97 if (!StringUtil.isFlag(items.get(0))) {
98 x=Double.parseDouble(items.remove(0));
99 flagSegs.remove(0);
100 }
101
102 // should get -s and -t flag lists
103 for (int j=0;j<2;j++) {
104 items=flagSegs.remove(0);
105 String lead=items.get(0);
106 if (lead.equals("-s")) {
107 items.remove(0);
108 sourceVerts=new NodeLink(extenderPD,items);
109 }
110 else if (lead.equals("-t")) {
111 items.remove(0);
112 targetVerts=new NodeLink(extenderPD,items);
113 }
114 }
115 } catch (Exception ex) {
116 throw new ParserException("didn't get data: "+ex.getMessage());
117 }
118
119 int snum=sourceVerts.size();
120 int tnum=targetVerts.size();
121 int min=snum;
122 min = (tnum<min) ? tnum: min;
123 for (int i=0;i<min;i++) {
124 int v=sourceVerts.get(i);
125 int w=targetVerts.get(i);
126 extenderPD.setRadius(w,x*extenderPD.getRadius(v));
127 count++;
128 }
129 return count;
130 }
131
132 // ============ set Radii ========================
133 else if (cmd.startsWith("setRadii")) {
134 boolean useRatio=false;
135 double []newRad=new double[extenderPD.nodeCount+1];
136
137 // check for 'r' flag to use ratios of radii; this is
138 // analogous to modifying the modulus of the derivative.
139 try {
140 items=flagSegs.remove(0);
141 if (items.get(0).contains("r"))
142 useRatio=true;
143 } catch(Exception ex) {}
144
145 // accumulate feedback
146 Iterator<FeedPacket> fdpk=depMatrix.iterator();
147 int v=0;
148 int w=0;
149 double x=1.0;
150 while (fdpk.hasNext()) {
151 FeedPacket packet=fdpk.next();
152 v=packet.toVert;
153 w=packet.fromVert;
154 x=packet.coeff;
155 if (useRatio)
156 newRad[v] +=x*extenderPD.getRadius(w)/maxEucl.getRadius(w);
157 else
158 newRad[v] +=x*extenderPD.getRadius(w);
159 }
160
161 // apply feedback
162 int icount=0;
163 for (int i=1;i<=extenderPD.nodeCount;i++) {
164 if (newRad[i]>0) {
165 extenderPD.setRadius(i,newRad[i]);
166 icount++;
167 }
168 }
169 return icount;
170 }
171
172 else if (cmd.startsWith("setVW")) {
173 int v=0;
174 int w=0;
175 double x=0.0;
176 try {
177 items=flagSegs.get(0);
178 v=NodeLink.grab_one_vert(extenderPD,items.get(0));
179 w=NodeLink.grab_one_vert(extenderPD,items.get(1));
180 x=Double.parseDouble(items.get(2));
181 } catch (Exception ex) {
182 Oops("usage: setVW v w {x}");
183 }
184 if (x<0.0 || v<1 || w<1 ||
185 v>extenderPD.nodeCount || w>extenderPD.nodeCount) {
186 Oops("usage: setDep v w x (x>=0.0)");
187 }
188
189 FeedPacket curP=chkPacket(v,w);
190 if (curP==null)
191 depMatrix.add(new FeedPacket(v,w,x));
192 else
193 curP.coeff=x;
194 return 1;
195 }
196
197 else if (cmd.startsWith("swapVW")) {
198 int v=0;
199 int w=0;
200 double x=0.0;
201 try {
202 items=flagSegs.get(0);
203 v=NodeLink.grab_one_vert(extenderPD,items.get(0));
204 w=NodeLink.grab_one_vert(extenderPD,items.get(1));
205 x=Double.parseDouble(items.get(2));
206 } catch (Exception ex) {
207 Oops("usage: setDep v w {x}");
208 }
209 if (x<0.0 || v<1 || w<1 ||
210 v>extenderPD.nodeCount || w>extenderPD.nodeCount) {
211 Oops("usage: setDep v w x (x>=0.0)");
212 }
213
214 FeedPacket curP=chkPacket(v,w);
215 if (curP==null)
216 depMatrix.add(new FeedPacket(v,w,x));
217 else
218 curP.coeff=x;
219 return 1;
220 }
221
222 else if (cmd.startsWith("getDM")) {
223 if (depMatrix==null || depMatrix.size()==0)
224 return 0;
225 Iterator<FeedPacket> fdpk=depMatrix.iterator();
226 while (fdpk.hasNext()) {
227 FeedPacket packet=fdpk.next();
228 if (packet.coeff>0.0)
229 this.msg("v="+packet.toVert+" w="+packet.fromVert+" x="+packet.coeff+"\n");
230 }
231 }
232
233 else if (cmd.startsWith("readDM")) {
234 boolean script_flag=false;
235 try {
236 items=flagSegs.remove(0);
237 if (items.get(0).startsWith("-s")) {
238 script_flag=true;
239 items.remove(0);
240 }
241 } catch (Exception ex) {}
242
243 String filename=StringUtil.reconItem(items);
244 BufferedReader fp=null;
245 File dir=null;
246
247 if (!script_flag) {
248 dir=CPFileManager.PackingDirectory;
249 try {
250 if (filename.startsWith("~/"))
251 filename=new String(CPFileManager.HomeDirectory+File.separator+filename.substring(2));
252 dir=new File(filename);
253 filename=dir.getName();
254 dir=new File(dir.getParent());
255 } catch (Exception ex) {
256 Oops("problem reading file");
257 }
258 }
259
260 try {
261 fp=CPFileManager.openReadFP(dir,filename,script_flag);
262 String line=null;
263 while((line=StringUtil.ourNextLine(fp))!=null) {
264 StringTokenizer tok = new StringTokenizer(line);
265 int v;
266 int w;
267 double x;
268 try {
269 v=Integer.parseInt(tok.nextToken());
270 w=Integer.parseInt(tok.nextToken());
271 x=Double.parseDouble(tok.nextToken());
272 } catch (Exception ex) {
273 return count;
274 }
275 depMatrix.add(new FeedPacket(v,w,x));
276 count++;
277 }
278 fp.close();
279 } catch (Exception ex) {
280 Oops("read error of some sort");
281 }
282 }
283
284 return count;
285 }

293 public FeedPacket chkPacket(int v,int w) {
294 if (depMatrix==null || depMatrix.size()==0)
295 return null;
296 Iterator<FeedPacket> fdpk=depMatrix.iterator();
297 while (fdpk.hasNext()) {
298 FeedPacket packet=fdpk.next();
299 if (packet.toVert==v && packet.fromVert==w)
300 return packet;
301 }
302 return null;
