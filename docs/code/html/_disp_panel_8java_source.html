98 public void actionPerformed(ActionEvent e){
99 String command = e.getActionCommand();
100 if (command.equals("reset")) {
101 PackControl.getActiveCPDrawing().dispOptions.reset();
102 flagField.setText("");
103 update(CirclePack.cpb.getActivePackNum(),CirclePack.cpb.getActivePackNum());
104 }
105 else if (command.equals("redraw")) {
106 try {
107 TrafficCenter.cmdGUI("disp -wr");
108 } catch (Exception ex) {}
109 PackControl.activeFrame.activeScreen.repaint();
110 }
111 else if (command.equals("redraw_flags")) {
112 try {
113 TrafficCenter.cmdGUI("disp "+ flagField.getText());
114 } catch (Exception ex) {}
115 PackControl.activeFrame.activeScreen.repaint();
116 }
117 else { // rest of actions can be used in call
118 Object obj=e.getSource();
119 if (obj instanceof JCheckBox) {
120 JCheckBox box=(JCheckBox)e.getSource();
121 PackControl.getActiveCPDrawing().dispOptions.setOnOff(command,box.isSelected());
122 }
123 }
