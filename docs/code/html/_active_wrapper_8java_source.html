250 //AF>>>//
251 // This implements cursor position dependent mouse wheel zooming.
252 // If the user rotates the mouse wheel away from them, the canvas
253 // will zoom in towards the cursor. If the user rotates the mouse wheel
254 // towards them, the canvas will zoom out away from the cursor.
255 public void mouseWheelMoved(MouseWheelEvent mwe) {
256 int mouseWheelClicks = mwe.getWheelRotation(); // Wheel rotation amount in clicks. Sign determines direction.
257 int width = mwe.getComponent().getWidth(); // Width of the firing component.
258 int height = mwe.getComponent().getHeight(); // Height of the firing component.
259 int x = (int) mwe.getPoint().getX(); // Current X position of mouse.
260 int y = (int) mwe.getPoint().getY(); // Current Y position of mouse.
261
262 // Get the mouse position in real-world values.
263 Point2D.Double realCurrentMousePoint = cpDrawing.pt2RealPt(new Point(x, y), width, height);
264 // Convert the mouse position to a complex number.
265 Complex complexCurrentMousePoint = new Complex(realCurrentMousePoint);
266 // Calculate the center of the canvas screen as a complex number.
267 Complex complexScreenCenter = cpDrawing.realBox.lz.add(cpDrawing.realBox.rz).divide(2.0D);
268 // Initialize translation vector; this will hold how much to translate the screen.
269 Complex scaledTranslationVector;
270
271//System.out.println("clicks = "+mouseWheelClicks);
272
273 // NOTE: I've changed these to only do one zoom, irrespective
274 // of number of clicks
275 // Mouse was rotated away from user; zoom in.
276 if (mouseWheelClicks < 0) {
277 // Calculate the translation vector and move the screen before zooming in.
278 scaledTranslationVector = complexCurrentMousePoint.
279 minus(complexScreenCenter).mult(1.0D - mouseWheelZoomInMultiplier);
280 cpDrawing.realBox.transView(scaledTranslationVector);
281 zoomIn(mouseWheelZoomInMultiplier);
282 }
283 // Mouse was rotated toward user; zoom out.
284 else if (mouseWheelClicks > 0) {
285 scaledTranslationVector = complexCurrentMousePoint.
286 minus(complexScreenCenter).mult(1.0D - mouseWheelZoomOutMultiplier);
287 cpDrawing.realBox.transView(scaledTranslationVector);
288 zoomOut(mouseWheelZoomOutMultiplier);
289 }
290 }
