139 public static Mobius normalizeTorus(PackData p) {
140
141 if (!TorusData.isTopTorus(p))
142 throw new ParserException("packing is not a torus");
143
144 // arrange two-side pairings only
145 if (p.packDCEL.pairLink==null || p.packDCEL.pairLink.size()!=5) {
146 CommandStrParser.jexecute(p,"newRed -t");
147 CommandStrParser.jexecute(p,"layout");
148 if (p.packDCEL.updatePairMob()==0)
149 throw new CombException("failed to update side-pairings");
150 }
151
152 // find the 4 locations for the corner vertex
153 Iterator<SideData> pdpl=
154 p.packDCEL.pairLink.iterator();
155 SideData epair=null;
156 epair=pdpl.next(); // first slot is empty
157 Complex[] Z=new Complex[4];
158 int j=0;
159 while(pdpl.hasNext()) {
160 epair=pdpl.next();
161 Z[j]=epair.startEdge.getCenter();
162 j++;
163 }
164
165 Mobius mob=new Mobius();
166 if (!isAffineTorus(p)) {
167 mob.b=Z[0].times(-1.0);
168 mob.d=Z[1].minus(Z[0]);
169 return mob;
170 }
171 // affine case
172 // In affine case, side-pairings are linear:
173 // z0 -> z3 and z1 -> z2
174 // z0 -> z1 and z3 -> z2
175 // The have a common finite fixed point.
176 // where 'fixed' is common fixed point,
177 Complex numtor=Z[0].times(Z[2]).minus(Z[1].times(Z[3]));
178 Complex denom=Z[0].add(Z[2]).minus(Z[1]).minus(Z[3]);
179 Complex fixed=numtor.minus(denom);
180 // So so mob(z)=az+b where b=fixed/(fixed-Z[0]) and a= 1/(Z[0]-fixed)
181 mob.a=new Complex(1.0).divide(Z[0].minus(fixed));
182 mob.b=fixed.divide(fixed.minus(Z[0]));
183 return mob;
184 }

205 public static boolean isAffineTorus(PackData p) {
206
207 if (!isTopTorus(p))
208 throw new CombException("This is not a torus");
209
210 // No side pairings?
211 if (p.packDCEL.pairLink==null)
212 throw new CombException("Not a torus or side-pairings are not set");
213
214 // check number of sides
215 int N=p.packDCEL.pairLink.size()-1;
216 if (N!=4 && N!=6)
217 throw new CombException("Incorrect number of side-pairings");
218
219 // compare radii of opposite vertex on each sidepair
220 double radcomp=1.0;
221 Iterator<SideData> plink=p.packDCEL.pairLink.iterator();
222 plink.next(); // first is empty
223 while (plink.hasNext()) {
224 SideData ep=(SideData)plink.next();
225 double rad=ep.endEdge.getRadius();
226 SideData oppep=p.packDCEL.pairLink.get(ep.mateIndex);
227 double ratio=rad/oppep.startEdge.nextRed.getRadius();
228 radcomp=(ratio>radcomp) ? ratio : radcomp;
229 }
230
231 // if significant difference in some pair of
232 // opposite radii, then must be affine.
233 if (radcomp>1.00001)
234 return true;
235
236 return false;
237 }

269 public static Complex Teich2Tau(Complex teich) {
270 if (teich.y<=0.0)
271 return null;
272 Complex w=new Complex(teich);
273 w.x=w.x-Math.floor(w.x)-
274 Math.floor(2*(w.x-Math.floor(w.x)));
275 int count=0;
276 Complex m_one=new Complex(-1.0);
277 while (count<1000 && (w.y<Math.sqrt(1.0-w.x*w.x))) {
278 count++;
279 w=m_one.divide(w);
280 w.x=w.x-Math.floor(w.x)-
281 Math.floor(2*(w.x-Math.floor(w.x)));
282 }
283 if (count==1000) { // error
284 return null;
285 }
286 return w;
287 }
