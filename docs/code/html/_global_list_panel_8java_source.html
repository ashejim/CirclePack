126 public void actionPerformed(ActionEvent aev) {
127 String cmd=aev.getActionCommand();
128 Component cmpt=VFETabbing.getSelectedComponent();
129 if (cmpt==null) return;
130 StringBuilder theCmd=new StringBuilder();
131
132 if (cmd.equals("getGlobal")) {
133 if (cmpt==(Component)vertTab) {
134 try {
135 Iterator<Integer> vlist=CPBase.Vlink.iterator();
136 while (vlist.hasNext() && theCmd.length()<2000) {
137 theCmd.append(Integer.toString((Integer)vlist.next())+" ");
138 }
139 VlistArea.setText(theCmd.toString());
140 } catch (Exception ex) {
141 VlistArea.setText("");
142 return;
143 }
144 }
145 else if (cmpt==(Component)edgeTab) {
146 try {
147 Iterator<EdgeSimple> elist=CPBase.Elink.iterator();
148 while (elist.hasNext() && theCmd.length()<2000) {
149 EdgeSimple edge=(EdgeSimple)elist.next();
150 theCmd.append(Integer.toString(edge.v)+" "+
151 Integer.toString(edge.w)+" ");
152 }
153 ElistArea.setText(theCmd.toString());
154 } catch (Exception ex) {
155 ElistArea.setText("");
156 return;
157 }
158 }
159 if (cmpt==(Component)faceTab) {
160 try {
161 Iterator<Integer> flist=CPBase.Flink.iterator();
162 while (flist.hasNext() && theCmd.length()<2000) {
163 theCmd.append(Integer.toString((Integer)flist.next())+" ");
164 }
165 FlistArea.setText(theCmd.toString());
166 } catch (Exception ex) {
167 FlistArea.setText("");
168 return;
169 }
170 }
171 return;
172 }
173
174 boolean append=false;
175 if (cmd.startsWith("append"))
176 append=true;
177
178 if (cmpt==(Component)vertTab) {
179 theCmd.append("set_Vlist ");
180 if (append) theCmd.append("Vlist ");
181 theCmd.append(VlistArea.getText());
182 }
183 else if (cmpt==(Component)edgeTab) {
184 theCmd.append("set_Elist ");
185 if (append) theCmd.append("Elist ");
186 theCmd.append(ElistArea.getText());
187 }
188 else if (cmpt==(Component)faceTab) {
189 theCmd.append("set_Flist ");
190 if (append) theCmd.append("Flist ");
191 theCmd.append(FlistArea.getText());
192 }
193
194 try {
195 TrafficCenter.cmdGUI(theCmd.toString());
196 } catch (Exception ex) {
