546 public static TileLink tileSpecs(PackData p,Vector<SelectSpec> specs) {
547 if (specs==null || specs.size()==0 || p.tileData==null)
548 return null;
549 SelectSpec sp=null;
550 int count=0;
551
552 // will store results in 'tmpFlag'
553 int[] tmpUtil=new int[p.tileData.tileCount+1];
554 // loop through all the specifications: these should alternate
555 // between 'specifications' and 'connectives', starting with
556 // the former, although typically there will be just one
557 // specification in the vector and no connective.
558 UtilPacket uPx=null;
559 UtilPacket uPy=null;
560 boolean isAnd=false; // true for '&&' connective, false for '||'.
561 for (int j=0;j<specs.size();j++) {
562 sp=(SelectSpec)specs.get(j);
563 if (sp.object!='t')
564 throw new ParserException(); // spec must be for tiles
565 try {
566 for (int t=1;t<=p.tileData.tileCount;t++) {
567
568 // success?
569 boolean outcome=false;
570 uPx=sp.node_to_value(p,t,0);
571 if (sp.unary) {
572 if (uPx.rtnFlag!=0)
573 outcome=sp.comparison(uPx.value,0);
574 }
575 else {
576 uPy=sp.node_to_value(p,t,1);
577 if (uPy.rtnFlag!=0)
578 outcome=sp.comparison(uPx.value, uPy.value);
579 }
580 if (outcome) { // yes, this value satisfies condition
581 if (!isAnd && tmpUtil[t]==0) { // 'or' situation
582 tmpUtil[t]=1;
583 count++;
584 }
585 }
586 else { // no, fails this condition
587 if (isAnd && tmpUtil[t]!=0) { // 'and' situation
588 tmpUtil[t]=0;
589 count--;
590 }
591 }
592 }
593 } catch (Exception ex) {
594 throw new ParserException();
595 }
596
597 // if specs has 2 or more additional specifications, the next must
598 // be a connective. Else, finish loop.
599 if ((j+2)<specs.size()) {
600 sp=(SelectSpec)specs.get(j+1);
601 if (!sp.isConnective)
602 throw new ParserException();
603 isAnd=sp.isAnd;
604 j++;
605 }
606 else j=specs.size(); // kick out of loop
607 }
608
609 if (count>0) {
610 TileLink nl=new TileLink(p.tileData);
611 for (int t=1;t<=p.tileData.tileCount;t++)
612 if (tmpUtil[t]!=0) nl.add(t);
613 return nl;
614 }
615 else return null;
616 }
