351 public static ArrayList<Integer> blue_red_color_ramp(ArrayList<Double> data) {
352 if (data==null || data.size()==0)
353 return null;
354 ArrayList<Integer> output=new ArrayList<Integer>(data.size());
355 // determine most negative, most positive
356 double miN=0.0;
357 double maX=0.0;
358 for (int i=0;i<data.size();i++) {
359 double dta=data.get(i);
360 miN = (dta<miN) ? dta : miN;
361 maX = (dta>maX) ? dta : maX;
362 }
363
364 // balance these
365 double range=maX;
366 double miNabs=Math.abs(miN);
367 range=(miNabs>maX) ? miNabs : range;
368 maX=range;
369 miN=-range;
370
371 // set output
372 for (int i=0;i<data.size();i++) {
373 double dta=data.get(i);
374 if (dta==0)
375 output.add(Integer.valueOf(100)); // white
376 else if (dta<0)
377 output.add(Integer.valueOf((int)(100.0-99.0*(dta/miN))));
378 else
379 output.add(Integer.valueOf((int)(100.0+99.0*(dta/maX))));
380 }

449 public static Vector<Color> richter_red_ramp(Vector<Double> data) {
450 int dsize;
451 if (data==null || (dsize=data.size())==0) return null;
452 Vector<Color> output=new Vector<Color>(dsize);
453 output.add(0,null); // first spot is background
454
455 // determine most negative, most positive for scaling purposes
456 double miN=0.0;
457 double maX=0.0;
458 for (int i=1;i<dsize;i++) {
459 double dta=data.get(i);
460 miN = (dta<miN) ? dta : miN;
461 maX = (dta>maX) ? dta : maX;
462 }
463 double M=maX-miN;
464 if (M<.0001) { // data too clustered, return all background
465 for (int i=1;i<dsize;i++) {
466 output.add(ColorUtil.getBGColor());
467 }
468 return output;
469 }
470 double denom=Math.log(M+1);
471 for (int i=1;i<dsize;i++) {
472 double newdata=Math.log(1+data.get(i)-miN)/denom;
473 if (newdata<.01)
474 newdata=0.0;

476 output.add(new Color(red[indx],green[indx],blue[indx]));
477 }
