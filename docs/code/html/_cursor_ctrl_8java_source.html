117 public static Cursor createScaledCursor(ImageIcon imageIcon, Point hotPt) {
118 Toolkit toolkit;
119// MediaTracker mediaTracker;
120 BufferedImage scaledCursorImage;
121 Graphics2D g2;
122 Point cursorHotSpot;
123 int origWidth;
124 int origHeight;
125 Image origImage=imageIcon.getImage();
126 origWidth = origImage.getWidth(null);
127 origHeight = origImage.getHeight(null);
128
129 toolkit = Toolkit.getDefaultToolkit();
130
131
132/* see if we can do without this
133
134 //load a MediaTracker and add the cursor image, then
135 //wait for it to load. getImage(String imagePath) may
136 //return before loading the image finishes, and often
137 //causes strange behavior if operations are attempted on
138 //the image before it successfully loads.
139 mediaTracker = new MediaTracker(CPBase.frame);
140 mediaTracker.addImage(origImage, 0);
141 try {
142 mediaTracker.waitForAll();
143 } catch (InterruptedException ie) {
144 ie.printStackTrace();
145 }
146*/
147
148 // get closest allowed dimension (e.g. Windows is 32x32 only
149 Dimension dim=toolkit.getBestCursorSize(origWidth,origHeight); // (22,22);
150
151 // if system does not allow custom cursors, default to crosshair
152 if (dim.width==0) {
153 return Cursor.getPredefinedCursor(Cursor.CROSSHAIR_CURSOR);
154 }
155
156 // create pixel transparent image dimension dim
157 scaledCursorImage = new BufferedImage(dim.width,dim.height,
158 BufferedImage.TYPE_INT_ARGB);
159 g2 = (Graphics2D) scaledCursorImage.getGraphics();
160 g2.setColor(new Color(0, 0, 0, 0)); // transparent
161 g2.drawRect(0, 0, dim.width,dim.height);
162
163 // if cursor image fit in area (I assume area and cursor are square)
164 if (origWidth <= dim.width && origHeight <= dim.height) {
165 // draw the cursor over the image starting in the top left corner
166 g2.drawImage(origImage, 0, 0, origWidth, origHeight, null);
167 cursorHotSpot = new Point(hotPt.x,hotPt.y);
168 }
169
170 // else rescale width and/or height
171 else {
172 double scaleX=1.0;
173 double scaleY=1.0;
174
175 if (origWidth > dim.width)
176 scaleX = (double)dim.width / (double) origWidth;
177 if (origHeight> dim.height)
178 scaleY = (double)dim.height/ (double)origHeight;
179 // draw the scaled-down image over the dim transparent
180 //image and recalculate the hot spot.
181 g2.drawImage(origImage,0,0,
182 (int)(scaleX * (double) origWidth),
183 (int)(scaleY * (double) origHeight), null);
184 cursorHotSpot = new Point((int)(scaleX * (double)hotPt.x),
185 (int)(scaleY * (double) hotPt.y));
186 }
187 g2.dispose();
188
189 try {
190 return toolkit.createCustomCursor(scaledCursorImage,cursorHotSpot,null); // cursorImageFile);
191 } catch (Exception ex) {
192 throw new MiscException("problem creating custom cursor"+ex.getMessage());
193 }
194 }
