255 public void cmdRecurseOC(CPTreeNode treeNode,boolean open) {
256 if (treeNode==null) return;
257 if (treeNode.stackBox.isOpen && treeNode.getChildCount()>0) {
258 int j=0;
259 while (j<treeNode.getChildCount()) {
260 CPTreeNode child=(CPTreeNode)treeNode.getChild(j);
261 j++;
262 if (child.tntype==CPTreeNode.COMMAND || child.tntype==CPTreeNode.MODE) {
263 if (open && !child.stackBox.isOpen)
264// child.stackBox.open();
265 child.stackBox.isOpen=true;
266 if (!open && child.stackBox.isOpen)
267// child.stackBox.close();
268 child.stackBox.isOpen=false;
269 }
270 if (child.tntype==CPTreeNode.LINEUP && open) {
271 child.stackBox.open();
272 cmdRecurseOC(treeNode,open);
273 }
274 else cmdRecurseOC(child,open);
275 }
276 }

310 public void actionPerformed(ActionEvent evt) {
311 if (evt.getActionCommand().equals("comboBoxChanged")
312 || evt.getActionCommand().equals("comboBoxEdited")) {
313 // already handling an event,
314 if (isInProcess) // || m_runner.getRunning())
315 return;
316
317 // 'addOKflag' is false, so don't want to load this script
318 if (!MemComboBox.addOKflag)
319 return;
320
321 // else, load the script
322 isInProcess=true;
323 String sUrl = ((String)m_locator.getSelectedItem()).trim();
324 URL url=FileUtil.parseURL(sUrl);
325 URL oldurl=FileUtil.parseURL(loadedCPS);
326
327 // this file is already loaded -- likely, an erroneous duplicate event
328 if (url == null || url.equals(oldurl)) {
329 isInProcess=false;
330 return;
331 }
332
333 // load the script
334 sUrl=url.toString();
335 if (manager.getScript(sUrl,sUrl,false)!=0) { // success
336 isInProcess=false;
337 m_locator.setSuccess();
338 }
339 else {
340 isInProcess=false;
341 m_locator.setFailure();
342 }
343 }
344 }
