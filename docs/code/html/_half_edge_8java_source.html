107 public Vertex []detach() {
108 Vertex []maroon=new Vertex[2];
109 maroon[0]=maroon[1]=null;
110 boolean hits=false;
111
112 // handle origin end
113 if (twin.next==this) { // marooned?
114 origin.halfedge=null;
115 maroon[0]=origin;
116 hits=true;
117 }
118 else { // detach this end
119 twin.next.prev=prev;
120 prev.next=twin.next;
121 origin.halfedge=prev.twin;
122 }
123
124 // handle far end
125 if (next==twin) { // marooned?
126 twin.origin.halfedge=null;
127 maroon[1]=twin.origin;
128 hits=true;
129 }
130 else { // detach far end
131 next.prev=twin.prev;
132 twin.prev.next=next;
133 twin.origin.halfedge=twin.prev.twin;
134 }
135
136 // null out 'prev' and 'next'
137 next=prev=null;
138 twin.next=twin.prev=null;
139
140 // return
141 if (!hits)
142 return null;
143 return maroon;
