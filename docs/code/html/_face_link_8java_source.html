1453 public static double []next_incident_face(PackData p,int f,
1454 double param,PathInterpolator pInt) {
1455 double []ans=new double[2];
1456
1457 // find min of radii of f and contiguous faces
1458 int[] fverts=p.getFaceVerts(f);
1459 double step=p.getRadius(fverts[0]);
1460 double rad=p.getRadius(fverts[1]);
1461 if (rad<step) step=rad;
1462 rad=p.getRadius(fverts[2]);
1463 if (rad<step) step=rad;
1464
1465 int v=p.find_common_left_nghb(fverts[1],fverts[0]);
1466 if (v>0 && (rad=p.getRadius(v))<step)
1467 step=rad;
1468 v=p.find_common_left_nghb(fverts[2],fverts[1]);
1469 if (v>0 && (rad=p.getRadius(v))<step)
1470 step=rad;
1471 v=p.find_common_left_nghb(fverts[0],fverts[2]);
1472 if (v>0 && (rad=p.getRadius(v))<step)
1473 step=rad;
1474
1475 if (step<=0)
1476 throw new DataException("problem with radii of face or neighbors");
1477
1478 step/=3; // this should guarantee that we don't step over all incident faces
1479
1480 Complex z=null;
1481 while (param<pInt.length) {
1482 param += step;
1483 if (param>pInt.length) param=pInt.length;
1484 z=pInt.sToZ(param);
1485 if (p.pt_in_tri(f,z)==0) { // no longer in f
1486 FaceLink flink=p.tri_search(z);
1487 if (flink==null || flink.size()==0) { // not in any face
1488 ans[0]=0;
1489 ans[1]=param;
1490 return ans;
1491 }
1492 Iterator<Integer> fls=flink.iterator();
1493 while (fls.hasNext()) {
1494 int fs=fls.next();
1495 if (p.faces_incident(fs,f)) {
1496 ans[0]=fs;
1497 ans[1]=param;
1498 return ans;
1499 }
1500 }
1501
1502 // no incident face found
1503 ans[0]=0;
1504 ans[1]=param;
1505 return ans;
1506 }
1507 } // end of stepping while
1508
1509 // nothing found
1510 ans[0]=0;
1511 ans[1]=param;
1512 return ans;
1513 }
