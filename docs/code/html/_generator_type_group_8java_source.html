153 public void computeGroup() {
154 // go over every word in the current group and add a letter
155 // let begin and end label the beginning and the end
156 // of the previous level computed
157 int begin = 0;
158 int nGenerators = numberOfGenerators();
159 int end = nGenerators;
160 // add generators
161 for (int j = 0; j < nGenerators; j++) {
162 groupValues.add(generatorValues.get(j));
163 groupKeys.add(generatorKeys.get(j));
164 }
165 for (int lev = 1; lev < depth; lev++) {
166 int count=0;
167 for (int i = begin; i < end; i++) {
168 GroupElement word = (GroupElement) groupValues.get(i);
169 String key = (String) groupKeys.get(i);
170 for (int j = 0; j < nGenerators; j++) {
171 GroupElement gen = (GroupElement) generatorValues.get(j);
172 String genKey = (String) generatorKeys.get(j);
173 if (isIrreducible(key, genKey)) {
174 groupKeys.add(key + genKey);
175 GroupElement newWord = word.rmultby(gen);
176 newWord.setLevel(word.getLevel()+1);
177 groupValues.add(newWord);
178 count++;
179 }
180 }
181 }
182 begin = end;
183 end = end + count;
184 }
185 isGroupComputed = true;
