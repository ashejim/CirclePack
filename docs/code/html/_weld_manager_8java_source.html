178 public int findWeldMap(PackData p,PackData q, int V, int W, int n) {
179 int count = 0;
180 boolean closeup = false;
181
182 if (!p.status || !q.status || p.hes >= 0 || q.hes >= 0 ||
183 p.euler != 1 || p.genus != 0 || q.euler != 1 || q.genus != 0) {
184 Oops("findWeldMap: packings must be "
185 + " topological discs in hyperbolic geometry");
186 }
187 if (V < 1 || V > p.nodeCount || !p.isBdry(V) ||
188 W < 1 || W > q.nodeCount || !q.isBdry(W)) {
189 Oops("findWeldMap: given vertices are improper");
190 }
191
192 // reset 'bdryStarts'
193 p.bdryStarts[1] = V;
194 q.bdryStarts[1] = W;
195
196 // find
197 NodeLink p_blist = new NodeLink(p, "b");
198 NodeLink q_blist = new NodeLink(q, "b");
199 // must reverse the order of q_blist and rotate last to first
200 q_blist = q_blist.reverseMe();
201 q_blist.removeLast();
202 q_blist.add(0,Integer.valueOf(W));
203
204 int npb = p_blist.size();
205 int nqb = q_blist.size();
206
207 if (npb < 3 || nqb < 3 || npb < n || (n > 0 && nqb < n)) { // asked for
208 // n matches
209 Oops("findWeldMap: not enough edges vis-a-vis given n = " + n);
210 }
211 if (n < 0 && npb != nqb) { // n<0 means match full boundaries
212 Oops("findWeldMap: edge counts don't agree.");
213 }
214 if (n < 0 || npb <= n)
215 closeup = true;
216
217 // Create bdry center list, reverse q, normalize to argument 0, check
218 // modulus
219 if (closeup) { // full bdry
220 weldmapDomain = new double[npb + 1]; // room to close up
221 weldmapRange = new double[npb + 1]; // room to close up
222 } else {
223 weldmapDomain = new double[n];
224 weldmapRange = new double[n];
225 }
226
227 double dist;
228 double maxx = -1000000;
229 double p_arg0, q_arg0;
230
231 // normalize args in both p and q to start at zero
232 weldmapDomain[0] = 0.0;
233 p_arg0 = p.getCenter(V).arg();
234 count = 1;
235 Iterator<Integer> pl = p_blist.iterator();
236 int v = (Integer)pl.next();
237 while (pl.hasNext()) {
238 v = (Integer) pl.next();
239 weldmapDomain[count] = p.getCenter(v).arg() - p_arg0;
240 while (weldmapDomain[count] < weldmapDomain[count - 1])
241 weldmapDomain[count] += 2.0 * Math.PI;
242 maxx = ((dist = Math.abs(1.0 - (p.getCenter(v)).abs())) > maxx) ? dist
243 : maxx;
244 if (maxx>.01)
245 System.out.println("p "+v);
246 count++;
247 }
248 if (closeup)
249 weldmapDomain[count] = 2.0 * Math.PI; // TODO: or do we want 0.0?
250
251 // reverse normalization for q (recall: ordering of q_blist already
252 // reversed)
253 weldmapRange[0] = 0.0;
254 q_arg0 = q.getCenter(W).arg();
255 count = 1;
256 Iterator<Integer> ql = q_blist.iterator();
257 v=(Integer)ql.next();
258 while (ql.hasNext()) {
259 v = (Integer) ql.next();
260 weldmapRange[count] = q_arg0 - q.getCenter(v).arg();
261 while (weldmapRange[count] < weldmapRange[count - 1])
262 weldmapRange[count] += 2.0 * Math.PI;
263 maxx = ((dist = Math.abs(1.0 - (q.getCenter(v)).abs())) > maxx) ? dist
264 : maxx;
265 if (maxx>.01)
266 System.out.println("q "+v);
267 count++;
268 }
269 if (closeup)
270 weldmapRange[count] = 2.0 * Math.PI; // TODO: or do we want 0.0?
271
272 // drifted too far from unit circle?
273 if (maxx > .01) {
274 Oops("findWeldMap: packings don't seem to be maximal.");
275 }
276 return count;
277 }

404 public PackData unweld(PackData p,EdgeLink elist) {
405
406 if (elist==null || elist.size()==0)
407 return null;
408 PackData q=p.copyPackTo();
409 HalfLink hlist=new HalfLink();
410 hlist.addSimpleEdges(q.packDCEL,elist);
411
412 // check that 'hlist' is simple and closed
413 int[] vhits=new int[q.nodeCount+1];
414 Iterator<HalfEdge> his=hlist.iterator();
415 int firstv=his.next().origin.vertIndx;
416 int lastv=firstv;
417 vhits[lastv]=-1; // on the path
418 NodeLink nghbs=new NodeLink();
419 while (his.hasNext()) {
420 HalfEdge he=his.next();
421 int v=he.origin.vertIndx;
422 if (v!=lastv || vhits[v]!=0) {
423 CirclePack.cpb.errMsg(
424 "unweld: closed path was not simple or not connected");
425 return null;
426 }
427 vhits[v]=-1;
428
429 // catalog vertices to the left of 'hlist'
430 int w=he.prev.origin.vertIndx;
431 if (vhits[w]>=0 && !q.isBdry(w)) {
432 vhits[w]=1;
433 nghbs.add(w);
434 }
435
436 lastv=v;
437 }
438
439 if (lastv!=firstv) {
440 CirclePack.cpb.errMsg("unweld: path did not close up");
441 return null;
442 }
443
444 // set alpha to be to the left of hlist
445 Iterator<Integer> nis=nghbs.iterator();
446 HalfEdge alp=null;
447 while (alp==null && nis.hasNext()) {
448 Vertex vert=q.packDCEL.vertices[nis.next()];
449 HalfLink spokes=vert.getSpokes(null);
450 Iterator<HalfEdge> sis=spokes.iterator();
451 while (alp==null && sis.hasNext()) {
452 HalfEdge he=sis.next();
453 int w=he.twin.origin.vertIndx;
454 if (vhits[w]==0 && !q.isBdry(w))
455 alp=he;
456 }
457 }
458 if (alp==null) {
459 CirclePack.cpb.errMsg(
460 "unweld: did not find an alpha left of path");
461 return null;
462 }
463 q.packDCEL.alpha=alp;
464 q.setAlpha(alp.origin.vertIndx);
465
466 // cookie out by using 'hlist' as the new red chain
467 int ans=CombDCEL.redchain_by_edge(
468 q.packDCEL,hlist,q.packDCEL.alpha,false);
469 if (ans<=0)
470 CirclePack.cpb.errMsg("unweld: failed to get red chain");
471
472 q.packDCEL.fixDCEL(q);
473 return q;
474 }

490 public int weld_map_to_list(PackData p, PackData q, int p_start_vert,
491 int q_start_vert, int p_count, int q_count, String weldmapfile,
492 boolean script_flag,int opt_flag) {
493 int j, n, pn, qn;
494 int[] p_vert = null; // counterclockwise closed list of 'p' boundary vertices
495 int[] q_vert = null; // clockwise closed list of 'q' boundary vertices
496 boolean full_flag = true;
497 double x1=0.0;
498 double x2=0.0;
499 double y1=0.0;
500 double y2=0.0;
501 double tolr;
502 double[] p_coord = null; // coordinates of p's bdry verts (on [0,1]).
503 double[] q_coord = null; // coordinates of q's bdry verts (on [0,1]).
504
505 // read the weld map of x-y coords.
506 Path2D.Double path_list = PathManager.readpath(weldmapfile,script_flag);
507 if (path_list == null) { // default to identity: [0,1] to [0,1]
508 path_list = new Path2D.Double();
509 path_list.moveTo(0.0, 0.0);
510 path_list.lineTo(0.5, 0.5);
511 path_list.lineTo(1.0, 1.0);
512 }
513
514 // TODO: need to tune this cutoff for rounding off a vertex
515 tolr = 0.1;
516
517 // TODO: should do more error checking
518 if (!p.isBdry(p_start_vert) || !q.isBdry(q_start_vert))
519 Oops("start vertices must be on the boundary");
520
521 // open temporary file for weld list
522 try {
523 BufferedWriter fp = CPFileManager.openWriteFP(new File(System
524 .getProperty("java.io.tmpdir")), weldListFileName, false);
525 if (fp == null) {
526 Oops("weld: couldn't open " + weldListFileName);
527 }
528
529 /*
530 * Create appropriate bdry lists in p_vert/q_vert vectors. Count =
531 * -1 indicates full bdry; we also set this if conditions
532 * force it.
533 */
534
535 // default to full boundaries; not full is not yet used
536 if (p_count > 2 && q_count > 2)
537 full_flag = false;
538
539 // find boundary counts
540 int p_hits = 0;
541 for (int vv = 1; vv <= p.nodeCount; vv++)
542 if (p.isBdry(vv))
543 p_hits++;
544 int q_hits = 0;
545 for (int vv = 1; vv <= q.nodeCount; vv++)
546 if (q.isBdry(vv))
547 q_hits++;
548
549 // create space
550 if (full_flag || p_count >= p_hits || q_count >= q_hits) {
551 // Not enough edges, have to do all of both bdry's
552 full_flag = true;
553 p_count = p_hits;
554 q_count = q_hits;
555 }
556 p_vert = new int[p_count + 1];
557 q_vert = new int[q_count + 1];
558
559 // for full bdry, close with repeated first entry
560 int last;
561 p_vert[0] = last = p_start_vert;
562 for (n = 1; n <= p_count; n++)
563 // p list positive (counterclockwise)
564 p_vert[n] = last = p.getFirstPetal(last);
565 q_vert[0] = last = q_start_vert;
566 for (n = 1; n <= q_count; n++)
567 // Recall: q list must be clockwise.
568 q_vert[n] = last = q.getLastPetal(last);
569
570 // Use opt_flag here (TODO: expect more options eventually).
571
572 // default: two hyperbolic max packings --- check that they
573 // appear to be maximal, use arguments of center points
574 // (increasing arguments normalized to [0,1]).
575 // eucl: use bdry edge length --- check that both are eucl;
576 // both are normalized to [0,1]. */
577
578 // if first bit not set and eucl, set 0001
579 if ((opt_flag & (0001 | 0002 | 0004)) == 0) {
580 if (p.hes == 0 && q.hes == 0)
581 opt_flag |= 0001;
582 }
583
584 // if first bit not now set, check that these are
585 // max packings --- bdry centers on unit circle?
586 double dist;
587 double pmax=0.0;
588 double qmax=0.0;
589 if ((opt_flag & (0001 | 0002 | 0004)) == 0) {
590 for (n = 0; n <= p_count; n++)
591 pmax = ((dist = Math.abs(1.0 - p.getCenter(p_vert[n]).abs()))
592 > pmax) ? dist : pmax;
593 for (n = 0; n <= q_count; n++)
594 qmax = ((dist = Math.abs(1.0 - q.getCenter(q_vert[n]).abs()))
595 > qmax) ? dist : qmax;
596
597 if (pmax > .01 || qmax > .01) {
598 Oops("weld_map: packings don't seem to be maximal.");
599 }
600
601 // put coords in vectors, normalize each to [0,1].
602 p_coord = new double[p_count + 1];
603 double p_ang0 = p.getCenter(p_vert[0]).arg();
604 p_coord[0] = 0.0;
605 double p_arg;
606 for (n = 1; n <= p_count; n++) {
607 p_arg = p.getCenter(p_vert[n]).arg();
608 p_coord[n] = p_arg - p_ang0;
609 // make arguments monotone
610 while (p_coord[n] < p_coord[n - 1])
611 p_coord[n] += 2.0*Math.PI;
612 }
613 for (n = 1; n <= p_count; n++)
614 p_coord[n] /= p_coord[p_count];
615
616 q_coord = new double[q_count + 2];
617 double q_ang0 = q.getCenter(q_vert[0]).arg();
618 q_coord[0] = 0.0;
619 double q_arg;
620 for (n = 1; n <= q_count; n++) {
621 q_arg = q.getCenter(q_vert[n]).arg();
622 q_coord[n] = q_ang0 - q_arg;
623 // make arguments monotone
624 while (q_coord[n] < q_coord[n - 1])
625 q_coord[n] += 2.0*Math.PI;
626 }
627 for (n = 1; n <= q_count; n++)
628 q_coord[n] /= q_coord[q_count];
629
630 } // end of opt_flag=0 setup
631
632 // use euclidean edge lengths, normalized to 1
633 // TODO: currently use sum of radii instead of distance
634 // between centers.
635 else if ((opt_flag & 0001) != 0) {
636 if (p.hes != 0 || q.hes != 0) { // not eucl
637 Oops("weld: packings should be euclidean for given option");
638 }
639 p_coord = new double[p_count + 2];
640 p_coord[0] = 0.0;
641 for (n = 1; n <= p_count; n++)
642 p_coord[n] = p_coord[n - 1] + p.getRadius(p_vert[n - 1])
643 + p.getRadius(p_vert[n]);
644 for (j = 1; j <= p_count; j++)
645 p_coord[j] /= p_coord[p_count]; // normalize
646 q_coord = new double[q_count + 2];
647 q_coord[0] = 0.0;
648 for (n = 1; n <= q_count; n++)
649 q_coord[n] = q_coord[n - 1] + q.getRadius(q_vert[n - 1])
650 + q.getRadius(q_vert[n]);
651 for (j = 1; j <= q_count; j++)
652 q_coord[j] /= q_coord[q_count]; // normalize
653 } // end of eucl case
654
655 /*
656 * Want to transfer p_coords/q_coords to common
657 * interval [0,1] so they can be compared. For
658 * balance we do this in a neutral way, in essence
659 * replacing the path x-->y by x-->(x+y)/2. This should
660 * maintain monotonicity, but not favor p or q.
661 * We simultaneously do the linear interpolation.
662 */
663
664 double[] coords = new double[2];
665 @SuppressWarnings("unused")
666 int type=0;
667
668 // first with p
669 PathIterator plist = path_list
670 .getPathIterator(new AffineTransform());
671
672 // have to get started with first couple points
673 type = plist.currentSegment(coords);
674 x1 = coords[0];
675 y1 = coords[1];
676 if (!plist.isDone()) {
677 plist.next();
678 type = plist.currentSegment(coords);
679 x2 = coords[0];
680 y2 = coords[1];
681 }
682 else {
683 Oops("weld_map_to_list: too few coords.");
684 }
685 pn = 1;
686
687 while (pn < p_count) {
688 // arrange x1<p_coord<x2
689 while (!plist.isDone() && (x2 < p_coord[pn])) {
690 x1 = x2;
691 y1 = y2;
692 plist.next();
693 type = plist.currentSegment(coords);
694 x2 = coords[0];
695 y2 = coords[1];
696 }
697 if (plist.isDone()) { // ran out of function?
698 Oops("weld_map_to_list: problem: coords don't match map.");
699 }
700 p_coord[pn] = ((y1 + (p_coord[pn] - x1)
701 * ((y2 - y1) / (x2 - x1))) + p_coord[pn]) / 2.0;
702 pn++;
703 }
704 // get the last p coord
705 plist = path_list.getPathIterator(new AffineTransform());
706 do {
707 plist.next();
708 type = plist.currentSegment(coords);
709 x1 = coords[0];
710 y1 = coords[1];
711 } while (!plist.isDone());
712 p_coord[p_count] = (x1 + y1) / 2.0;
713
714 // same with q
715 plist = path_list.getPathIterator(new AffineTransform());
716
717 // have to get started with first couple points
718 type = plist.currentSegment(coords);
719 x1 = coords[0];
720 y1 = coords[1];
721 if (!plist.isDone()) {
722 plist.next();
723 type = plist.currentSegment(coords);
724 x2 = coords[0];
725 y2 = coords[1];
726 } else {
727 Oops("weld_map_to_list: too few coords.");
728 }
729 pn = 1;
730
731 while (pn < q_count) {
732 // arrange y1<p_coord<y2
733 while (!plist.isDone() && (y2 < q_coord[pn])) {
734 x1 = x2;
735 y1 = y2;
736 plist.next();
737 type = plist.currentSegment(coords);
738 x2 = coords[0];
739 y2 = coords[1];
740 }
741 if (plist.isDone()) { // ran out of function?
742 Oops("weld_map_to_list: problem: coords don't match map.");
743 }
744// q_coord[pn] = ((y1 + (q_coord[pn] - x1)
745// * ((y2 - y1) / (x2 - x1))) + q_coord[pn]) / 2.0;
746 q_coord[pn]=((x1+(q_coord[pn]-y1)*((x2-x1)/(y2-y1)))+q_coord[pn])/2.0;
747 pn++;
748 }
749 // get the last p coord
750 plist = path_list.getPathIterator(new AffineTransform());
751 do {
752 plist.next();
753 type = plist.currentSegment(coords);
754 x1 = coords[0];
755 y1 = coords[1];
756 } while (!plist.isDone());
757 q_coord[q_count] = (x1 + y1) / 2.0;
758
759 // Now for the determination of new vertex creation
760
761 fp.write("Vv\n"); // Always weld p_start_vert to q_start_vert for now;
762 // might change the convention later
763
764 /*
765 * Heart of the algorithm: walk around bdry of p and q looking at
766 * locations of p_vert and q_vert. (We always weld p onto q.)
767 *
768 * If q_vert maps "past" p_vert, then we could add vertex to q so
769 * that p_vert has a vertex to glue onto. If q_vert maps to a
770 * location "before" p_vert, then we could add a vertex to p so that
771 * q_vert has a vertex to glue onto.
772 *
773 * Since p_coord/q_coord values are now intermingled on same
774 * interval [0,1], just have to tick off new/existing verts in
775 * order. Hard part is deciding when two are close enough to be
776 * identified.
777 */
778
779 pn = qn = 1;
780 while (pn <= p_count && qn <= q_count) {
781 /*
782 * TODO: very hard to decide how to parse this.
783 * In the theory you need to avoid "thin triangles",
784 * mainly occurring when new verts generated are
785 * too close to existing ones. To avoid this,
786 * try to put interval of attraction around each
787 * vert to tell when new verts should be identified.
788 * Simple rule now: bracket each vert by "tolr"
789 * proportion of distance to previous/next vert,
790 * if they intersect, identify those verts. (This
791 * can be preempted if other verts intervene.)
792 * Also have to take care in approaching the
793 * end verts.
794 */
795
796 if (pn==p_count && qn==q_count) { // ends?, identify
797 fp.write("Vv\n");
798 pn++;
799 qn++;
800 break;
801 }
802 if (pn == p_count) { // end of p, must add new to q
803 fp.write("Nv\n");
804 qn++;
805 } else if (qn == q_count) { // end of q, must add to p

852 int weld(PackData p, PackData q, int v, int w, int opt_flag) {
853 int ans = 1;
854 int count = 0;
855 int v_next, w_next;
856 int v_orig = v;
857 int w_orig = w;
858
859 if (!p.status || !q.status || (v > p.nodeCount) ||
860 (w > q.nodeCount) || v < 1 || w < 1 ||
861 !p.isBdry(v) || !q.isBdry(w)) {
862 Oops("weld: improper data; e.g., verts " + v + " and " + w
863 + " cannot be used");
864 }
865 // TODO: might save orig p and q in case this bombs
866
867 // Augment p and q
868 try {
869 BufferedReader fpr = CPFileManager.openReadFP(new File(System
870 .getProperty("java.io.tmpdir")), weldListFileName, false);
871 if (fpr == null) {
872 throw new DataException("weld file " + weldListFileName
873 + " failed to open.");
874 }
875 String line = StringUtil.ourNextLine(fpr);
876 if (!line.contains("V") || !line.contains("v")) {
877 Oops("weld list: first line must be 'Vv'.");
878 }
879 v_next = p.getFirstPetal(v);
880 w_next = q.getLastPetal(w);
881 while ((line = StringUtil.ourNextLine(fpr)) != null) {
882 count = count + 1;
883
884 // check 'p' situation (upper case letters)
885 if (line.contains("V")) {
886 v = v_next;
887 v_next = p.getFirstPetal(v);
888 } else if (line.contains("N")) {
889 if (add_between(p, v, v_next) == 0)
890 return (-1);
891 v = p.getFirstPetal(v);
892 // v = newly added new vertex
893 } else { /* extraneous stuff? */
894 Oops("weld: weld file format problem, upper case");
895 }
896
897 if (line.contains("v")) {
898 w = w_next;
899 w_next = q.getLastPetal(w);
900 } else if (line.contains("n")) {
901 if (add_between(q, w_next, w) == 0)
902 return -1;
903 w = q.getLastPetal(w);
904 // w = newly added new vertex
905 } else { /* extraneous stuff? */
906 Oops("weld: weld file format problem, lower case");
907 }
908 } // end of while
909 } catch (Exception ex) {
910 throw new DataException("weld: problem reading list: "
911 + ex.getMessage());
912 }
913
914 String buf = null;
915 if (v == p.getFirstPetal(v_orig)) {
916 buf = new String("b");
917 } else {
918 v = p.getLastPetal(v);
919 buf = new String("b(" + v_orig + " " + v + ")");
920 }
921 p.vlist = new NodeLink(p, buf);
922 p.elist = new EdgeLink(p, buf);
923 if (w == q.getLastPetal(w_orig)) {
924 buf = new String("b");
925 } else {
926 w = q.getFirstPetal(w);
927 buf = new String("b(" + w + " " + w_orig + ")");
928 }
929 q.vlist = new NodeLink(q, buf);
930 q.elist = new EdgeLink(q, buf);
931
932 // Adjoin p and q?
933 if (opt_flag>0) {
934 q.bdryStarts[1] = w_orig;
935
936 p.packDCEL=CombDCEL.adjoin(p.packDCEL,
937 q.packDCEL, v_orig, w_orig, count);
938 p.packDCEL.fixDCEL(p);
939 // TODO: note that 'adjoin' pastes p clockwise, q
940 // counterclockwise; no problem for full bdry, but
941 // with partial bdry's, have to do something else.
942 // TODO: use to pass back 'Oldnew' to define vertex_map of q.
943 for (v = 1; v <= p.nodeCount; v++)
944 p.setPlotFlag(v,1);
945 }
946 else {
947 msg("weld: two packs appear ready to adjoint:\n"
948 + " designated bdry vertices are " + v + " and " + w
949 + ", resp.," + " alpha vertex of outer pack is " + q.getAlpha());
950 }
951 return ans;
952 }

966 public int add_between(PackData p, int v, int v_next) {
967 // Basic checks
968 HalfEdge hedge=p.packDCEL.findHalfEdge(new EdgeSimple(v,v_next));
969 if (!p.isBdry(v) || !p.isBdry(v_next) ||
970 hedge==null || !hedge.isBdry()) {
971 CirclePack.cpb.errMsg(
972 "add_between: bad data");
973 return 0;
974 }
975
976 if (hedge.face.faceIndx<0)
977 hedge=hedge.twin;
978
979 // if no interior petal, have to split opposite edge first
980 if (hedge.next.isBdry()) {
981 HalfEdge opp=hedge.next.next;
982 if (opp.isBdry() ||
983 RawManip.splitEdge_raw(p.packDCEL,opp)==null) {
984 CirclePack.cpb.errMsg(
985 "add_between: opp edge is bdry or failed to split");
986 return 0;
987 }
988 }
989
990 // add new vertex between v and v_next
991 if (RawManip.splitEdge_raw(p.packDCEL,hedge)==null) {
992 CirclePack.cpb.errMsg(
993 "add_between: failure in splitting "+hedge);
994 return 0;
995 }
996
997 p.packDCEL.fixDCEL(p);
998 return p.nodeCount;
999
