47 public void adjustParameters(String specStr) {
48
49 // process 'specs' string: -m {min} -M {max} -ftn {ftn} -Pi *pi -cmd {str}
50 Vector<Vector<String>> specSegs=StringUtil.flagSeg(specStr);
51 Iterator<Vector<String>> segs=specSegs.iterator();
52 Vector<String> items=null;
53 while (segs.hasNext()) {
54 try {
55 items=segs.next();
56 String str=items.remove(0);
57 if (str.equals("-m")) { // minimum
58 minValue=Double.parseDouble(items.get(0));
59 }
60 else if (str.equals("-M")) { // max
61 maxValue=Double.parseDouble(items.get(0));
62 }
63 else if (str.equals("-cmd")) { // command (only if nonempty)
64 commandActive=true;
65 // can turn off by setting without string
66 if (items.size()>0 && items.get(0).length()>0) {
67 command=StringUtil.reconItem(items);
68 }
69 else
70 commandActive=false;
71 }
72 else if (str.equals("-ftn")) { // function
73 functionActive=true;
74
75 // may leave 'function' at default 'z*pi'
76 if (items.size()>0)
77 function=StringUtil.reconItem(items);
78 }
79 else if (str.equals("-Pi")) { // set function to z*pi
80 functionActive=true;
81 function="z*pi";
82 }
83 } catch (Exception ex) {
84 CirclePack.cpb.errMsg("slider spec problem with '"+items.get(0));
85 }
86
87 } // end of while to get specs
