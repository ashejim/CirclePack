59 // TODO: have problems in changing underlying packData; so for now,

78 // ----- hex ---- cookie Gamma from a regular hex, radii x
79 else if (cmd.startsWith("????hex")) {
80 double r;
81 try {

83 } catch(Exception ex) {
84 throw new ParserException("usage: hex {x} for radii size");
85 }
86 int n=(int)(((1/r)-1.0)/2.0); // number of generations to fill unit disc
87
88 // cookie out from the regular hex and display
89 return cpCommand(extenderPD,"seed;add_gen "+n+" 6;gamma 10;"+
90 "set_rad "+r+" a;layout;cookie;disp -w -c -g");
91 }
92
93 // ----- choose ---
94 else if (cmd.startsWith("choose")) {
95 theChosen=PackData.resample(extenderPD,CPBase.ClosedPath,mode, maxThin);
96 if (theChosen==null)
97 throw new ParserException("filed to choose 'theChosen'");
98 extenderPD.vlist=theChosen.makeCopy();
99 return theChosen.size();
100 }
101
102 // ------ pack
103 else if (cmd.startsWith("pack")) {
104 outputData=PackData.sampledSubPack(extenderPD,theChosen);
105 if (outputData==null)
106 return 0;
107 return outputData.nodeCount;
108 }
109
110 // ========== copy <pnum>
111 else if (cmd.startsWith("copy")) { // copy 'outputData' in some pack
112 try {
113 items=(Vector<String>)flagSegs.get(0);
114 int pnum=Integer.parseInt((String)items.get(0));
115 if (pnum==extenderPD.packNum)
116 return 0;
117 CirclePack.cpb.swapPackData(outputData,pnum,false);
118 } catch (Exception ex) {
119 return 0;
120 }
121 return 1;
122 }
123
124 return super.cmdParser(cmd, flagSegs);
