272 public static int reapRadii(PackData p,NodeLink vlist,int mode) {
273 PackDCEL pdcel=p.packDCEL;
274 if (pdcel.triData==null ||
275 pdcel.triData.length<(pdcel.faceCount+1))
276 throw new ParserException("'triData' allocation problem");
277 if (vlist==null || vlist.size()==0)
278 vlist=new NodeLink(p,"a");
279 Iterator<Integer> vis=vlist.iterator();
280 int count=0;
281 while (vis.hasNext()) {
282 combinatorics.komplex.Vertex vert=pdcel.vertices[vis.next()];
283 if (vert.redFlag) {
284 HalfLink spokes=vert.getEdgeFlower();
285 Iterator<HalfEdge> sis=spokes.iterator();
286 while (sis.hasNext()) {
287 HalfEdge he=sis.next();
288 int f=he.face.faceIndx;
289 int indx=pdcel.triData[f].edgeIndex(he);
290 if (he.myRedEdge!=null) {
291 if (mode==2)
292 pdcel.setRad4Edge(he,pdcel.triData[f].labels[indx]);
293 else
294 pdcel.setRad4Edge(he,pdcel.triData[f].radii[indx]);
295 }
296 }
297 }
298 else { // store from first face containing v
299 int v=vert.vertIndx;
300 combinatorics.komplex.DcelFace face=pdcel.vertices[v].halfedge.face;
301 TriData vtd=pdcel.triData[face.faceIndx];
302 int vindx=vtd.vertIndex(v);
303 if (mode==2)
304 p.setRadius(v,vtd.labels[vindx]);
305 else
306 p.setRadius(v,vtd.radii[vindx]);
307 }
308 count++;
309 }
310 p.fillcurves();
311 return count;
312 }
