627 public static TileData getRulesTD(TileRule tRule, EdgeLink type2Size,int md) {
628
629 // initiate TileData and its tiles, start with 'vert's set to -1.
630 TileData tileData = new TileData(tRule.childCount,md);
631 for (int j = 1; j <= tRule.childCount; j++) {
632 int sz = type2Size.findW(tRule.childType[j]);
633 tileData.myTiles[j] = new Tile(sz);
634 tileData.myTiles[j].tileIndex = j;
635 tileData.myTiles[j].tileType = tRule.childType[j];
636 tileData.myTiles[j].mark = tRule.childMark[j];
637 tileData.myTiles[j].tileFlower = new int[sz][2];
638 for (int k = 0; k < sz; k++) {
639 tileData.myTiles[j].vert[k] = -1; // set negative to start
640 tileData.myTiles[j].tileFlower[k][0] = tRule.childFlower[j][k][0];
641 tileData.myTiles[j].tileFlower[k][1] = tRule.childFlower[j][k][1];
642 }
643 }
644
645 // last vertex index used
646 int lastv = 0;
647 int[] doneTiles = new int[tileData.tileCount + 1];
648
649 // keep two lists of tile indices in processing
650 NodeLink curr = new NodeLink();
651 NodeLink next = new NodeLink();
652 next.add(1);
653 while (next.size() > 0) {
654 curr = next;
655 next = new NodeLink();
656
657 while (curr.size() > 0) {
658
659 // next tile to handle if not done
660 int base_tile = curr.remove(0);
661 if (base_tile > 0 && doneTiles[base_tile] <= 0) {
662
663 Tile tile = tileData.myTiles[base_tile];
664
665 int safety = 10 * tile.vertCount;
666 while (safety > 0 && doneTiles[base_tile] <= 0) {
667 safety--;
668
669 // check if we're done with all its vertices
670 boolean done = true;
671 for (int j = 0; (j < tile.vertCount && done); j++)
672 if (tile.vert[j] < 0)
673 done = false;
674
675 if (done) {
676 doneTiles[base_tile] = 1;
677 break;
678 }
679
680 // process vertices that aren't finished
681 for (int j = 0; j < tile.vertCount; j++) {
682 int base_v = tile.vert[j];
683
684 if (base_v < 0) {
685 base_v=tile.vert[j]=++lastv;
686
687 // ======= go ccw through faces first
688
689 // we always have current situation and next situation
690 // use einfo = (nghbtile, nghbindex) to move ccw around 'base_v'
691 int curr_tile = base_tile;
692 Tile cTile = tileData.myTiles[curr_tile];
693 int curr_edge = (j - 1 + cTile.vertCount)% cTile.vertCount; // upstream edge
694 int[] einfo = cTile.tileFlower[curr_edge];
695 int next_tile = einfo[0];
696 int next_edge = einfo[1];
697
698 // touch this tile for future processing
699 if (next_tile>0 && next.containsV(next_tile)<0)
700 next.add(next_tile);
701
702 // proceed until reaching bdry edge or returning to base_tile across edge j
703 while (next_tile > 0 && (next_tile!=base_tile || next_edge!=j )) {
704
705 Tile nTile = tileData.myTiles[next_tile];
706
707 // check vert at base_v on next_tile
708 int nv = nTile.vert[next_edge];
709 if (nv > 0 && nv != base_v)
710 throw new DataException("error: doesn't match base_v");
711 if (nv < 0)
712 nv = nTile.vert[next_edge] = base_v;
713
714 // done? base_tile, edge downstream from base_v?
715 // Note: may be okay to reenter base_tile on some other edge
716 if (next_tile == base_tile && next_edge == j) {
717 next_tile = 0;
718 break;
719 }
720
721 // else reset curr/next.
722 curr_tile = next_tile;
723 cTile = tileData.myTiles[curr_tile];
724 curr_edge = (next_edge - 1 + cTile.vertCount)%cTile.vertCount;
725 einfo = cTile.tileFlower[curr_edge];
726 next_tile = einfo[0];
727 next_edge = einfo[1];
728
729 // touch this tile for future processing
