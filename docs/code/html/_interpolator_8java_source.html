135 public ZRhold interpolate(ZRhold zr1,ZRhold zr2,double x,int hes) {
136 Complex z1=zr1.getCenter();
137 double r1=(double)zr1.getRadius();
138 Complex z2=zr2.getCenter();

140 Double rad=Double.valueOf(r1*(1-x)+r2*x);
141 Complex Z=null;
142 if (hes<=0) {
143 Z= new Complex(z1.times(1-x).
144 add(z2.times(x)));
145 }
146 if (hes>0) {
147 Point3D mid=new Point3D(0.0,0.0,0.0);
148 Point3D p1=z1.getAsPoint();
149 Point3D p2=z2.getAsPoint();
150 mid.x=p1.x*(1-x)+p2.x*x;
151 mid.y=p1.y*(1-x)+p2.y*x;
152 mid.z=p1.z*(1-x)+p2.z*x;
153 Z=new Complex(mid.getTheta(),mid.getPhi());
154 }
