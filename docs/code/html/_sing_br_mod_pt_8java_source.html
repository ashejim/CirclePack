105 public int modifyPackData() {
106 faceEdge=new HalfEdge[3];
107 secondEdge=new HalfEdge[3];
108 corner=new Vertex[3];
109 connect=new Vertex[3];
110 guide=new Vertex[3];
111
112 faceEdge[0]=myEdge;
113 faceEdge[1]=faceEdge[0].next;
114 faceEdge[2]=faceEdge[1].next;
115 for (int j=0;j<3;j++) {
116 corner[j]=faceEdge[j].origin;
117 HalfEdge he=faceEdge[(j+1)%3];
118 connect[j]=he.twin.prev.origin;
119 }
120
121 // create 'eventHorizon'
122 eventHorizon=new HalfLink();
123 for (int j=0;j<3;j++) {
124 int v=corner[j].vertIndx;
125 int w=corner[(j+1)%3].vertIndx;
126 int u=connect[(j+2)%3].vertIndx;
127 EdgeSimple es=new EdgeSimple(v,u);
128 eventHorizon.add(pdc.findHalfEdge(es));
129 es=new EdgeSimple(u,w);
130 eventHorizon.add(pdc.findHalfEdge(es));
131 }
132
133 // create 'vanishing'
134 int van=RawManip.addBary_raw(pdc,faceEdge[0],false);
135 vanishing=pdc.vertices[van];
136
137 // successively split 'faceEdge's
138 for (int j=0;j<3;j++) {
139 RawManip.splitEdge_raw(pdc,faceEdge[j]);
140 guide[(j+2)%3]=faceEdge[j].twin.origin;
141 secondEdge[(j+1)%3]=faceEdge[j].twin.prev.twin.prev;
142 }
143
144 // get encircling link for local holonomy
145 myHoloBorder=HalfLink.leftsideLink(pdc,eventHorizon);
146 myHoloBorder.add(0,eventHorizon.get(0));
147
148 // add 'myHoloBorder' to 'layoutAddons'
149 layoutAddons=new HalfLink();
150 layoutAddons.abutMore(myHoloBorder);
151
152 // record exclusions
153 myExclusions=new ArrayList<combinatorics.komplex.Vertex>();
154 Iterator<HalfEdge> eis=eventHorizon.iterator();
155 while (eis.hasNext()) {
156 myExclusions.add(eis.next().origin);
157 }
158 myExclusions.add(vanishing);
159 myExclusions.add(guide[0]);
160 myExclusions.add(guide[1]);
161 myExclusions.add(guide[2]);
162
163 return pdc.vertCount;
164 }

182 public int setParameters(Vector<Vector<String>> flagSegs) {
183 int count=0;
184 boolean gotovers=false;
185 double []ovlp=new double[2];
186
187 // get flags: -a aim, -o o1 o2
188 Iterator<Vector<String>> fit=flagSegs.iterator();
189 while (fit.hasNext()) {
190 Vector<String> items=fit.next();
191 try {
192 if (!StringUtil.isFlag(items.get(0)))
193 throw new ParserException("usage: -a aim -o o1 o2");
194 String str=items.remove(0);
195 switch (str.charAt(1)) {
196 case 'a': // new aim
197 {
198 myAim=Double.parseDouble(items.get(0));
199 count++;
200 break;
201 }
202 case 'o': // overlaps
203 {
204 for (int i=0;i<2;i++) {
205 ovlp[i]=Double.parseDouble(items.remove(0));
206 if (ovlp[i]<=0 || ovlp[i]>=1.0)
207 throw new ParserException("overlap not in (0,1)");
208 count++;
209 }
210 if ((ovlp[0]+ovlp[1])>.99999999)
211 throw new ParserException("sum of o1 o2 overlaps not in (0,1)");
212 gotovers=true;
213 break;
214 }
215 } // end of switch
216 } catch(Exception ex) {
217 throw new ParserException("usage: -a aim, -o o1 o2");
218 }
219 } // end of while
220
221 // set overlaps
222 if (gotovers)
223 count+=resetOverlaps(ovlp[0],ovlp[1]);
224
225 return count;
226 }

236 public int resetOverlaps(double o0,double o1) {
237 if (o0<=0.0 || o0>=1.0 || o1<=0.0 || o1>=1.0 || (o0+o1)>=1.0)
238 throw new DataException(
239 "'singular' usage: 2 overlaps in [0,1], sum <= 1");
240 overlaps[0]=o0;
241 overlaps[1]=o1;
242 overlaps[2]=1-(overlaps[0]+overlaps[1]);
243 for (int j=0;j<3;j++) {
244 faceEdge[j].setInvDist(Math.cos(overlaps[j]*Math.PI));
245 secondEdge[j].setInvDist(Math.cos(overlaps[j]*Math.PI));
246 }
247 return 1;

308
309 } // h/y circle case
310

312 // the branch value. This option draws those edges.

314 dispCode="-e";
315 hit=true;
316 target=new String(" "+vanishing+" corner[0] "+

331 newFlagSegs.add(items);
