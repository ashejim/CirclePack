689 // TODO: should we allow color/thickness info override??
690 int numSides = -1;
691 if (p.getSidePairs() == null || (numSides = p.getSidePairs().size()) == 0)
692 break;
693 boolean do_mate = false;
694 boolean do_circle = false;
695 boolean do_label = false;
696 if (sub_cmd.length() > 0) { // additional options
697 for (int j = 0; j < sub_cmd.length(); j++) {
698 char cc = sub_cmd.charAt(j);
699 if (cc == 'p')
700 do_mate = true;
701 else if (cc == 'n')
702 do_label = true;
703 else if (cc == 'c')
704 do_circle = true;
705 }
706 }
707 Vector<Integer> verts = new Vector<Integer>(numSides);
708 // default or 'a' to all
709 if (items == null || items.size() == 0
710 || ((String) items.get(0)).contains("a")) {
711 for (int j = 0; j < numSides; j++)
712 verts.add(j, j);
713 } else {
714 for (int j = 0; j < items.size(); j++) {
715 try {
716 int n = Integer.parseInt((String) items.get(j));
717 if (n >= 0 && n < numSides)
718 verts.add(j, n);
719 } catch (Exception ex) {
720 }
721 }
722 }
723 int n, k;
724 for (int j = 0; j < verts.size(); j++) {
725 n = (Integer) verts.get(j);
726 SideData epair = p.getSidePairs().get(n);
727 count += p.post_bdry_seg(pF, n, do_label, do_circle,
728 epair.color, tx);
729 if (do_mate) { // do paired edge?
730 SideData ep =p.getSidePairs().get(n);
731 if ((k = ep.mateIndex) >= 0)
732 p.post_bdry_seg(pF, k, do_label, do_circle,
733 epair.color, tx);
734 }
735 }
736 break;
737 }
738 case 't': // trinket (dots,crosses,etc.) at cir centers
739 {
740 // format: 't{x}' or 't{x}c{col}' where 'x' is trinket code.
741 // Note: should already have set 'trinket' code.
742 NodeLink nodeLink = new NodeLink(p, items);
743 if (nodeLink.size() <= 0)
744 break; // nothing in list
745
746 // record 'mark' definition in the PostScript file
747 pF.post_shape(trinket);
748 // for scaling trinkets
749 double diam = 5 / cpDrawing.pixFactor;
750
751 Iterator<Integer> vit = nodeLink.iterator();
752 while (vit.hasNext()) {
753 v = (Integer) vit.next();
754 z = p.getCenter(v);
755 boolean front = true;
756 if (p.hes > 0) { // move z to visual plane
757 z = cpDrawing.sphView.toApparentSph(z);
758 if (Math.cos(z.x) < 0)
759 front = false; // on back
760 else
761 z = SphView.s_pt_to_visual_plane(z);
762 }
763 if (p.hes <= 0 || front) {
764 if (dispFlags.colBorder) {
765 if (!dispFlags.colorIsSet)
766 dispFlags.setColor(p.packDCEL.vertices[v].color);
767 pF.postColorTrinket(z, diam, dispFlags.getColor());
768 } else
769 pF.postTrinket(z, diam); // background
770 count++;
771 }
772 }
773 break;
774 }
775 case 'T': // tiles (if they exist) (note: 'ConformalTiling' extender
776 // has more complete options)
777 // TODO: emulating DisplayParser; not yet done (7/9/2019)
778 {
779 if (p.tileData!=null && p.tileData.tileCount>0) {
780 // default to 'all'
781 TileLink tileLink=new TileLink(p.tileData,items);
782 if (tileLink==null || tileLink.size()==0)
783 for (int t=1;t<=p.tileData.tileCount;t++)
784 tileLink.add(t);
785 if (tileLink.size() <= 0)
786 break; // nothing in list
787
788 Iterator<Integer> vit = tileLink.iterator();
789 while (vit.hasNext()) {
790 int t = (Integer) vit.next();
791 if (t>0 && t<=p.tileData.tileCount) {
792 Tile tile=p.tileData.myTiles[t];
793 if (tile==null)
794 break;
795
796 if (!dispFlags.colorIsSet) {
797 dispFlags.setColor(tile.color);
798 }
799
800 // get list of tile border and make axis-extended edgelist
801 NodeLink cornlist=tile.tileBorderLink();
802 EdgeLink tedgelist=
803 EdgeLink.verts2edges(p.packDCEL,cornlist,true);
804 Iterator<EdgeSimple> tel=tedgelist.iterator();
805
806 // set face/bdry colors
807 Color fcolor=null;
808 Color bcolor=null;
809 if (dispFlags.fill) {
810 if (!dispFlags.colorIsSet)
811 dispFlags.setColor(tile.color);
812 fcolor=dispFlags.getFillColor();
813 }
814 if (dispFlags.draw) {
815 if (dispFlags.colBorder)
816 bcolor=dispFlags.getColor();
817 else
818 bcolor=ColorUtil.getFGColor();
819 }
820
821 Complex []Z=new Complex[tedgelist.size()];
822 int tick=0;
823 while (tel.hasNext()) {
824 EdgeSimple edge=tel.next();
825 int vv=edge.v;
826 Z[tick]=p.getCenter(vv);
827 if (p.hes>0)
828 Z[tick]=p.cpDrawing.sphView.toApparentSph(Z[tick]);
829 tick++;
830 }
831 pF.post_Poly(p.hes, Z, fcolor, bcolor, tx);
832
833 if (dispFlags.label) { // put at approximate center
834 Complex wc=null;
835 // if there is a 'baryVert', use its center
836 int bv=tile.baryVert;
837 if (bv>0 && bv<=p.nodeCount) {
838 wc=p.getCenter(tile.baryVert);
839 }
840
841 // else use average of corner verts centers
842 else {
843 Vector<Complex> cz=new Vector<Complex>(0);
844 for (int jj=0;jj<tile.vertCount;jj++)
845 cz.add(p.getCenter(tile.vert[jj]));
846
847 // for sphere, compute via vectors --- may end up at antipodal point
848 if (p.hes>0) {
849 double xc=0.0;
850 double yc=0.0;
851 double zc=0.0;
852 Iterator<Complex> czit=cz.iterator();
853 while (czit.hasNext()) {
854 double []xyz=SphericalMath.s_pt_to_vec(czit.next());
855 xc+=xyz[0];
856 yc+=xyz[1];
857 zc+=xyz[2];
858 }
859 xc /=tile.vertCount;
860 yc /=tile.vertCount;
861 zc /=tile.vertCount;
862 wc=SphericalMath.proj_vec_to_sph(xc,yc,zc);
863 }
864 else {
865 Iterator<Complex> zpts=cz.iterator();
866 wc=new Complex(0.0);
867 while (zpts.hasNext()) {
868 wc=wc.add(zpts.next());
869 }
870 wc=wc.divide((double)tile.vertCount);
871 }
872 }
873
874// Complex wc=new Complex(0.0);
875// for (int vj=0;vj<tile.vertCount;vj++)
876// wc = wc.add(p.rData[tile.vert[vj]].center);
877// wc=wc.divide((double)tile.vertCount);
878
879 cpDrawing.drawIndex(wc,tile.tileIndex, 1);
880 count++;
881 } // end of label display
882
883 count++;
884 }
885 } // end of while
886 } // done with tiles
887 break;
888 } // done with case 'T'
889 case 'u': // the unit circle
890 {
891 z = new Complex(0.0);
892 if (p.hes > 0)
893 z = cpDrawing.sphView.toApparentSph(z);
894 pF.postUnitCircle(p.hes, z);
895 break;
896 }
897
898 } // end of main switch
899 } // end of while
900 return count;
901 }
