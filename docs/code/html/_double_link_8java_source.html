105 public int addDoubleLinks(Vector<String> items) {
106
107 if (items==null || items.size()==0)
108 return 0;
109
110 int count=0;
111
112 String str=items.get(0);
113 boolean thereIsFlag=false;
114
115 // check for single-letter flag; only
116 // first entry can be flag. if there is
117 // an accidental '-', remove it
118 char a=str.charAt(0);
119 if (StringUtil.isFlag(str)) {
120 str=str.substring(1);
121 a=str.charAt(0);
122 items.remove(0);
123 thereIsFlag=true;
124 }
125 else if (a!='-' && a!='.' && a!='+' && !Character.isDigit(a)) {
126 items.remove(0);
127 thereIsFlag=true;
128 }
129
130 // self call if str is a variable with value a double
131 if (str.startsWith("_")) {
132 count+=addDoubleLinks(CPBase.varControl.getValue(str));
133 return count;
134 }
135
136 // look for 'Dlist'
137 else if (str.substring(1).startsWith("list")) {
138 DoubleLink dlink=null;
139
140 if ((str.startsWith("Dli") &&
141 (dlink=CPBase.Dlink)!=null && dlink.size()>0)) {
142 String[] b_string;
143 String brst;
144 String strdata=str.substring(5).trim(); // remove '?list'
145
146 // check for parens listing range of indices
147 int lsize=dlink.size()-1;
148 int[] irange=StringUtil.get_int_range(strdata, 0,lsize);
149 if (irange!=null) {
150 int aa=irange[0];
151 int bb=(irange[1]>lsize) ? lsize : irange[1];
152 for (int j=aa;j<=bb;j++) {
153 add(dlink.get(j));
154 count++;
155 }
156 }
157
158 // else check for brackets
159 else if ((b_string=StringUtil.get_bracket_strings(strdata))!=null
160 && (brst=b_string[0])!=null) {
161 if (brst.startsWith("r")) { // rotate: copy first at end
162 dlink.add(dlink.getFirst());
163 }
164 if (brst.startsWith("r")
165 || brst.startsWith("n")) { // use an remove first
166 Double d=dlink.removeFirst();
167 add(d);
168 count++;
169 }
170 if (brst.startsWith("l")) { // last
171 add(dlink.getLast());
172 count++;
173 }
174 else { // else specified index
175 try{
176 int n=MathUtil.MyInteger(brst);
177 if (n>=0 && n<dlink.size()) {
178 add(dlink.get(n));
179 count++;
180 }
181 } catch (NumberFormatException nfe) {}
182 }
183 }
184 // else just adjoin the current list
185 else {
186 int n=size();
187 addAll(n,dlink);
188 count +=dlink.size();
189 }
190 return count;
191 }
192 else // no appropriate list found
193 return count;
194 } // done with 'Dlist' processing
195
196 if (!thereIsFlag && (items==null || items.size()==0)) // nothing to do?
197 return 0;
198
199 // If there is a flag, we get it, process, and return.
200 if (thereIsFlag) {
201 switch(a) {
202 case 'f': // run string through 'function',
203 // if defined in function window. CAUTION:
204 // only add real part of function output
205 {
206 if (PackControl.newftnFrame.ftnField.getText().trim().length()==0) {
207 CirclePack.cpb.errMsg("'Function' frame is not set");
208 return 0;
209 }
210
211 DoubleLink dlink=new DoubleLink(items);
212 if (dlink==null || dlink.size()==0)
213 throw new ParserException("failed in getting Doubles");
214 Iterator<Double> dts=dlink.iterator();
215 while (dts.hasNext()) {
216 try {
217 Complex z=CirclePack.cpb.getFtnValue(new Complex(dts.next()));
218 if (add(z.x)) // use only real part
219 count++;
220 } catch (Exception ex) {}
221 }
222 break;
223 }
224 case 's': // schwarzians for given list of edges
225 case 'u':
226 {
227 HalfLink elist=new HalfLink(packData,items);
228 if (elist==null || elist.size()==0)
229 return count;
230 Iterator<HalfEdge> elst=elist.iterator();
231 while(elst.hasNext()) {
232 HalfEdge he=elst.next();
233 double s=he.getSchwarzian();
234 if (a=='s')
235 add(s);
236 if (a=='u')
237 add(1.0-s);
238 count++;
239 }
240 break;
241 }
242 } // end of switch
243 return count;
244 } // end of flag processing
245
246 // reaching here, just pull off doubles
247 Iterator<String> its=items.iterator();
248 while (its!=null && its.hasNext()) {
249 str=its.next();
250 try {
251 double x=Double.parseDouble(str);
252 if (add(Double.valueOf(x)))
253 count++;
254 } catch (Exception ex) {
255 CirclePack.cpb.errMsg("format error reading Doubles, count "+count);
256 return count;
257 }
258 } // end of getting x y points
259 return count;
260 }

334 public static DoubleLink rotateMe(DoubleLink link,int indx) {
335 int sz=0;
336 if (link==null || (sz=link.size())==0 || link.get(0)!=link.get(sz-1))
337 return null; // not closed list
338 DoubleLink dlink=new DoubleLink();
339 int i=indx;
340 while (i<(sz-1)) { // last one is a repeat
341 dlink.add(link.get(i));
342 i++;
343 }
344 i=0;
345 while (i<=indx) {
346 dlink.add(link.get(i));
347 i++;
348 }
349 return dlink;
