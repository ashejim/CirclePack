162 public double []sortEdges() {
163 EdgeLink elist=new EdgeLink(extenderPD,"a");
164 double l2error=0.0;
165 int count=0;
166 edgeData =new EdgeData[elist.size()];
167 Iterator<EdgeSimple> elst=elist.iterator();
168
169 int hits=0;
170 while (elst.hasNext()) {
171 EdgeSimple edge=elst.next();
172 double x=0.0;
173 try {
174 x=getLegality(edge.v,edge.w);
175 } catch(Exception ex) {
176 System.err.println("v w "+edge.v+" "+edge.w);
177 }
178 if (x>0) {
179 l2error += x*x;
180 count++;
181 }
182 edgeData[hits++]=new EdgeData(edge,x,100);
183 }
184
185 // sort
186 Comparator<EdgeData> EDComparator = new EDComparator();
187 Arrays.sort(edgeData,EDComparator);
188
189 // set color ramp
190 double mx=0.0;
191 double mn=0.0;
192 for (int j=0;j<edgeData.length;j++) {
193 mx = (edgeData[j].illegality>mx) ? edgeData[j].illegality : mx;
194 mn = (edgeData[j].illegality<mn) ? edgeData[j].illegality : mn;
195 }
196 for (int j=0;j<edgeData.length;j++) {
197 if (edgeData[j].illegality>0)
198 edgeData[j].color=(int)(100.0+95.0*(edgeData[j].illegality/mx));
199 else if (edgeData[j].illegality<0)
200 edgeData[j].color=(int)(100.0-95.0*(edgeData[j].illegality/mn));
201 }
202
203 double []ans=new double[2];
204 ans[0]=(double)count;
205 ans[1]=Math.sqrt(l2error/extenderPD.nodeCount);
206 return ans;

213 public int gogo(int N) {
214 int count=0;
215 int flipCount=0;
216 int node=extenderPD.nodeCount;
217 while (count<N) {
218 // random vert, random petal
219 int v=rand.nextInt(node)+1;
220 int j=-1;
221 int[] petals=extenderPD.packDCEL.vertices[v].getPetals();
222 int num=petals.length;
223 if (extenderPD.isBdry(v)) { // degree must be >3 for interior, >2 for bdry
224
225 if (num<3)
226 continue;
227 j=1+rand.nextInt(num-1);
228 }
229 else if (num>3)
230 j=rand.nextInt(num);
231
232 if (j>-1) {
233 int w=petals[j];
234 flipCount += flip2Legal(v,w);
235 }
236 count++;
237 }
238 return flipCount;
239 }

245 public int goOrder(int N) {
246 int count=0;
247 int flipCount=0;
248 if (edgeData==null) sortEdges();
249 N = (edgeData.length<N) ? edgeData.length : N;
250 int tick=0;
251 while (count<N && tick<extenderPD.nodeCount &&
252 edgeData[tick].illegality>0) {
253 int v=edgeData[tick].edge.v;
254 int w=edgeData[tick].edge.w;
255 if (extenderPD.flipable(v,w)) {
256 flipCount += flip2Legal(v,w);
257 }
258 count++;
259 tick++;
260 }
261 return flipCount;
262 }

289 public double[] getQuadAngles(int v,int w) {
290 if ((extenderPD.isBdry(v) && extenderPD.isBdry(w)) ||
291 extenderPD.nghb(v,w)<0)
292 return null;
293 int []corn_vert=new int[4];
294
295 HalfEdge he=extenderPD.packDCEL.findHalfEdge(new EdgeSimple(v,w));
296 if (he==null || extenderPD.packDCEL.isBdryEdge(he))
297 return null;
298 corn_vert[0]=v;
299 corn_vert[1]=he.twin.next.twin.origin.vertIndx; // clw edge
300 corn_vert[2]=w;
301 corn_vert[3]=he.prev.origin.vertIndx; // cclw edge
302
303 // affine map based on Beltrami coeff at centroid of these 4 verts
304 Complex midZ=extenderPD.getCenter(v).add(extenderPD.getCenter(corn_vert[1])).
305 add(extenderPD.getCenter(w)).add(extenderPD.getCenter(corn_vert[3]));
306 midZ=midZ.times(0.25);
307
308 // get 2x2 real matrix
309 double []affine=getAffine(getCoefficient(midZ));
310 Complex []corner=new Complex[4];
311 for (int i=0;i<4;i++) {
312 Complex Z=extenderPD.getCenter(corn_vert[i]);
313 // apply affine
314 double x=affine[0]*Z.x+affine[1]*Z.y;
315 double y=affine[2]*Z.x+affine[3]*Z.y;
316 corner[i]=new Complex(x,y);
317 }
318
319 return EuclMath.QuadAngles(corner[0],corner[1],corner[2],corner[3]);
320 }

328 public int flip2Legal(int v,int w) {
329 if ((extenderPD.isBdry(v) && extenderPD.isBdry(w)) ||
330 extenderPD.nghb(v,w)<0)
331 return 0;
332 double x=getLegality(v,w);
333 if (x>0) {
334 HalfEdge he=extenderPD.packDCEL.findHalfEdge(new EdgeSimple(v,w));
335 if (he==null)
336 return 0;
337 HalfEdge newhe=RawManip.flipEdge_raw(extenderPD.packDCEL, null);
338 if (newhe==null)
339 return 0;
340 extenderPD.packDCEL.fixDCEL(extenderPD);
341 return 1;
342 }
343 return 0;
