1094 while (slst.hasNext()) {
1095 int v=slst.next();
1096 if (book[v]==-2)
1097 throw new CombException("'seed' and 'target' sets intersect");
1098 book[v]=currgen;

1102 NodeLink nextGen=null;
1103 int safty=pdcel.vertCount*10;
1104 boolean hit=true;

1115 int k=he.next.origin.vertIndx;
1116

1118 if (book[k]==0) {
1119 hit=true;
1120 book[k]=currgen;
1121 nextGen.add(k);
1122 }
1123
1124 // hit a target?
1125 if (book[k]==-2) {
1126 hit=true;
1127 book[k]=currgen;
1128 maxdist=currgen; // won't need to search further,
1129 // but continue to find competitors
1130 if (theOne==0)
1131 theOne=k;
1132 }
1133 he=he.prev.twin; // cclw
1134 } while (he!=vert.halfedge);
1135 } // inner while

1137 prevGen=nextGen;
1138 } // outer while

1142 }
1143 if (currgen<maxdist) {

1145 "'seed' and 'target' separated, no geodesic");
1146 }

1151 int gen=book[theOne]-1;
1152 while (gen>0) {
1153 int ed=-1;

1157 int k=he.next.origin.vertIndx;

1167 start=ed;

1169 }
