70 public void serializeNode(Node node, Writer writer, String indentLevel) throws IOException {
71 // Determine action based on node type
72 switch(node.getNodeType()){
73 case Node.DOCUMENT_NODE:
74 writer.write("<?xml version=\"1.0\"?>");
75 writer.write(lineSeparator);
76
77 // recurse on each child
78 NodeList nodes = node.getChildNodes();
79 if(nodes != null){
80 for(int i=0; i<nodes.getLength(); i++){
81 serializeNode(nodes.item(i), writer, "");
82 }
83 }
84 break;
85
86 case Node.ELEMENT_NODE:
87 String name = node.getNodeName();
88 writer.write(lineSeparator);
89 writer.write(indentLevel + "<" + name);
90 NamedNodeMap attributes = node.getAttributes();
91 for(int i=0; i<attributes.getLength(); i++){
92 Node current = attributes.item(i);
93 writer.write(" " + current.getNodeName() + "=\"" + current.getNodeValue() + "\"");
94 }
95 writer.write(">");
96
97 // recurse on each child
98 NodeList children = node.getChildNodes();
99 if(children != null){
100 if((children.item(0) != null)
101 && (children.item(0).getNodeType() == Node.ELEMENT_NODE)){
102 writer.write(lineSeparator);
103 }
104 for(int i=0; i<children.getLength(); i++) {
105 serializeNode(children.item(i), writer, indentLevel + indent);
106 }
107 if((children.item(0) != null) &&
108 (children.item(children.getLength()-1).getNodeType()
109 == Node.ELEMENT_NODE)) {
110 writer.write(indentLevel);
111 }
112 }
113
114 writer.write("</" + name + ">\n");
115 writer.write(lineSeparator);
116 break;
117
118 case Node.TEXT_NODE:
119 writer.write(node.getNodeValue());
120 break;
121
122 case Node.CDATA_SECTION_NODE:
123 writer.write("<![CDATA[" + node.getNodeValue() + "]]>");
124 break;
125
126 case Node.COMMENT_NODE:
127 writer.write(indentLevel + "<!-- " + node.getNodeValue() + " -->");
128 writer.write(lineSeparator);
129 break;
130
131 case Node.PROCESSING_INSTRUCTION_NODE:
132 writer.write("<?" + node.getNodeName() + " " + node.getNodeValue() + "?>");
133 writer.write(lineSeparator);
134 break;
135
136 case Node.ENTITY_REFERENCE_NODE:
137 writer.write("&" + node.getNodeName() + ";");
138 break;
139
140 case Node.DOCUMENT_TYPE_NODE:
141 DocumentType docType = (DocumentType)node;
142 writer.write("<!DOCTYPE " + docType.getName());
143 if(docType.getPublicId() != null){
144 System.out.print(" PUBLIC \"" + docType.getPublicId() + "\" ");
145 }
146 else{
147 writer.write(" SYSTEM ");
148 }
149 writer.write("\"" + docType.getSystemId() + "\">");
150 writer.write(lineSeparator);
151 break;
152 }
153 }
