206 public void keyReleased(KeyEvent ke) {
207
208 // hide completion list if a key is hit
209 if (compList != null)
210 compList.hide();
211
212 int code=ke.getKeyCode();
213 // send command string for processing
214 switch(code) {
215
216 case KeyEvent.VK_ENTER: { // send command string for processing
217 String cmd = cmdline.getText().trim();
218 if(cmd.length()==0) return;
219
220 CPBase.trafficCenter.parseWrapper(cmd,CirclePack.cpb.getActivePackData(),
221 true,true,0,this);
222
223 // leave and select the command for changes or to replace with new typing
224 cmdline.selectAll();
225 break;
226 }
227
228 // set the entry field text to next command
229 case KeyEvent.VK_UP: {
230 if (ShellManager.cmdHistoryIndex < ShellManager.cmdHistory.size()-1)
231 ShellManager.cmdHistoryIndex++;
232 cmdline.setText(ShellManager.cmdHistory.get(ShellManager.cmdHistoryIndex));
233 break;
234 }
235
236 // set the entry field text to previous command
237 case KeyEvent.VK_DOWN: {
238 if (ShellManager.cmdHistoryIndex > -1) ShellManager.cmdHistoryIndex--;
239 if (ShellManager.cmdHistoryIndex == -1) cmdline.setText("");
240 else cmdline.setText(ShellManager.cmdHistory.get(ShellManager.cmdHistoryIndex));
241 break;
242 }
243
244 // do command completion with TAB
245 case KeyEvent.VK_TAB: {
246 StringBuilder cmdbuf = new StringBuilder(cmdline.getText());
247 int caret=cmdline.getCaretPosition();
248 if (caret<1)
249 break;
250 String preCaret=cmdbuf.substring(0,caret-1);
251 int lastsemi=preCaret.lastIndexOf(';');
252 if (lastsemi==caret-1)
253 break;
254 lastsemi=(lastsemi<0) ? 0:lastsemi;
255 // 'hold' is for reinstating later; add a ';'
256 String hold=cmdbuf.substring(0,lastsemi);
257 if (hold==null)
258 hold="";
259 else if (hold.length()>0 && hold.charAt(hold.length()-1)!=';')
260 hold=hold+";";
261 String leadchars=null;
262 if (hold.length()==0)
263 leadchars=cmdbuf.substring(lastsemi,caret).trim();
264 else
265 leadchars=cmdbuf.substring(lastsemi+1,caret).trim();
266 // is there anything to complete?
267 if (leadchars.length()==0)
268 break;
269 String postCaret=cmdbuf.substring(caret);
270 String resp[] = complete(leadchars);
271 String cmd=resp[0];
272 if (cmd==null)
273 cmd="";
274 // build completed command with pre/post included
275 cmdline.setText(hold+cmd+postCaret);
276 cmdline.setCaretPosition(hold.length()+cmd.length());
277
278 // creates a tooltip and places it directly below the cmdline box
279 JToolTip tip = cmdline.createToolTip();
280 tip.setTipText("<html>" + resp[1] + "</html>");
281 Rectangle rect = cmdline.getBounds();
282 Point loc = cmdline.getLocationOnScreen();
283 compList = PopupFactory.getSharedInstance().getPopup(cmdline, tip,
284 loc.x, loc.y + rect.height - 2);
285 compList.show();
286
287 // hide tip after 5 secs
288 compTimer.start();
289
290 break;
291 }
292 } // end of switch
293 }

308 public String []complete(String command) {
309 int nFound=0;
310 String newcmd=null;
311 String possible="";
312 String complist[] = new String[commandMap.size()];
313 commandMap.keySet().toArray(complist);
314 int cmdLength=command.length();
315
316 // modify 'PackExtender' call prefixes to caps for lookup
317 if (cmdLength>1 && command.charAt(0)=='|') {
318 StringBuilder strbuf=new StringBuilder(command);
319 if (cmdLength>2)
320 strbuf.setCharAt(1,Character.toUpperCase(strbuf.charAt(1)));
321 if (cmdLength>3)
322 strbuf.setCharAt(2,Character.toUpperCase(strbuf.charAt(2)));
323 command=strbuf.toString();
324 }
325
326 // cut down length until you get a match
327 while (cmdLength>0 && nFound==0) {
328 command=command.substring(0,cmdLength);
329 for(int i=0; i<complist.length; i++) {
330 if(complist[i].startsWith(command)) {
331 possible+=" "+complist[i];
332 nFound++;
333
334 if( newcmd == null ) {
335 newcmd=complist[i];
336 }
337 else { // already found a match, now we only want max substring
338 int sz=complist[i].length();
339 int j=0;
340 if( newcmd.length()<sz ) sz=newcmd.length();
341 for(j=0; j<sz; j++) {
342 if( newcmd.charAt(j) != complist[i].charAt(j) )
343 break;
344 }
345 newcmd=complist[i].substring(0,j);
346 }
347 }
348 } // end of for
349 cmdLength--;
350 } // end of while
351
352 if( nFound==1 ) {
353 // gets a brief usage message
354 possible = commandMap.get(newcmd);
355 }
356 else if(possible.length()!=0) {
357 possible = possible.replace(" "+newcmd," <b>"+newcmd+"</b>").substring(1);
358 }
359 else {
360 possible = "No completions available";
361 }
362 String ret[] = new String[2];
363 ret[0]=newcmd;
364 ret[1]=possible;
365 return ret;
366 }

373 public static void add2CmdCompletion(String str) {
374 if (str==null || str.trim().length()==0)
375 return;
376 String newstr=str.trim();
377
378 int k=str.indexOf(' ');
379
380 // no spaces? then no complications
381 if (k<0) {
382 commandMap.put(newstr,new String("<b>"+newstr+"</b> "));
383 return;
384 }
385
386 // pick off possible prefix and 'info'
387 String pref=null;
388 // Is this a 'PackExtender' command: eg. "|as| dah"?
389 if (k>0 && newstr.charAt(k-1)=='|') { // get prefix
390 pref=newstr.substring(0,k);
391 newstr=newstr.substring(k,newstr.length()).trim();
392 }
393
394 k=newstr.indexOf(' ');
395 if (k<0) { //
396 if (pref==null)
397 commandMap.put(new String(newstr),"<b>"+newstr+"</b> ");
398 else
399 commandMap.put(new String(pref+" "+newstr),"<b>"+newstr+"</b> ");
400 return;
401 }
402 String cmd=newstr.substring(0,k);
403 String info=newstr.substring(k,newstr.length());
404 if (pref==null)
405 commandMap.put(cmd,new String("<b>"+cmd+"</b> "+info));
406 else
407 commandMap.put(new String(pref+" "+cmd),new String("<b>"+
408 pref+" "+cmd+"</b>"+info));
409 return;
410 }
