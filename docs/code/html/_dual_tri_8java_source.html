70 public void setInCirclePts() {
71 double []len=new double[3];
72 double []r=new double[3];
73 double mindist;
74
75 if (hes<0) { // hyp
76 int tick=0;
77
78 // check legality, number on unit circle
79 for (int j=0;j<3;j++) {
80 if (corners[j].abs()>1.0001) { // clearly illegal, error
81 TangPts=null;
82 return;
83 }
84 if (corners[j].abs()>1.0) { // roundoff? proj to unit circle
85 corners[j]=corners[j].divide(corners[j].abs());
86 tick++;
87 }
88 if (corners[j].abs()>.99999)
89 tick++;
90 }
91 if (tick>0) { // some point(s) on the unit circle
92
93 try {
94 // first case: all three on unit circle
95 if (tick==3) {
96 for (int j=0;j<3;j++)
97 corners[j]=corners[j].divide(corners[j].abs());
98
99 // If verts were at 1, +-i, then tangency points
100 // are t1=origin, t0=(c)+(c)i, t2=(c)-(c)i, where
101 // c=1-1/sqrt(2).
102 // Strategy: find Mobius m mapping verts to 1, i, -i,
103 // tangency points are inverses of tj's.
104 //
105 Complex z1=corners[1].divide(corners[0]);
106 Complex z2=corners[2].divide(corners[0]);
107 double v1=z1.y/(1-z1.x);
108 double v2=z2.y/(1-z2.x);
109 // phi(z)=(1+z)/(1-z) maps 1 to inf, zj to (vj)i
110 double a=2/(v1-v2);
111 double b=(v2+v1)/(v2-v1);
112 // psi(z)=(a)(i)z + b(i): phi(z1)=i, phi(z2)=-i
113 Mobius rot=new Mobius(corners[0].reciprocal(),
114 new Complex(0.0),new Complex(0.0),
115 new Complex(1.0));
116 Mobius phi=new Mobius(new Complex(1.0),new Complex(1.0),
117 new Complex(-1.0),new Complex(1.0));
118 Mobius psi=new Mobius(new Complex(0,a),new Complex(0,b),
119 new Complex(0.0),new Complex(1.0));
120 // Here's m
121 Mobius m=(Mobius)rot.lmultby(phi.lmultby(psi).lmultby(phi.inverse()));
122 Mobius m_inv=(Mobius)m.inverse();
123 double c=1-1/Math.sqrt(2);
124 TangPts[0]=m_inv.apply(new Complex(c,c));
125 TangPts[1]=m_inv.apply(new Complex(0.0));
126 TangPts[2]=m_inv.apply(new Complex(c,-c));
127 return;
128 }
129
130 // handle second case: two points on unit circle
131 int offset=-1;
132 if (tick==2) {
133 for (int j=0;j<3;j++)
134 if (offset<0 && corners[j].abs()<=.99999)
135 offset=j;
136 if (offset<0)
137 throw new DataException("error with points on unit circle");
138 Complex z0=corners[offset];
139 Complex z1=corners[(offset+1)%3];
140 Complex z2=corners[(offset+2)%3];
141
142 // first map z0 to origin z1 to 1
143 Complex lad=new Complex(1.0);
144 lad=lad.minus(z0.conj().times(z1));
145 lad=lad.divide(new Complex(z1.x-z0.x,z1.y-z0.y));
146 Mobius phi=new Mobius(lad,lad.times(z0).times(-1.0),
147 z0.conj().times(-1.0),new Complex(1.0));
148 Complex nz2=phi.apply(z2);
149 double ag=nz2.arg()/2.0;
150 Mobius rot=new Mobius(new Complex(Math.cos(ag),Math.sin(-ag)),
151 new Complex(0.0),new Complex(0.0),new Complex(1.0));
152 Mobius m=(Mobius)phi.lmultby(rot);
153 Mobius m_inv=(Mobius)m.inverse();
154 // Mobius m maps z0 to origin, z1, z2 equidistant from 1
155
156 double h=Math.sin(ag);
157 double s=h/(1+h);
158 double rad=1-2*s; // radius of circle at origin
159
160 // In this standard position, tangency points are
161 double rr=Math.pow(rad,1.5);
162 Complex t0=new Complex(rr/(rad+s),-rad*h);
163 Complex t2=t0.conj();
164 Complex t1=new Complex(Math.sqrt(rad)); // this is real
165
166 // Set tangency points
167 TangPts[offset]=m_inv.apply(t0);
168 TangPts[(offset+1)%3]=m_inv.apply(t1);
169 TangPts[(offset+2)%3]=m_inv.apply(t2);
170 return;
171 }
172
173 // last case, one on boundary
174 if (tick==1) {
175 for (int j=0;j<3;j++)
176 if (offset<0 && corners[j].abs()>.99999)
177 offset=j;
178 if (offset<0)
179 throw new DataException("error with points on unit circle");
180 Complex z0=corners[offset];
181 Complex z1=corners[(offset+1)%3];
182 Complex z2=corners[(offset+2)%3];
183
184 // Mobius m maps disc to upper half plane, z0 to infinity
185 Mobius m=new Mobius(new Complex(0.0,1.0),new Complex(-z0.y,z0.x),
186 new Complex(-1.0),z0);
187 double h_len=HyperbolicMath.h_dist(z1,z2);
188 Complex w1=m.apply(z1);
189 Complex w2=m.apply(z2);
190 double lg=Math.log(w1.y/w2.y);
191 // hyperbolic radii
192 double r1=(0.5)*(h_len-lg);
193 double r2=(0.5)*(h_len+lg);
194 double Y1=Math.exp(r1)*w1.y;
195 double Y2=Math.exp(r2)*w2.y;
196 Complex t0=new Complex(w1.x,Y1);
197 Complex t2=new Complex(w2.x,Y2);
198
199 Mobius m_inv=(Mobius)m.inverse();
200 // Set tangency points
201 TangPts[offset]=m_inv.apply(t0);
202 TangPts[(offset+2)%3]=m_inv.apply(t2);
203
204 // For tangency between finite points, need x-radii
205 if(r1 > 0.0001)
206 r1 = 1-Math.exp(-2.0*r1); // x=1-exp(-2h)
207 else
208 r1 = 2.0*r1*(1.0 - r1*(1.0-2.0*r1/3.0));
209 if(r2 > 0.0001)
210 r2 = 1-Math.exp(-2.0*r2);
211 else
212 r2 = 2.0*r2*(1.0 - r2*(1.0-2.0*r2/3.0));
213 TangPts[(offset+1)%3]=HyperbolicMath.hyp_tangency(z1, z2, r1, r2);
214 return;
215 }
216 } catch(Exception ex) {
217 throw new DataException("Mobius processing error:"+ex.getMessage());
218 }
219 }
220 len[0]=mindist=HyperbolicMath.h_dist(corners[0],corners[1]);
221 len[1]=mindist=HyperbolicMath.h_dist(corners[1],corners[2]);
222 len[2]=mindist=HyperbolicMath.h_dist(corners[2],corners[0]);
223 }
224
225 else if (hes>0) { // sph
226 len[0]=mindist=SphericalMath.s_dist(corners[0],corners[1]);
227 len[1]=mindist=SphericalMath.s_dist(corners[1],corners[2]);
228 len[2]=mindist=SphericalMath.s_dist(corners[2],corners[0]);
229 }
230
231 else { // eucl
232 len[0]=mindist=corners[1].minus(corners[0]).abs();
233 len[1]=corners[2].minus(corners[1]).abs();
234 len[2]=corners[0].minus(corners[2]).abs();
235 }
236

238 mindist=(mindist<len[1]) ? mindist : len[1];
239 mindist=(mindist<len[2]) ? mindist : len[2];
240 if (mindist<OKERR) {
241 TangPts=null;
242 return;
243 }
244
245 // find "effective" radii at vertices: radii of circles
246 // that would form this triangles.
247 for (int j=0;j<3;j++) {
248 double op=len[(j+1)%3];
249 r[j]=(len[j]+len[(j+2)%3]-op)/2.0;
250 if (r[j]<0.000001*op) { // extremely thin triangle
251 TangPts=null;
252 return;
253 }
254 }
255
256 // TODO: work on case of one or more points on unit circle
257 if (hes<0) {
258 // hyperbolic: need these as x-radii
259 for (int j=0;j<3;j++) {
260 if(r[j] > 0.0001)
261 r[j] = 1-Math.exp(-2.0*r[j]); // x=1-exp(-2h)

271 }
272 return;
