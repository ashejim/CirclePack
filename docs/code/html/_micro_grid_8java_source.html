818 ans=(intensityField[ix][jy]+intensityField[ix+1][jy]+
819 intensityField[ix+1][jy]+intensityField[ix+1][jy+1])/4.0;
820 } catch (Exception ex) {
821 System.out.println(" problem setting intensityField.");
822 }
823 return ans;

2343 int count=0;
2344 for (int f=1;f<=q.faceCount;f++) {
2345 boolean marked=false;
2346 int[] vert=q.packDCEL.faces[f].getVerts();
2347 double[] radii=new double[3];
2348 for (int j=0;j<3;j++) {
2349 radii[j]=q.getRadius(vert[j]);
2350 }

2352 // first check if the 3 radii are not (essentially) equal
2353 boolean maybe=false;
2354 for (int j=0;(j<3 && !maybe);j++)
2355 if (Math.abs(radii[j]-radii[(j+1)%3])>.02*radii[j])
2356 maybe=true;

2359 if (maybe) {
2360 for (int j=0;(j<3 && !marked);j++) {
2361 // an edge overlap exceeds the given threshold?
2362 if (q.getInvDist(vert[j],vert[(j+1)%3])>thresh)
2363 marked=true;

2372 Complex bcent=pts[0].add(pts[1].add(pts[2])).divide(3.0);
