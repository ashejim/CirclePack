258 public boolean prepData() {
259 boolean hit=false;
260 if (pdcel.triData==null) {
261 pdcel.triData=new TriData[pdcel.faceCount+1];
262 for (int f=1;f<=pdcel.faceCount;f++) {
263 pdcel.triData[f]=new TriData(pdcel,pdcel.faces[f]);
264 if (pdcel.triData[f].hasInvDist())
265 hit=true;
266 }
267 }
268 else {
269 hit=pdcel.updateTriDataRadii();
270 }
271
272 // create 'vNum', 'findices', 'vindices' after a minimal
273 // check on whether they already exist
274 if (findices==null || findices.length!=p.nodeCount+1) {
275 vNum=new int[p.nodeCount+1];
276 findices=new int[p.nodeCount+1][];
277 vindices=new int[p.nodeCount+1][];
278 for (int v=1;v<=p.nodeCount;v++) {
279 vNum[v]=p.countFaces(v);
280 findices[v]=new int[vNum[v]];
281 vindices[v]=new int[vNum[v]];
282 HalfEdge he=pdcel.vertices[v].halfedge;
283 int tick=0;
284 do {
285 findices[v][tick]=he.face.faceIndx;
286 vindices[v][tick++]=he.face.getVertIndx(v);
287 he=he.prev.twin;
288 } while(he!=pdcel.vertices[v].halfedge && he.face.faceIndx>=0);
289 }
290 }
291 return hit;
292 }
