98 public double []LaplaceIt(double []values) {
99 double []lp=new double[extenderPD.nodeCount+1];
100 for (int v=1;v<=extenderPD.nodeCount;v++) {
101 int[] flower=extenderPD.getFlower(v);
102 double tot=0.0;
103 for (int j=0;j<flower.length;j++)
104 tot += conductance[v][j];
105 double avg=0.0;
106 for (int j=0;j<flower.length;j++)
107 avg += values[flower[j]]*conductance[v][j]/tot;
108 lp[v]=avg-values[v];
109 }
110 return lp;

136 public boolean compFaceMobs() {
137 if (!domainData.status || !rangeData.status ||
138 domainData.nodeCount != rangeData.nodeCount ||
139 domTPs==null || ranTPs==null)
140 Oops("domain/range incompatible or tangency points not updated");
141 faceMobs=new Mobius[domainData.faceCount+1];
142 for (int f=1;f<=domainData.faceCount;f++) {
143 Complex a=domTPs[f].getTP(0);
144 Complex b=domTPs[f].getTP(1);
145 Complex c=domTPs[f].getTP(2);
146 Complex A=ranTPs[f].getTP(0);
147 Complex B=ranTPs[f].getTP(1);
148 Complex C=ranTPs[f].getTP(2);
149
150 Mobius m=Mobius.standard3Point(a, b, c);
151 Mobius M=Mobius.standard3Point(A,B,C);
152 GroupElement MInverse=(GroupElement)M.inverse();
153 faceMobs[f]=(Mobius)m.lmultby(MInverse);
154 }
155 return true;

539 public int SR_parameterize(double aparam) {
540 for (int v=1;v<=extenderPD.nodeCount;v++) {
541 double r=domainData.getRadius(v);
542 double expal;
543 // at z=0 replace |f(z)/z| by |f'(z)| (typically, 1)
544 if (domainData.getCenter(v).abs()<.0000001) {
545 expal=Math.exp(aparam*Math.log(extenderPD.getRadius(v)/r));
546 }
547 else {
548 double rc_dc=extenderPD.getCenter(v).
549 divide(domainData.getCenter(v)).abs();
550 expal=Math.exp(aparam*Math.log(rc_dc));
551 }
552 outputData.setRadius(v,extenderPD.getRadius(v)/expal);
553 }
554 return 1;
555 }

568 public static double[][] setConductances(PackData domData) {
569 if (domData==null || domData.nodeCount<=0 || domData.hes!=0
570 || domData.status==false) {
571 throw new ParserException("packing not set or not suitable");
572 }
573 double[] slengths=null; // spoke lengths
574 Complex[] inCenters=null;
575 double[][] conductance=new double[domData.nodeCount+1][];
576 Complex f1=null;
577 Complex f2=null;
578 for (int v=1;v<=domData.nodeCount;v++) {
579 int[] flower=domData.getFlower(v);
580 int num=domData.countFaces(v);
581 Complex z=domData.getCenter(v);
582 slengths=new double[num+1];
583 inCenters=new Complex[num];
584
585 // closed for interior v
586 conductance[v]=new double[num+1];
587
588 // store edge lengths, incenters
589 f2=domData.getCenter(flower[0]);
590 slengths[0]=z.minus(f2).abs();
591 CircleSimple sc=null;
592 for (int j=1;j<=num;j++) {
593 f1=f2;
594 f2=domData.getCenter(flower[j]);
595 sc=EuclMath.eucl_tri_incircle(z,f1,f2);
596 slengths[j]=z.minus(f2).abs();
597 inCenters[j-1]=sc.center;
598 }
599
600 // store conductances
601
602 // for bdry, use ratio of inRad/length for first and last edges
603 if (domData.isBdry(v)) {
604 f1=domData.getCenter(flower[0]);
605 f2=domData.getCenter(flower[1]);
606 double inRad=EuclMath.eucl_tri_inradius(slengths[0],slengths[1],f1.minus(f2).abs());
607 conductance[v][0]=inRad/slengths[0];
608 f1=domData.getCenter(flower[num-1]);
609 f2=domData.getCenter(flower[num]);
610 inRad=EuclMath.eucl_tri_inradius(slengths[num-1],slengths[num],f1.minus(f2).abs());
611 conductance[v][num]=inRad/slengths[num];
612 }
613 else { // interior: first conductance repeated in last
614 conductance[v][0]=
615 conductance[v][num]=inCenters[num-1].minus(inCenters[0]).abs()/slengths[0];
616 }
617
618 // now the rest
619 for (int j=1;j<num;j++) {
620 conductance[v][j]=inCenters[j-1].minus(inCenters[j]).abs()/slengths[j];
621 }
622 } // end of loop on v
623 return conductance;
624 }

636 public int discreteDeriv(PackData dData,PackData rData, int mode) {
637 if (rData==null || rData.nodeCount!=dData.nodeCount) {
638 errorMsg("rangeData has not been loaded");
639 return 0;
640 }
641 if (conductance==null) {
642 errorMsg("conductances are not set");
643 return 0;
644 }
645 Complex []domSpokes=null;
646 Complex []ranSpokes=null;
647 try {
648 for (int v=1;v<=dData.nodeCount;v++) {
649
650 // data at v
651 int num=dData.countFaces(v);
652 Complex z=dData.getCenter(v);
653 Complex w=rData.getCenter(v);
654
655 // check if packings match here
656 if (num!=rData.countFaces(v) || num!=dData.countFaces(v)
657 || dData.getBdryFlag(v)!=rData.getBdryFlag(v)
658 || dData.getBdryFlag(v)!=rData.getBdryFlag(v))
659 throw new DataException("combinatorics of packings do not agree");
660 // for the data
661 domSpokes=new Complex[num+1];
662 ranSpokes=new Complex[num+1];
663
664 double totalWeight=0.0;
665
666 // store complex edge vectors, compute node conductance
667 int[] petals=dData.getPetals(v);
668 for (int j=0;j<petals.length;j++) {
669 domSpokes[j]=z.minus(dData.getCenter(petals[j]));
670 ranSpokes[j]=w.minus(rData.getCenter(petals[j]));
671 totalWeight+=conductance[v][j];
672 }
673
674 Complex deriv=new Complex(0.0);
675 for (int j=0;j<petals.length;j++)
676 deriv=deriv.add(ranSpokes[j].divide(domSpokes[j]).
677 times(conductance[v][j]));
678 outputData.setCenter(v,deriv.divide(totalWeight));
679 }
680 } catch(Exception ex) {
681 errorMsg("error in computing derivative: "+ex.getMessage());
682 return 0;
683 }
684
685 // Also compute the radii based solely on the computed centers.
686 if (EuclMath.effectiveRad(outputData,null)==0) {
687 errorMsg("error in setting 'effective' radii.");
688 }
689 return 1;
690 }
