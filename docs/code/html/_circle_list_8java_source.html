47 public static Vector<Point2D.Double> getArcList(Complex center,
48 double rad,double startAng,double extent,int N) {
49 double zr,zi,wr,wi;
50 Point2D.Double startPt,endPt;
51
52 if (N<1) N=1;
53
54 zr=rad*Math.cos(startAng);
55 zi=rad*Math.sin(startAng);
56 startPt=new Point2D.Double(zr+center.x,zi+center.y);
57 endPt=new Point2D.Double(rad*Math.cos(startAng+extent)+center.x,
58 rad*Math.sin(startAng+extent)+center.y);
59 // Proportion of those steps to use based on 'extent' an 'N'
60 int n=(int)(Math.floor(N*Math.abs(extent)/m2pi));
61 Vector<Point2D.Double> vect=new Vector<Point2D.Double>(n+2);
62 // Start with first point
63 vect.add(startPt);
64 // Perhaps just add line segment
65 if (n<=1) {
66 vect.add(endPt);
67 return vect;
68 }
69 // A 2x2 matrix propogates points of circle based on N divisions
70 double step=m2pi/N;
71 if (extent<0.0) step = -step;
72 double ss=step*step/4.0;
73 double a11=(1-ss)/(1+ss);
74 double a12= (-step)/(1+ss);
75 double a21= (-a12);
76 double a22=a11;
77
78 for (int j=0;j<n;j++) {
79 wr=zr;
80 wi=zi;
81 zr=a11*wr+a12*wi;
82 zi=a21*wr+a22*wi;
83 vect.add(new Point2D.Double(zr+center.x,zi+center.y));
84 }
85 // finish with endpoint
86 vect.add(endPt);
87 return vect;
88 }

181 public static void buildSphList(Point3D A,Point3D B,Point3D Cent,
182 double sphrad,Point3D end1,Point3D end2,Path2D.Double gp,
183 boolean start,int N,CPdrawing cpS) {
184
185 double a1,extent;
186
187 // full circle?
188 if (end1==null) {
189 a1=0.0;
190 extent=2.0*Math.PI;
191 }
192 else {
193 a1=Math.atan2(Point3D.DotProduct(B,end1),
194 Point3D.DotProduct(A,end1));
195 double a2=Math.atan2(Point3D.DotProduct(B,end2),
196 Point3D.DotProduct(A,end2));
197 extent=MathComplex.radAngDiff(a1,a2);
198 }
199 double rad=Math.sin(sphrad);
200 // Here's a 2D circle in AB-coords as seen from Cent direction:
201 Vector<Point2D.Double> vec=
202 CircleList.getArcList(MathComplex.ZERO,rad,a1,extent,N);
203
204 // Now convert to 3D using AB-coords, then project back to
205 // viewer's yz-coords to put in path.
206 double Y,Z;
207 int sz=vec.size();
208 int init_i=0;
209 double cosr=Math.cos(sphrad);
210 double Cy=cosr*Cent.y;
211 double Cz=cosr*Cent.z;
212 if (start) { // gp is to be reset and started again
213 gp.reset();
214 if (sz==0)
215 return;
216 else { // set first point
217 Point2D.Double pt=(Point2D.Double)vec.get(0);
218 Y=Cy+pt.x*A.y+pt.y*B.y;
219 Z=Cz+pt.x*A.z+pt.y*B.z;
220 gp.moveTo(cpS.toPixX(Y),cpS.toPixY(Z));
221 }
222 init_i=1;
223 }
224 for (int i=init_i;i<vec.size();i++) {
225 Point2D.Double pt=(Point2D.Double)vec.get(i);
226 Y=Cy+pt.x*A.y+pt.y*B.y;
227 Z=Cz+pt.x*A.z+pt.y*B.z;
228 gp.lineTo(cpS.toPixX(Y),cpS.toPixY(Z));
229 }

286 CircleList.circlePoints(vec,gp,start,cpS);
287 }
288 else {
289 // find an orthonormal basis; note it is easier
290 // here for geodesic circles than for other circles.
291 Point3D C=new Point3D(sg.center);
292 double plen=Math.sqrt(C.y*C.y+C.z*C.z);
293 // unit orthog to yz-proj of C
294 Point3D A=new Point3D(0.0,-C.z/plen,C.y/plen);
295 Point3D B=Point3D.CrossProduct(C,A);
296
297 // both ends should be visible
298 Point3D end1=new Point3D(sg.z1);
299 Point3D end2=new Point3D(sg.z2);
300
301 // create path
302 CircleList.buildSphList(A,B,new Point3D(0,0,0),
303 CPBase.piby2,end1,end2,gp,start,N,cpS);
304 }
