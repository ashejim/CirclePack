439 public void actionPerformed(ActionEvent evt) {
440
441 String cmd=evt.getActionCommand();
442
443 if (cmd.equals("Slider Info")) {
444 JFrame auxHelpFrame=new JFrame();
445 auxHelpFrame.setTitle("Help for SliderFrame");
446 JTextArea helpText=new JTextArea();
447 helpText.setText(helpInfo.toString());
448 auxHelpFrame.add(helpText);
449 helpText.setLineWrap(true);
450 helpText.setWrapStyleWord(true);
451 util.EmacsBindings.addEmacsBindings(helpText);
452 helpText.setPreferredSize(new Dimension(500,500));
453 auxHelpFrame.pack();
454 auxHelpFrame.setVisible(true);
455 }
456 else if (cmd.equals("Update")) {
457 downloadData();
458 }
459 else if (type!=SCHFLOWER && cmd.equals("add object")) {
460 addObject(addField.getText().trim());
461 }
462 else if (type!=SCHFLOWER && cmd.equals("remove object")) {
463 removeObject(removeField.getText().trim());
464 }
465 else if (cmd.equals("set minimum")) {
466 double min=minValue.getValue();
467 this.resetMin(min);
468 }
469 else if (cmd.equals("set maximum")) {
470 double max=maxValue.getValue();
471 this.resetMax(max);
472 }
473 else if (cmd.equals("optional cmd")) {
474 String cmdStr=optCmdField.getText();
475 if (cmdStr.length()>0) {
476 ResultPacket rP=new ResultPacket(packData,cmdStr);
477 CPBase.trafficCenter.parseCmdSeq(rP,0,null);
478 }
479 }
