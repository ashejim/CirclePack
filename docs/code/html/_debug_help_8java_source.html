76 public static int debugSolverData(SolverData sdata,String fname) {
77 BufferedWriter dbw=CPFileManager.openWriteFP(tmpdir,fname,false);
78 if (sdata==null || sdata.nz_entries<=0 || sdata.intNum<=0 ||
79 sdata.Aentries==null || sdata.Ai==null || sdata.Ap==null ||
80 sdata.rhsX==null || sdata.rhsY==null)
81 return 0;
82 try {
83 // write various counts
84 dbw.write("intNum = "+sdata.intNum+"\n");
85 dbw.write("bdryNum = "+sdata.bdryNum+"\n");
86 dbw.write("nz_entries = "+sdata.nz_entries+"\n");
87
88 // write Ap
89 dbw.write("Ap = [\n");
90 for (int i=0;i<=sdata.intNum;i++)
91 dbw.write(sdata.Ap[i]+" ");
92 dbw.write("\n];\n");
93
94 // write Ai
95 dbw.write("Ai = [\n");
96 for (int i=0;i<sdata.nz_entries;i++)
97 dbw.write(sdata.Ai[i]+" ");
98 dbw.write("\n];\n");
99
100 // write Aentries
101 dbw.write("Aentries = [\n");
102 for (int i=0;i<sdata.nz_entries;i++)
103 dbw.write(sdata.Aentries[i]+" ");
104 dbw.write("\n];\n");
105
106 // write rhsX as column vector
107 dbw.write("rhsX = [\n");
108 for (int i=0;i<sdata.intNum;i++)
109 dbw.write(sdata.rhsX[i]+";");
110 dbw.write("\n];\n");
111
112 // write rhsY as column vector
113 dbw.write("rhsY = [\n");
114 for (int i=0;i<sdata.intNum;i++)
115 dbw.write(sdata.rhsY[i]+";");
116 dbw.write("\n];\n");
117
118 // adjust indexing for matlab's use
119 dbw.write("Ai_m = Ai+1;\n");
120
121 // solve using matlab
122 dbw.write("\n% convert to systems A*x=rhsX, A*x=rhsY and solve\n");
123 dbw.write("A=zeros(intNum,intNum);\n");
124 dbw.write("tick=1;\nindxaccum=0;\n");
125 dbw.write("for j=2:intNum+1\n n=Ap(j);\n for ii=1:(n-indxaccum)\n");
126 dbw.write(" A(Ai_m(ii),j-1)=Aentries(tick);\n tick=tick+1;\n end\n");
127 dbw.write(" indxaccum=indxaccum+Ap(j);\nend\n\n");
128 dbw.write("Zreal=A\\rhsX;\nZimag=A\\rhsY;\n\n");
129 dbw.write("Z=Zreal+Zimag*1i;\n");
130
131 dbw.flush();
132 dbw.close();
133 } catch (Exception ex) {
134 throw new DataException("error writing sdata to matlab: "+ex.getMessage());
135 }
136
137 return sdata.nz_entries;
138 }
