109 public int addPointLinks(Vector<String> items) {
110
111 if (items==null || items.size()==0)
112 return 0;
113
114 int count=0;
115
116 String str=items.get(0);
117 boolean thereIsFlag=false;
118
119 // check for flag; only first entry can be flag.
120 // if there is an accidental '-', remove it
121 char a=str.charAt(0);
122 if (StringUtil.isFlag(str)) {
123 str=str.substring(1);
124 a=str.charAt(0);
125 items.remove(0);
126 thereIsFlag=true;
127 }
128 else if (a!='-' && a!='.' && !Character.isDigit(a)) {
129 items.remove(0);
130 thereIsFlag=true;
131 }
132
133 // self call if str is a variable
134 if (str.startsWith("_")) {
135 count+=addPointLinks(CPBase.varControl.getValue(str));
136 }
137
138 // look for 'zlist' or 'Zlist'
139 else if (str.substring(1).startsWith("list")) {
140 PointLink zlink=null;
141
142 if ((str.startsWith("zli") &&
143 (zlink=packData.zlist)!=null && zlink.size()>0) ||
144 (str.startsWith("Zli") &&
145 (zlink=CPBase.Zlink)!=null && zlink.size()>0)) {
146 String[] b_string;
147 String brst;
148 String strdata=str.substring(5).trim(); // remove '?list'
149
150 // check for parens listing range of indices
151 int lsize=zlink.size()-1;
152 int[] irange=StringUtil.get_int_range(strdata, 0,lsize);
153 if (irange!=null) {
154 int aa=irange[0];
155 int bb=(irange[1]>lsize) ? lsize : irange[1];
156 for (int j=aa;j<=bb;j++) {
157 add(zlink.get(j));
158 count++;
159 }
160 }
161
162 // else check for brackets
163 else if ((b_string=StringUtil.get_bracket_strings(strdata))!=null
164 && (brst=b_string[0])!=null) {
165 if (brst.startsWith("r")) { // rotate: copy first at end
166 zlink.add(zlink.getFirst());
167 }
168 if (brst.startsWith("r")
169 || brst.startsWith("n")) { // use an remove first
170 Complex z=zlink.removeFirst();
171 add(z);
172 count++;
173 }
174 if (brst.startsWith("l")) { // last
175 add(zlink.getLast());
176 count++;
177 }
178 else { // else specified index
179 try{
180 int n=MathUtil.MyInteger(brst);
181 if (n>=0 && n<zlink.size()) {
182 add(zlink.get(n));
183 count++;
184 }
185 } catch (NumberFormatException nfe) {}
186 }
187 }
188 // else just adjoin the current list
189 else {
190 int n=size();
191 addAll(n,zlink);
192 count +=zlink.size();
193 }
194 }
195 else // no appropriate list found
196 return count;
197 }
198
199 if (!thereIsFlag && (items==null || items.size()==0)) // nothing to do?
200 return 0;
201
202 // If there is a flag, we get it, process, and return.
203 if (thereIsFlag) {
204 String str2=items.get(0);
205 char aa=str2.charAt(0);
206
207 // because of recursion, must check for second flag:
208 // should be only points
209 if (StringUtil.isFlag(str2) || (aa!='-' && aa!='.' &&
210 !Character.isDigit(aa))) {
211 // check for 'Zlist' and 'zlist' again (since they may follow a flag)
212 if (str2.startsWith("Zli")) {
213 abutMore(CPBase.Zlink);
214 return CPBase.Zlink.size();
215 }
216 else if (str2.startsWith("zli")) {
217 if (packData==null)
218 throw new ParserException("no packdata with this 'zlist'");
219 abutMore(packData.zlist);
220 return packData.zlist.size();
221 }
222
223 // else, failure
224 throw new ParserException("'PointLink' data can have only one flag");
225 }
226
227 // recursive call to process the rest of the input
228 // (should be just x y pairs)
229 PointLink ptlink=new PointLink(items);
230 if (ptlink==null || ptlink.size()==0)
231 throw new ParserException("failed in getting complex points");
232
233 switch(a) {
234 case 'f': // run string through 'function', if defined in function window
235 {
236 if (PackControl.newftnFrame.ftnField.getText().trim().length()==0) {
237 CirclePack.cpb.errMsg("'Function' frame is not set");
238 return 0;
239 }
240
241 Iterator<Complex> pts=ptlink.iterator();
242 while (pts.hasNext()) {
243 try {
244 if (add(CirclePack.cpb.getFtnValue(pts.next())))
245 count++;
246 } catch (Exception ex) {}
247 }
248 break;
249 }
250 case 'g': // run string through 'parameter path',
251 // if defined in function window
252 {
253 if (PackControl.newftnFrame.paramField.getText().trim().length()==0) {
254 CirclePack.cpb.errMsg("'Parameter' field in Function Frame is not set");
255 return 0;
256 }
257
258 Iterator<Complex> pts=ptlink.iterator();
259 while (pts.hasNext()) {
260 try {
261 if (add(CirclePack.cpb.getParamValue(pts.next().x)))
262 count++;
263 } catch (Exception ex) {}
264 }
265 break;
266 }
267 case 'z': // generally from mouse location
268 {
269 // read the x y pairs
270 Iterator<Complex> pts=ptlink.iterator();
271 while (pts.hasNext()) {
272 if (add(pts.next()))
273 count++;
274 }
275 break;
276 }
277 case 'c': // centers of specified vertices
278 {
279 if (packData==null)
280 throw new ParserException("can't get centers, no packing given");
281 NodeLink vlit=new NodeLink(packData,items);
282 Iterator<Integer> vlt=vlit.iterator();
283 while (vlt.hasNext()) {
284 if (add(packData.getCenter(vlt.next())))
285 count++;
286 }
287 break;
288 }
289 case 'd': // dual objects
290 {
291 if (str.length()==2)
292 break;
293 char c2=str.charAt(2);
294 switch (c2) {
295 case 'c': // face incenters
296 {
297 // TODO: process and return
298 break;
299 }
300 case 'p': // points of tangency
301 {
302 // TODO: process and return
303 break;
304 }
305 case 'z': // zlist? Zlist?
306 case 'Z':
307 {
308 if (str.startsWith("zlist") || str.startsWith("Zlist"))
309 break;
310 }
311 } // end of dual switch
312
313 break;
314 }
315 } // end of switch

317 return count;
318 } // end of flag processing

321 Iterator<String> its=items.iterator();
322 while (its!=null && its.hasNext()) {
323 str=its.next();
324 try {

413 if (link==null || (sz=link.size())==0 || link.get(0)!=link.get(sz-1))
414 return null;

416 int i=indx;
