406 int left=verts3.get(2);
407 int right=verts3.get(0);
408 Complex leftZ=p.getCenter(left);
409 Complex rightZ=p.getCenter(right);
410 double bottomLength=rightZ.minus(leftZ).abs();
411
412 // linked list along bottom; need to toss first 2 and last 2
413 NodeLink spots=new NodeLink(p,"b("+left+" "+right+")");
414 spots.remove(0);
415 spots.remove(0);
416 spots.remove(spots.size()-1);
417 spots.remove(spots.size()-1);
418
419 // start data in matlab form: vector 'data', 'vertCount' rows, each with
420 // entry x,s, where x is barycentric coord of point along edge, s is

422 StringBuilder outputData=new StringBuilder("%% CirclePack, percolation trials: nodecount "+

701 public int runWalker(int v) {
702 int[] flower=extenderPD.getFlower(v);
703 int mySpot=v;
704
705 // is initial vertex already pinned?
706 if (!ColorUtil.equalColors(extenderPD.getCircleColor(mySpot),ColorUtil.FG_Color))
707 return 0;
708
709 // run the walk; inherit the mark of first hit
710 while (ColorUtil.equalColors(extenderPD.getCircleColor(mySpot),ColorUtil.FG_Color)) {
711 double x=rand.nextDouble();
712 mySpot=flower[petalTrans[mySpot].whichPetal(x)];
713 }
714 int hitColor=ColorUtil.col_to_table(extenderPD.getCircleColor(mySpot));
715
716 // hit fresh vert? use same color
717 if (hitColor>200) {
718 extenderPD.setCircleColor(v,ColorUtil.coLor(hitColor));
719 return hitColor;
720 }
721
722 // else, determine what color to set
723 int myColor=0;
724 switch (hitColor) {
725 case 1: {
726 myColor=244;
727 break;
728 }
729 case 2: {
730 myColor=244;
731 break;
732 }
733 case 199: {
734 myColor=243;
735 break;
736 }

738 myColor=243;
739 break;
740 }
741 } // end of switch
742
743 extenderPD.setCircleColor(v,ColorUtil.cloneMe(ColorUtil.coLor(myColor)));

755 public void setPetalTrans() {
756 for (int v=1;v<=extenderPD.nodeCount;v++) {
757 int num=extenderPD.countFaces(v)+extenderPD.getBdryFlag(v)-1;
758//System.out.println(" vert "+v+" num "+num);
759 petalTrans[v]=new PetalTrans(num);
760
761 double totalCond=0.0;
762 for (int j=0;j<=num;j++) {
763 totalCond += conductances[v][j];
764 }
765
766 double accum=0.0;
767 for (int j=0;j<num;j++) {
768 accum+=conductances[v][j]/totalCond;
769 petalTrans[v].portions[j]=accum;
