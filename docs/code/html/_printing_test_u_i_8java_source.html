95 public void actionPerformed(ActionEvent e) {
96
97 //Browse button is pressed. Open JFileChooser.
98 if ("browse".equals(e.getActionCommand())) {
99 chooser = new JFileChooser();
100 chooser.setFileFilter(new JPGHTMLFilter());
101 int returnVal = chooser.showOpenDialog(null);
102 if (returnVal == JFileChooser.APPROVE_OPTION) {
103 String extension = CPFileManager.getFileExt(chooser.getSelectedFile());
104 if (extension != null) {
105 if (extension.equalsIgnoreCase("jpg") ||
106 extension.equalsIgnoreCase("jpeg")) {
107 //If file is JPG, initialize the JPG Printable.
108 jpgPrintable = new JPGPrintable(chooser.getSelectedFile());
109 printerJob.setPrintable(jpgPrintable, pageFormat);
110 }
111 else if (extension.equalsIgnoreCase("html") ||
112 extension.equalsIgnoreCase("htm")) {
113 //If file is HTML, initialize the HTML Printable.
114 htmlPrintable = new HTMLPrintable(chooser.getSelectedFile());
115 printerJob.setPrintable(htmlPrintable, pageFormat);
116 }
117 else {
118 //User has somehow selected an invalid file.
119 System.exit(1);
120 }
121 }
122 fileLabel.setText(chooser.getSelectedFile().getName());
123 //Set the job name to the selected file name.
124 printerJob.setJobName(chooser.getSelectedFile().getName());
125 //Enable the page format and print buttons.
126 pageFormatButton.setEnabled(true);
127 printButton.setEnabled(true);
128 }
129 }
130 //Page format button is pressed.
131 if ("pageformat".equals(e.getActionCommand())) {
132 /* pageDialog() brings up the page dialog window and returns the
133 PageFormat instance with any changes. */
134 pageFormat = printerJob.pageDialog(pageFormat);
135 }
136 //Print button is pressed.
137 if ("print".equals(e.getActionCommand())) {
138 if (printerJob.printDialog()) {
139 try {
140 /* print() brings up the print dialog window and attempts to
141 print according to the current Printable on confirmation. */
142 printerJob.print();
