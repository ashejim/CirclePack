739 public static Complex string2Complex(String str)
740 throws VarException {
741 if (str.charAt(0) == '-' && StringUtil.isFlag(str))
742 throw new VarException(
743 "this seems to be a flag, not a complex number");
744 if (str.charAt(0) == '+') // this should preclude '_' at start
745 str = str.substring(1).trim();
746 boolean minusflag = false;
747 int spot_i = str.indexOf('i');
748 int spot_I = str.indexOf('I');
749
750 // Tokenize on whitespace to identify real and imaginary parts
751 String[] pieces = str.split(" ");
752 int numPieces = pieces.length;
753 if (numPieces == 0)
754 throw new VarException("empty string");
755 if ((spot_i < 0 && spot_I < 0 && numPieces > 2) || (spot_i >= 0 && spot_I >= 0))
756 throw new VarException("improper complex number format");
757
758 // --- Single token: might need splitting at an internal '+' or '-' ---
759 // Care is taken to skip the '-' in scientific-notation exponents
760 // like "3.5E-02".
761 if (numPieces == 1) {
762 int splitspot = -1;
763 int tick = 1;
764 while (splitspot < 0 && tick < str.length() - 2) {
765 if (str.charAt(tick) == '+') {
766 char c = str.charAt(tick - 1);
767 if (c != 'e' && c != 'E') {
768 splitspot = tick;
769 break;
770 }
771 }
772 if (str.charAt(tick) == '-') {
773 char c = str.charAt(tick - 1);
774 if (c != 'e' && c != 'E') {
775 splitspot = tick;
776 break;
777 }
778 }
779 tick++;
780 }
781
782 if (splitspot > 0) {
783 pieces = new String[2];
784 pieces[0] = str.substring(0, splitspot);
785 pieces[1] = str.substring(splitspot);
786 if (pieces[1].charAt(0) == '+') // shuck '+' sign
787 pieces[1] = pieces[1].substring(1);
788 numPieces = 2;
789 }
790
791 // Remaining single token: either a real number or pure imaginary
792 else {
793 if (spot_i >= 0 || spot_I >= 0) {
794 Double imag = getImagPart(pieces[0]); // this also looks for variable
795 if (imag == null)
796 throw new VarException("pure imaginary format error");
797 return new Complex(0.0, (double) imag);
798 }
799 // Check for a CirclePack variable reference
800 if (str.charAt(0) == '_') {
801 Double real = null;
802 try {
803 real = Double.parseDouble(CPBase.varControl.getValue(str));
804 return new Complex((double) real);
805 } catch (Exception ex) {
806 throw new VarException("not read: " + ex.getMessage());
807 }
808 }
809 try {
810 return new Complex(Double.parseDouble(str));
811 } catch (Exception ex) {
812 throw new VarException("not a number :" + ex.getMessage());
813 }
814 }
815 }
816
817 // --- More than 2 tokens: reduce by consuming separators ---
818 // Strips stray '*', '+', and '-' tokens and repositions 'i'/'I'
819 // until exactly two meaningful tokens remain.
820 while (numPieces > 2) {
821// String endone=pieces[numPieces-1];
822
823 // unneeded '*'
824 if (pieces[1].equals("*")) {
825 for (int j = 1; j < numPieces - 1; j++)
826 pieces[j] = pieces[j + 1];
827 numPieces--;
828 }
829
830 // unneeded ' + '
831 if (pieces[1].equals("+")) {
832 for (int j = 1; j < numPieces - 1; j++)
833 pieces[j] = pieces[j + 1];
834 numPieces--;
835 }
836
837 // ' - ', hold for later
838 else if (pieces[1].equals("-")) {
839 minusflag = true;
840 for (int j = 1; j < numPieces - 1; j++)
841 pieces[j] = pieces[j + 1];
842 numPieces--;
843 }
844
845 else if (pieces[1].startsWith("i") || pieces[1].startsWith("I")) {
846 // Rewrite so 'i' is a suffix (e.g. "i*3" -> "3i")
847 if (pieces[1].charAt(1) == '*') {
848 if (pieces[1].length() > 2)
849 pieces[1] = new String(pieces[1].substring(2) + "i");
850 }
851 else if (pieces[1].length() > 1) {
852 pieces[1] = new String("-" + pieces[1].substring(1) + "i");
853 }
854 else { // or remove this piece and put 'i' at end of next one
855 for (int j = 1; j < numPieces - 1; j++)
856 pieces[j] = pieces[j + 1];
857 numPieces--;
858 pieces[1] = new String(pieces[1] + "i");
859 }
860 }
861
862 // if last string is just 'i' or '*i', add 'i' to previous (should be number)
863 else if (pieces[numPieces - 1].equalsIgnoreCase("*i") ||
864 pieces[numPieces - 1].equalsIgnoreCase("i")) {
865 pieces[numPieces - 2] = new String(pieces[numPieces - 1] + "i");
866 pieces[numPieces - 1] = null;
867 numPieces--;
868 }
869
870 else
871 throw new VarException("improper format");
872 }
873
874 // --- Down to exactly two tokens: assign real and imaginary parts ---
875 Double realpart = null;
876 Double impart = null;
877
878 spot_i = pieces[0].indexOf("i");
879 spot_I = pieces[0].indexOf("I");
880
881 // Imaginary part could be first
882 if (spot_i >= 0 || spot_I >= 0) {
883 impart = getImagPart(pieces[0]);
884 if (impart == null) {
885 throw new VarException("error in getting first segment as imaginary part");
886 }
887 }
888 // else try to interpret as the real part
889 else {
890 try {
891 if (pieces[0].charAt(0) == '_')
892 realpart = Double.parseDouble(CPBase.varControl.getValue(pieces[0]));
893 else realpart = Double.parseDouble(pieces[0]);
894 } catch (Exception ex) {
895 throw new VarException("error in real part: " + ex.getMessage());
896 }
897 }
898
899 spot_i = pieces[1].indexOf("i");
900 spot_I = pieces[1].indexOf("I");
901 if ((spot_i >= 0 || spot_I >= 0)) {
902 if (impart != null)
903 throw new VarException("both parts think they are the imaginary part");
904 impart = getImagPart(pieces[1]);
905 if (impart == null) {
906 throw new VarException("missing real or imaginary part");
907 }
908 }
909
910 // If one part is still unresolved, try parsing the remaining token
911 // as a plain number
912 if (realpart == null || impart == null) {
913 try {
914 if (realpart != null)
915 impart = Double.parseDouble(pieces[1]);
916 else if (impart != null) {
917 if (pieces[1].charAt(0) == '_')
918 realpart = Double.parseDouble(CPBase.varControl.getValue(pieces[1]));
919 else realpart = Double.parseDouble(pieces[1]);
920 }
921 } catch (Exception ex) {
922 throw new VarException("failed to get real or imaginary part");
923 }
924 }
925
926 if (realpart == null || impart == null)
927 throw new VarException("missing real or imaginary part");
928
929 double real = (double) realpart;
930 double img = (double) impart;
931 if (minusflag) img *= -1.0;
932 return new Complex(real, img);
933 }

951 public static Double getImagPart(String imag) {
952
953 // first, check for leading negative sign
954 boolean negSign = false;
955 if (imag.startsWith("-")) {
956 negSign = true;
957 imag = new String(imag.substring(1));
958 }
959
960 int spot_i = imag.indexOf('i');
961 int spot_I = imag.indexOf('I');
962
963 // just 'i' or 'I' alone?
964 if (imag.length() == 1) {
965 if (negSign)
966 return Double.valueOf(-1.0);
967 return Double.valueOf(1.0);
968 }
969
970 // check for '*' separating the coefficient from 'i'
971 int st = imag.indexOf("*");
972 if (st < 0) { // no '*'
973 // starts with 'i' or 'I' — coefficient follows
974 if (spot_i == 0 || spot_I == 0) {
975 try {
976 if (imag.charAt(1) == '_') // variable?
977 return Double.parseDouble(CPBase.varControl.getValue(imag.substring(1)));
978 return Double.parseDouble(imag.substring(1));
979 } catch (Exception ex) {
980 return null;
981 }
982 }
983 // ends with 'i' or 'I' — coefficient precedes
984 if (spot_i == imag.length() - 1 || spot_I == imag.length() - 1) {
985 try {
986 return Double.parseDouble(imag.substring(0, imag.length() - 1));
987 } catch (Exception ex) {
988 return null;
989 }
990 }
991 }
992
993 // There is a '*': expect "i*<coeff>", "I*<coeff>", "<coeff>*i", or "<coeff>*I"
994 if (imag.startsWith("i*") || imag.startsWith("I*")) {
995 try {
996 double nd;
997 if (imag.charAt(2) == '_') // variable?
998 nd = Double.parseDouble(CPBase.varControl.getValue(imag.substring(2)));
999 else nd = Double.parseDouble(imag.substring(2));
1000 if (negSign)
1001 nd *= -1.0;
1002 return Double.valueOf(nd);
1003 } catch (Exception ex) {
1004 return null;
1005 }
1006 }
1007 if (imag.endsWith("*i") || imag.endsWith("*I")) {
1008 double nd;
1009 try {
1010 nd = Double.parseDouble(imag.substring(0, imag.length() - 2));
1011 } catch (Exception ex) {
1012 return null;
1013 }
1014 if (negSign)
1015 nd *= -1.0;
1016 return Double.valueOf(nd);
1017 }
1018
1019 // should have exhausted all properly formatted cases
1020 return null;
1021 }
