617 public double reportEnergy() {
618 double totalE=0.0;
619// double carbonE=0.0;
620 double topRingE=0.0;
621 int topRingV=0;
622 double topCarbonE=0.0;
623 int topCarbonFace=0;
624 double topBondE=0.0;
625 int topAngSumV=0;
626 double topAngSumE=0.0;
627 EdgeSimple topBondEdge=new EdgeSimple(0,0);
628
629 for (int f = 1; f <= extenderPD.faceCount; f++) {
630 CarbonEnergy cE = carbonEnergies.get(f);
631 double newC = cE.atomEnergy();
632
633 // look for largest carbon
634 if (newC > topCarbonE) {
635 topCarbonE = newC;
636 topCarbonFace = f;
637 }
638
639 int k = -1;
640 int myf=cE.faceIndx;
641 int[] verts=extenderPD.packDCEL.faces[myf].getVerts();
642 for (int j = 0; (j < 3 && k < 0); j++) {
643 int vj = verts[j];
644 int oppf = extenderPD.face_opposite(myf, vj);
645 if (oppf >= 0) { // ignore phantom bonds
646 double L = cE.bondLengths[j];
647 L = L * L;
648 double newB = (bondParam / 4.0) * (L - 1.0) * (L - 1.0);
649 if (newB > topBondE && myf < oppf) {
650 topBondE = newB;
651 topBondEdge = new EdgeSimple(myf, oppf);
652 k = j;
653 }
654 }
655 }
656 }
657
658 // identify largest ring, angle sum energy and its vert
659 for (int v = 1; v <= extenderPD.nodeCount; v++) {
660 double newE = ringEnergy(v);
661 double newASE=0.0;
662 if (!extenderPD.isBdry(v)) {
663 Vertex vert=extenderPD.packDCEL.vertices[v];
664 double angsum=extenderPD.packDCEL.getVertAngSum(vert);
665 double term=2.0*Math.PI-angsum;
666 newASE=planarParam*term*term;
667 }
668 if (newE > topRingE) {
669 topRingE = newE;
670 topRingV = v;
671 }
672 if (!extenderPD.isBdry(v) && (newASE>topAngSumE)) {
673 topAngSumE=newASE;
674 topAngSumV=v;
675 }
676 totalE += newE+newASE;
677 }
678
679 msg("Total energy is " + String.format("%.8e", totalE) + "\n"
680 + " Largest ring energy: " + String.format("%.6e", topRingE)
681 + " at " + topRingV + "\n"
682 + " Largest anglesum energy: " + String.format("%.6e", topAngSumE)
683 + " at " + topAngSumV + "\n"
684 + " Largest carbon (angle) energy: "
685 + String.format("%.6e", topCarbonE) + " at " + topCarbonFace
686 + "\n" + " Largest bond energy:"
687 + String.format("%6e", topBondE) + ", dual edge <"
688 + topBondEdge.v + "," + topBondEdge.w + ">");
689
690 updateColors();
691 return totalE;
692 }

723 public double ringEnergy(int v) {
724 int[] fflower=extenderPD.getFaceFlower(v);
725 double bondE=0.0;
726 double angE=0.0;
727 for (int j=0;j<extenderPD.countFaces(v);j++) {
728 CarbonEnergy cE=carbonEnergies.get(fflower[j]);
729 if (j>0 || !extenderPD.isBdry(v)) {// skip first for bdry
730 bondE+=cE.getBondEnergy(v)/2.0;
731 }
732 angE += cE.getAngleEnergy(v);
733 }
734 double planarE=0.0;
735 if (!extenderPD.isBdry(v)) { // interior?
736 Vertex vert=extenderPD.packDCEL.vertices[v];
737 double angsum=extenderPD.packDCEL.getVertAngSum(vert);
738 double angleErr=Math.PI*2-angsum;
739 planarE=planarParam*(angleErr*angleErr);
740 }
741 return (bondE+angE+planarE);
742 }

769 public static double L(double r,double s,double t,double u) {
770 return Math.sqrt((r*s*t)/(r+s+t))+Math.sqrt((r*u*t)/(r+u+t));
771 }

778 public static double C(double r,double t,double u) {
779 return (2.0*(t/(r+t))*(u/(r+u))-1.0);
780 }

782 public static double dCdr(double r,double t,double u) {
783 return (-2.0*t*u/((r+t)*(r+u))*(1/(r+t)+1/(r+u)));
784 }
785
786 public static double dCdt(double r,double t,double u) {
787 return (2*r*u/((r+t)*(r+t)*(r+u)));
788 }
789
790 public static double dCdu(double r,double t,double u) {
791 return dCdt(r,u,t);
792 }

799 public double A(double r,double s,double t,double u,double v) {
800 double term=L(r,s,t,u)*L(r,t,u,v)*C(r,t,u)+0.5;
801 return angleParam*term*term;

897 for (int vertindx=1;vertindx<=extenderPD.nodeCount;vertindx++) {

903 double angsum=extenderPD.packDCEL.getVertAngSum(
904 extenderPD.packDCEL.vertices[vertindx],r);
905 double ASfactor=(-2.0)*planarParam*(Math.PI*2.0-angsum);

908 // get data for faces {r,s,t}, {r,t,u}, and {r,u,v}
909 for (int j=0;j<num;j++) {

914 int uv=extenderPD.getPetal(vertindx,(j+1)%num);
915 double u=extenderPD.getRadius(uv);

917 double v=extenderPD.getRadius(vv);
918
919 if (j==0 && bdryvert) {
920 s=u; // reflected data for phantom bond
921 }
922 else if (j==(num-1) && bdryvert) {
923 v=t; // reflected data for phantom bond
924 }
925
926 if ((j>0 || !bdryvert) && (j<(num-1) || !bdryvert)) {
927 // bond contributions (cut in half due to duplication)
928 grad[vertindx] += 0.5*dBdr(r,s,t,u);
929 if (j==0) { // not bdry,
930 grad[sv] += 0.5*dBds(r,s,t,u);
931 }
932 grad[tv] += 0.5*dBdt(r,s,t,u);
933 if (j==(num-1)) // not bdry
934 grad[uv] += 0.5*dBdu(r,s,t,u);

949 }
