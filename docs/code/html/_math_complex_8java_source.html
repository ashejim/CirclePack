293 public static Complex ln(Complex t) {
294 double phi = 0d;
295 double rho = abs(t);
296 double tempx = Math.log(rho);
297 // Determine the argument in [0, 2π) by quadrant analysis
298 if (Math.abs(t.real()) > tol)
299 if(t.real()>0)
300 if(t.imag()>=0)
301 phi = Math.atan(t.imag() / t.real());
302 else
303 phi = 2*Math.PI+Math.atan(t.imag() / t.real());
304 else

461 public static double distancePL(Complex p0, Complex p1, Complex p2) {
462 // Direction vector of the segment
463 double ax = p2.real() - p1.real();
464 double ay = p2.imag() - p1.imag();
465 // Parameter t for the foot of the perpendicular from p0 onto the line
466 double t = ( (p0.real() - p1.real()) * ax + (p0.imag() - p1.imag()) * ay) /
467 (ax * ax + ay * ay);
468 // Coordinates of the foot
469 double xopt = p1.real() + ax * t;
470 double yopt = p1.real() + ay * t;
471 // Check whether the foot lies between p1 and p2 by examining
472 // whether the vectors from the endpoints to the foot point in
473 // opposite directions
474 // atemp1 from p1 to opt
475 double atemp1x = xopt - p1.real();
476 double atemp1y = yopt - p1.real();
477 // atemp2 from p2 to opt
478 double atemp2x = xopt - p2.real();
479 double atemp2y = yopt - p2.imag();
480
481 // atemp1 and atemp2 should have opposite sign if foot is between endpoints
482 if ( (atemp1x * atemp2x <= 0) && (atemp1y * atemp2y <= 0))
483 return Math.sqrt( (p0.real() - xopt) * (p0.real() - xopt) +
484 (p0.imag() - yopt) * (p0.imag() - yopt));
485 else
486 return Math.min(Math.sqrt( (p0.real() - p1.real()) * (p0.real() - p1.real()) +
487 (p0.imag() - p1.imag()) * (p0.imag() - p1.imag())),
488 Math.sqrt( (p0.real() - p2.real()) * (p0.real() - p2.real()) +
489 (p0.imag() - p2.imag()) * (p0.imag() - p2.imag())));
490 }
