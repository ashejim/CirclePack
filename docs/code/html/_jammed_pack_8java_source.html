206 double r=extenderPD.getRadius(iflower[j]);
207 sqrs=sqrs+r*r;
208 mean=mean+r;
209 }

211 double var=sqrs/N-mean*mean;
212 double dev=Math.sqrt(var)/mean; // standard deviation as fraction of mean

334 errorMsg("vertices "+v+" and "+w+" are not original vertices");

433 public HalfEdge addEdge(int v,int w,int b) {
434 if (v==w || homePack.getVertMark(v)>=0 || homePack.getVertMark(w)>=0 ||
435 homePack.getVertMark(b)<0 || extenderPD.isBdry(b))
436 return null;
437
438 PackDCEL pdcel=extenderPD.packDCEL;
439 // if red chain runs through 'b', then toss it
440 if (pdcel.vertices[b].redFlag)
441 pdcel.redChain=null;
442
443 // first split the flower at b, giving new vertex
444 // and edge connecting it to b
445 HalfEdge vedge=pdcel.findHalfEdge(new EdgeSimple(b,v));
446 HalfEdge wedge=pdcel.findHalfEdge(new EdgeSimple(b,w));
447 if (vedge==null || wedge==null)
448 return null;
449 HalfEdge tmpEdge=RawManip.splitFlower_raw(pdcel,vedge,wedge);
450 if (tmpEdge==null)
451 return null;
452
453 // now flip that new edge
454 HalfEdge newEdge=RawManip.flipEdge_raw(pdcel,tmpEdge);
455 return newEdge;
456 }
