224 public static EdgeLink equil2tri(int a,int b, int c,int [][]v2m,int d) {
225 int []A=v2m[a];
226 int []B=v2m[b];
227 int []C=v2m[c];
228 int []hold;
229
230 // In our slanted hex arrangement, triangle <A,B,C> has one of
231 // two shapes and we can arrnage that A, B are horizontal, with
232 // C above or below B. Make it so.
233 if (A[1]!=B[1]) {
234 if (A[1]==C[1]) { // rotate clw
235 hold=C;
236 C=B;
237 B=A;
238 A=hold;
239 }
240 else { // B, C form horizontal, rotate cclw
241 hold=C;
242 C=A;
243 A=B;
244 B=hold;
245 }
246 }
247
248 // If C above A, all set; B to right of A, step increment = +1.
249 // But if C is below, then B is left of A, step increment = -1;
250 EdgeLink elink=new EdgeLink();
251 if (C[1]>A[1]) {
252 for (int i=1;i<d;i++) {
253 for (int j=0;j<=i;j++)
254 elink.add(new EdgeSimple(A[0]+i,A[1]+j) );
255 elink.add(new EdgeSimple(B[0],B[1]+i));
256 }
257 }
258 else {
259 for (int i=1;i<d;i++) {
260 for (int j=0;j<=i;j++)
261 elink.add(new EdgeSimple(A[0]-i,A[1]-j) );
262 elink.add(new EdgeSimple(B[0],B[1]-i));
263 }
264 }
265
266 return elink;
267 }

610 public static EdgeSimple uw2UWBox(int u,int w,int pn,int n,int d) {
611 int N=n*d;
612 if (n<2 || d<1 || u<-N || u>N || w>(u+N) || w<-N || w>N || w<(u-N))
613 return null;
614 int uu=u;
615 int ww=w;
616 // find (uu,ww) relative to verts of form (d*x,d*y)
617 int x=(int)Math.floor((double)uu/(double)d);
618 int us=x*d;
619 int y=(int)Math.floor((double)ww/(double)d);
620 int ws=y*d;
621 return new EdgeSimple(us,ws);
622 }
