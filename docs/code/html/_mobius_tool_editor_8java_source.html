254 public void initMoreFields(MyTool theTool) {
255 reset_abcd();
256 int indx=getCPIconIndx(theTool.getCPIcon());
257 if (indx<0) iconCombo.iconBox.setSelectedIndex(randomCPIcon());
258 else iconCombo.iconBox.setSelectedIndex(indx);
259
260 // parse the current command to set the fields:
261 // Should be form 'appMob a.x a.y b.x b.y c.x c.y d.x d.y [-1]'
262 oriented=true;
263 String cmds[]=theTool.getCommand().split("\\s+");
264 Vector<Double> parts=new Vector<Double>(10);
265 try {
266 for(int i=1;i<=8;i++) {
267 parts.add(Double.parseDouble(cmds[i]));
268 }
269 if (cmds.length>9) {
270 double flip=Double.parseDouble(cmds[9]);
271 if (flip<0) oriented=false;
272 }
273 } catch(Exception ex) {
274 throw new ParserException("Error setting MobiusToolEditor");
275 }
276 orientBox.setSelected(oriented);
277
278 try {
279 aField.setValue(new Complex(parts.get(0),parts.get(1)));
280 bField.setValue(new Complex(parts.get(2),parts.get(3)));
281 cField.setValue(new Complex(parts.get(4),parts.get(5)));
282 dField.setValue(new Complex(parts.get(6),parts.get(7)));
283 } catch (NumberFormatException e) {
284 reset_abcd();
285 return;
286 }
287 }
