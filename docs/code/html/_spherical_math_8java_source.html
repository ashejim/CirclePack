495 public static Complex sph_tangency(Complex z1,Complex z2,
496 double r1,double r2) {
497 double dratio=(s_dist(z1,z2))*r1/(r1+r2);
498 // try first direction
499 Point3D T3=new Point3D(sph_tangent(z1,z2));
500 Complex tp1=sph_shoot(z1,T3,dratio);
501 double err1=Math.abs(s_dist(z1,tp1)-r1);
502 if (err1<S_TOLER) // looks good
503 return tp1;
504
505 // try opposite direction
506 T3=T3.times(-1.0); // flip tangent
507 Complex tp2=sph_shoot(z1,T3,dratio);
508 double err2=Math.abs(s_dist(z1,tp2)-r1);
509 if (err2<S_TOLER) // looks good
510 return tp2;
511
512 // Should be tp1
513 if (err1<=10000.0*err2)
514 return tp1;
515 return tp2;
516 }

533 public static CircleSimple s_compcenter(Complex z0,Complex z1,
534 double r0,double r1,double r2,double ivd0,double ivd1,double ivd2) {
535 double[] vec0=s_pt_to_vec(z0);
536 // side lengths
537 double s0=s_ivd_length(r0,r1,ivd0);
538 double s1=s_ivd_length(r1,r2,ivd1);
539 double s2=s_ivd_length(r2,r0,ivd2);
540 // angle is how far around from TV we will rotate
541 double angle=Math.acos(( Math.cos(s1)-Math.cos(s2)*Math.cos(s0) )/
542 ( Math.sin(s2)*Math.sin(s0) ));
543 // TV is a tangent vector at z0
544 double[] TV=sph_tangent(z0,z1);
545 // N = z0 x T
546 double[] N=crossProduct(vec0,TV);
547 // P will point toward the new center
548 double []P=new double[3];
549 P[0]=Math.cos(angle)*TV[0]+Math.sin(angle)*N[0];
550 P[1]=Math.cos(angle)*TV[1]+Math.sin(angle)*N[1];
551 P[2]=Math.cos(angle)*TV[2]+Math.sin(angle)*N[2];
552
553 double []mtan=new double[3];
554 mtan[0]=Math.cos(r0+r2)*vec0[0]+Math.sin(r0+r2)*P[0];
555 mtan[1]=Math.cos(r0+r2)*vec0[1]+Math.sin(r0+r2)*P[1];
556 mtan[2]=Math.cos(r0+r2)*vec0[2]+Math.sin(r0+r2)*P[2];
557 Complex z=new Complex(0.0);
558 if(mtan[2]<=(1.0-S_TOLER)) {
559 z=new Complex(Math.atan2(mtan[1],mtan[0]),Math.acos(mtan[2]));
560 }
561 return new CircleSimple(z,r2,1);

588 public static Complex s_shoot(Complex z,Complex w,double dist) {
589 // if very close, just return z
590 if (Math.abs(dist)<S_TOLER)
591 return new Complex(z);
592
593 // adjust mod 2*pi until dist lies in [0,2*pi]
594 double pi2=2.0*Math.PI;
595 while (dist<0) dist+=pi2;
596 while (dist>pi2) dist -=pi2;
597
598 double[] T=sph_tangent(z,w);
599 double[] V=s_pt_to_vec(z);
600 double[] A=new double[3];
601 double cosd=Math.cos(dist);
602 double sind=Math.sin(dist);
603 A[0]=cosd*V[0]+sind*T[0];
604 A[1]=cosd*V[1]+sind*T[1];
605 A[2]=cosd*V[2]+sind*T[2];
606 return proj_vec_to_sph(A[0],A[1],A[2]);
607 }

643 public static double[] sph_tangent(Complex ctr1,Complex ctr2) {
644 double d,vn;
645 double[] A;
646 double[] B;
647 double[] P=new double[3];
648 double[] T=new double[3];
649
650 A=s_pt_to_vec(ctr1);
651 B=s_pt_to_vec(ctr2);
652 d=dot_prod(A,B);
653 // Find projection of B on plane normal to A
654 P[0]=B[0]-d*A[0]; P[1]=B[1]-d*A[1]; P[2]=B[2]-d*A[2];
655
656 // A and B essentially parallel?
657 if ((vn=vec_norm(P))<S_TOLER)
658 {
659 // if A is not N or S, point in horizontal direction
660 double pn=Math.sqrt(A[0]*A[0]+A[1]*A[1]);
661 if (pn>.0000001) {
662 // get orthogonal, X coord 0
663 T[0]=A[1]/pn;
664 T[1]=-A[0]/pn;
665 T[2]=0;
666 }
667 // otherwise, point toward (0,1,0)
668 else {
669 T[0]=0.0;
670 T[1]=1.0;
671 T[2]=0.0;
672 }
673 return T;
674 }
675 T[0]=P[0]/vn;T[1]=P[1]/vn;T[2]=P[2]/vn;
676 return T;
677 }

728 public static CircleSimple s_to_e_data(Complex z,double r) {
729 int flipflag=1; // set to -1 if south pole enclosed
730 double er; // new radius
731 Complex e; // new center
732
733 double[] V=s_pt_to_vec(z); // unit 3-vector
734
735 // essentially passes through infinity?
736 if (Math.abs(z.y+r-Math.PI)<S_TOLER) // increment r slightly, proceed
737 r += 2.5*S_TOLER;
738
739 // encloses infinity?
740 if ((z.y+r)>=(Math.PI+S_TOLER)) {
741 r=Math.PI-r; // fake radius
742 V[0] *= (-1.0);
743 V[1] *= (-1.0);
744 V[2] *= (-1.0);
745 z=proj_vec_to_sph(V[0],V[1],V[2]); // fake center
746 flipflag=-1;
747 }
748
749 // z-coords of bdry points below/above center
750 double up=z.y+r; // below
751 double down=z.y-r; // above
752
753 // essentially centered at np
754 if (Math.abs(z.y)<S_TOLER) {
755 er=Math.sin(up)/(1.0+Math.cos(up));
756 e=new Complex(0.0);
757 return new CircleSimple(e,er,flipflag);
758 }
759
760 // essentially centered at sp, but not flipped.
761 if (Math.abs(z.y-Math.PI)< S_TOLER && flipflag!=-1) {
762 // radius must be extremely small, treat as though
763 // it encloses sp, so fake center and huge fake radius.
764 er=100000;
765 e=new Complex(0.0);
766 flipflag=-1;
767 return new CircleSimple(e,er,flipflag);
768 }
769
770 // essentially passes through infinity, so decrease 'up'.
771 if (Math.abs(up-Math.PI)< .00001) {
772 up -= .000015;
773 }
774
775 // proceed
776 double RR=Math.sin(up)/(1.0+Math.cos(up));
777 double rr=Math.sin(down)/(1.0+Math.cos(down));
778 er=Math.abs(RR-rr)/2.0;
779 double m=(RR+rr)/2.0;
780 double sny=Math.sin(z.y);
781 e=new Complex(V[0]*m/sny,V[1]*m/sny);
782
783 return new CircleSimple(e,er,flipflag);
784 }

800 public static CircleSimple e_to_s_data(Complex ez,double er) {
801
802 Complex sz=new Complex(0.0); // will be sph center
803 double[] P3=new double[3]; // 3D point on sphere
804
805 double rr=Math.abs(er); // note, r negative handled later
806
807 // if er too small, project center, set sr=er unchanged.
808 if (rr<S_TOLER) {
809 double denom=ez.absSq()+1.0;
810 P3[0]=(2*ez.x)/denom;
811 P3[1]=(2*ez.y)/denom;
812 P3[2]=(2.0-denom)/denom;
813 if(P3[2]>(1.0-S_TOLER)) { // near N pole
814 ez.x=ez.y=0.0;
815 return new CircleSimple(sz,er,0);
816 }
817 if (P3[2]<(S_TOLER-1.0)) { // near S pole
818 sz.x=0.0;
819 sz.y=Math.PI;
820 return new CircleSimple(sz,er,0);
821 }
822 sz.y=Math.acos(P3[2]);
823 sz.x=Math.atan2(P3[1],P3[0]);
824 return new CircleSimple(sz,er,1);
825 }
826
827 // General strategy: project 3 equally spaced points of the
828 // euclidean circle to 3 points on the sphere.
829
830 Complex[] epts=new Complex[3];
831 epts[0]=ez.plus(rr);
832 epts[1]=ez.plus(CPBase.omega3[1].times(rr));
833 epts[2]=ez.plus(CPBase.omega3[2].times(rr));
834
835 Complex[] spts=new Complex[3];
836 for (int j=0;j<3;j++)
837 spts[j]=proj_pt_to_sph(epts[j]);
838
839 // for er<0, reverse orientation
840 if (er<0) {
841 Complex h=spts[0];
842 spts[0]=spts[2];
843 spts[2]=h;
844 }
845
846 // find the circle through the 3 spherical points
847 CircleSimple cS=circle_3_sph(spts[0],spts[1],spts[2]);
848
849 return cS;
850 }
