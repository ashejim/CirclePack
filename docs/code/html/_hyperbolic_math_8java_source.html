836 w0 = z1;
837 w1 = z2;
838 w2 = z0;
839 } else if (w0 == null)
840 z1 = z1.divide(z1.abs());
841 if (w0 == null && z2.absSq() < .9999) {
842 w0 = z2;
843 w1 = z0;
844 w2 = z1;

962 Complex z2;
963 Mobius map; // normalize: z0 at origin or at -1, z1 on + real axis
964 Mobius remap; // return to original
965 double R2=.20;
966
967 // (1) Most Common: all finite. Assume first at origin,
968 // second on pos x-axis; compute angle at first,
969 // dist from origin of third to get normalized center,
970 // then apply remap to carry first/second to original
971 // positions.
972 if (x0>0 && x1>0 && x2>0) {
973 map=Mobius.standard_mob(z0, z1);
974 remap=(Mobius)map.inverse();
975 double ch0=h_ivd_cosh(x0,x1,ivd0);
976 double ch1=h_ivd_cosh(x1,x2,ivd1);
977 double ch2=h_ivd_cosh(x2,x0,ivd2);
978 // compute angle at first by law of cosines:
979 // recall sinh^2=cosh^2-1; see 'h_comp_cos'
980 double cosang=(ch2 * ch0 - ch1) /
981 Math.sqrt((ch2 * ch2 - 1.0) * (ch0 * ch0 - 1.0));
982 double sinang=Math.sqrt(1.0-cosang*cosang);
983 double len01=h_ivd_length(x0,x2,ivd2);
984 double exph=Math.exp(-len01);
985 R2=(1-exph)/(1+exph);
986 Complex z=new Complex(R2*cosang,R2*sinang);
987 z2=remap.apply(z);
988 return new CircleSimple(z2, x2, 1);
989 }
990
991 CircleSimple csIn;
992 CircleSimple csOut=new CircleSimple(); // instantiate
993
994 // First and second finite, third horocycle
995 if (x0>0 && x1>0 && x2<0) {
996 map=Mobius.standard_mob(z0, z1); // map.apply(z1);
997 remap=(Mobius)map.inverse();
998 double R0=(2-x0 - 2*Math.sqrt(1-x0)) / x0; // eucl radius at origin
999 double len0=h_ivd_length(x0,x1,ivd0);
1000 double s=Math.exp(-len0);
1001 double edist=(1-s)/(1+s);
1002 Complex hypz=new Complex(edist); // absolute value of hyp center
1003 double R1=h_to_e_data(new CircleSimple(hypz,x1)).rad;
1004 R2=(1.0-R0*R0)/(2.0+2.0*R0*ivd2); // eucl rad of third circle
1005 double cosang=EuclMath.e_cos_overlap(R0, R1, R2,ivd0,ivd1,ivd2);
1006 double sinang=Math.sqrt(1.0-cosang*cosang);
1007 Complex z=new Complex(cosang,sinang); // z.abs();
1008 csIn=new CircleSimple(z,-R2);
1009 Mobius.mobius_of_circle(remap, -1, csIn, csOut, true);
1010 return csOut;
1011 }
1012
1013 // first two are horocycles
1014 if (x0<0 && x1<0) { // first and second are horocycles
1015 return h_horo_center(z0,z1,x0,x1,x2,ivd0,ivd1,ivd2);
1016 }
1017
1018 int flag=1; // 1=normal; -1, interchange first/second;
1019
1020 // (3) first horocycle, second finite; we interchange them
1021 if (x0<0 && x1>0) {
1022 Complex holdz=z1;
1023 z1=z0;
1024 z0= holdz;
1025 double hold = x0;
1026 x0 = x1;
1027 x1 = hold;
1028 hold = ivd1;
1029 ivd2 = ivd1;
1030 ivd1 = hold;
1031 flag=-1;
1032 }
1033
1034 // First is now finite, second is horocycle, third may be
1035 // finite or infinite. Convert to eucl situation.
1036 // HyperbolicMath.x_to_h_rad(x2);
1037 double r0=(2-x0 - 2*Math.sqrt(1-x0)) / x0; // eucl radius at origin
1038 double R1=(1-r0*r0)/(2.0+2.0*r0*ivd0); // eucl rad of horocycle
1039
1040 // compute
1041 double len2=0.0; // need hyp length later if third is finite
1042
1043 // if third is horocycle
1044 if (x2<0) {
1045 R2=(1-r0*r0)/(2.0+2.0*r0*ivd2); // eucl rad of third
1046 }
1047 else { // third is finite, need to find its eucl radius
1048 len2=h_ivd_length(x0,x2,ivd2);
1049 double s=Math.exp(-len2);
1050 double edist=(1-s)/(1+s);
1051 Complex hypz=new Complex(edist); // absolute value of hyp center
1052 // now position on positive x-axis just to find eucl radius R2
1053 CircleSimple cSe=h_to_e_data(new CircleSimple(hypz,x2));
1054 R2=cSe.rad;
1055 }
1056
1057 double cosang=EuclMath.e_cos_overlap(r0,R1,R2,ivd0,ivd1,ivd2);
1058 double sinang=Math.sqrt(1.0-cosang*cosang);
1059 z2=new Complex(cosang,sinang); // center's direction
1060
1061 // Finish up
1062 if (x2>0) { // third is finite
1063 double s=Math.exp(-len2); // convert len2 to eucl distance
1064 double edist=(1-s)/(1+s);
1065 z2=z2.times(edist); // hyp center
1066 if (flag==-1)
1067 z2=z2.conj();
1068 csIn=new CircleSimple(z2,x2);
1069 }
1070 else {
1071 if (flag==-1)
1072 z2=z2.conj();
1073 csIn=new CircleSimple(z2,(-1.0)*R2);
1074 }
1075 map=Mobius.standard_mob(z0, z1);
1076 remap=(Mobius)map.inverse();
1077 Mobius.mobius_of_circle(remap, -1, csIn, csOut, true);
1078 return csOut;

1100 double x0,double x1,double x2,
1101 double ivd0,double ivd1,double ivd2) {
1102 CircleSimple sici;
1103
1104 double s0 = HyperbolicMath.x_to_s_rad(x0);
1105 double s1 = HyperbolicMath.x_to_s_rad(x1);
1106 double s2 = HyperbolicMath.x_to_s_rad(x2);

1108 // If any eucl radii are invalid, return with an error
1109 if (s0 <= (-1.0) || s1 <= (-1.0) || s2<=(-1.0))
1110 throw new DataException("error computing center: "+
1111 "horocycle eucl rad <= -1.0");

1113 Complex a = z0;
1114 Complex b = z1;
1115 double sgn = 1; // signal orientation
1116 if ((s0 <= 0) && (s1 > 0)) { // second circle finite, first not?
1117 // then interchange order
1118 a = z1;
1119 b = z0;
1120 double hold = s0;
1121 s0 = s1;
1122 s1 = hold;
1123 hold = x0;
1124 x0 = x1;
1125 x1 = hold;
1126 hold = ivd1;
1127 ivd2 = ivd1;
1128 ivd1 = hold;

1130 }
1131 if (s0 > 0) { // first is now finite
1132 double cc = h_comp_cos(x0, x1, x2, ivd0, ivd1, ivd2);
1133
1134 // if third circle is finite
1135 if (s2 > 0) {
1136 double x13 = x0 * (x2);
1137 double x1p3 = x0 + (x2);
1138 double s13 = s0 * s2;
1139 double acstuff = (x1p3 - x13) /
1140 (s13 * (1 + s13)) - (2 * x1p3 - (1 + ivd1) * x13) /
1141 (4 * s13);
1142 if (acstuff < 0.0) // error
1143 return new CircleSimple(false);
1144 double side_p1 = acstuff + Math.sqrt(acstuff * (acstuff + 2));
1145
1146 double ahc = side_p1 / (side_p1 + 2); // abs value of hyp center
1147 // center as if z0 at origin
1148 Complex z3 = new Complex(cc * ahc, sgn *
1149 Math.sqrt(1 - cc * cc) * ahc);
1150
1151 if (a.x == b.x && a.y == b.y)
1152 z3.x *= ahc;
1153
1154 z3 = Mobius.mobDiscInvValue(z3, a, b); // move to right place
1155 return new CircleSimple(z3, x2, 1);
1156 }
1157
1158 // else third circle is a horocycle
1159
1160 double r = (1 - s0) / (1 + s0);
1161 double sc = (r * r + 1 + 2 * r * ivd1) /
1162 (2 * (1 + r * ivd1)); // abs value of eucl center
1163 Complex c = new Complex();
1164 c.x = sc * cc;
1165 double cc2;
1166 if ((cc2 = cc * cc) > 1.0) // error
1167 return new CircleSimple(false);
1168 c.y = sc * sgn * Math.sqrt(1 - cc2);
1169 double rad = 1 - sc; // Now have c and its radius
1170 Complex w0 = new Complex(c.x - rad, c.y);
1171 Complex w1 = new Complex(c.x + rad, c.y);
1172 Complex w2 = new Complex(c.x, c.y + rad); // 3 points on the circle
1173 w0 = Mobius.mobDiscInvValue(w0, a, b);
1174 w1 = Mobius.mobDiscInvValue(w1, a, b);
1175 w2 = Mobius.mobDiscInvValue(w2, a, b); // 3 pts on new circle
1176 sici = EuclMath.circle_3(w0, w1, w2);
1177 sici.rad *= -1.0; // store neg of eucl radius as x_rad
1178 // get hyp center on unit circle
1179 sici.center = sici.center.times(1.0 / (sici.center.abs()));
1180 }
1181
1182 // first two horocycles, third finite (Note: generally prefer
1183 // to avoid this situation)
1184 else if (s2 > 0) {
1185 // Idea: build triple with circle 3 centered at origin,
1186 // then apply a Mobius to get its euclidean radius and
1187 // tangency points right to original values, see where
1188 // circle 3 ends up
1189 // here get eucl rad of circle 3 if centered at origin
1190 double erad = (1.0 - s2) / (1.0 + s2);
1191 // Compute eucl radii of horocycles with correct inv dist
1192 // to circle 3 (at origin)
1193 double hororad1 = (1.0 - erad * erad) / (2.0 * (1 + erad * ivd1));
1194 double hororad2 = (1.0 - erad * erad) / (2.0 * (1 + erad * ivd2));
1195
1196 // Assume circle 1 tangent at z=1, compute tangency
1197 // point of circle 2.
1198 double dist12 = EuclMath.e_ivd_length(hororad1, hororad2, ivd1);
1199 double d1 = 1.0 - hororad1;
1200 double d2 = 1.0 - hororad2;
1201 double theta = Math.acos((d1 * d1 + d2 * d2 - dist12 * dist12) /
1202 (2.0 * d1 * d2));
1203
1204 // Mobius that fixes +-1 and moves d1=1-2*hororad1 to 1+2*s1
1205 // (recalling that s1 is euclidean rad with negative sign).
1206 // Mobius is form (z-t)/(1-tz), and since d1 gets carried
1207 // to d2, can compute t. Apply it to origin to see where
1208 // hyp center of circle 3 is and to tangency point of circle 2.
1209 d1 = 1.0 - 2.0 * hororad1;
1210 d2 = 1.0 + 2.0 * s0;
1211 double t = (d2 - d1) / (d2 * d1 - 1.0);
1212 Mobius m1 = new Mobius(new Complex(1.0), new Complex(-t),
1213 new Complex(-t), new Complex(1.0));
1214 Complex newZ = m1.apply(new Complex(0.0));
1215 theta = m1.apply(new Complex(0.0, theta).exp()).arg();
1216
1217 // Next want parabolic fixing 1 and mapping tangency point
1218 // of circle 2 so it makes original angle between the
1219 // original horocycles. Use map (z+1)/(z-1). This maps
1220 // disc to left half plane, is own inverse. The parabolics
1221 // are shift in halfplane by t in imaginary direction.
1222 // Mobius is ((2+it)z-it)/(itz+(2-it)).
1223 // Under this map, exp{a} goes to -sin(a)/(1-cos(a)) time i.
1224 // So want t to equal to difference between old image and new.
1225 double origTheta = z1.divide(z0).arg();
1226 t = Math.sin(theta) / (1.0 - Math.cos(theta))- Math.sin(origTheta)/
1227 (1.0 - Math.cos(origTheta));
1228 Mobius m2 = new Mobius(new Complex(2.0, t), new Complex(0.0, -t),
1229 new Complex(0.0, t),new Complex(2.0, -t));
1230 Complex nextZ = m2.apply(newZ);
1231
1232 // Final step is to rotate to put tangency point of circle 1
1233 // where it was originally
1234 sici = new CircleSimple(true);
1235 sici.center = nextZ.times(z0); // recall z1 should be on unit circle
1236 sici.rad = x2;
1237 }
1238
1239 // remaining case: all three are horocycles. Expect -eucl
1240 // radius as s-rad of target circle.
1241 else {
1242 try {
1243 sici = old_h_horo_center(z0, z1, -s1, ivd0, ivd1, ivd2);
1244 } catch (MobException mex) {
1245 throw new MobException("error computing horocycle center");
1246 }
1247 }
1248
1249 return sici;
1250 }
