301 public void editAction(String command) {
302 if (command.equals("edit")) {
303 startEdit();
304 }
305 else if (command.startsWith("add_above") || command.startsWith("add_below")) {
306 if (this instanceof DataSBox || this instanceof FileSBox) {
307 // can only add a file below: give user chooser dialog
308 manager.insertNewTN(tNode,CPTreeNode.FILE,1);
309 return;
310 }
311 else if (this instanceof ScriptSBox && command.startsWith("add_above")) return;
312
313 int abbel=1; // below (downstream)
314 if (command.startsWith("add_above")) abbel=0; // upstream (i.e., insert at current spot)
315 // type is: 0=COMMAND, 1=TEXT,2=SECTION
316 int tNtype=CPTreeNode.COMMAND;
317 if (command.endsWith("text")) tNtype=CPTreeNode.TEXT;
318 if (command.endsWith("section")) tNtype=CPTreeNode.SECTION;
319 manager.insertNewTN(tNode,tNtype,abbel);
320 return;
321 }
322 else if (command.equals("Next script cmd")) {
323 if (tNode.tntype!=CPTreeNode.COMMAND &&
324 tNode.tntype!=CPTreeNode.EDIT_CMDorMODE &&
325 tNode.tntype!=CPTreeNode.SECTION &&
326 tNode.tntype!=CPTreeNode.EDIT_SECTION &&
327 tNode.tntype!=CPTreeNode.TEXT &&
328 tNode.tntype!=CPTreeNode.EDIT_TEXT)
329 return;
330 CPTreeNode pred=(CPTreeNode)tNode.getParent();
331 int indx;
332 if ((indx=pred.getIndex(tNode))<0) return;
333 if (indx>0)
334 pred=(CPTreeNode)pred.getChild(indx-1);
335
336 // is there a previous next to redisplay?
337 if (manager.nextCmdNode!=null) {
338 StackBox oldsb=manager.nextCmdNode.stackBox;
339 manager.nextCmdNode=null;
340 if (oldsb instanceof CmdSBox)
341 oldsb.redisplaySB(oldsb.myWidth);
342 }
343 manager.resetNextCmdNode(pred);
344 }

459 public void actionPerformed(ActionEvent e){
460 String command = e.getActionCommand();
461
462 if (command.equals("edit")) {
463 startEdit();
464 }
465 else if (command.equals("cancel_edit"))
466 cancelEdit();
467 else if (command.equals("export_file")) {
468 if (this instanceof FileSBox) {
469 File dataOutFile=null;
470 FileSBox fsbox=(FileSBox)this;
471
472 // run a file chooser
473 JFileChooser dbox = new JFileChooser();
474 dbox.setCurrentDirectory(CPFileManager.PackingDirectory);
475 dbox.setSelectedFile(new File(CPFileManager.PackingDirectory+File.separator+fsbox.includedFile.origName));
476 dbox.setFileFilter(new DataFileFilter());
477 int result = dbox.showDialog(
478 (Component) PackControl.activeFrame,"Export data file");
479 if (result == JFileChooser.APPROVE_OPTION) {
480 dataOutFile = dbox.getSelectedFile();
481 CPFileManager.PackingDirectory = dbox.getCurrentDirectory(); // change directory
482 }
483 try {
484 CPFileManager.copyFile(fsbox.includedFile.tmpFile,dataOutFile);
485 } catch (Exception ex) {
486 CirclePack.cpb.errMsg("Failed in copying "+fsbox.includedFile.tmpFile.getName());
487 return;
488 }
489 CirclePack.cpb.msg("Exported data file to "+dataOutFile.getPath());
490 }
491 else return;
492 }
493 else if (command.equals("delete_node")) {
494 if (this instanceof SectionSBox) {
495 int result=queryDeletion();
496 if (result==0) // cancel deletion
497 return;
498
499 //AF>>>//
500 // Lock the viewport so GUI changes don't jitter and relocate.
501 ((LockableJViewport) PackControl.scriptHover.stackScroll.getViewport()).setLocked(true);
502 //<<<AF//
503
504 CPTreeNode parTN=(CPTreeNode)tNode.getParent();
505 if (result==-1) { // remove everything
506 deleteChildNodes();
507 if (parTN!=null) parTN.remove((MutableTreeNode)tNode);
508 }
509 else { // save children as siblings under parent
510 if (parTN==null) {
511 //AF>>>//
512 // Unlock the viewport.
513 EventQueue.invokeLater(new Runnable() {
514 public void run() {
515 ((LockableJViewport) PackControl.scriptHover.stackScroll.getViewport()).setLocked(false);
516 }
517 });
518 //<<<AF//
519 return;
520 }
521 int index=parTN.getIndex(tNode);
522 // Note: child vec depleted by one each step
523 int ccount=tNode.getChildCount();
524 for (int i=0;i<ccount;i++)
525 parTN.insert((CPTreeNode)tNode.getChild(0),index+i+1);
526 parTN.remove(index);
527 }
528 if (!manager.hasChanged) {
529 manager.hasChanged=true;
530 PackControl.scriptHover.scriptTitle(manager.scriptName,true);
531 }
532 manager.repopulateRecurse(parTN);
533
534 //AF>>>//
535 // Unlock the viewport.
536 EventQueue.invokeLater(new Runnable() {
537 public void run() {
538 ((LockableJViewport) PackControl.scriptHover.stackScroll.getViewport()).setLocked(false);
539 }
540 });
541 //<<<AF//
542 }
543 else deleteNode();
544 }
545 else if (command.equals("accept_edit"))
546 acceptEdit();
547 else if (command.equals("open_node"))
548 open();
549 else if (command.equals("close_node"))
550 close();
551
552 // for CmdSBox's, this executes the command as currently edited.
553 else if (command.equals("trial_exec") && currentMode!=DISPLAY) {
554 CmdSBox csb=(CmdSBox)(this.tNode.stackBox);
555 String str=csb.getCmdText();
556 CPBase.trafficCenter.parseWrapper(str,CirclePack.cpb.getActivePackData(),false,true,0,null);
557 }
558 else editAction(command);
559 }
