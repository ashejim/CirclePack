543 public int pasteNext() {
544 boolean debug=false; // true
545 PasteCode pc=nextPasteCode();
546
547 // no remaining 'PasteCode'? ERROR or BUILT?
548 if (pc==null) {
549 if (domainPack.getBdryCompCount()>0) {
550 msg("RM: all pastings done, but the boundary is non empty");
551 rmState=ERROR;
552 return -1;
553 }
554 msg("We seem to have done all the pastings");
555 rmState=BUILT;
556 domainPack.status=true;
557 domainPack.hes=1;
558 domainPack.set_aim_default();
559 domainPack.vertexMap=masterMap.makeCopy();
560 return 1;
561 }
562
563 // debug
564 if (debug) {
565 displayES();
566 msg("Try next pasteCode: <ss,sc,ts,tc>: "+
567 pc.sourceSheet+" "+pc.sourceEdge+
568 " "+pc.targetSheet+" "+pc.targetEdge);
569 }
570
571 // both sheets already attached to base? 'edgeSeg's
572 // should be in 'masterESlist'
573 if (sheetCheck[pc.sourceSheet]<0 && sheetCheck[pc.targetSheet]<0) {
574 EdgeSeg sES=findMasterES(pc.sourceSheet,pc.sourceEdge);
575 EdgeSeg tES=findMasterES(pc.targetSheet,pc.targetEdge);
576 if (pasteEm(sES,tES)>0) {
577 pasteCount++;
578 masterMap=VertexMap.followedBy(domainPack.vertexMap,
579 masterMap);
580 domainPack.vertexMap=masterMap.makeCopy();
581 if (debug)
582 msg("success (target sheet already attached)");
583 }
584 else if (debug) {
585 msg("failed with this pasteCode");
586 return -1;
587 }
588 }
589
590 // one sheet is attached to base, one isn't
591 else if (sheetCheck[pc.sourceSheet]*sheetCheck[pc.targetSheet]<0) {
592 EdgeSeg aES; // the one attached
593 int toSlit; // the slit number for the unattached sheet
594 int newSheetNum; // the number of this new sheet
595 if (sheetCheck[pc.sourceSheet]<0 &&
596 sheetCheck[pc.targetSheet]>0) {
597 aES=findMasterES(pc.sourceSheet,pc.sourceEdge);
598 toSlit=pc.targetEdge;
599 newSheetNum=pc.targetSheet;
600 }
601 else {
602 aES=findMasterES(pc.targetSheet,pc.targetEdge);
603 toSlit=pc.sourceEdge;
604 newSheetNum=pc.sourceSheet;
605 }
606
607 // do the attachment
608 if (attachIt(aES,newSheetNum,toSlit)>0) {
609 pasteCount++;
610 sheetCheck[newSheetNum]=-1;
611 if (debug)
612 msg("New sheet "+newSheetNum+
613 " attached via slit "+toSlit);
614 }
615 else if (debug) {
616 msg("Failed to attach new sheet "+newSheetNum);
617 return -1;
618 }
619 }
620 if (pasteCodes.indexOf((Object)pc)>=0) {
621// if (debug) msg("remove this pasteCode.");
622 pasteCodes.remove(pc);
623 }
624
625 // are we done?
626 if (domainPack.getBdryCompCount()==0) {
627 rmState=BUILT;
628 domainPack.status=true;
629 domainPack.hes=1;
630 domainPack.set_aim_default();
631 domainPack.vertexMap=masterMap.makeCopy();
632 }
633 return 1;
634 }

640 public int pasteEm(EdgeSeg es1,EdgeSeg es2) {
641 if (es1==null || es2==null) return 0;
642 int length=1;
643 int v=es1.endV;
644 int next=domainPack.getFirstPetal(v);
645 int safety=domainPack.nodeCount;
646 while (next!=es1.startV && next!=es1.endV && safety>0) {
647 v=next;
648 next=domainPack.getFirstPetal(v);
649 length++;
650 safety--;
651 }
652 if (safety<=0 || next==es1.endV)
653 return 0;
654 domainPack.packDCEL=CombDCEL.adjoin(domainPack.packDCEL,
655 domainPack.packDCEL, es1.startV,es2.endV,length);
656 domainPack.vertexMap=domainPack.packDCEL.oldNew;
657 domainPack.packDCEL.fixDCEL(domainPack);
658
659 // remove these 'EdgeSeg's from 'masterESlist'
660 masterESlist.remove((EdgeSeg)es1);
661 masterESlist.remove((EdgeSeg)es2);
662
663 // adjust numbering of remaining EdgeSeg's
664 VertexMap vM=domainPack.vertexMap;
665 Iterator<EdgeSeg> ml=masterESlist.iterator();
666 EdgeSeg es=null;
667 while (ml.hasNext()) {
668 es=(EdgeSeg)ml.next();
669 if (!es.convertIndices(vM))
670 throw new ParserException("Missing vertexMap entry");
671 }
672
673 // fix 'masterMap': should <dp,rp> pairs, dp in
674 // domainPack to rp in rangePack
675 VertexMap tmpVM=new VertexMap();
676 for (int i=1;i<=domainPack.nodeCount;i++) {
677 int oi=domainPack.vertexMap.findV(i); // find former index
678 tmpVM.add(new EdgeSimple(oi,masterMap.findW(oi)));
679 }
680 masterMap=tmpVM;
681
682 return length;
683 }

694 public int attachIt(EdgeSeg es,int tSheet,int tSlit) {
695 if (es==null)
696 return 0;
697
698 // 'domainPack' is the base packing we attach to.
699 int holdCount=domainPack.nodeCount;
700
701 // check validity, length of this edge segment
702 int length=es.validate(domainPack);
703
704 // need pointer 'EdgeSeg' in 'slits' linked list
705 Iterator<EdgeSeg> esl=slits.iterator();
706 EdgeSeg tES=null;
707 while (tES==null && esl.hasNext()) {
708 EdgeSeg ees=(EdgeSeg)esl.next();
709 if (ees.slitNumber==tSlit)
710 tES=ees;
711 }
712 if (tES==null) {
713 throw new CombException("Error: given slit not located");
714 }
715 int len=tES.validate(slitPack);
716 if (len==0 || len!=length) { // not matching lengths
717 errorMsg("'edgeSeg' lengths didn't match");
718 return 0;
719 }
720
721 // do the 'adjoin'
722 domainPack.packDCEL=CombDCEL.adjoin(domainPack.packDCEL,
723 slitPack.packDCEL, es.startV,tES.endV,length);
724 VertexMap holdvm=domainPack.packDCEL.oldNew;
725 domainPack.packDCEL.fixDCEL(domainPack);
726 domainPack.vertexMap=holdvm;
727
728 // must remove the 'EdgeSeg' from 'masterESlist'
729 masterESlist.remove(es);
730
731 // clone 'slits', shift indices, remove tES, add rest to
732 // 'masterESlist'; hold conversion from 'slitPack'
733 Iterator<EdgeSeg> sls=slits.iterator();
734 VertexMap vM=domainPack.vertexMap;
735 while (sls.hasNext()) {
736 EdgeSeg ees=(EdgeSeg)sls.next();
737 if (ees.slitNumber!=tSlit) {
738 ees=ees.clone(tSheet);
739 ees.convertIndices(vM);
740 masterESlist.add(ees);
741 }
742 }
743
744 // augment 'masterMap' for new vertices
745 Iterator<EdgeSimple> doml=domainPack.vertexMap.iterator();
746 while (doml.hasNext()) {
747 EdgeSimple edge=(EdgeSimple)doml.next();
748 if (edge.w>holdCount) { // this is a new vertex
749 int w=slitMap.findW(edge.v); // w=image vert in 'rangePack'
750 masterMap.add(new EdgeSimple(edge.w,w));
751 }
752 }
753
754 // set 'mark' of 'domainPack' to sheet number
755 for (int v=(holdCount+1);v<=domainPack.nodeCount;v++) {
756 domainPack.setVertMark(v,tSheet);
757 }
758 return 1;
759 }

780 public int readRMData(BufferedReader fp) {
781 try {
782 String line=null;
783 String nexttok=null;
784 StringTokenizer tok;
785 line=StringUtil.ourNextLine(fp);
786 while (line!=null) {
787 tok = new StringTokenizer(line);
788 nexttok=tok.nextToken();
789 if (nexttok.equals("SLITCOUNT:")) {
790
791 // get number of slits
792 numSlits=Integer.parseInt(tok.nextToken());
793
794 // get that many pairs of complex slits endpts
795 // (delineated by linebreaks)
796 slitLinks=new ArrayList<HalfLink>();
797 // keep list of endpoints of slits
798 branchList=new NodeLink(rangePack);
799 line=StringUtil.ourNextLine(fp);
800 int tick=0;
801
802 // with 'nonos' we try to insure each new geo
803 // misses previous
804 NodeLink nonos=new NodeLink(rangePack);
805 while (tick<numSlits && line!=null) {
806 // line of form "v1 v2 ... vn" or "x1 y1 x2 y2"
807 // check if first entry is integer.
808 boolean zform=false;
809 try {
810 String first=StringUtil.grabNext(line);
811 int dumy=Integer.parseInt(first);
812 if (dumy<1 || dumy>rangePack.nodeCount)
813 zform=true;
814 } catch(NumberFormatException nfe) {
815 zform=true;
816 }
817
818 NodeLink nlk=null;
819 try {
820 if (zform) { // create geodesic based on endpoints
821 String []splits=line.split("\\s+"); // whitespace
822 Complex z1=new Complex(Double.parseDouble(splits[0]),
823 Double.parseDouble(splits[1]));
824 Complex z2=new Complex(Double.parseDouble(splits[2]),
825 Double.parseDouble(splits[3]));
826 int v1=0;
827 int v2=0;
828 try {
829 v1=rangePack.cir_closest(z1,false).get(0);
830 v2=rangePack.cir_closest(z2,false).get(0);
831 } catch (Exception ex) {
832 throw new DataException(
833 "Didn't find closest circle to an endpoint");
834 }
835
836 // get geodesic halflinks (may throw exception)
837 HalfLink elk=HalfLink.getCombGeo(rangePack.packDCEL,
838 new NodeLink(rangePack,v1),
839 new NodeLink(rangePack,v2),nonos);
840
841 // convert to vertlist
842 nlk=new NodeLink(rangePack,
843 elk.get(0).origin.vertIndx);
844 Iterator<HalfEdge> hlst=elk.iterator();
845 while (hlst.hasNext()) {

939 // leverage this.

942 if (downV>rangePack.nodeCount) { // 'downV' is new
943 oppV=pdcel.vertices[slitPack.vertexMap.findV(downV)];
944 }
945 HalfEdge he=oppV.halfedge;
946 endV=he.twin.origin.vertIndx;
947

949 for (int j=0;j<length;j++)
950 he=he.twin.next.twin;
951 startV=he.twin.origin.vertIndx;
952 edgeSeg=new EdgeSeg(++tick,startV,endV);
953 edgeSeg.setLength(length);
954 slits.add(edgeSeg);
955 } // end of while
