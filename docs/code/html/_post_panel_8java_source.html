358 public String formPostFlags() {
359 StringBuilder flags=new StringBuilder("");
360 StringBuilder frag=null;
361 if (((Boolean)(checks.get("unitcir"))).booleanValue())
362 flags.append(" -u");
363 if (((Boolean)(checks.get("circles"))).booleanValue()) {
364 frag=new StringBuilder("");
365 if (((Boolean)(checks.get("circolor"))).booleanValue())
366 frag.append("c");
367 if (((Boolean)(checks.get("cirfill"))).booleanValue())
368 frag.append("f");
369 flags.append(" -c"+frag.toString()+" ");
370 }
371 if (((Boolean)(checks.get("faces"))).booleanValue()) {
372 frag=new StringBuilder("");
373 if (((Boolean)(checks.get("facecolor"))).booleanValue())
374 frag.append("c");
375 if (((Boolean)(checks.get("facefill"))).booleanValue())
376 frag.append("f");
377 flags.append(" -f"+frag.toString()+" ");
378 }
379 frag=new StringBuilder("");
380 if (((Boolean)(checks.get("cirlabels"))).booleanValue())
381 frag.append(" -cn");
382 if (((Boolean)(checks.get("facelabels"))).booleanValue())
383 frag.append(" -fn");
384 if (((Boolean)(checks.get("path"))).booleanValue())
385 flags.append(" -g");
386 if (flags.length()==0) return null; // didn't get any flags
387
388 return (flags.toString());
389 }

420 public void actionPerformed(ActionEvent e){
421 String command = e.getActionCommand();
422 File aboutFile=null;
423
424 if (command.equals("canvas2JPG")) {
425 String []jpgFileName=jpgOutputDialog();
426 if (jpgFileName==null) return;
427 try {
428 File file = new File(CPFileManager.ImageDirectory,jpgFileName[1]);
429 ImageIO.write(PackControl.getActiveCPDrawing().packImage, "jpg", file);
430 } catch (Exception exc) {
431 return;
432 }
433 return;
434 }
435 if (command.equals("aboutJPG") &&
436 (aboutFile=PackControl.scriptManager.getAboutTmpFile())!=null) {
437 String []jpgFileName=aboutOutputDialog();
438 if (jpgFileName==null)
439 return;
440 File file=null;
441 try {
442 file = new File(CPFileManager.ImageDirectory,jpgFileName[1]);
443 CPFileManager.copyFile(aboutFile, file);
444 } catch (Exception exc) {
445 CirclePack.cpb.errMsg("problem copying 'AboutImage' to "+file);
446 }
447 return;
448 }
449 if (command.equals("GO")
450 || (command.equals("useFlags") && flagField.getText().trim().length()>0)) {
451 String []psFileInfo=new String[2];
452 psFileInfo[1]=new String("");
453 // some commands need us to open file
454 if (((Boolean)(checks.get("save"))).booleanValue()
455 || ((Boolean)(checks.get("append"))).booleanValue()
456 || ((Boolean)(checks.get("print"))).booleanValue()) {
457 psFileInfo=postOutputDialog();
458 if (psFileInfo==null) return;
459 }
460
461 // psFileInfo[1] should have just the file name; 'PostDirectory' should have
462 // been set in chooser call.
463 buildCmd=new StringBuilder(createPrefix(psFileInfo[1]));
464 String flags;
465 if (command.equals("useFlags")) flags=flagField.getText();
466 else flags=formPostFlags();
467 if (flags.length()<1) return;
468 buildCmd.append(" "+flags+" "+createSuffix());
469 try {
470 TrafficCenter.cmdGUI(buildCmd.toString());
471 } catch (Exception ex) {return;}
472 return;
473 }
474 Object obj=e.getSource();
475 if (obj instanceof JCheckBox) {
476 JCheckBox jbox=(JCheckBox)e.getSource();
477 checks.put(jbox.getActionCommand(),Boolean.valueOf(jbox.isSelected()));
478 }
479 }
