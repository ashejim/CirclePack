73 public int addBaryLinks(Vector<String>items) {
74 if (items==null || items.size()==0)
75 return 0;
76
77 int count=0;
78
79 String str=items.get(0);
80
81 // check for flag character; only first entry can be flag
82
83 // if there is an accidental '-', remove it
84 if (StringUtil.isFlag(str)) {
85 str=str.substring(1);
86 }
87
88 // check for Blink or blist
89 if (str.substring(1).startsWith("list")) {
90 BaryLink blink=null;
91
92 if ((str.contains("Bli") &&
93 (blink=CPBase.Blink)!=null && blink.size()>0) ||
94 (str.contains("Bli") &&
95 (blink=CPBase.Blink)!=null && blink.size()>0)) {
96
97 String[] b_string;
98 String brst;
99
100 String strdata=str.substring(5).trim(); // remove '?list'
101
102 // check for parens listing range of indices
103 int lsize=blink.size()-1;
104 int[] irange=StringUtil.get_int_range(strdata, 0,lsize);
105 if (irange!=null) {
106 int a=irange[0];
107 int b=(irange[1]>lsize) ? lsize : irange[1];
108 for (int j=a;j<=b;j++) {
109 add(blink.get(j));
110 count++;
111 }
112 }
113 // else check for brackets
114 else if ((b_string=StringUtil.get_bracket_strings(strdata))!=null
115 && (brst=b_string[0])!=null) {
116 if (brst.startsWith("r")) { // rotate: copy first at end
117 blink.add(blink.getFirst());
118 }
119 if (brst.startsWith("r")
120 || brst.startsWith("n")) { // use an remove first
121 add(blink.removeFirst());
122 count++;
123 }
124 if (brst.startsWith("l")) { // last
125 add(blink.getLast());
126 count++;
127 }
128 else { // else specified index
129 try{
130 int n=MathUtil.MyInteger(brst);
131 if (n>=0 && n<blink.size()) {
132 add(blink.get(n));
133 count++;
134 }
135 } catch (NumberFormatException nfe) {}
136 }
137 }
138 else {
139 abutMore(CPBase.Blink);
140 return CPBase.Blink.size();
141 }
142 }
143 else // there was no appropriate list
144 return count;
145 }
146
147 // If there is a flag, we get it, process, and return.
148 if (StringUtil.isFlag(str) || (str.charAt(0)!='-' &&
149 !Character.isDigit(str.charAt(0)) && str.charAt(0)!='.')) {
150 items.remove(0); // toss, but still held in str
151 if (items==null || items.size()==0)
152 return 0;
153 char c=str.charAt(0);
154 switch(c) {
155 case 'f': // faces
156 {
157 if (packData==null) {
158 throw new ParserException(
159 "there is no PackData for this BaryLink");
160 }
161
162 // get x y points from rest of strings
163 PointLink ptlink=new PointLink(items);
164 Iterator<Complex> pts=ptlink.iterator();
165
166 // create 'BaryPoint's for any triangles containing x,y.
167 while (pts.hasNext()) {
168 Complex z=new Complex(pts.next());
169 FaceLink faces=packData.tri_search(z);
170 Iterator<Integer> fcs=faces.iterator();
171 while (fcs.hasNext()) {
172 int f=fcs.next();
173 int[] verts=packData.packDCEL.faces[f].getVerts();
174 BaryPoint bp=null;
175 if (packData.hes==0) { // eucl
176 bp=EuclMath.e_pt_to_bary(z,
177 packData.getCenter(verts[0]),
178 packData.getCenter(verts[1]),
179 packData.getCenter(verts[2]));
180 }
181 else if (packData.hes<0) { // hyp
182 bp=HyperbolicMath.h_pt_to_bary(z,
183 packData.getCenter(verts[0]),
184 packData.getCenter(verts[1]),
185 packData.getCenter(verts[2]));
186 }
187 if (bp!=null) {
188 bp.face=f;
189 if (add(bp))
190 count++;
191 }
192 }
193 }
194 break;
195 }
196 case 'i': // interstice
197 {
198 if (packData==null) {
199 throw new ParserException(
200 "there is no PackData for this BaryLink");
201 }
202 if (packData.hes>0) {
203 throw new ParserException(
204 "spherical interstice barycentric coods not yet computed");
205 }
206
207 // get x y points from rest of strings
208 PointLink ptlink=new PointLink(items);
209 Iterator<Complex> pts=ptlink.iterator();
210
211 // find any triangles containing x,y.
212 while (pts.hasNext()) {
213 Complex z=new Complex(pts.next());
214 FaceLink faces=packData.tri_search(z);
215 Iterator<Integer> fcs=faces.iterator();
216 while (fcs.hasNext()) {
217 int f=fcs.next();
218 combinatorics.komplex.DcelFace face=packData.packDCEL.faces[f];
219 Complex []m=new Complex[3];
220
221 // find tangency points (m[j] opposite to w[j])
222 m[2]=packData.tangencyPoint(face.edge);
223 m[0]=packData.tangencyPoint(face.edge.next);
224 m[1]=packData.tangencyPoint(face.edge.next.next);
225
226 // eucl circle through them: map to the unit disc
227 CircleSimple incir=EuclMath.circle_3(m[0],m[1],m[2]);
228 Complex []nz=new Complex[3];
229 for (int j=0;j<3;j++) {
230 nz[j]=m[j].minus(incir.center);
231 nz[j]=nz[j].divide(nz[j].abs());
232 }
233 Complex Z=z.minus(incir.center).divide(incir.rad);
234
235 // now find angle-type barycentric coords for z
236 // relative to ideal triangle
237 try {
238 BaryPoint bpt=HyperbolicMath.ideal_bary(Z,
239 nz[0],nz[1], nz[2]);
240 bpt.face=f;
241
242 // check that pt is in the closed interstice
243 if (bpt.b0>=0 && bpt.b1>=0 && bpt.b2>=0) {
244 add(bpt);
245 count++;
246 }
247 } catch (Exception ex) {}
248 } // end of while for faces
249 } // end of while for points
250
251 break;
252 }
253 } // end of switch
254
255 return count;
256 }
257
258 // else, raw data of form "f b1 b2"
259 while (items.size()>2) {
260 try {
261 int f=Integer.parseInt(items.remove(0));
262 double c1=Double.parseDouble(items.remove(0));
263 double c2=Double.parseDouble(items.remove(0));
264 BaryPoint bp=new BaryPoint(c1,c2);
265 if (f>0)
266 bp.face=f;
267 if (add(bp))
268 count++;
269 } catch (Exception ex) {
270 throw new ParserException("BaryPoint input usage: f c1 c2");
271 }
272 }
273
274 return count;
275 }
