748 public int drawTileRecurs(SubdivisionRules sRules,
749 int tiletype,Complex []base,int depth,DispFlags dflags) {
750 TileRule topRule=sRules.tileRules.get(tiletype-4);
751
752 // for transformations
753 Complex origin=base[0];
754 Complex basedir=base[1].minus(base[0]);
755
756 // first draw yourself, then position, recursively draw children
757 double []stdC=new double[2*topRule.stdCorners.length];
758 for (int j=0;j<topRule.stdCorners.length;j++) {
759 Complex z=new Complex(topRule.stdCorners[j].
760 times(basedir).add(origin));
761 stdC[2*j]=z.x;
762 stdC[2*j+1]=z.y;
763 }
764 try {
765 drawClosedPoly(topRule.stdCorners.length,stdC,dflags);
766 } catch(Exception ex) {
767 throw new InOutException("failed in drawing polygon.");
768 }
769 int count=1;
770
771 // recurse through children
772 if (depth > 0) {
773 for (int n = 1; n <= topRule.childCount; n++) {
774 Complex []subtileBase=new Complex[2];
775 subtileBase[0] = new Complex(topRule.tileBase[n][0].
776 times(basedir).add(origin));
777 subtileBase[1] = new Complex(topRule.tileBase[n][1].
778 times(basedir).add(origin));
779 int rslt = drawTileRecurs(sRules, topRule.childType[n],
780 subtileBase, depth - 1,dflags);
781 if (rslt <= 0)
782 return 0;
783 count += rslt;
784 }
785 }
786 return count;
787 }
