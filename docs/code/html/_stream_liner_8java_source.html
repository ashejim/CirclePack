54 public BaryCoordLink getStreamline(Complex pt, boolean uphill) {
55 BaryCoordLink outLink=new BaryCoordLink();
56
57 // what face are we in?
58 FaceLink flk=basePack.tri_search(pt);
59 if (flk==null || flk.size()==0)
60 return null;
61 combinatorics.komplex.DcelFace face=basePack.packDCEL.faces[flk.get(0)];
62 int[] vert=face.getVerts();
63 Complex v0=basePack.getCenter(vert[0]);
64 Complex v1=basePack.getCenter(vert[1]);
65 Complex v2=basePack.getCenter(vert[2]);
66
67 // find the first 'BaryPoint'
68 BaryPoint startBpt=BaryPoint.complex2bp(basePack.hes, pt, v0, v1, v2);
69 startBpt.face=face.faceIndx;
70
71 // chase along
72 int safety=1000;
73 while (startBpt!=null && safety>0) {
74 safety--;
75
76 // decide which way to go from 'startBpt'
77 startBpt=whichWay(startBpt,uphill);
78
79 // did we reach an error or a stopping point?
80 if (startBpt==null) {
81 if (outLink!=null && outLink.size()>0)
82 return outLink;
83 else return null;
84 }
85 BaryPacket newBP=faceTransit(startBpt,uphill);
86
87 if (newBP==null) {
88 if (outLink!=null && outLink.size()>0)
89 return outLink;
90 return null;
91 }
92
93 outLink.add(newBP);
94 startBpt=newBP.end.clone();
95 } // end of while
96
97 if (outLink!=null && outLink.size()>0)
98 return outLink;
99 return null;
100 }

111 public BaryPacket faceTransit(BaryPoint spt,boolean uphill) {
112 if (spt==null || spt.face<1)
113 return null;
114
115 // grad
116 Complex grad=new Complex(-normals[spt.face].x,-normals[spt.face].y);
117 if (!uphill)
118 grad=grad.times(-1.0);
119 combinatorics.komplex.DcelFace face=basePack.packDCEL.faces[spt.face];
120 int[] vert=face.getVerts();
121
122 Complex vec=BaryPoint.vec2simplex(grad,basePack.getCenter(vert[0]),
123 basePack.getCenter(vert[1]),basePack.getCenter(vert[2]));
124 double []baryc=BaryPoint.upGrad(spt.b0,spt.b1,vec);
125 if (baryc==null)
126 return null;
127
128 BaryPoint exitpt=new BaryPoint(baryc[0],baryc[1]);
129 exitpt.face=spt.face;
130
131 BaryPacket outPacket= new BaryPacket(vert[0], vert[1], vert[2]);
132 outPacket.faceIndx = spt.face;
133 outPacket.start = spt.clone();
134 outPacket.start.face=spt.face;
135 outPacket.end = exitpt;
136 return outPacket;
137 }

209 public BaryPoint whichWay(BaryPoint inpt,boolean uphill) {
210 if (inpt.face<=0)
211 return null;
212
213 int ins=BaryPoint.baryPtInside(inpt);
214
215 // outside? error
216 if (ins<0)
217 return null;
218
219 // interior? use same face
220 if (ins==0) {
221 BaryPoint outpt=new BaryPoint(inpt);
222 outpt.face=inpt.face;
223 return outpt;
224 }
225
226 // at vertex?
227 if (ins<4) {
228 int v=basePack.packDCEL.faces[inpt.face].getVerts()[ins-1];
229 // if boundary vertex, stop
230 if (basePack.isBdry(v))
231 return null;
232
233 int num=basePack.countFaces(v);
234 int[] flower=basePack.getFlower(v);
235 Complex []edgevec=new Complex[num+1];
236 for (int j=0;j<num;j++) {
237 edgevec[j]=basePack.getCenter(flower[j]).minus(basePack.getCenter(v));
238 }
239 edgevec[num]=edgevec[0];
240
241 // what is the largest gradient that goes into its face?
242 int bestindx=-1;
243 double bestgrad=-1.0;
244 int[] faceFlower=basePack.getFaceFlower(v);
245 for (int j=0;j<num;j++) {
246 Complex grad=new Complex(-normals[faceFlower[j]].x,-normals[faceFlower[j]].y);
247 if (!uphill)
248 grad=grad.times(-1.0);
249 if (gradFaceCheck(grad,edgevec[j],edgevec[j+1])) {
250 if (grad.abs()>bestgrad) {
251 bestgrad=grad.abs();
252 bestindx=j;
253 }
254 }
255 }
256
257 // if there's none, then look for best edge
258 if (bestindx<0 || bestgrad<=0.0) {
259 bestgrad=-1;
260 bestindx=-1;
261 for (int j=0;j<num;j++) {
262 double jump=dataValues[flower[j]]-dataValues[v];
263 if (!uphill)
264 jump *= -1.0;
265 jump /= edgevec[j].abs();
266 if (jump>bestgrad) {
267 bestgrad=jump;
268 bestindx=j;
269 }
270 }
271 }
272
273 // did we find some direction to go?
274 if (bestgrad>0.0 && bestindx>=0) {
275 int bestface=faceFlower[bestindx];
276 int vindx=basePack.packDCEL.faces[bestface].getVertIndx(v);
277 double b1=0.0;
278 double b2=0.0;
279 if (vindx==0)
280 b1=1.0;
281 else if (vindx==1)
282 b2=1.0;
283 BaryPoint outpt=new BaryPoint(b1,b2);
284 outpt.face=bestface;
285 return outpt;
286 }
287
288 // vert must be extremum
289 return null;
290 } // end of vertex
291
292 // on edge?
293 if (ins>10) {
294 combinatorics.komplex.DcelFace face=basePack.packDCEL.faces[inpt.face];
295 int []vert=face.getVerts();
296 int v=-1;
297 int w=-1;
298 int v_indx; // which index in 'vert'?
299 if (ins==12) {
300 v_indx=0;
301 v=vert[0];
302 w=vert[1];
303 }
304 else if (ins==13) {
305 v_indx=2;
306 v=vert[2];
307 w=vert[0];
308 }
309 else {
310 v_indx=1;
311 v=vert[1];
312 w=vert[2];
313 }
314
315 // stop if this is boundary edge
316 if (basePack.isBdry(v) && basePack.isBdry(w))
317 return null;
318
319 // inpt.face is on the left of vw_edge
320 int face_l=inpt.face;
321 HalfEdge vwhe=basePack.packDCEL.findHalfEdge(new EdgeSimple(v,w));
322 int face_r=vwhe.twin.face.faceIndx;
323 Complex vw_edge=basePack.getCenter(w).minus(basePack.getCenter(v));
324
325 // have to decide if going left (this same face) or going right (neighboring
326 // face) are viable, then compare them.
327 Complex grad_l=new Complex(-normals[face_l].x,-normals[face_l].y);
328 Complex grad_r=new Complex(-normals[face_r].x,-normals[face_r].y);
329 if (!uphill) {
330 grad_l=grad_l.times(-1.0);
331 grad_r=grad_r.times(-1.0);
332 }
333 boolean ck_left=true;
334 boolean ck_right=true;
335
336 // do the gradients point into the faces?
337 double arg_l=grad_l.divide(vw_edge).arg();
338 double arg_r=grad_r.divide(vw_edge).arg();
339 if ((arg_l<0.0 && arg_l>-Math.PI) || grad_l.abs()<PackData.TOLER)
340 ck_left=false;
341 if (arg_r>=0 || grad_r.abs()<PackData.TOLER)
342 ck_right=false;
343
344 // neither face? perhaps follow edge, so return current face
345 if (!ck_left && !ck_right) {
346 return inpt.clone();
347 }
348
349 if (!ck_right) {
350 BaryPoint outpt=new BaryPoint(inpt);
351 outpt.face=face_l;
352 return outpt;
353 }
354
355 // right is our choice; must adjust face
356 vert=basePack.packDCEL.faces[face_r].getVerts();
357 int k=-1;
358 for (int j=0;(j<3 && k<0);j++) {
359 if (vert[j]==w)
360 k=j;
361 }
362 if (k<0)
363 return null;
364
365 // have to find bary coords from face_l
366 double bv=0.0;
367 double bw=0.0;
368 if (v_indx==0) {
369 bv=inpt.b0;
370 bw=inpt.b1;
371 }
372 else if (v_indx==1) {
373 bv=inpt.b1;
374 bw=inpt.b2;
375 }
376 else {
377 bv=inpt.b2;
378 bw=inpt.b0;
379 }
380
381 // set them w.r.t. face_r
382 BaryPoint outpt=null;
383 if (k==0) {
384 outpt=new BaryPoint(bw,bv);
385 }
386 else if (k==1) {
387 outpt=new BaryPoint(0,bw);
388 }
389 else {
390 outpt=new BaryPoint(bv,0);
391 }
392 outpt.face=face_r;
393
394 return outpt;
395 } // done with edge
396
397 // shouldn't reach here
398 return null;
399 }
