169 scanner.close();

173 // get the title code

178 // find line nums, first/last lines of data targets.

184 while(scanner.hasNext()) {
185 String line=scanner.nextLine();
186 line_num++;

194 about_b=line_num;

196
197 // look for description first

199 if (des_a>0 && des_b>=des_a) {
200 try {
201 scanner.close();
202 scanner=new Scanner(file);
203 } catch (Exception iox) {
204 System.err.println("Failed to open cps file "+file.getPath()+" for precessing or to start a file 'scanner'");
205 return false;
206 }
207 int tick=1;
208 while (tick<des_a) {
209 scanner.nextLine();
210 tick++;
211 }
212 tick++;
213 String desline=scanner.nextLine(); // starts here
214 int k1=desline.indexOf('>')+1;
215
216 // description on single line
217 if (des_a==des_b) {
218 if (k1>0) {
219 int k2=desline.indexOf('<',k1);
220 String dtion=desline.substring(k1,k2).trim();
221 if (dtion.length()>0)
222 description.append(dtion);
223 }
224 }
225 else {
226 description.append(desline.substring(k1));
227 while (tick<des_b) {
228 description.append(scanner.nextLine());
229 tick++;
230 }
231 desline=scanner.nextLine();
232 int k2=desline.indexOf('<');
233 if (k2>0) {
234 desline=desline.substring(0,k2).trim();
235 if (desline.length()>0)
236 description.append(desline);
237 }
238 }
239 } // done with 'description', but may be empty
240 if (description.length()==0)
241 description.append("No description provided");
242
243 // next get AboutImage
244 StringBuilder aboutImage=new StringBuilder();
245 if (about_a>0 && about_b>about_a) {
246 try {
247 scanner.close();
248 scanner=new Scanner(file); // reopen
249 } catch (Exception iox) {
250 System.err.println("Failed to open cps file "+file.getPath()+" for precessing or to start a file 'scanner'");
251 return false;
252 }
253 // skip to actual image
254 int tick=0;
255 while (tick<about_a) {
256 scanner.nextLine();
257 tick++;
258 }
259 String about=scanner.nextLine(); // this is the image
260 if (about.length()>0) {
261 aboutImage.append("<a href="+file.getPath()+"> \n");
262 aboutImage.append("<img src=\"data:image/jpeg;base64,");
263 aboutImage.append(about);
264 if (mode==1) // standalone mode size

267 aboutImage.append("\" alt=\"HTML5\" style=\"width:100px;height:auto\"></a>");
