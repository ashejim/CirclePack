79 public int addObject(String objstr) {

81 if (nl==null || nl.size()==0)

83 ActiveSlider[] tmpSliders=new ActiveSlider[+sliderCount+nl.size()];
84 for (int j=0;j<sliderCount;j++)

92 String str=Integer.toString(v);

94 tmpSliders[sliderCount+hit]=

100 if (hit>0) {
101 sliderCount += hit;
102 mySliders=new ActiveSlider[sliderCount];
103 for (int j=0;j<sliderCount;j++)
104 mySliders[j]=tmpSliders[j];

106 this.pack();

110 public int removeObject(String objstr) {
111 NodeLink nl=new NodeLink(packData,objstr);
112 if (nl==null || nl.size()==0)
113 return 0;
114 ActiveSlider[] tmpSliders=new ActiveSlider[sliderCount];
115 for (int j=0;j<sliderCount;j++)
116 tmpSliders[j]=mySliders[j];
117 Iterator<Integer> vls=nl.iterator();
118 int hit=0;
119 while (vls.hasNext()) {
120 int v=vls.next();
121 int vindx=-1;
122 if ((vindx=verts.containsV(v))<0) // no such object
123 continue;
124 for (int j=(vindx+1);j<(sliderCount-hit);j++) {
125 tmpSliders[j-1]=tmpSliders[j];
126 tmpSliders[j-1].setIndex(j-1);

129 sliderPanel.remove(mySliders[vindx]);

131 hit++;
132 }
133
134 if (hit>0) {
135 sliderCount -= hit;
136 mySliders=new ActiveSlider[sliderCount];
137 for (int j=0;j<sliderCount;j++)
138 mySliders[j]=tmpSliders[j];
