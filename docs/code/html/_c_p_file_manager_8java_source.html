189 public static int trailingFile(Vector<Vector<String>> fseg,StringBuilder strbld) {
190 if (strbld==null || strbld.length()>0 || fseg==null || fseg.size()==0)
191 return 0;
192
193 int code=01;
194
195 // get the last flag sequence in 'fsa'
196 Vector<String> tems=(Vector<String>)fseg.remove(fseg.size()-1);
197 if (tems.size()==0)
198 return 0;
199 String flag=(String)tems.get(0);
200
201 // check for flag(s); if unexpected, return null
202 if (StringUtil.isFlag(flag)) {
203 tems.remove(0); // remove the flag string
204 // inappropriate flag?
205 if (!flag.startsWith("-s") && !flag.startsWith("-f") && !flag.startsWith("-a")) {
206 Vector<String> vecstr=new Vector<String>(1);
207 vecstr.add(flag); // put this flag back in as the last segment
208 }
209 else {
210 // script file?
211 if (flag.contains("s"))
212 code |= 04;
213 if (flag.contains("a"))
214 code |=02;
215 }
216
217 // rest is interpreted as the name, possibly with directory
218 if (tems.size()>0) {
219 strbld.append(StringUtil.reconItem(tems));
220 }
221 else if ((code & 04)==04) // no name, but want 'script' mode set
222 return code;
223 }
224 // no flag? take as the name
225 else
226 strbld.append(StringUtil.reconItem(tems));
227
228 // name may include '~' and/or directory
229 if (strbld.length()==0)
230 return 0;
231 if (strbld.charAt(0)=='~') {
232 strbld.deleteCharAt(0);
233 strbld.insert(0,HomeDirectory);
234 }
235
236 // name may include "[_n]" structure; find/insert number or literal
237 int k=-1;
238 if ((k=strbld.indexOf("["))>=0 && (k+1)<strbld.length()) {
239 int kk=-1;
240 Double nbr=null;
241 if ((kk=strbld.indexOf("]",k+1))>0) {
242 String vstr=strbld.substring(k+1,kk);
243 if (vstr!=null && vstr.length()>0) {
244 try {
245 nbr=Double.parseDouble(vstr);
246 } catch (Exception ex) {
247 // attempt failed, just return what we have
248 strbld.replace(k,kk+1,vstr);
249 return code;
250 }
251 Integer intnum=Integer.valueOf((int)Math.floor(nbr.doubleValue()));
252 strbld.replace(k,kk+1,intnum.toString());
253 }
254 }
255 }
256
257 return code;
258 }

273 public static String getFileName(StringBuilder strbuf) {
274 int length=strbuf.length();
275 if (length<=0)
276 return null;
277
278 // work backwards to find important characters
279 int spot=length;
280 boolean hit=false; // set to true when some legitimate char is found
281 while (spot>0) {
282 spot--;
283 char c=strbuf.charAt(spot);
284 switch(c) {
285 case ' ': {
286 if (spot==0) {
287 if (length>0) // should have been trimmed
288 return strbuf.substring(1);
289 else
290 return null;
291 }
292 if (strbuf.charAt(spot-1)=='\\')
293 spot--; // continue, this was escaped space
294 break;
295 }
296 case '\\': { // should be separation character
297 if (hit && spot<length-1)
298 return strbuf.substring(spot+1);
299 return null; // shouldn't be '\' at end
300 }
301 case '/': { // should be separation character
302 if (hit && spot<length-1)
303 return strbuf.substring(spot+1);
304 return null; // shouldn't be '\' at end
305 }
306 default: {
307 hit=true;
308 }
309 } // end of switch
310 } // end of while
311
312 if (hit)
313 return strbuf.substring(0).trim();
314 return null; // no legitimate character found
315 }

366 public static int readDataFile(PackData p,String filename,boolean script_flag,int mode) {
367 BufferedReader fp=openReadFP(filename,script_flag);
368 if (fp==null) return 0;
369 String line;
370
371 // ====================== various modes =============================
372
373 if (mode==1) { // read xyz location data for all vertices
374 int N=0;
375 try {
376 // Find keyword 'POINTS:' and number 'N' first
377 while((line=StringUtil.ourNextLine(fp))!=null) {
378 StringTokenizer tok = new StringTokenizer(line);
379 while(tok.hasMoreTokens()) {
380 String mainTok = tok.nextToken();
381 if(mainTok.equals("POINTS:")) {
382 try {
383 N=Integer.parseInt((String)tok.nextToken());
384 if (N<1 || N>p.nodeCount) throw new ParserException();
385 } catch(Exception ex) {
386 N=p.nodeCount;
387 }
388 }
389 }
390 }
391
392 // reaching here, must be ready to read N xy[z] locations for
393 if (p.xyzpoint==null)
394 p.xyzpoint=new Point3D[p.nodeCount+1];
395 int count=0;
396 try{
397 while((line=StringUtil.ourNextLine(fp))!=null && count<N) {
398 StringTokenizer tok = new StringTokenizer(line);
399 String str=(String)tok.nextToken();
400 double X=Double.parseDouble(str);
401 double Y=Double.parseDouble((String)tok.nextToken());
402
403 // may have just 2 coords
404 double Z;
405 try {
406 Z=Double.parseDouble((String)tok.nextToken());
407 } catch(Exception ex) {
408 Z=0.0;
409 }
410
411 count++;
412 p.xyzpoint[count]=new Point3D(X,Y,Z);
413 }
414 return count;
415 } catch (Exception ex) {
416 if (count>0) {
417 CirclePack.cpb.msg(
418 "read "+count+" xyz points from file "+filename);
419 return count;
420 }
421 else throw new InOutException();
422 }
423
424 } catch (Exception ex) {
425 CirclePack.cpb.myErrorMsg("Exception in reading xyz data from file.");
426 return 0;
427 }
428 } // end of mode 1, (xyz points)
429 return 0;
430 }
