488 public void drawCall(PairWrapper pWrapper,boolean cf_flag,double x,double y) {
489 String zpt=new String("z "+x+" "+y);
490 PackData p=domainCPS.getPackData();
491 PackData q=rangeCPS.getPackData();
492 if (pwIsDomain(pWrapper)!=0) { // else swap
493 PackData hold=q;
494 q=p;
495 p=hold;
496 }
497 if (cf_flag) { // circles
498 NodeLink vertlist=new NodeLink(p,zpt);
499 Iterator<Integer> vlist=vertlist.iterator();
500 int v;
501 while (vlist.hasNext()) {
502 v=(Integer)vlist.next();
503 p.circle_map_action(q,v,true);
504 }
505 }
506 else { // faces
507 FaceLink facelist=new FaceLink(p,zpt);
508 Iterator<Integer> flist=facelist.iterator();
509 int f;
510 while (flist.hasNext()) {
511 f=(Integer)flist.next();
512 p.face_map_action(q,f,true);
513 }
514 }
515 domainScreen.repaint();
516 rangeScreen.repaint();
517 }

565 public void actionPerformed(ActionEvent e){
566
567 // one of combo boxes??
568 if (e.getSource() instanceof JComboBox<?>) {
569 JComboBox<?> cb = (JComboBox<?>)e.getSource();
570 int i=cb.getSelectedIndex();
571 if (cb==domainCB) {

574 colorBorders();
575 setTitle("Mapping Window: P"+i+" P"+getRangeNum());
576 domainScreen.repaint();
577 }
578 else if (cb==rangeCB) {
579 rangeCPS=CPBase.cpDrawing[i];
580 PackControl.canvasRedrawer.changeRange(i);
581 colorBorders();
582 setTitle("Mapping Window: P"+getDomainNum()+" P"+i);
583 rangeScreen.repaint();
584 }
585 return;
586 }
587
588 String command = e.getActionCommand();
589
590 if (command.endsWith("pairDump")) {
591 PackControl.screenCtrlFrame.imagePanel.storeCPImage(null);
