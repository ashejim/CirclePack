81 public EdgeSimple chooseFlip() {
82 // "default":
83 // * if deg <= 6, just move
84 // * else choose randomly among largest edge energy if >= 2
85 noflip_count++;
86 int[] hflower=p.getFlower(homeVert);
87 if (flipStrategy.toLowerCase().startsWith("default")) {
88 if (homeDegree<=6)
89 return null;
90
91 // find the vector of petal indices for the biggest energy
92 Vector<Integer> biggest=bigEnergyEdges();
93 if (biggest.size()==0 || edgeEnergies[biggest.get(0)]<2)
94 return null;
95 int J=biggest.get(0);
96
97 // choose a random one among the biggest
98 if (biggest.size()>1) {
99 J=biggest.get(new Random().nextInt(biggest.size()));
100 }
101 int V=hflower[J];
102
103 // return this edge for flipping
104 noflip_count=0;
105 return new EdgeSimple(homeVert,V);
106 }
107 if (flipStrategy.toLowerCase().startsWith("hhl")) {
108 int num=p.countFaces(homeVert);
109 if (num<7) { // just move
110 return null;
111 }
112 int pre=p.nghb(homeVert, previousHome);
113 if (pre<0) { // if problem, choose random petal index
114 Random rand=new Random();
115 pre=hflower[rand.nextInt(num)];
116 }
117 int j=(pre-4+num)%num;
118
119 // return edge between hex axis and next vert to right
120 // TODO: careful, don't flip (home,previous)
121 noflip_count=0;
122 return new EdgeSimple(homeVert,hflower[j]);
123 }
124
125 // Ian Francis idea (6/2014)
126 if (flipStrategy.toLowerCase().startsWith("if")) {
127
128 // first choice: largest with energy >= 3
129 Vector<Integer> candidates=getCandidates(3);
130 if (candidates==null) {
131 // second choice: 'homeVert' and a petal have deg 7,
132 // commond nghbs degree 6.
133 candidates=getCandidates(1);
134 if (candidates==null) {
135 // third choice: 'homVert' deg 7, nghb degree 6,
136 // and common nghbs degree 6 and 5.
137 candidates=getCandidates(2);
138 if (candidates==null) {
139 // third choice: 'homVert' deg 6, nghb degree 6,
140 // and common nghbs degree 6 and 4.
141 candidates=getCandidates(5);
142 }
143 }
144 }
145 if (candidates!=null) {
146 int sz=candidates.size();
147 Random rand=new Random();
148 int j=rand.nextInt(sz);
149 noflip_count=0;
150 return new EdgeSimple(homeVert,candidates.get(j));
151 }
152 return null; // didn't get any suitable candidates
153 }
154
155 return null;
156 }

175 public int chooseMove(int home, int previous, EdgeSimple newEdge) {
176
177 int[] hflower=p.getFlower(homeVert);
178 if (moveStrategy.toLowerCase().startsWith("default")) {
179
180 // there was a flip
181 if (newEdge!=null) {
182
183 // find end's degrees (former left and right neighbors', resp.)
184 int deg_left=p.countFaces(newEdge.v)+p.getBdryFlag(newEdge.v);
185 int deg_right=p.countFaces(newEdge.w)+p.getBdryFlag(newEdge.w);
186
187 // if we move, which way is best?
188 int best=newEdge.v;
189 if (deg_right>deg_left)
190 best=newEdge.w;
191 else if (deg_left==deg_right && new Random().nextBoolean()) // flip coin
192 best=newEdge.w;
193
194 // don't move unless degree is higher than homeDegree
195 if (homeDegree>=(p.countFaces(best)+p.getBdryFlag(best)))
196 return 0;
197
198 // okay, this is the way to move
199 return best;
200 }
201
202 // if no flip, move to neighbor with larger degree
203 Vector<Integer> bigpets=getCandidates(0);
204 if (bigpets==null || bigpets.size()==0)
205 return 0; // some error occurred
206 int deg=bigpets.get(0);
207 if (petalDegrees[deg]>=homeDegree) {
208 int n=new Random().nextInt(bigpets.size());
209 return hflower[n];
210 }
211 return 0;
212 }
213
214 if (moveStrategy.toLowerCase().startsWith("hhl")) {
215 int num=p.countFaces(home);
216 int pre=p.nghb(home, previous);
217 if (pre>=0)
218 return hflower[(pre-3+num)%num];
219 else
220 throw new CombException("in move, 'home' and 'previous' not neighbors");
221 }
222
223 // Ian Francis idea, 6/2014: if there's been a flip, move to 'otherEnd' (the
224 // original opposite end of the flipped edge). Else, move to neighbor across
225 // highest energy edge.
226 if (moveStrategy.toLowerCase().startsWith("if")) {
227
228 if (otherEnd>0 && otherEnd!=homeVert)
229 return otherEnd;
230 Vector<Integer> cand=getCandidates(4); // largest energy, even if negative
231 int n=0;
232 if (cand!=null && cand.size()>0) {
233 n=new Random().nextInt(cand.size());
234 return cand.get(n);
235 }
236 n=new Random().nextInt(p.countFaces(homeVert));
237 return hflower[n];
238 }
239
240 return 0;
241 }

256 public Vector<Integer> getCandidates(int mode) {
257 int tick=0;
258 int[] hflower=p.getFlower(homeVert);
259
260 // all petal verts
261 if (mode==-1) {
262 Vector<Integer> ptls=new Vector<Integer>(0);
263 for (int j=0;j<homeDegree+bdryFlag;j++) {
264 ptls.add(hflower[j]);
265 }
266 return ptls;
267 }
268
269 // petal verts having largest degree.
270 // Caution: add 1 if bdryFlag is set, so result is not
271 // always the same data as 'petalDegrees'.
272 if (mode==0) {
273 int big=0;
274 for (int j=0;j<=homeDegree;j++) {
275 int k=hflower[j];
276 int nb=p.countFaces(k)+p.getBdryFlag(k);
277 big=(nb>big) ? nb : big;
278 }
279
280 Vector<Integer> bigpets=new Vector<Integer>(0);
281 for (int j=0;j<=homeDegree;j++) {
282 int k=hflower[j];
283 int nb=p.countFaces(k)+p.getBdryFlag(k);
284 if (nb==big)
285 bigpets.add(hflower[j]);
286 }
287 return bigpets;
288 }
289
290 // IanFrancis exception 1: homeVert have degree 7, find petals
291 // of deg 7 with petals on each side of deg 6.
292 if (mode==1) {
293 if (homeDegree!=7)
294 return null;
295 Vector<Integer> ans=new Vector<Integer>();
296 for (int j=1;j<=(homeDegree-bdryFlag);j++) {
297 int lpd=petalDegrees[(j+1)%homeDegree];
298 int rpd=petalDegrees[j-1];
299 if (petalDegrees[j]==7 && lpd==6 && rpd==6) {
300 ans.add(hflower[j]);
301 tick++;
302 }
303 }
304 if (tick==0)
305 return null;
306 return ans;
307 }
308
309 // IanFrancis exception2: homeVert degree 7, find petals
310 // of deg 6 with neighbors deg 5 and deg 6.
311 if (mode==2) {
312 if (homeDegree!=7)
313 return null;
314 Vector<Integer> ans=new Vector<Integer>();
315 for (int j=1;j<=(homeDegree-bdryFlag);j++) {
316 int lpd=petalDegrees[(j+1)%homeDegree];
317 int rpd=petalDegrees[j-1];
318 if (petalDegrees[j]==6 && ((lpd==5 && rpd==6) || (lpd==6 && rpd==5))) {
319 ans.add(hflower[j]);
320 tick++;
321 }
322 }
323 if (tick==0)
324 return null;
325 return ans;
326 }
327
328 // IanFrancis exception3: homeVert degree 6, find petals of deg 6
329 // with neighbors deg 4 and deg 6.
330 if (mode==5) {
331 if (homeDegree!=6)
332 return null;
333 Vector<Integer> ans=new Vector<Integer>();
334 for (int j=1;j<=(homeDegree-bdryFlag);j++) {
335 int lpd=petalDegrees[(j+1)%homeDegree];
336 int rpd=petalDegrees[j-1];
337 if (petalDegrees[j]==6 && ((lpd==4 && rpd==6) || (lpd==6 && rpd==4))) {
338 ans.add(hflower[j]);
339 tick++;
340 }
341 }
342 if (tick==0)
343 return null;
344 return ans;
345 }
346
347 // largest energies >= 3
348 if (mode==3) {
349 Vector<Integer> biggest=bigEnergyEdges(); // always non-empty
350 int j=biggest.get(0);
351 if (edgeEnergies[j]<3) // none qualify
352 return null;
353
354 // convert to vertex indices
355 Vector<Integer> verts=new Vector<Integer>();
356 Iterator<Integer> bgv=biggest.iterator();
357 while (bgv.hasNext())
358 verts.add(hflower[bgv.next()]);
359 return verts;
360 }
361
362 // largest energies, pos/neg/zero
363 if (mode==4) {
364 Vector<Integer> biggest=bigEnergyEdges(); // always non-empty
365
366 // convert to vertex indices
367 Vector<Integer> verts=new Vector<Integer>();
368 Iterator<Integer> bgv=biggest.iterator();
369 while (bgv.hasNext())
370 verts.add(hflower[bgv.next()]);
371 return verts;
372 }
373
374 throw new ParserException("'getCandidates': no valid mode given");
375 }

382 public int update() {
383
384 // am I on the boundary? This is 0 (no) or 1 (yes)
385 bdryFlag=p.getBdryFlag(homeVert);
386
387 // my degree? ('num' faces; add 1 if a bdry vertex)
388 homeDegree=p.countFaces(homeVert)+bdryFlag;
389 int[] hflower=p.getFlower(homeVert);
390
391 // my petals' degrees?
392 petalDegrees=new int[p.countFaces(homeVert)+1];
393 for (int j=0;j<=p.countFaces(homeVert);j++) {
394 int k=hflower[j];
395 petalDegrees[j]=p.countFaces(k)+p.getBdryFlag(k);
396 }
397
398 // petal 'energies'? (interior edges only)
399 edgeEnergies=new int[p.countFaces(homeVert)+1];
400 edgeEnergies[0]=edgeEnergies[homeDegree]=0;
401 for (int j=1;j<p.countFaces(homeVert);j++) {
402 edgeEnergies[j]=
403 homeDegree+petalDegrees[j]-(petalDegrees[j-1]+petalDegrees[j+1]);
404 }
405 if (bdryFlag==0) { // if closed, initial edge also
406 edgeEnergies[0]=edgeEnergies[homeDegree]=
407 homeDegree+petalDegrees[0]-(petalDegrees[1]+petalDegrees[homeDegree]);
408 }
409 return 1;
410 }
