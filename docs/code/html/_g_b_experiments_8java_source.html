71 public int runSweep(Vector<Vector<String>> flagSegs) {
72
73 // --- Parse arguments with defaults ---
74 int fixedVertex = -1;
75 double aim1 = 4.0; // fixed BP aim (multiple of pi)
76 double aim2 = 4.0; // sweep BP aim (multiple of pi)
77 double xmin = -1.0, xmax = 1.0, ymin = -1.0, ymax = 1.0;
78 int gridSize = 20;
79 String outputFile = "sweep_output.csv";
80
81 try {
82 for (Vector<String> items : flagSegs) {
83 if (items.isEmpty())
84 continue;
85 String str = items.get(0);
86 if (StringUtil.isFlag(str)) {
87 char c = str.charAt(1);
88 switch (c) {
89 case 'v': // fixed vertex
90 fixedVertex = Integer.parseInt(items.get(1));
91 break;
92 case 'a': // aim for fixed BP
93 aim1 = Double.parseDouble(items.get(1));
94 break;
95 case 'A': // aim for sweep BP
96 aim2 = Double.parseDouble(items.get(1));
97 break;
98 case 'b': // bounding box
99 xmin = Double.parseDouble(items.get(1));
100 xmax = Double.parseDouble(items.get(2));
101 ymin = Double.parseDouble(items.get(3));
102 ymax = Double.parseDouble(items.get(4));
103 break;
104 case 'n': // grid size
105 gridSize = Integer.parseInt(items.get(1));
106 break;
107 case 'o': // output file
108 outputFile = items.get(1);
109 break;
110 }
111 }
112 }
113 } catch (Exception ex) {
114 CirclePack.cpb.errMsg("sweep: error parsing arguments: "
115 + ex.getMessage());
116 return 0;
117 }
118
119 // --- Validate ---
120 if (fixedVertex <= 0) {
121 CirclePack.cpb.errMsg(
122 "sweep: must specify fixed vertex with -v {vertex}");
123 return 0;
124 }
125 if (fixedVertex > gmb.refPack.nodeCount) {
126 CirclePack.cpb.errMsg(
127 "sweep: vertex " + fixedVertex + " out of range");
128 return 0;
129 }
130 if (gmb.refPack.isBdry(fixedVertex)) {
131 CirclePack.cpb.errMsg(
132 "sweep: vertex " + fixedVertex + " is a boundary vertex");
133 return 0;
134 }
135
136 gmb.msg("sweep: fixed vertex=" + fixedVertex
137 + " aim1=" + aim1 + "pi"
138 + " aim2=" + aim2 + "pi"
139 + " grid=" + gridSize + "x" + gridSize
140 + " bounds=[" + xmin + "," + xmax + "]x["
141 + ymin + "," + ymax + "]");
142
143 // --- Run the sweep ---
144 double dx = (gridSize > 1) ? (xmax - xmin) / (gridSize - 1) : 0;
145 double dy = (gridSize > 1) ? (ymax - ymin) / (gridSize - 1) : 0;
146 int successCount = 0;
147 int failCount = 0;
148 int totalPoints = gridSize * gridSize;
149
150 StringBuilder csvData = new StringBuilder();
151 csvData.append("x,y,holonomy_error\n");
152
153 for (int i = 0; i < gridSize; i++) {
154 double x = xmin + i * dx;
155 for (int j = 0; j < gridSize; j++) {
156 double y = ymin + j * dy;
157
158 double hError = evaluatePoint(fixedVertex, aim1,
159 aim2, x, y);
160
161 if (hError >= 0) {
162 csvData.append(String.format("%.8f,%.8f,%.8f%n",
163 x, y, hError));
164 successCount++;
165 } else {
166 failCount++;
167 }
168 }
169
170 // Progress update every row
171 if ((i + 1) % 5 == 0 || i == gridSize - 1) {
172 gmb.msg("sweep progress: row " + (i + 1) + "/"
173 + gridSize + " (" + successCount
174 + " ok, " + failCount + " failed)");
175 }
176 }
177
178 // --- Write output ---
179 try {
180 File outFile = new File(outputFile);
181 BufferedWriter writer = new BufferedWriter(
182 new FileWriter(outFile));
183 writer.write(csvData.toString());
184 writer.close();
185 gmb.msg("sweep: wrote " + successCount
186 + " points to " + outFile.getAbsolutePath());
187 } catch (Exception ex) {
188 CirclePack.cpb.errMsg("sweep: failed to write output: "
189 + ex.getMessage());
190 // Print to console as fallback
191 System.out.println(csvData.toString());
192 }
193
194 // --- Revert to clean state ---
195 gmb.revert();
196 gmb.holoBorder = HalfLink.HoloHalfLink(
197 gmb.extenderPD.packDCEL, -1);
198
199 gmb.msg("sweep complete: " + successCount + "/"
200 + totalPoints + " points succeeded"
201 + " (" + failCount + " failed)");
202
203 return successCount;
204 }

225 public double evaluatePoint(int fixedVertex, double aim1,
226 double aim2, double x, double y) {
227 try {
228 // 1. Revert to clean packing
229 gmb.revert();
230
231 // Reinitialize holoBorder for the new DCEL
232 // (revert swaps extenderPD, so old holoBorder
233 // references stale half-edges)
234 gmb.holoBorder = HalfLink.HoloHalfLink(
235 gmb.extenderPD.packDCEL, -1);
236
237 // 2. Place fixed traditional branch point
238 TradBrModPt tbp = new TradBrModPt(gmb,
239 gmb.branchPts.size(),
240 aim1 * Math.PI, fixedVertex);
241 if (gmb.installBrPt(tbp) == 0)
242 return -1.0;
243
244 // 3. Repack and layout
245 if (!repackAndLayout())
246 return -1.0;
247
248 // 4. Place sweep branch point at (x,y) via click
249 Complex pt = new Complex(x, y);
250 double[] data = gmb.getClickData(gmb.refPack, pt);
251 if (data == null)
252 return -1.0;
253
254 int mode = (int) data[0];
255 int indx = (int) data[1];
256
257 if (mode < 0 || mode > 3 || indx <= 0
258 || indx > gmb.extenderPD.nodeCount)
259 return -1.0;
260
261 GenBrModPt bpt = null;
262 if (mode == 2) { // chaperone (inside a circle)
263 bpt = new ChapBrModPt(gmb,
264 gmb.branchPts.size(),
265 aim2 * Math.PI, indx,
266 (int) data[2], (int) data[3],
267 data[4], data[5]);
268 } else if (mode == 3) { // traditional (near center)
269 bpt = new TradBrModPt(gmb,
270 gmb.branchPts.size(),
271 aim2 * Math.PI, indx);
272 } else if (mode == 1) { // singular (interstice)
273 bpt = new SingBrModPt(gmb,
274 gmb.branchPts.size(),
275 aim2 * Math.PI, indx,
276 data[2], data[3]);
277 }
278
279 if (bpt == null || !bpt.success)
280 return -1.0;
281
282 if (gmb.installBrPt(bpt) == 0)
283 return -1.0;
284
285 // 5. Repack and layout
286 if (!repackAndLayout())
287 return -1.0;
288
289 // 6. Compute holonomy
290 Mobius holomob = PackData.holonomyMobius(
291 gmb.extenderPD, gmb.holoBorder);
292 if (holomob == null)
293 return -1.0;
294
295 double frobNorm = Mobius.frobeniusNorm(holomob);
296 return frobNorm;
297
298 } catch (Exception ex) {
299 // Silently skip points that cause exceptions
300 return -1.0;
301 }
302 }

314 if (repackResult < 0)
315 return false;
316 CommandStrParser.jexecute(gmb.extenderPD, "layout");
317 return true;
318 } catch (Exception ex) {
319 return false;
320 }
321 }
