197 String cmd_str=(String)mt.getCommand();
198 if (cmd_str!=null)
199 writer.write(indent+"<cmd>"+cmd_str+"</cmd>\n");
200 String data=(String)mt.getCPIcon().getIconName();

205 writer.write(indent+"<tooltip>"+data+"</tooltip>\n");

208 writer.write("</MyTool>\n\n");

455 public void parseMyTool(Element tE) {
456 // careful: these find first instances, even in subelements.
457 String name= tE.getAttribute("name");
458 String yesno=tE.getAttribute("dropable");
459// String needsC=tE.getAttribute("needsC");
460 String mnemonic=tE.getAttribute("mnemonic");
461 String tooltype=tE.getAttribute("type");
462 String handy=tE.getAttribute("handy");
463
464 String tip=getTextValue(tE,"tooltip");
465 String cmd_text=getTextValue(tE,"cmd");
466 String cmd2_text=getTextValue(tE,"cmd_m2");
467 String cmd3_text=getTextValue(tE,"cmd_m3");
468 String iconname=getTextValue(tE,"iconname");
469
470 // elements initiating canvass 'modes'
471 String modename=getTextValue(tE,"modename");
472 String canvasmode=getTextValue(tE,"canvasmode"); // short description
473 String cursorpoint=getTextValue(tE,"cursorpoint");
474
475 // is this the right 'type' for this toolbar? (MISC: is okay for any toolbar)
476 if (tooltype!=null && !tooltype.startsWith(this.toolPrefix)
477 && !tooltype.startsWith("MISC:") && tooltype.length()>0) return;
478
479 // First see if this is an 'active mode'; i.e., gives a cursor,
480 // cursor sweet spot, activemode string, and invokes a mode
481 // in the active canvas.
482 if (canvasmode!=null) {
483 Point point=null;
484 if (cursorpoint!=null) { // cursor sweet spot may be given
485 String coords[]=cursorpoint.split("\\s+");
486 if (coords.length==2) {
487 int x=Integer.valueOf(coords[0]).intValue();
488 int y=Integer.valueOf(coords[1]).intValue();
489 point=new Point(x,y);
490 }
491 }
492
493 boolean hndy=false;
494 if (handy!=null && handy.equals("yes"))
495 hndy=true;
496
497 CPIcon cpIcon= new CPIcon(iconname);
498 MyCanvasMode tm=new MyCanvasMode(modename,cpIcon,point,
499 cmd_text,cmd2_text,cmd3_text,canvasmode,tip,
500 toolPrefix,hndy);
501
502 // put 'menuItem' in mode vector; script handled separately
503 if (!toolPrefix.equals("SCRIPT:"))
504 tm.updateMenuItem();
505
506 return;
507 }
508
509 // check for 'menu' entries: can only accept one top level menu
510 PopupBuilder popUp=null;
511 NodeList nl=tE.getElementsByTagName("menu");
512 if (nl!=null && nl.getLength()>0) {
513 if (this instanceof ACTIVEHandler) {
514 try {
515 popUp=new PopupBuilder(nl.item(0),((ACTIVEHandler)this).activeWrapper);
516 cmd_text=new String(name); // have to have identifier
517 try {
518 String attachTo=nl.item(0).getAttributes().
519 getNamedItem("attachTo").getFirstChild().getNodeValue().trim();
520 if (attachTo.contains("canv")) {
521 ACTIVEHandler aH=(ACTIVEHandler)this;
522 aH.activeWrapper.button3Popup=popUp;
523 }
524 } catch (Exception ex) {}
525 } catch (Exception ex) {
526 CirclePack.cpb.errMsg("Failed building menu "+name+": "+ex.getMessage());
527 return;
528 }
529 }
530 return; // don't build the tool. (This could be left for redundancy)
531 }
532
533 // otherwise is a typical tool
534 boolean dropable=true;
535 if (yesno!=null && yesno.equals("no")) dropable=false;
536
537 // create the tool
538 CPIcon cpIcon= new CPIcon(iconname);
539 createTool(cpIcon,cmd_text,name,mnemonic,tip,dropable,popUp);
540 }
