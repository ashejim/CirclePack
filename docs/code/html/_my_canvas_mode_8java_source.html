211 // rotate sphere: 'ACTIVEHandler' holds data; this rotates so
212 // handStartX/Y becomes handX/Y
213 public static void rotate(ActiveWrapper aW) {
214 try {
215 CPdrawing cpS=aW.getCPDrawing();
216 ACTIVEHandler mH=aW.activeHandler;
217 Dimension dim=aW.getSize();
218 Point2D.Double pt1=cpS.pt2RealPt(new Point(mH.handStartX,mH.handStartY), dim.width,dim.height);
219 Point2D.Double pt2=cpS.pt2RealPt(new Point(mH.handX,mH.handY), dim.width,dim.height);
220 if (pt1.x * pt1.x + pt1.y * pt1.y > 1 || pt2.x * pt2.x + pt2.y * pt2.y > 1)
221 return;
222 double x1 = Math.sqrt(1 - pt1.x * pt1.x - pt1.y * pt1.y);
223 double x2 = Math.sqrt(1 - pt2.x * pt2.x - pt2.y * pt2.y);
224 // double theta1=Math.atan2(y1,x1);
225 // double theta2=Math.atan2(y2,x2);
226 // rotate by
227 Point3D pnt1 = new Point3D(x1, pt1.x, pt1.y);
228 Point3D pnt2 = new Point3D(x2, pt2.x, pt2.y);
229 Point3D n = math.Point3D.CrossProduct(pnt1, pnt2);
230 n = n.divide(n.norm());
231 double theta = n.getTheta();
232 double phi = n.getPhi();
233 double alpha = Math.acos(Point3D.DotProduct(pnt1, pnt2));
234 Matrix3D matrixTheta = Matrix3D.FromEulerAnglesXYZ(0, 0, theta);
235 Matrix3D matrixPhi = Matrix3D.FromEulerAnglesXYZ(0, phi, 0);
236 Matrix3D part2 = Matrix3D.times(matrixTheta, matrixPhi);
237 Matrix3D part1 = Matrix3D.Inverse(part2);
238 Matrix3D rot = Matrix3D.times(part2, Matrix3D.times(Matrix3D
239 .FromEulerAnglesXYZ(0, 0, alpha), part1));
240 Matrix3D matrix=Matrix3D.times(rot,cpS.sphView.viewMatrix);
241 if (!Matrix3D.isNaN(matrix)) {
242 cpS.sphView.viewMatrix=new Matrix3D(matrix);
243 TrafficCenter.cmdGUI(cpS.getPackData(),"disp -wr");
244 }
245 } catch (Exception ex) {return;}
246 }
