641 public static int ifThenWrapper(PackData p,String wholeStr,MyConsole myc,ResultPacket rP) {
642 int ifk=wholeStr.indexOf("IF");
643 int thenk=wholeStr.indexOf("THEN");
644 int elsek=wholeStr.indexOf("ELSE");
645
646 // Is this an 'IF-THEN' statement
647 if (ifk!=0 || thenk<0)
648 return -1;
649
650 // get the condition string
651 String condition=wholeStr.substring(ifk+2, thenk).trim();
652 if (condition.contains(";"))
653 return -1;
654
655 // get then and else commands
656 String thenStr=null;
657 String elseStr=null;
658 if (elsek<0) {
659 thenStr=wholeStr.substring(thenk+4).trim();
660 elseStr="break";
661 }
662 else {
663 thenStr=wholeStr.substring(thenk+4,elsek).trim();
664 elseStr=wholeStr.substring(elsek+4).trim();
665 }
666
667 Boolean outcome=BoolParser.trueFalse(p,condition);
668 if (outcome==null) {
669 throw new ParserException("if-then condition failed to evaluate correctly");
670 }
671
672 // carry out the 'then' or 'else' commands
673 ResultPacket innerRP=null;
674 if (outcome)
675 innerRP=new ResultPacket(p,thenStr);
676 else
677 innerRP=new ResultPacket(p,elseStr);
678
679
680 CPBase.trafficCenter.parseCmdSeq(innerRP,0,myc);
681 // pass along interrupt signal (e.g. from 'break')
682 if (innerRP.interrupt) {
683 rP.interrupt=true;
684 }
685 rP.cmdCount +=innerRP.cmdCount;
686 return innerRP.cmdCount;
