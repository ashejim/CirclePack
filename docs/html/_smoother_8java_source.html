215 public void calcForce(int v) {
216
217 PackData p=myPackData;
218 Complex F=new Complex(0.0);
219 double radF=0.0;
220 int N=0;
221 int num=p.countFaces(v)+p.getBdryFlag(v);
222 int[] flower=p.packDCEL.vertices[v].getFlower(true);
223 for (int j=0;j<num;j++) {
224 int w=flower[j];
225 double inv_d=EuclMath.inv_dist(newCenters[v],newCenters[w],
226 newRadii[v],newRadii[w]);
227 if ((p.isBdry(v) && bdryMode>0) || inv_d<min_inv_dist || inv_d>max_inv_dist) {
228 double x=TARGET_INV_DIST-inv_d; // currently working with tangency
229 double k=1.0; // ?? for future use?
230 Complex uvec=newCenters[w].minus(newCenters[w]);
231 double uvecabs=uvec.abs();
232 F.add(uvec.times(k*x/uvecabs));
233 radF =+ k*(uvecabs-newRadii[v]-newRadii[w]);
234 N++;
235 }
236 }
237
238 if (p.isBdry(v) && bdryMode>0) {
239 Complex bdryPt=PathUtil.getClosestPoint(newCenters[v],myPolygon);
240 Complex vec=bdryPt.minus(newCenters[v]);
241 F.add(vec);
242 radF += (vec.abs()-newRadii[v]);
243 N++;
244 }
245// else { // ??? doesn't seem to be used
246// double desiredRad=parentMG.getRadius(newCenters[v]);
247// }
248 ctr_force[v]=F;
249 rad_force[v]=radF/N;
250 }

257 public void applyForce(int v,double t) {
258 newRadii[v] += rad_force[v]*t;
259 if (newRadii[v]<=minRadius)

261 else if (newRadii[v]>=maxRadius)
262 newRadii[v]=maxRadius;
263 newCenters[v]=newCenters[v].add(ctr_force[v].times(t));
