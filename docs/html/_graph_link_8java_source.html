171 public int []componentize() {
172 boolean acyclic=true;
173
174 // First, set up 'mark' vector: =-1 if index isn't used,
175 // otherwise, smallest node it's connected to.
176 int []mark=new int[maxIndex+1];
177 for (int i=0;i<=maxIndex;i++)
178 mark[i]=-1; // -1 indicates i doesn't appear as node
179 Iterator<EdgeSimple> git=this.iterator();
180 EdgeSimple edge=null;
181 while (git.hasNext()) {
182 edge=new EdgeSimple(git.next());
183 int min=(edge.v<edge.w) ? edge.v :edge.w;
184 if (min==0) // have to watch for (0,v) root.
185 min=edge.w;
186 if (mark[edge.v]<0) mark[edge.v]=min;
187 if (mark[edge.w]<0) mark[edge.w]=min;
188 int mm=(mark[edge.v]<mark[edge.w]) ? mark[edge.v]:mark[edge.w];
189 mark[edge.v]=mark[edge.w]=mark[mm];
190 }
191 // pass through while some mark is adjusted down
192 boolean hit=true;
193 while (hit) {
194 hit=false;
195 for (int i=1;i<=maxIndex;i++) {
196 if (mark[i]>0 && mark[i]<=i) {
197 int cur=mark[i];
198 while (mark[cur]<cur) {
199 hit=true;
200 mark[i]=cur=mark[cur];
201 }
202 }
203 }
204 git=this.iterator();
205 while (git.hasNext()) {
206 edge=git.next();
207 if (mark[edge.v]!=mark[edge.w]) {
208 hit=true;
209 mark[edge.v]=mark[edge.w]=(mark[edge.v]<mark[edge.w]) ? mark[edge.v] : mark[edge.w];
210 }
211 }
212 }
213
214 // 'gen' builds generations of nodes from their roots.
215 int []gen=new int[maxIndex+1];
216 int compCount=0;
217 // look for smallest marks; these are all roots
218 for (int i=1;i<=maxIndex;i++) {
219 if (mark[i]==i) {
220 gen[i]=1;
221 compCount++;
222 }
223 }
224
225 // check acyclic: pass through edges as long as 'gen' is being reset
226 hit=true;
227 while (hit) {
228 hit=false;
229 git=this.iterator();
230 while (git.hasNext()) {
231 edge=git.next();
232 if (gen[edge.v]>0 || gen[edge.w]>0) {
233 if (gen[edge.v]==gen[edge.w] ||
234 (gen[edge.v]>0 && gen[edge.w]>0 &&
235 Math.abs(gen[edge.v]-gen[edge.w])>1))
236 acyclic=false;
237 if (gen[edge.v]==0) {
238 gen[edge.v]=gen[edge.w]+1;
239 hit=true;
240 }
241 else if (gen[edge.w]==0) {
242 gen[edge.w]=gen[edge.v]+1;
243 hit=true;
244 }
245 }
246 }
247 } // done with while
248
249 // return results
250 mark[0]=compCount;
251 if (!acyclic) mark[0] *= -1;
252 return mark;
253 }
