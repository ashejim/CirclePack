143 public double evaluate_h(Complex z) {
144 double ans=1.0;
145 if (ftn_mode==1) { // recip of Poincare density is unit disc
146 ans=(1.0-z.times(z).abs())/2.0;
147 }
148 else if (ftn_mode==2) { // Example 5.4, Bauer/Kraus/Roth/Wegert.
149 double absz=z.abs();
150 if (absz<=2.0)
151 ans=Math.sqrt(2*absz*absz+1.0);
152 else if (absz<=3.0)
153 ans=3.0;
154 else if (absz<=6.0)
155 ans=absz;
156 else ans=6.0;
157 }
158 else if (ftn_mode==3) {
159 ans=2-(z.abs()-1)*(z.abs()-1);
160 }
161 else { // use 'Function' tab in main GUI.
162 ans=CirclePack.cpb.getFtnValue(z).abs();
163 }
164
165 return ans;
166 }

176 public int beurling_flow(double t,double xpnt,int option) {
177 int count=0;
178 double CPhz,e_rad,cur,factor=1.0;
179 Complex z;
180
181 // cycle through bdry pr
182 for (int v=1;v<=extenderPD.nodeCount;v++) {
183 if (extenderPD.isBdry(v)) {
184 z=extenderPD.getCenter(v);
185 e_rad=domainData.getRadius(v);
186 CPhz=evaluate_h(z);
187
188 // Set the new radius
189 if (xpnt!=1.0) CPhz=Math.exp(xpnt*Math.log(CPhz));
190 cur=extenderPD.getRadius(v)/e_rad;
191 factor=Math.exp(t*Math.log(CPhz/cur));
192 double vrad=extenderPD.getRadius(v);
193 if (option==1) {
194 if (factor>1.0) // increases only
195 extenderPD.setRadius(v,factor*vrad);
196 }
197 else if (option==2) {
198 if (factor<1.0) // decreases only
199 extenderPD.setRadius(v,factor*vrad);
200 }
201 else
202 extenderPD.setRadius(v,factor*vrad);
203 count++;
204 }
205 } // end of for loop
206 return count;
207 }
