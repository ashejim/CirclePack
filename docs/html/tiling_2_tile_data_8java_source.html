228 public static PackData tiles2packing(TileData td) {
229
230 // check if we have enough data
231 if (td==null || td.tileCount<=0 || td.myTiles==null)
232 throw new CombException("Tile data is missing");
233 if (td.myTiles[1].vert[0]<=0 && td.myTiles[1].tileFlower==null)
234 throw new CombException(
235 "Tile data doesn't seem to have vertices or tileFlowers");
236
237 // in 'TILEFLOWERS' case, need to set consistent vert lists first
238 if (td.myTiles[1].vert[0]<=0) {
239 try {
240 if (TileBuilder.tileflowers2verts(td)<=0)
241 throw new CombException();
242 } catch (Exception ex) {
243 throw new CombException("failed using 'tileFlower'");
244 }
245 }
246
247 // Are there unigons, digons, slits, self-pasting?
248 boolean special=false;
249 for (int t=1;t<=td.tileCount && !special;t++) {
250 Tile tile=td.myTiles[t];
251 for (int j=0;(j<tile.vertCount && !special);j++) {
252 int v=tile.vert[j];
253 if (tile.vertCount==1 || tile.vertCount==2) // uni/digon?
254 special=true;
255 }
256 // any self-pasting, repeated verts?
257 if (!special) {
258 int vcnt=tile.vertCount;
259 for (int k=0;(k<vcnt && !special);k++) {
260 // slit? i.e., successive edges self-pasting
261 if (tile.tileFlower[k][0]==tile.tileIndex)
262 special=true;
263 }
264 if (!special) {
265 for (int k=0;(k<vcnt && !special);k++) {
266 int v=tile.vert[k];
267 for (int j=1;(j<vcnt && !special);j++)
268 if (tile.vert[(k+j)%vcnt]==v)
269 special=true;
270 }
271 }
272 }
273 }
274
275 // if yes, then need full barycentric refinement;
276 // NOTE: If there are unigons, this will fail, as we
277 // have no consistent way to handle them.
278 if (special) {
279 TileBuilder tileBuilder=new TileBuilder(td);
280 PackData p=tileBuilder.newfromFlowers();
281 if (p==null || p.tileData==null || p.nodeCount<=0)
282 throw new CombException(
283 "'tileBuilder' seems to have failed in 'tile2packing'");
284 return p;
285 }
286
287 // find vertices that occur among tile 'vert' lists;
288 // not guaranteed to start at 1 or be contiguous.
289 int maxnode=0;
290 for (int f=1;f<=td.tileCount;f++) {
291 Tile tile=td.myTiles[f];
292 for (int k=0;k<tile.vertCount;k++)
293 maxnode = (tile.vert[k]>maxnode) ? tile.vert[k]:maxnode;
294 }
295 int tmpNodeCount=maxnode+td.tileCount+1;
296
297 // building the bouquet takes several steps:
298 // 1. build unordered 'rawflower' of tiles for each vert.
299 // 2. then create ordered 'prequet' of petals
300 // 3. then create 'midquet': add flowers for tile
301 // barycenters and intersperse in 'prequet' to
302 // get 'midquet'
303 // 4. there are some gaps, so reorder to end up with 'bouquet'
304
305 // 1. build 'rawflower':
306
307 // 'tilehits' counts tiles for each vert to allocate space
308 int []tilehits=new int[maxnode+1];
309 for (int t=1;t<=td.tileCount;t++) {
310 Tile tile=td.myTiles[t];
311 for (int j=0;j<tile.vertCount;j++) {
312 tilehits[tile.vert[j]]++;
313 }
314 }
315 int [][]rawflower=new int[maxnode+1][];
316 for (int v=1;v<=maxnode;v++)
317 rawflower[v]=new int[tilehits[v]+1];
318
319 // 0-entry accumulates the number of entries
320 for (int t=1;t<=td.tileCount;t++) {
321 Tile tile=td.myTiles[t];
322 for (int j=0;j<tile.vertCount;j++) {
323 int v=tile.vert[j];
324 rawflower[v][0] +=1; // increment
325 rawflower[v][rawflower[v][0]]=t; // add t to list
326 }
327 }
328
329 // 2. go vert-by-vert to build 'prequet'
330 int[][] prequet=new int[tmpNodeCount+1][];
331 for (int v=1;v<=maxnode;v++) {
332 int numt=rawflower[v][0];
333 if (numt>0) {
334 // catalog edges from v in its tiles: edge = (u, w)
335 // means (v,u) and (w,v) are cclw edges of the tile
336 EdgeSimple[] nextprev=new EdgeSimple[numt];
337 for (int j=1;j<=numt;j++) {
338 Tile tile=td.myTiles[rawflower[v][j]];
339 int vj=tile.vertIndx(v);
340 int u=tile.vert[(vj+1)%tile.vertCount];
341 int w=tile.vert[(vj-1+tile.vertCount)%tile.vertCount];
342 nextprev[j-1]=new EdgeSimple(u,w);
343 }
344
345 // build a vector putting nghb'ing tiles in order
346 Vector<Integer> flow=new Vector<Integer>();
347 int[] tickoff=new int[numt+1];
348
349 // add first cclw pair of petals
350 int first=nextprev[0].v;
351 int last=nextprev[0].w;
352 flow.add(first);
353 flow.add(last);
354 tickoff[0]=1;
355
356 // find contiguous petal
357 int hits=1;
358 int safety=1000;
359 while (hits<numt && safety>0) {
360 safety--;
361 for (int pj=1;pj<numt;pj++) {
362 if (tickoff[pj]==0) {
363 EdgeSimple edge=nextprev[pj];
364 if (edge.v==last) {
365 flow.add((last=edge.w));
366 tickoff[pj]=1;
367 hits++;
368 }
369 else if (edge.w==first) {
370 flow.insertElementAt((first=edge.v),0);
371 tickoff[pj]=1;
372 hits++;
373 }
374 }
375 }
376 } // end of while
377
378 // might fail to use all tiles if the faces around
379 // v form more than one linked component
380 if (hits<numt) {
381 CirclePack.cpb.errMsg(
382 "tile flower for v="+v+" is incomplete");
383 }
384
385 // did we bomb?
386 if (safety==0)
387 throw new CombException(
388 "safety'ed out in building flower for v="+v);
389
390 // build 'prequet'
391 prequet[v]=new int[numt+1];
392 for (int k=0;k<=numt;k++) {
393 prequet[v][k]=flow.get(k);
394 }
395
396// if (prequet[v][0]==prequet[v][numt])
397// thePack.setBdryFlag(v,0);
398// else
399// thePack.setBdryFlag(v,1);
400// thePack.rData[v]=new RData();
401// thePack.rData[v].rad=.5;
402 }
403
404 // TODO: we do not check that we get a connected pattern or
405 // that we use all tiles.
406
407 } // end of loop for 'prequet'
408
409
410 // 3. Create 'midquet': add flowers for tile barycenters
411 // and interpose these barycenters in vertex flowers.
412 // Note that skipped indices are left with null array
413 int[][] midquet=prequet;
414 for (int t=1;t<=td.tileCount;t++) {
415 Tile tile=td.myTiles[t];
416 int tnode=t+maxnode; // barycenter index
417 midquet[tnode]=new int[tile.vertCount+1];
418 for (int j=0;j<tile.vertCount;j++) {
419 int v=tile.vert[j];
420 midquet[tnode][j]=v;
421 int preV=tile.vert[(j+1)%tile.vertCount];
422 midquet[v]=insert_index(midquet[v],tnode,preV);
423 }
424 midquet[tnode][tile.vertCount]=midquet[tnode][0];
425 } // end of 'while' for tiles
426
427 // 4. create 'bouquet': toss empty arrays, re-index
428 int[] newIndx=new int[tmpNodeCount+1];
429 int tick=0;
430 for (int v=1;v<=tmpNodeCount;v++)
431 if (midquet[v]!=null)
432 newIndx[v]=++tick;
433 int newNodeCount=tick;
434
435 int[][] bouquet=new int[newNodeCount+1][];
436 for (int v=1;v<=tmpNodeCount;v++) {
437 int newv=newIndx[v];
438 if (newv!=0) {
439 for (int j=0;j<midquet[v].length;j++)
440 midquet[v][j]=newIndx[midquet[v][j]];
441 }
442 bouquet[newv]=midquet[v];
443 }
444
445 // complete the packing
446 PackDCEL pdcel=CombDCEL.getRawDCEL(bouquet);
447 if (pdcel==null)
448 throw new CombException(
449 "Failed to create using 'bouquet'");
450 PackData thePack=new PackData(null);
451 pdcel.fixDCEL(thePack);
452 thePack.status=true;
453
454 // duplicate tile data, but duals, quads, wgTiles, and
455 // augmented vertices are set to null
456 thePack.tileData=td.copyMyTileData();
457 TileData newtd=thePack.tileData; // point to new copy
458 newtd.dualTileData=newtd.quadTileData=null;
459 newtd.wgTiles=null;
460 for (int t=1;t<=newtd.tileCount;t++)
461 newtd.myTiles[t].augVert=null;
462
463 // reset tile vertices to match its barycenter's flower
464 for (int t=1;t<=newtd.tileCount;t++) {
465 Tile tile=newtd.myTiles[t];
466 tile.baryVert=newIndx[t+maxnode];
467 tile.vert=thePack.getFlower(tile.baryVert);
468 }
469
470 // allocate 'tileFlowers'
471 for (int t=1;t<=newtd.tileCount;t++) {
472 Tile tile=newtd.myTiles[t];
473 tile.tileFlower=new int[tile.vertCount][2];
474 }
475
476 // catalog the tiles
477 int tbase=thePack.nodeCount-newtd.tileCount; // barycenter indices
478 for (int t=1;t<=newtd.tileCount;t++) {
479 Tile tile=newtd.myTiles[t];
480 HalfLink spokes=
481 thePack.packDCEL.vertices[tile.baryVert].getSpokes(null);
482
483 // go around vertices, see if there's a tile across edge.
484 for (int k=0;k<spokes.size();k++) {
485 HalfEdge spoke=spokes.get(k);
486 HalfEdge he=spoke.next;
487 if (!he.isBdry()) { // should be a tile s across edge
488 int w=he.twin.origin.vertIndx;
489 he=he.twin.next.next;
490 int s=he.origin.vertIndx-tbase; // opposite tile
491 if (s<=0)
492 throw new CombException("should be an opposite tile");
493 Tile nghbtile=newtd.myTiles[s];
494 tile.tileFlower[k][0]=s;
495 int nghbind=thePack.nghb(nghbtile.baryVert,w); // index of w in s
496 tile.tileFlower[k][1]=nghbind;
497 nghbtile.tileFlower[nghbind][0]=t;
498 nghbtile.tileFlower[nghbind][1]=k;
499 }
500 else { // on bdry?
501 tile.tileFlower[k][0]=0;
502 tile.tileFlower[k][1]=0;
503 }
504 }
505 } // done cataloging
506
507 // wrap up and return
508 thePack.setAlpha(newtd.myTiles[1].baryVert);
509 thePack.setGamma(newtd.myTiles[1].vert[0]);
510 thePack.hes=thePack.intrinsicGeom;
511 thePack.set_aim_default();
512
513 return thePack;
514 }

526 public static int[] insert_index(int[] array,int newV, int preV) {
527 int n=array.length;
528 int spot=-1;
529 for (int j=0;(j<n && spot<0);j++)
530 if (array[j]==preV)
531 spot=j;
532 if (spot<0)
533 return null;
534 int[] newarray=new int[n+1];
535 for (int k=0;k<=spot;k++)
536 newarray[k]=array[k];
537 newarray[spot+1]=newV;
538 for (int k=spot+2;k<=n;k++)
539 newarray[k]=array[k-1];
540 return newarray;
541 }

587 public static PackData tiles2FullBary(TileData td) {
588 if (td==null || td.tileCount<=0)
589 return null;
590
591 // do we have vertices that are indexed?
592 // May not, eg., when data was read in
593 // 'TILEFLOWERS:' form.
594 // This call indexes the vertices from 'tileFlowers'.
595 // Throw exception if these don't exist or give
596 // exception
597 if (td.myTiles[1].vert[0]<=0) {
598 try {
599 if (TileBuilder.tileflowers2verts(td)<=0)
600 throw new CombException();
601 } catch (Exception ex) {
602 throw new CombException("Failed in using 'tileFlowers'");
603 }
604 }
605
606 // check for:
607 // * any tile that is unigon or bigon
608 // * any tile with two edges identified
609 boolean special=false;
610 for (int t=1;(t<=td.tileCount && !special);t++) {
611 Tile tile=td.myTiles[t];
612 if (tile.vertCount<3)
613 special=true;
614 for (int j=0;(j<(tile.vertCount-1) && !special);j++)
615 for (int k=j+1;(k<tile.vertCount && !special);k++)
616 if (tile.vert[j]==tile.vert[k])
617 special=true;
618 }
619 if (special) {
620 TileBuilder tileBuilder=new TileBuilder(td);
621 PackData p=tileBuilder.newfromFlowers();
622 if (p==null || p.tileData==null || p.nodeCount<=0)
623 throw new CombException("'tileBuilder' seems to have failed");
624 // see 'slitSpecial' for one case
625// throw new CombException("Not yet ready for these 'special' tile combinatorics");
626 }
627
628 // create minimal packing for given 'TileData'
629 PackData newPD=TileData.tiles2packing(td); // minimal packing
630 TileData workingTD=newPD.tileData;
631
632 // clear 'mark's
633 for (int i=1;i<=newPD.nodeCount;i++)
634 newPD.setVertMark(i,0);
635
636 // go through tiles, split edges; splitting edge
637 // should prevent it being selected to be split
638 // again from the other side.
639 int origcount=newPD.nodeCount;
640 for (int t=1;t<=workingTD.tileCount;t++) {
641 Tile tile=workingTD.myTiles[t];
642 newPD.setVertMark(tile.baryVert,1);
643 int w=tile.vert[0];
644 int v=w;
645 for (int j=1;j<=tile.vertCount;j++) {
646 v=w;
647 if (j==tile.vertCount)
648 w=tile.vert[0];
649 else
650 w=tile.vert[j];
651 HalfEdge he=newPD.packDCEL.findHalfEdge(v, w);
652 if (he!=null)
653 RawManip.splitEdge_raw(newPD.packDCEL,he);
654 else { // old edge; already split
655// DCELdebug.printBouquet(newPD.packDCEL);
656 continue;
657 }
658 }
659 }
660 newPD.packDCEL.fixDCEL(newPD);
661 // DCELdebug.printBouquet(newPD.packDCEL);
662
663 // set marks: 1=barycenters, 2=tile verts, 3=new vertices
664 for (int t=1;t<=newPD.tileData.tileCount;t++) {
665 Tile tile=newPD.tileData.myTiles[t];
666 newPD.setVertMark(tile.baryVert,1);
667 for (int j=0;j<tile.vertCount;j++)
668 newPD.setVertMark(tile.vert[j],2);
669 }
670 for (int v=origcount+1;v<=newPD.nodeCount;v++)
671 newPD.setVertMark(v,3);
672
673 // prepare for 'quadTile' and 'dualTile'
674 int dcount=0;
675 int qcount=0;
676 for (int v=1;v<=newPD.nodeCount;v++) {
677 int k=newPD.getVertMark(v);
678 if (k==2)
679 dcount++;
680 else if (k==3)
681 qcount++;
682 }
683 workingTD.dualTileData=new TileData(dcount,td.builtMode);
684 workingTD.quadTileData=new TileData(qcount,td.builtMode);
685
686 // bary_refine: each face in barycentric subdivision of
687 // tiling is bary_refined, hence is a 6-flower: these
688 // new barycenters are numbered starting at 'bcbase'.
689 // Save lots of work by building master list 'wgTiles' as
690 // as they are created, indexed from 1 by using the index
691 // of the barycenter circle and subtracting 'bcbase'.
692 workingTD.wgTileCount=newPD.faceCount;
693 RawManip.hexBaryRefine_raw(newPD.packDCEL,true);
694 newPD.packDCEL.fixDCEL(newPD);
695 int bcbase=newPD.nodeCount-workingTD.wgTileCount;
696 // indexed from 1 to wgTileCount
697 workingTD.wgTiles=new Tile[workingTD.wgTileCount+1];
698
699 // for each tile create 'augVert' list for tile,
700 // create and save 'wgTiles'
701 int wgTick=0; // for indices of wgTiles in master list
702 for (int t=1;t<=workingTD.tileCount;t++) {
703 Tile tile=workingTD.myTiles[t];
704 tile.wgIndices=new int[2*tile.vertCount];
705 int num=newPD.countFaces(tile.baryVert);
706
707 // check consistency
708 if (newPD.isBdry(tile.baryVert) || num!=(4*tile.vertCount)) {
709 CirclePack.cpb.errMsg("Tile "+t+" (center "+tile.baryVert+") "+
710 "flower has problem");
711 break; // out of 'for' loop
712 }
713
714 // arrange that halfedge of baryVert is direction to vert[0].
715 // TODO: to handle dual tilings consistently, have to consider
716 // case that baryVert is on the boundary (and tile bdry
717 // goes through it. Best is vert[0] is required to be
718 // flower[0] in this case from the beginning.
719 Vertex vert=newPD.packDCEL.vertices[tile.baryVert];
720 HalfLink spokes=vert.getSpokes(null);
721 vert.halfedge=null;
722 Iterator<HalfEdge> sis=spokes.iterator();
723 while (sis.hasNext()) {
724 HalfEdge he=sis.next();
725 int m=he.twin.origin.vertIndx;
726 if (newPD.countFaces(m)==4 && newPD.nghb(m, tile.vert[0])>=0) {
727 vert.halfedge=he;
728 break;
729 }
730 }
731 if (vert.halfedge==null) {
732 CirclePack.cpb.errMsg("Tile "+t+" (center "+tile.baryVert+") "+
733 "refined flower has some problem");
734 break; // out of 'for' loop
735 }
736
737 // build augmented vertices by visiting sectors in succession
738 int newAugCount=4*tile.vertCount;
739 int []augVs=new int[newAugCount];
740 int [] myflower=vert.getFlower(true);
741 for (int j=0;j<tile.vertCount;j++) {
742
743 // two hex flower subfaces, first is white (positively oriented)
744 Tile wgtile=new Tile(3);
745 wgtile.augVertCount=6;
746 wgtile.augVert=new int[6];
747 wgtile.mark=1; // positively oriented
748 int baryV=myflower[4*j+1]; // its barycenter
749
750 // put wgTile in master list and record its index
751 tile.wgIndices[2*j]=++wgTick; // point to wgTile index
752 workingTD.wgTiles[tile.wgIndices[2*j]]=wgtile; // put wgTile into master list
753
754 // set halfedge of baryV to point to tile.baryVert
755 newPD.packDCEL.vertices[baryV].halfedge=
756 newPD.packDCEL.findHalfEdge(baryV,tile.baryVert);
757 int[] itsflower=newPD.getFlower(baryV);
758 for (int k=0;k<3;k++) {
759 wgtile.vert[k]=wgtile.augVert[2*k]=itsflower[2*k];
760 wgtile.augVert[2*k+1]=itsflower[2*k+1];
761 }
762 augVs[4*j]=itsflower[2];
763 augVs[4*j+1]=itsflower[3];
764
765 // second is grey (negatively oriented)
766 wgtile=new Tile(3);
767 wgtile.augVertCount=6;
768 wgtile.augVert=new int[6];
769 wgtile.mark=-1; // negatively oriented
770 baryV=newPD.getFlower(tile.baryVert)[4*j+3]; // its barycenter
771 tile.wgIndices[2*j+1]=++wgTick; // point to wgTile index
772 workingTD.wgTiles[tile.wgIndices[2*j+1]]=wgtile; // put wgTile into master list
773
774 // set halfedge of baryV to point to tile.baryVert
775 newPD.packDCEL.vertices[baryV].halfedge=
776 newPD.packDCEL.findHalfEdge(baryV,tile.baryVert);

1036 public int identifyTiles() {
1037 int nextTile=0;
1038 int []vtrack=new int[packData.nodeCount+1];
1039 int [][]tflower=new int[packData.nodeCount+1][];
1040
1041 // paralleling 'flower': keeps track of hits
1042 for (int vv=1;vv<=packData.nodeCount;vv++)
1043 tflower[vv]=new int[packData.countFaces(vv)+1];
1044
1045 // cycle between two lists
1046 NodeLink curr=new NodeLink(packData);
1047 NodeLink next=new NodeLink(packData);
1048 next.add(1);
1049
1050 // go through list of verts and identify tiles
1051 while (next!=null && next.size()>0) {
1052 curr=next;
1053 next=new NodeLink(packData);
1054 Iterator<Integer> crt=curr.iterator();
1055
1056 // process this vertex
1057 while (crt.hasNext()) {
1058 int v=crt.next();
1059 int num=packData.countFaces(v);
1060 int[] vflower=packData.getFlower(v);
1061 if (vtrack[v]!=num) { // not done with this vert
1062 for (int j=0;j<num;j++) {
1063
1064 // start a tile in this direction?
1065 if (tflower[v][j]==0) {
1066 Vector<Integer> tileVerts=new Vector<Integer>(3);
1067 tileVerts.add(v);
1068 int u=vflower[j];
1069 int safety=1000;
1070 int w=v;
1071
1072 // go around edges to find the tile vertices
1073 while (u!=v && safety>0) {
1074 safety--;
1075 tileVerts.add(u);
1076 int indx_uw=packData.nghb(u,w);
1077 int nm=packData.countFaces(u);
1078 tflower[u][indx_uw] += 1;
1079 tflower[w][packData.nghb(w,u)] += 1;
1080
1081 // add this to next list
1082 if (vtrack[u]==0)
1083 next.add(u);
1084 vtrack[u]++;
1085
1086 // shift
1087 w=u;
1088 if (indx_uw==0) // must be closed flower
1089 indx_uw=nm-1;
1090 indx_uw=(indx_uw-1+nm)%nm;
1091 u=packData.getFlower(v)[indx_uw];
1092 } // while around tile
1093
1094 // probably an error
1095 if (safety==0) {
1096 throw new CombException("error finding tile for "+v);
1097 }
1098 // create the tile vertex list
1099 int sz=tileVerts.size();
1100 Tile tile=new Tile(packData,this,sz);
1101 for (int ii=0;ii<sz;ii++)
1102 tile.vert[ii]=tileVerts.get(ii);
1103
1104 // counting
1105 vtrack[v]++; // have new tile for v
1106 tile.tileIndex=++nextTile;
1107 myTiles[nextTile]=tile;
1108 }
1109
1110 } // done with petals of v
1111 } // done with v
1112 } // end of 'while' through 'curr'
1113 }
1114
1115 tileCount=nextTile;
1116 return tileCount;

1135 return null;

1138 NodeLink finalList=new NodeLink(p);

1140
1141 // cycle between two lists
1142 NodeLink curr=null;
1143 NodeLink next=new NodeLink(p,V);
1144 int safety=2*p.nodeCount;
1145 while(next!=null && next.size()>0 && next.size()>0 && safety>0) {
1146 safety--;
1147 curr=next;
1148 next=new NodeLink(p);
1149 Iterator<Integer> cl=curr.iterator();
1150 while (cl.hasNext() && safety>0) {
1151 safety--;

1153 if (util[v]==0 && !p.isBdry(v)) {
1154 // want this v
1155 finalList.add(v);
1156 util[v]=1;
1157
