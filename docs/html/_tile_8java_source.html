408 public int checkDepth() {
409 int n=0;
410
411 // no 'TileData'? then depth is zero
412 if (myTileData==null || myTileData.tileCount<=0)
413 return n;
414 boolean someneg=false;
415 int min=myTileData.myTiles[1].checkDepth();
416 if (min<0)
417 someneg=true;
418 min=Math.abs(min);
419 for (int t=2;t<=myTileData.tileCount;t++) {
420 int m=myTileData.myTiles[t].checkDepth();
421 if (m<0)
422 someneg=true;
423 m=Math.abs(m);
424 if (m!=min)
425 someneg=true;
426 min=(m<min)?m:min;
427 }
428
429 // new depth is min+1
430 min++;
431 if (someneg)
432 min *= -1;
433 return min;

474 public NodeLink findAugEdge(int indx) {
475 indx=Math.abs(indx)%vertCount;
476 int v=vert[indx];
477 int w=vert[(indx+1)%vertCount];
478
479 NodeLink out=new NodeLink();
480
481 // no 'augVert'? just return v,w
482 if (augVert==null) {
483 out.add(v);
484 out.add(w);
485 return out;
486 }
487
488 // Because vert indices can be repeated, we
489 // have to identify indices of edge starts from
490 // 0th edge up until we reach indx edge.
491 int[] starts=new int[vertCount]; // indices of edge starts
492 starts[0]=0;
493 int tick=0;
494 while (tick<indx) {
495 int hit=vert[(tick+1)%vertCount];
496 int k=starts[tick]+1;
497 while (k<augVertCount) {
498 if (augVert[k]==hit) {
499 tick++;
500 starts[tick]=k;
501 break;
502 }
503 k++;
504 }
505 }
506 int j=starts[tick];
507 out.add(augVert[j]);
508 j++;// next possible edge starting
509 int ticker=(tick+1)%vertCount; // last edge, have to wrap to 0
510 while (j<augVertCount &&
511 augVert[j]!=vert[ticker]) {
512 out.add(augVert[j]);
513 j++;
514 }
515 out.add(vert[ticker]); // add the last one
516 return out;
517 }

650 public static Vector<EdgeSimple> tile2tileMatch(int[][] t,
651 int tindx,int[][] nt,int ntindx) {
652
653 // check hints for minimal consistency
654 boolean hints=false;
655
656 int tnt=0;
657 int tfirstmiss=-1;
658 for(int j=0;j<t.length;j++) {
659 if (t[j][0]==ntindx) {
660 tnt++;
661 int dx;
662 if ((dx=t[j][1])>=0) {
663 if (!hints)
664 hints=true; // there's a matching hint
665 if (dx>=nt.length || nt[t[j][1]][0]!=tindx ||
666 (nt[dx][1]>=0 && nt[dx][1]!=j))
667 throw new DataException("inconsistency between entries and hint "+dx);
668
669 // can set t[j][1] based on index given in nt
670 if (t[j][1]<0 && nt[dx][1]==j)
671 t[j][1]=dx;
672 }
673 }
674 else if (tfirstmiss<0) // find first miss
675 tfirstmiss=j;
676
677 }
678
679 int ntt=0;
680 int ntfirstmiss=-1;
681 for(int j=0;j<nt.length;j++) {
682 if (nt[j][0]==tindx) {
683 ntt++;
684 int dx;
685 if ((dx=nt[j][1])>=0) {
686 if (!hints)
687 hints=true; // there's some hint for matchin
688 if (dx>=t.length || t[nt[j][1]][0]!=ntindx ||
689 (t[dx][1]>=0 && t[dx][1]!=j))
690 throw new DataException("inconsistency other way between entries and hints "+dx);
691
692 // can set nt[j][1] based on index given in t
693 if (nt[j][1]<0 && t[dx][1]==j)
694 nt[j][1]=dx;
695 }
696 }
697 else if (ntfirstmiss<0)
698 ntfirstmiss=j;
699 }
700
701 // inconsistency?
702 if (tnt!=ntt)
703 throw new DataException("match counts between lists are not equal");
704
705 // no matches?
706 if (tnt==0)
707 return null;
708
709 // most common situation
710 if (tnt==1) {
711 int thit=-1;
712 for (int j=0;(j<t.length && thit<0);j++)
713 if (t[j][0]==ntindx)
714 thit=j;
715 int nthit=-1;
716 for (int j=0;(j<t.length && nthit<0);j++)
717 if (nt[j][0]==tindx)
718 nthit=j;
719 Vector<EdgeSimple> ans=new Vector<EdgeSimple>();
720 ans.add(new EdgeSimple(thit,nthit));
721 return ans;
722 }
723
724 // complete string
725 if ((tnt==t.length || ntt==nt.length) && !hints) {
726 throw new DataException("all edges, but no hints to help match");
727 }
728
729 // find 'tstarts' of starts of contiguous matches to 'nt' within 't'
730 Vector<EdgeSimple> tstarts=new Vector<EdgeSimple>();
731 for (int i=0;i<t.length;i++) {
732 if (t[i][0]==ntindx && t[(i-1+t.length)%t.length][0]!=ntindx)
733 tstarts.add(new EdgeSimple(i,0));
734 }
735
736 // and 'ntstarts' of starts of contiguous matches to 't' within 'nt'
737 Vector<EdgeSimple> ntstarts=new Vector<EdgeSimple>();
738 for (int i=0;i<nt.length;i++) {
739 if (nt[i][0]==tindx && nt[(i-1+nt.length)%nt.length][0]!=tindx)
740 ntstarts.add(new EdgeSimple(i,0));
741 }
742
743 // determine/store lengths of contiguous sets of matches
744 for (int k=0;k<tstarts.size();k++) {
745 int st=tstarts.get(k).v;
746 int tick=1;
747 while (t[(st+tick)%t.length][0]==ntindx)
748 tick++;
749 tstarts.get(k).w=tick;
750 }
751 for (int k=0;k<ntstarts.size();k++) {
752 int nst=ntstarts.get(k).v;
753 int tick=1;
754 while (nt[(nst+tick)%nt.length][0]==tindx)
755 tick++;
756 ntstarts.get(k).w=tick;
757 }
758
759 if (tstarts.size()==0 || ntstarts.size()==0)
760 throw new DataException("screwed up one of 'tstarts' or 'ntstarts'");
761
762 // Now have segs for 't' and 'nt': need to match as best we can
763 Vector<EdgeSimple> ans=new Vector<EdgeSimple>();
764
765 // most typical: single segment on each side (necessarily the same length)
766 if (tstarts.size()==1 && ntstarts.size()==1) {
767 EdgeSimple ts1=tstarts.get(0);
768 EdgeSimple nts1=ntstarts.get(0);
769
770 // number of entries
771 int tcount=ts1.w;
772 int ntcount=nts1.w;
773 if (tcount!=ntcount || tcount<=0)
774 throw new DataException("some inconsistency in 'tstarts' and 'ntstarts'");
775 for (int j=0;j<tcount;j++) {
776 ans.add(new EdgeSimple((ts1.v+j)%(t.length),(nts1.v+tcount-j-1+nt.length)%(nt.length)));
777 }
778 return ans;
779 }
780
781 // TODO: Here we need more sophisticated matching strategies;
782 // For now, pass back pairs <j,-3> so we don't look at this matching again
783 else {
784 EdgeSimple ts1=tstarts.get(0);
785 for (int j=0;j<ts1.w;j++) {
786 ans.add(new EdgeSimple((ts1.v+j)%(t.length),-3));
787 }
788 return ans;
789 }
