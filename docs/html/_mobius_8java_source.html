469 public static Mobius standard3Point(Complex a, Complex b, Complex c) {
470
471 Mobius mob = new Mobius(); // initiate as identity
472
473 // compute min of approximate spherical distances
474 double sdst = a.minus(b).abs() / (1 + a.absSq());
475 double hd = 1.0;
476 sdst = (sdst < (hd = b.minus(c).abs() / (1 + b.absSq()))) ? sdst : hd;
477 sdst = (sdst < (hd = c.minus(a).abs() / (1 + c.absSq()))) ? sdst : hd;
478
479 if (sdst < 10 * MOB_TOLER)
480 return mob; // identity
481
482 // check if any are at infinity
483 int inf = -1;
484 if (a.absSq() > 10000000000.0)
485 inf = 0;
486 if (b.absSq() > 10000000000.0) {
487 if (inf == 0)
488 return mob; // two points are too close
489 inf = 1;
490 }
491 if (c.absSq() > 10000000000.0) {
492 if (inf >= 0)
493 return mob; // two points are too close
494 inf = 2;
495 }
496
497 if (inf == 2) { // c=inf-->inf, for Az+B
498 mob.c = new Complex(0.0);
499 mob.d = new Complex(1.0);
500 mob.a = new Complex(1.0);
501 mob.a = mob.a.divide(b.sub(a));
502 mob.b = a.divide(a.sub(b));
503 } else if (inf == 0) { // a at infinity; like above, then reciprocate
504 mob.a = new Complex(0.0);
505 mob.b = new Complex(1.0);
506 mob.c = new Complex(1.0);
507 mob.c = mob.c.divide(b.sub(c));
508 mob.d = c.divide(c.sub(b));
509 } else if (inf == 1) { // b at infinity; [1 -a;1 -c]
510 mob.a = new Complex(1.0);
511 mob.b = new Complex(-a.x, -a.y);
512 mob.c = new Complex(1.0);
513 mob.d = new Complex(-c.x, -c.y);
514 } else { // general case:
515 mob.a = b.minus(c);
516 mob.b = a.times(mob.a).times(-1.0);
517 mob.c = b.sub(a);
518 mob.d = c.times(mob.c).times(-1.0);
519 }
520
521 mob.normalize();
522 return mob;
523 }

967 Mobius mob = new Mobius();
968 Mobius M1 = standard_mob(a, b);
969 Mobius M2 = standard_mob(A, B);
970 M2.inverse();
971 mob = (Mobius) M2.rmultby(M1); // M = M2 * M1
972
973 // if we switch, have to switch back
974 if (aa >= MOD1)
975 mob.inverse();
976 mob.normalize();
977 return mob;
978 }
979 case 2: {
980 return trans_abAB(a, b, A, B, new Complex(0.0), new Complex(0.0));
981 }
982 case 0: {
983 int err_hit = 0;
984 Mobius M1, M2, mob, tmp;
985
986 mob = new Mobius();
987 M1 = standard_mob(a, b);
988 err_hit = M1.util;
989 tmp = standard_mob(A, B);
990 err_hit += tmp.util;
991 if (err_hit > 0) {
992 mob.util = 1;
993 mob.normalize();
994 return mob;
995 }
996 M2 = (Mobius) tmp.inverse();
997 mob = (Mobius) M2.rmultby(M1); // M = M2 * M1
998 mob.error = B.sub(mob.apply(b)).abs();
999 mob.normalize();
1000 return mob;
1001 }
1002 } // end of switch
1003 throw new MobException("failed to find method");

1021 Complex c, Complex C) throws MobException {

1024 if (a.abs() < MOD1 || b.abs() < MOD1 || A.abs() < MOD1
1025 || B.abs() < MOD1)
1026 return null;

1028 Mobius mob = new Mobius();

1117 public static Mobius norm_abc(Complex a, Complex b, Complex c)
1118 throws MobException {
1119
1120 Complex A, B, One, img;
1121 Mobius R, T, M, W;
1122
1123 // is data valid??
1124 if (a.abs() < MOD1 || b.abs() < MOD1 || c.abs() >= MOD1) {
1125 throw new MobException();
1126 }
1127 // make sure they're on the unit circle
1128 A = a.mult(a.abs());
1129 B = b.mult(b.abs());
1130 One = new Complex(1.0);
1131 // R to map unit disc to right half plane, 0->1, -A->0, A->infty.
1132 // i.e., R(z)=(z+A)/(-z+A)
1133 R = new Mobius(One, A, new Complex(-1.0), A);
1134 img = R.apply(B);
1135 // T translates by -img to put image of B at 0.
1136 T = new Mobius(One, new Complex(0.0, -img.y), new Complex(0.0), One);
1137 // compose these
1138 M = (Mobius) T.rmultby(R);
1139 // scale by reciprocal of image of c
1140 M = (Mobius) M.scale(M.apply(c).reciprocal());
1141 // W maps back to the disc: 1->0, 0->-1,infty->1; ie. z->(z-1)/(z+1)
1142 W = new Mobius(One, new Complex(-1.0), One, One);
1143 // compose for final map
1144 return (Mobius) W.rmultby(M);
1145 }

1341 Mobius M2 = new Mobius();

1391 /*
1392 * Non-antipodal: involves interesting geometry due to fact that sph
1393 * circle centers are not preserved under Mobious.
1394 */
1395 /*
1396 * Consider the plane containing the origin, zN, and zS. Intersection
1397 * with sphere gives disc bounded by great circle thru zN and zS; we
1398 * treat this as the unit disc and locate points via angular arguments
1399 * (sph distance). Going counterclockwise, locate pts w2, zS, w1, z1,
1400 * zN, z2, where the w's and z's are, resp., on the circles (zS,rS) and
1401 * (zN,rN). Let g be the hyp geodesic orthogonal to both the geodesics
1402 * w1 to w2 and z1 to z2. The endpoints of g on the unit circle are the
1403 * precise points which we want to send to N and S. They are on the
1404 * fixed point set of anticonformal involution of the disc interchanging
1405 * z1, z2 and w1, w2.
1406 */
1407
1408 /*
1409 * In the disc, the idea is to put w2 at 1 as base for measuring the
1410 * others using angles (sph distances). Then move to the upper
1411 * half-plane with T, where T(1)=infinity, T(w1)=0, and T(z1)=1. In the
1412 * half-plane, g is orthog to the imaginary axis and to the geo through
1413 * T(z1)=1 and T(z2)>1 (draw a picture!). g ends at points +/-
1414 * sqrt(T(z2)). Carry these back to unit circle under T^{-1} and then
1415 * onto sphere using angle and tangent direction from zS to zN.
1416 */
1417
1418 else {
1419 /*
1420 * locate points by putting w2 at 1, progress through ew1, ez1, ez2
1421 * counterclockwise using distances radii info.
1422 */
1423 ew1 = new Complex(0.0, 2 * rS).exp();
1424 ez1 = new Complex(0.0, dist + rS - rN).exp();
1425 ez2 = new Complex(0.0, dist + rS + rN).exp();
1426 /*
1427 * trans to upper half-plane is
1428 * T(z)=coef*fac=[(ez1-1.0)/(ez1-ew1)]*[(z-ew1)/(z-1.0)], and we
1429 * only need T(ez2), which should be positive.
1430 */
1431 coef = ez1.minus(MathComplex.ID).divide(ez1.minus(ew1));
1432 tmp = ez2.minus(ew1).divide(ez2.minus(MathComplex.ID));
1433 T_ez2 = coef.times(tmp);
1434 lam = Math.sqrt(T_ez2.x);
1435 /*
1436 * pts corresp to ends of g on real line are lam and -lam. Have to
1437 * map these back to unit circle under inverse of T, up to factor
1438 * this is TT(z)=(-z+ew1*coef)/(-z+coef), and then find their
1439 * arguments.
1440 */
1441 s = new Complex(-lam);
1442 tmp = s.add(ew1.times(coef)).divide(s.add(coef));
1443 if ((argz = (tmp.arg() - rS)) < -Math.PI)
1444 argz += 2.0 * Math.PI;
1445 s.x = lam;
1446 tmp = s.add(ew1.times(coef)).divide(s.add(coef));
1447 if ((argw = (tmp.arg() - rS)) < 0)
1448 argw += 2.0 * Math.PI;
1449 /*
1450 * argw and argz are the sph distance from zS on circle to desired
1451 * pts between w1 and w2 and between z1 and z2, resp.
1452 */
1453 V = SphericalMath.s_pt_to_vec(zS); // vector pointing to zS
1454 T = SphericalMath.sph_tangent(zS, zN); // tangent vector at zS in
1455 // direction zN
1456
1457 /*
1458 * here's the point, first in vec form, then projected to eucl
1459 * plane, which should go to the South pole. (only need a radius for
1460 * its sign -- tell if we want the outside.)
1461 */

1463 C[1] = V[1] * Math.cos(argw) + T[1] * Math.sin(argw);
1464 C[2] = Srad = V[2] * Math.cos(argw) + T[2] * Math.sin(argw);

1466 if (denom < MOB_TOLER) { // at one of the poles?
1467 Sctr = new Complex(0.0);
1468 Smod = 0.0;
1469 } else {
1470 Smod = denom / (1.0 + C[2]); // modulus of eucl center
1471 Sctr = new Complex(C[0] * Smod / denom, C[1] * Smod / denom);
1472 }
1473

1475 C[0] = V[0] * Math.cos(argz) + T[0] * Math.sin(argz);
1476 C[1] = V[1] * Math.cos(argz) + T[1] * Math.sin(argz);
1477 C[2] = Nrad = V[2] * Math.cos(argz) + T[2] * Math.sin(argz);
1478 denom = Math.sqrt(C[0] * C[0] + C[1] * C[1]);
1479 if (denom < MOB_TOLER) { // at one of the poles
1480 Nctr = new Complex(0.0);
1481 Nmod = 0.0;
1482 } else {
1483 Nmod = denom / (1.0 + C[2]); // modulus of eucl center
1484 Nctr = new Complex(C[0] * Nmod / denom, C[1] * Nmod / denom);
1485 }
1486

1488
1489 if (Nmod < MOB_TOLER || Smod < MOB_TOLER) { // at least one at a
1490 // pole
1491 if (Nmod < MOB_TOLER) {
1492 if (Smod < MOB_TOLER) { // Sctr is at other
1493 if (Nrad < 0) { // simply have to invert
1494 MM.a = new Complex(0.0);
1495 MM.b = new Complex(1.0);
1496 MM.c = new Complex(1.0);
1497 MM.d = new Complex(0.0);

1500 } else
1501 MM = new Mobius(); // identity
1502 } else {
1503 if (Nrad < 0) { /*
1504 * Nctr at south pole, need to invert
1505 * first, get new Sctr
1506 */

1508 M1.b = new Complex(1.0);

1519 M2.a = new Complex(0.0);

1530 * Sctr at north pole, need to invert first,

1552 } else { // generic case; move Nctr to origin, Sctr to infty

1554 MM.b = new Complex(-Nctr.x, -Nctr.y);
1555 MM.c = new Complex(1.0);
1556 MM.d = new Complex(-Sctr.x, -Sctr.y);

1558 MM.normalize();

1560 }

1564 mobius_of_circle(MM, 1, zN, rN, sc, true);

1566 mobius_of_circle(MM, 1, zS, rS, sc, true);

1571 mobius_of_circle(MM, 1, zE, rE, sc, true);

1587 phi = zzE.y + rrE;

1602 */

1608 * ab should be positive, but
1609 * fabs() is cautionary
1610 */
1611

1613 M1.b = new Complex(0.0);
1614 }
1615 M1.c = new Complex(0.0);
1616 M1.d = new Complex(1.0);
1617 M1.oriented = true;
1618 M1.normalize();
