90 public int cmdParser(String cmd,Vector<Vector<String>> flagSegs) {
91 Vector<String> items=null;
92
93 // ============ help ===============
94 if (cmd.startsWith("help")) {
95 helpInfo();
96 return 1;
97 }
98
99 // ============ export ============
100 if (cmd.startsWith("export")) {
101 try {
102 items=(Vector<String>)flagSegs.get(0);
103 int pnum=Integer.parseInt((String)items.get(0));
104 CPdrawing cpS=CPBase.cpDrawing[pnum];
105 if (cpS!=null) {
106 PackData p=extenderPD.copyPackTo();
107 return CirclePack.cpb.swapPackData(p, pnum, false).nodeCount;
108 }
109 } catch (Exception ex) {}
110 return 0;
111 }
112
113 errorMsg("|"+extensionType+"| command not found");
114 return 0;
115 }

137 public int swapExtenderPD(PackData newPD) {
138 if (newPD==null)
139 return 0;
140
141 CPdrawing holdcpS=extenderPD.cpDrawing;
142 int holdpnum=extenderPD.packNum;
143 extenderPD=newPD.copyPackTo();
144 extenderPD.packNum=holdpnum;
145 extenderPD.packExtensions=new Vector<PackExtender>(1);
146 extenderPD.packExtensions.add(this);
147 pdc=newPD.packDCEL;
148
149 // reconnect pack and screen
150 extenderPD.cpDrawing=holdcpS;
151 holdcpS.setPackData(extenderPD);
152 extenderPD.setGeometry(newPD.hes);
153
154 return extenderPD.nodeCount;
155 }
