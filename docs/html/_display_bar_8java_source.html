137 public void mouseClicked(MouseEvent evt) {
138
139 // for angsum bar, allow setting any aim up to 5*Pi.
140 if (evt.getButton() == MouseEvent.BUTTON3 || (evt.getButton()==MouseEvent.BUTTON1 &&
141 (evt.getModifiersEx() & ActionEvent.SHIFT_MASK)==
142 ActionEvent.SHIFT_MASK)) {
143 if (!mode) { // for aims
144 if (parent.lock[vertNum]) { // rad is locked
145 CirclePack.cpb.msg("Unlock radius to reset aim.");
146 return;
147 }
148 int y = evt.getPoint().y;
149 double newAim = (double) (bar_length - y) / factor;
150 parent.setValue(newAim, vertNum, mode);
151 placePointer(newAim);
152 parent.displayAngError();
153 } else { // for radii, this means to toggle lock state and color
154 if (parent.lock[vertNum]) {
155 parent.lock[vertNum] = false;
156 barArea.setBackground(Color.blue);
157 } else {
158 parent.lock[vertNum] = true;
159 barArea.setBackground(Color.magenta); // new
160 // Color(220,220,255));
161 }
162 }
163 }
164
165 else if (evt.getButton() == MouseEvent.BUTTON1) {
166 int y = evt.getPoint().y;
167 double value;
168 if (!mode) { // on left button, only allow 2*Pi or 4*Pi.
169 if (parent.lock[vertNum]) { // rad is locked
170 return;
171 }
172 value = (double) (bar_length - y) / factor;
173 if (value > 3 * Math.PI)
174 value = 4.0 * Math.PI;
175 else
176 value = 2.0 * Math.PI;
177 parent.packData.setAim(vertNum,value);
178 placePointer(value);
179 }
180 else { // for radii
181 // convert (reverse) to log factor
182 double rbl = (double) SphWidget.RAD_BAR_HEIGHT;
183 value = (double) (y * SphWidget.RAD_BAR_POWER * Math.log(2.0))
184 / rbl;
185 // System.err.println("y = "+y+" and value =
186 // "+String.format("%.8e",value));
187 value = Math.PI * (1.0 / Math.exp(value));
188
189 if (parent.lock[vertNum]) { // rad is locked
190 CirclePack.cpb.msg("Vert "+vertNum+" is locked");
191 return;
192 }
193 parent.setValue(value, vertNum, mode);
194 setBarHeight(value);
195 parent.displayArea();
196 }
197 parent.displayAngError();
198 }
