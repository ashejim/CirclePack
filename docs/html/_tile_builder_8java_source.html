665 public void updateTileVerts(TileData td,VertexMap vmap) {
666 for (int t=1;t<=td.tileCount;t++) {
667 Tile tile=td.myTiles[t];

669 continue;
670
671 // adjust vert
672 for (int j=0;j<tile.vertCount;j++) {
673 int newv=vmap.findW(tile.vert[j]);
674 if (newv>0)
675 tile.vert[j]=newv;

677
678 // adjust augVert
679 if (tile.augVert!=null) {
680 for (int j=0;j<tile.augVertCount;j++) {
681 int newv=vmap.findW(tile.augVert[j]);

683 tile.augVert[j]=newv;
684 }
685 }
686
687 // adjust baryVert

689 if (newv>0)
690 tile.baryVert=newv;

703
704 Tile tile=null;
705 Tile otile=null;
706
707 for (int t=1;t<=growTD.tileCount;t++) {
708 tile=growTD.myTiles[t];
709 otile=null;

712 if (tile!=null) {
713 otile=origTD.myTiles[tile.tileIndex];

770 public boolean swallowOneUnigon() {
771
772 Tile tile=null;
773 Tile otile=null;
774
775 for (int t=1;t<=growTD.tileCount;t++) {
776 tile=growTD.myTiles[t];
777 otile=null;
778
779 if (tile != null) { // this should imply corresponding origTD tile is not null
780 otile=origTD.myTiles[tile.tileIndex];
781 int ecount = tile.vertCount;
782 for (int e = 0; e < ecount; e++) {
783
784 // skip if edge barycenter vert is not on bdry of tPack
785 int eb = tile.augVert[4*e+2];
786 if (masterPack.isBdry(eb))
787 continue;
788
789 // nghb invalid or already incorporated?
790 int nghb = otile.tileFlower[e][0];
791 if (nghb == 0 || tileAdded[nghb] > 0)
792 continue;
793
794 // do edge ends match in origTD?
795 if (otile.vert[e] == otile.vert[(e + 1) % ecount]) {
796 PackData newp = otile.singleCanonical(3);
797 int w=newp.tileData.myTiles[1].vert[0];
798 int v=tile.vert[(e+1)%ecount]; // need the next vertex to go clockwise
799
800 // 'adjoin' is complicated. If ends already match in 'tile'
801 // then just usual adjoin
802 if (tile.vert[e]==v) {
803 masterPack.packDCEL=CombDCEL.adjoin(
804 masterPack.packDCEL,newp.packDCEL, v,w,4);
805 masterPack.vertexMap=masterPack.packDCEL.oldNew;
806 // fix the new tile data
807 updateTileVerts(newp.tileData,masterPack.vertexMap);
808 growTD.myTiles[t] = newp.tileData.myTiles[1];
809 if (copyFlowers(origTD.myTiles[t],growTD.myTiles[t])<=0)
810 throw new ParserException("failed in loop work to copy tileFlower");
811 growTD.myTiles[t].tileIndex = 1;
812 tileAdded[t] = 1;
813 }
814 else { // must be done in two steps:
815 // (1) adjoin along 3 edges only: update tile verts,
816 // add to growTD, transfer flowers, etc.
817 // (2) then zip last edge: update growTD
818 // this will become tip of final slit
819 int tip=newp.tileData.myTiles[1].augVert[3];
820 masterPack.packDCEL=CombDCEL.adjoin(
821 masterPack.packDCEL,newp.packDCEL, v,w,3);
822 masterPack.vertexMap=masterPack.packDCEL.oldNew;
823 tip=masterPack.vertexMap.findW(tip);
824
825 // fix the new tile data
826 updateTileVerts(newp.tileData,masterPack.vertexMap);
827 growTD.myTiles[t] = newp.tileData.myTiles[1];
828 if (copyFlowers(origTD.myTiles[t],growTD.myTiles[t])<=0)
829 throw new ParserException("failed in loop work to copy tileFlower");
830 growTD.myTiles[t].tileIndex = 1;
831 tileAdded[t] = 1;
832 masterPack.packDCEL=CombDCEL.adjoin(
833 masterPack.packDCEL,masterPack.packDCEL,tip,tip,1);
834 masterPack.vertexMap=masterPack.packDCEL.oldNew;
835 updateTileVerts(growTD,masterPack.vertexMap);
836 }
837 return true;
838 }
839 }
840 }
841 }
842 return false;

919 public static int []tilePastings(TileData tData,Tile t1,Tile t2) {
920 int []ans=new int[4];
921 if (t1.tileFlower==null || t2.tileFlower==null)
922 return null;
923
924 // check if t1 and t2 share full boundaries
925 boolean t1full=true;
926 boolean t2full=true;
927 for (int j=0;(j<t1.vertCount && t1full);j++) {
928 int ti=t1.vert[j];
929 if (ti!=t2.tileIndex)
930 t1full=false;
931 }
932 for (int j=0;(j<t2.vertCount && t2full);j++) {
933 int ti=t2.vert[j];
934 if (ti!=t1.tileIndex)
935 t2full=false;
936 }
937
938 if (t1full && t2full) {
939
940 // TODO: if t1 is single cell complementary to a tree on the sphere,
941 // then need more information to decide where to start pasting.
942 if (t1.tileIndex==t2.tileIndex)
943 throw new CombException("tile share all edges with itself; ambiguity, tree complement");
944 if (t1.vertCount!=t2.vertCount)
945 throw new CombException("tiles share all edges, but not equal");
946 ans[0]=ans[2]=0;
947 ans[1]=ans[3]=t1.vertCount;
948 return ans;
949 }
950 else if (t1full || t2full) // one is full, but not both
951 throw new CombException("ambiguity: where to attach if all edges involved?");
952
953 // find largest index of edge t1 shares with t2
954 int hit2=-1;
955 for (int j=(t1.vertCount-1);(j>=0 && hit2<0);j--) {
956 int ti=t1.vert[j];
957 if (ti==t2.tileIndex)
958 hit2=j;
959 }
960 // find largest index of edge t2 shares with t1
961 int hit1=-1;
962 for (int j=0;(j<t2.vertCount && hit2<0);j--) {
963 int ti=t1.vert[j];
964 if (ti==t2.tileIndex)
965 hit2=j;
966 }
967
968 // do t1 analysis first.
969 if (hit2==t1.vertCount) { // this is last edge, look ccw
970 int tick=0;
971 while ((tick<(t1.vertCount-1)) && Math.abs(t1.tileFlower[tick][0])==t2.tileIndex)
972 tick++;
973 hit2=tick;
974 }
975
976 // now count number of clockwise edges shared with t2
977 int t1length=0;
978 int ptr=(hit2-1+t1.vertCount)%t1.vertCount;
979 while (Math.abs(t1.tileFlower[ptr][0])==t2.tileIndex) {
980 t1length++;
981 ptr=(ptr-1+t1.vertCount)%t1.vertCount;
982 }
983
984
985 // now do t2
986 if (hit1==0 && Math.abs(t2.tileFlower[t2.vertCount-1][0])==t1.tileIndex) { // need to back up
987 hit1=t2.vertCount-1;
988 while (Math.abs(t2.tileFlower[(hit1-1+t2.vertCount)%t2.vertCount][0])==t1.tileIndex)
989 hit1=(hit1-1+t2.vertCount)%t2.vertCount;
990 }
991
992 // count number of ccw edges shared with t1
993 int t2length=0;
994 ptr=hit1;
995 while (Math.abs(t2.tileFlower[ptr][0])==t1.tileIndex) {
996 t2length++;
997 ptr=(ptr+1)%t2.vertCount;
998 }
999
1000
1001 // some checks are these the same tile in 'tData'
1002 // TODO: this only allows for pasting of a slit
1003 if (t1.tileIndex==t2.tileIndex) {
1004 // lengths must be even and the same
1005 if (t1length!=t2length || 2*(int)(t1length/2)!=t1length) {
1006 t1length=t1length/2;
1007 t2length=t2length/2;
1008 }
1009 else
1010 throw new CombException("self pasting problem");
1011 }
1012
1013 // For t1, start at 'vert[ans[0]]' (end vertex for first edge shared with t2)
1014 // and go clockwise ans[1] edges.
1015 ans[0]=hit2;
1016 ans[1]=t1length;
1017 // For t2, start at 'vert[ans[2]]' (beginning vert for first edge shared with t1)
1018 // and go ccw ans[3] edges.
1019 ans[2]=hit2;
1020 ans[3]=t1length;
1021
1022 return ans;
1023 }

1040 public static int tileflowers2verts(TileData td) {
1041
1042 // bad data?
1043 if (td.myTiles==null || td.tileCount<=0)
1044 return -1;
1045 for (int t=1;t<=td.tileCount;t++) {
1046 Tile tile=td.myTiles[t];
1047 if (tile.tileFlower==null)
1048 return -1;
1049 td.myTiles[t].vert=new int[tile.vertCount]; // toss old indices
1050 }
1051
1052 // process tile-by-tile, finishing all vertices of each
1053 // before going on.
1054 // Within a tile, proceed through the vertices in order,
1055 // labeling each vertex as its full tile neighborhood is
1056 // completed.
1057 int[] doneTiles = new int[td.tileCount + 1]; // 1 when tile is done
1058
1059 // Alternate between curr/next lists of tiles visited.
1060 NodeLink next=new NodeLink();
1061 NodeLink curr = new NodeLink();
1062 int tick = 1;
1063 next.add(tick);
1064 while (next.size() > 0) {
1065 curr = next;
1066 next = new NodeLink();
1067 int vert = 0;
1068 Iterator<Integer> tls = curr.iterator();
1069 while (tls.hasNext()) {
1070 Tile tile = td.myTiles[tls.next()];
1071 if (doneTiles[tile.tileIndex] > 0)
1072 continue;
1073 int vcount = tile.vertCount;
1074
1075 for (int i = 0; i < vcount; i++) {
1076 if (tile.vert[i]<=0) {
1077 vert = tile.vert[i] = tick++; // set the index
1078
1079 // proceed ccw, starting with edge before vert
1080 int myedge = (i - 1 + vcount) % vcount;
1081 Tile prevtile = tile;
1082 int nextIndex = -1;
1083
1084 // go until we get to bdry or return to 'tile' across edge 'i'.
1085 while (prevtile != null
1086 && (nextIndex = prevtile.tileFlower[myedge][0]) != 0
1087 && (nextIndex != tile.tileIndex || prevtile.tileFlower[myedge][1]!=i)) {
1088 int nghbedge = prevtile.tileFlower[myedge][1];
1089 int prevIndex = prevtile.tileIndex;
1090 Tile nghbtile = td.myTiles[nextIndex];
1091 next.add(nextIndex);
1092 int nghbcount = nghbtile.vertCount;
1093 // check compatibility
1094 if (nghbtile.tileFlower[nghbedge][0] != prevIndex
1095 || nghbtile.tileFlower[nghbedge][1] != myedge)
1096 throw new CombException("faulty tile nghb relationship");
1097
1098 // here's goal: set correct index in nghbtile.vert
1099 nghbtile.vert[nghbedge] = vert;
1100 // Is nghbtile a unigon? done
1101 if (nghbcount == 1)
1102 break;
1103
1104 myedge = (nghbedge - 1 + nghbcount) % nghbcount;
1105 prevtile = nghbtile;
1106 } // end of while through counterclockwise
1107
1108 // if not done, go clockwise about v
1109 if (nextIndex != tile.tileIndex) {
1110 myedge = i;
1111 prevtile = tile;
1112
1113 // in this direction we must eventually hit the bdry
1114 while (prevtile != null
1115 && (nextIndex = prevtile.tileFlower[myedge][0]) != 0) {
1116 int nghbedge = prevtile.tileFlower[myedge][1];
1117 int oldindex = prevtile.tileIndex;
1118 Tile nghbtile = td.myTiles[nextIndex];
1119 next.add(nextIndex);
1120 int nghbcount = nghbtile.vertCount;
1121
1122 // check compatibility
1123 if (nghbtile.tileFlower[nghbedge][0] != oldindex
1124 || nghbtile.tileFlower[nghbedge][1] != myedge)
1125 throw new CombException("faulty tile nghb relationship");
1126
1127 // here's goal: set correct index in nghbtile.vert
1128 nghbtile.vert[(nghbedge + 1) % nghbcount] = vert;
1129
1130 // Is nghbtile a unigon? done
1131 if (nghbcount == 1)
1132 break;
1133
1134 myedge = (nghbedge + 1) % nghbcount;
1135 prevtile = nghbtile;
1136 } // end of while through clockwise
1137 }
1138 } // end of for through vertices
1139
1140 doneTiles[tile.tileIndex] = 1;
1141 }
1142 } // end of while going through curr
1143 } // done, 'next' is now empty
1144
1145 return tick-1;
1146 }
