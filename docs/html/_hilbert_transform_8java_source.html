95 public int cmdParser(String cmd, Vector<Vector<String>> flagSegs)
96 {
97 String str;
98
99 if (cmd.startsWith("set_fct"))
100 {
101 // set the function to be transformed
102 Iterator<Vector<String>> iter = flagSegs.iterator();
103
104 while(iter.hasNext())
105 {
106 Vector<String> items = iter.next();
107 str = items.remove(0);
108 char c = str.charAt(1);
109
110 switch (c)
111 {
112 case 'z':
113 {
114 var = "z";
115 fct = items.remove(0);
116 }
117 case 't':
118 {
119 var = "t";
120 fct = items.remove(0);
121 }
122 }
123 }
124
125 CirclePack.cpb.FtnSpecification=new StringBuilder(fct);
126 return 1;
127 }
128
129 else if (cmd.startsWith("calc_Hilbert_transform")
130 || cmd.startsWith("calc_HT"))
131 {
132 // calculate the Hilbert transform
133 // is there a lambda given?
134 double lambda = 1;
135 if (flagSegs != null)
136 {
137 Vector<String> item = flagSegs.get(0);
138 Iterator<String> items = item.iterator();
139 String option = items.next();
140 if (option.compareToIgnoreCase("-l") == 0)
141 {
142 if (items.hasNext())
143 lambda = Double.valueOf(items.next());
144 else
145 lambda = 1;
146 }
147 }
148
149 // create a copy of the packing
150 PackData newPack = extenderPD.copyPackTo();
151 CirclePack.cpb.swapPackData(newPack, 1, false);
152
153 // modify the radii of the boundary circles
154 if (CPBase.GUImode!=0)
155 PackControl.newftnFrame.setFunctionText();
156 Complex val;
157 for(int k = 0; k < sz_bndry; k++)
158 {
159 val = CirclePack.cpb.getFtnValue(cz_hyper[k]);
160 double krad=newPack.getRadius(k);
161 newPack.setRadius(bndry[k],krad* Math.exp(lambda * val.real()));
162 values_fct[k] = val.real();
163 }
164
165 // copy the new packing to the panel 1 and then repack it
166 cpCommand(newPack, "repack");
167 cpCommand(newPack, "layout");
168 cpCommand(newPack, "set_screen -a");
169 cpCommand(newPack, "disp -w -c");
170 cpCommand(extenderPD, "Map 0 1 -o");
171
172 // calculate the change of the angle of the boundary edges
173 Complex z1, z2, z3, w1, w2, w3, z, w;
174 double arg1, arg2, shift;
175 double mean = 0;
176 double tau1, tau2;
177 //values_HilbTransform = new double[sz_bndry];
178 //tau = new double[sz_bndry];
179 for(int k = 0; k < sz_bndry; k++)
180 {
181 if (k <= 0)
182 z1 = extenderPD.getCenter(bndry[sz_bndry - 1]);
183 else
184 z1 = extenderPD.getCenter(bndry[k - 1]);
185 z2 = extenderPD.getCenter(bndry[k]);
186 z3 = extenderPD.getCenter(bndry[(k + 1) % sz_bndry]);
187
188 z = z1.minus(z2);
189 w = z3.minus(z2);
190 z = z.divide(w);
191 shift = Math.atan2(z.y, z.x);
192 if (shift < 0)
193 shift += 2 * Math.PI;
194 arg1 = Math.atan2(w.y, w.x) + shift / 2;
195 arg1 = Math.atan2(Math.sin(arg1), Math.cos(arg1)); // normalize to [-pi, pi]
196
197 if (k <= 0)
198 w1 = newPack.getCenter(bndry[sz_bndry - 1]);
199 else
200 w1 = newPack.getCenter(bndry[k - 1]);
201 w2 = newPack.getCenter(bndry[k]);
202 w3 = newPack.getCenter(bndry[(k + 1) % sz_bndry]);
203
204 z = w1.minus(w2);
205 w = w3.minus(w2);
206 z = z.divide(w);
207 shift = Math.atan2(z.y, z.x);
208 if (shift < 0)
209 shift += 2 * Math.PI;
210 arg2 = Math.atan2(w.y, w.x) + shift / 2;
211 arg2 = Math.atan2(Math.sin(arg2), Math.cos(arg2)); // normalize to [-pi, pi]
212
213 // elimate a possible jump in the argument
214 if ((arg1 >= 3 * Math.PI / 4) && (arg2 <= - 3 * Math.PI / 4))
215 arg2 += 2 * Math.PI;
216 if ((arg2 >= 3 * Math.PI / 4) && (arg1 <= - 3 * Math.PI / 4))
217 arg1 += 2 * Math.PI;
218
219 values_HilbTransform[k] = (arg2 - arg1) / lambda;
220
221 tau1 = Math.atan2(cz_hyper[k].y, cz_hyper[k].x);
222 tau[k] = tau1;
223 tau2 = Math.atan2(cz_hyper[(k + 1) % sz_bndry].y, cz_hyper[(k + 1) % sz_bndry].x);
224 // mean value of the Hilbert transform
225 if ((tau2 - tau1) < 0)
226 mean += (arg2 - arg1) * (tau2 - tau1 + 2 * Math.PI) / 2 / Math.PI;
227 else
228 mean += (arg2 - arg1) * (tau2 - tau1) / (2 * Math.PI);
229 }
230 for(int k = 0; k < sz_bndry; k++)
231 values_HilbTransform[k] -= mean / lambda;
232
233 return 1;
234 }
235
236 else if (cmd.startsWith("show"))
237 {
238 // show the function and its Hilbert transform
239 Dimension scrsz = Toolkit.getDefaultToolkit().getScreenSize();
240 int left = (scrsz.width - 670) / 2;
241 int top = (scrsz.height - 670) / 2;
242
243 if (frame == null)
244 frame = new ShowFrame(tau, values_fct, values_HilbTransform);
245 else
246 frame.setValues(tau, values_fct, values_HilbTransform);
247
248 frame.setTitle("Hilbert transform (red) and original function (blue)");
249 frame.setSize(670, 670);
250 frame.setLocation(left, top);
251 frame.setResizable(false);
252 frame.setVisible(true);
253 frame.repaint();
254
255 return 1;
256 }
257
258 return super.cmdParser(cmd, flagSegs);
259 }

298 private void sortBoundary()
299 {
300 // sort the boundary such that bndry[k] and bndry[k+1] are neighbors
301
302 int[] nb;
303 for(int k = 1; k < sz_bndry; k++)
304 {// the first is unchanged
305 nb = BoundaryNeighbors(bndry[k - 1]);
306
307 if (k <= 1)
308 // decide for one direction
309 bndry[k] = nb[0];
310 else if (Math.abs(bndry[k - 2] - nb[0]) <= 0)
311 // check which of the neighbors is not yet considered
312 bndry[k] = nb[1];
313 else
314 bndry[k] = nb[0];
315 }
316 }

318 private int[] BoundaryNeighbors(int vert)
319 {
320 // determine which neighbor of vert belongs to the boundary
321 int[] nb = new int[2];
322 int sz_flwr;
323 int count = 0;
324
325 int[] flwr = extenderPD.getFlower(vert);
326 if (extenderPD.isBdry(vert))
327 // vert is a boundary vertex
328 sz_flwr = flwr.length;
329 else
330 sz_flwr = flwr.length - 1;
331
332 for(int k = 0; k < sz_flwr; k++) {
333 if (extenderPD.isBdry(flwr[k])) {
334 nb[count] = flwr[k];
335 count++;
336 }
337 }
338 return nb;
339 }
