66 public int readCurves(BufferedReader fp,boolean addon) {
67 if (fp==null) return 0;
68 if (!addon) // restart;
69 restCurves=new Vector<RH_curve>(50);
70 String line;
71 int safety=10000;
72 Vector<Complex> pts=null;
73 ReadState state=ReadState.FIND_CURVE;
74 String str=null;
75 String xstr=null;
76
77 // look for 'PATH' or 'CIRCLE' (may reenter in one of these modes already)
78 while(safety>0 && (state==ReadState.FIND_CURVE
79 || state==ReadState.PATH || state==ReadState.CIRCLE)
80 && (line=StringUtil.ourNextLine(fp))!=null) {
81 StringTokenizer tok = new StringTokenizer(line);
82 if (state==ReadState.FIND_CURVE) {
83 while (tok.hasMoreTokens()) {
84 str=(String)tok.nextToken();
85 if (str.startsWith("PATH:")) {
86 state=ReadState.PATH;
87 }
88 else if (str.startsWith("CIRCLE"))
89 state=ReadState.CIRCLE;
90 }
91 }
92 if (state==ReadState.PATH) {
93 state=ReadState.READ_XY;
94 while (safety>0 && state==ReadState.READ_XY
95 && (line=StringUtil.ourNextLine(fp))!=null) {
96 tok = new StringTokenizer(line);
97 while(tok.hasMoreTokens()) {
98 // expect two doubles per line
99 try {
100 xstr=(String)tok.nextToken().trim();
101 double x=Double.parseDouble(xstr);
102 double y=Double.parseDouble((String)tok.nextToken());
103 if (pts==null) { // new vector
104 pts=new Vector<Complex>(25);
105 pts.add(new Complex(x,y));
106 }
107 else {
108 pts.add(new Complex(x,y));
109 }
110 } catch (NumberFormatException nfe) { // 'xstr' should have last string
111 if (xstr.startsWith("PATH"))
112 state=ReadState.PATH;
113 else if (xstr.startsWith("CIRCLE"))
114 state=ReadState.CIRCLE;
115 else state=ReadState.FIND_CURVE;
116 }
117 }
118 } // end of while
119 if (safety<=0) {
120 CirclePack.cpb.myErrorMsg("RH error in reading a curve.");
121 return 0;
122 }
123 if (pts!=null && pts.size()>=3) {
124 RH_curve rhc=new RH_curve(pts);
125 restCurves.add(rhc);
126 }
127 pts=null;
128 } // end of PATH case
129 else if (state==ReadState.CIRCLE) {
130 tok = new StringTokenizer(line);
131 while(tok.hasMoreTokens()) {
132 // expect 'x y rad'
133 try {
134 xstr=(String)tok.nextToken().trim();
135 double x=Double.parseDouble(xstr);
136 double y=Double.parseDouble((String)tok.nextToken());
137 double rad=Double.parseDouble((String)tok.nextToken());
138 Complex z=new Complex(x,y);
139 RH_curve rhc=new RH_curve(z,rad);
140 restCurves.add(rhc);
141 state=ReadState.FIND_CURVE;
142 } catch (NumberFormatException nfe) { // 'xstr' should have last
143 if (xstr.startsWith("PATH"))
144 state=ReadState.PATH;
145 else if (xstr.startsWith("CIRCLE"))
146 state=ReadState.CIRCLE;
147 else state=ReadState.FIND_CURVE;
148 }
149 } // end of while
150 }
151 } // end of outer 'while'
152 return restCurves.size();
153 }

199 public double centerDistance(int v) {
200 RH_curve rhc=curveForVert(v);
201 Complex cent=null;
202 if (extenderPD.hes<0) {
203 CircleSimple sc =HyperbolicMath.h_to_e_data(
204 extenderPD.getCenter(v),extenderPD.getRadius(v));
205 // what if sc.flag==-1? outside as disc
206 cent=sc.center;
207 }
208 else if (extenderPD.hes>0) {
209 CircleSimple sc =SphericalMath.s_to_e_data(
210 extenderPD.getCenter(v),extenderPD.getRadius(v));
211 cent=sc.center;
212 }
213 else cent=extenderPD.getCenter(v);
214 if (rhc.isCircle) {
215 return rhc.rad-rhc.center.minus(cent).abs();
216 }
217 double dist=PathUtil.gpDistance(rhc.restCurve,cent);
218 if (rhc.restCurve.contains(cent.x,cent.y)) return dist;
219 return -dist;

228 public double circleDistance(int v) {
229 RH_curve rhc=curveForVert(v);
230 Complex cent=null;
231 if (extenderPD.hes<0) {
232 CircleSimple sc =HyperbolicMath.h_to_e_data(
233 extenderPD.getCenter(v),extenderPD.getRadius(v));
234 cent=sc.center;
235 }
236 else if (extenderPD.hes>0) {
237 CircleSimple sc =SphericalMath.s_to_e_data(
238 extenderPD.getCenter(v),extenderPD.getRadius(v));
239 // what if sc.flag==-1? outside as disc
240 cent=sc.center;
241 }
242 else cent=extenderPD.getCenter(v);
243 double radius=extenderPD.getRadius(v);
244 double dist;
245 if (rhc.isCircle) {
246 dist=radius+rhc.center.minus(cent).abs();
247 return rhc.rad-dist;
248 }
249 dist=PathUtil.gpDistance(rhc.restCurve,cent);
250 if (dist>=0.0) return dist-radius;
251 return dist-radius;

266 NodeLink bdrylist=null;

268 CirclePack.cpb.myErrorMsg("RiemHilbert: the packing does not have boundary");

273 restCurves=new Vector<RH_curve>(50);
274 restCurves.add(defaultCurve);
275 }
276 if (v<0 || v>extenderPD.nodeCount || !extenderPD.isBdry(v)) {
277 v=extenderPD.bdryStarts[1];
278 }
279
280 /* proceed around the boundary; use as many curves as needed, clone
281 * curves as necessary if there are more bdry vertices.*/
282 vertCurve=new VertexMap();
283 Iterator<Integer> blist=bdrylist.iterator();
284 int w;
285 int count=0;
286 int num=restCurves.size();
287 while (blist.hasNext()) {
288 w=(Integer)blist.next();
289 extenderPD.setCircleColor(w,ColorUtil.spreadColor(count%16));
290 if (count>=num) // need to clone to get new 'RH_curve's
291 restCurves.add(restCurves.get(count%num).clone());
292 vertCurve.add(new EdgeSimple(w,count));
293 Color col=extenderPD.getCircleColor(w);
294 restCurves.get(count).color=new Color(col.getRed(),col.getGreen(),col.getBlue());
295 count++;
296 }

371 int old_thickness=extenderPD.cpDrawing.linethickness;

391 }
392

396 extenderPD.getRadius(v), dflags);

417 public int cmdParser(String cmd,Vector<Vector<String>> flagSegs) {
418 int count=0;
419 boolean script_flag=false;
420 Vector<String> items=null;
421
422 // ============ read_curves ===============
423 if (cmd.startsWith("read_cur") || cmd.startsWith("infile_cur")) {
424 if (cmd.startsWith("i")) script_flag=true;
425 StringBuilder namebuf=new StringBuilder("");
426 CPFileManager.trailingFile(flagSegs,namebuf);
427 File file=new File(namebuf.toString());
428 String dir=file.getParent();
429 if (dir==null)
430 dir=CPFileManager.CurrentDirectory.toString();
431 try {
432 BufferedReader fp=CPFileManager.openReadFP(new File(dir),file.getName(),script_flag);
433 count=readCurves(fp,false); // for now, overwrite previous curves.
434 fp.close();
435 if (count>0 && extenderPD.status) linkPackCurves();
436 return count;
437 } catch (Exception ex) {
438 return 0;
439 }
440 }
441
442 // ============= link =================
443 // designate starting boundary vertex for linking
444 else if (cmd.startsWith("link")) {
445 int v;
446 try {
447 v=NodeLink.grab_one_vert(extenderPD,flagSegs);
448 if (v!=0 && extenderPD.isBdry(v))
449 return linkPackCurves(v);
450 } catch (Exception ex) {}
451 return linkPackCurves(extenderPD.bdryStarts[1]);
452 }
453
454 // ============= draw_curves ===========
455 else if (cmd.startsWith("draw_cur")) {
456 if (flagSegs==null || flagSegs.size()==0)
457 return drawRestCurves(new NodeLink(extenderPD,"b")); // default to all
458 items=(Vector<String>)flagSegs.get(0);
459 return drawRestCurves(new NodeLink(extenderPD,items));
460 }
461
462 // ============= shade =========
463 else if (cmd.startsWith("shade")) {
464 if (flagSegs==null || flagSegs.size()==0)
465 return shadeCircles(new NodeLink(extenderPD,"b")); // default to all
466 items=(Vector<String>)flagSegs.get(0);
467 return shadeCircles(new NodeLink(extenderPD,items));
468 }
469
470 // ============= scale =========
471 else if (cmd.startsWith("scale")) {
472 items=(Vector<String>)flagSegs.get(0);
473 double factor=1.0;
474 try {
475 factor=Double.parseDouble((String)items.remove(0));
476 if (factor<=0) return 0;
477 if (items.size()>0)
478 return scaleCurves(new NodeLink(extenderPD,items),factor);
479 else return scaleCurves(new NodeLink(extenderPD,"b"),factor);
480 } catch (Exception ex) {}
481 }
482
483 // ============= rotate =========
484 else if (cmd.startsWith("rotate")) {
485 if (flagSegs.size()==0) return 0;
486 items=(Vector<String>)flagSegs.get(0);
487 double arg=0.0;
488 try {
489 arg=Double.parseDouble((String)items.remove(0));
490 if (items.size()>0)
491 return rotateCurves(new NodeLink(extenderPD,items),arg);
492 else return rotateCurves(new NodeLink(extenderPD,"b"),arg);
493 } catch (Exception ex) {}
494 return 0;
495 }
496
497 return super.cmdParser(cmd, flagSegs);
498 }
