47 public static ArrayList<Complex> layoutPolygon(PackDCEL pdcel, HalfLink hlink, int hes) {
48 if (hlink == null || hlink.size() <= 1)
49 return null;
50 ArrayList<Complex> Zlist = new ArrayList<Complex>();
51
52 // check if any red vertices are involved
53 boolean nored = true;
54 Iterator<HalfEdge> his = hlink.iterator();
55 while (his.hasNext() && nored) {
56 HalfEdge he = his.next();
57 if (he.origin.redFlag)
58 nored = false;
59 }
60
61 // easy case, accumulate centers
62 if (nored) {
63 his = hlink.iterator();
64 while (his.hasNext()) {
65 HalfEdge he = his.next();
66 Zlist.add(pdcel.getVertCenter(he));
67 }
68 // if not closed, add last vertex
69 if (hlink.getLast().twin.origin != hlink.getFirst().origin)
70 Zlist.add(pdcel.getVertCenter(hlink.getLast().next));
71 return Zlist;
72 }
73
74 // hard case; need layout to recompute centers
75
76 // if closed, don't need last edge
77 if (hlink.getLast().twin.origin == hlink.getFirst().origin)
78 hlink.removeLast();
79 HalfLink leftlink = HalfLink.leftsideLink(pdcel, hlink);
80 if (leftlink == null)
81 return null;
82
83 // create triAspects for these faces
84 ArrayList<TriAspect> aspectlist = new ArrayList<TriAspect>();
85 his = leftlink.iterator();
86 while (his.hasNext()) {
87 aspectlist.add(new TriAspect(pdcel, his.next().face));
88 }
89
90 // start with both ends of the first edge
91 HalfEdge firsthe = hlink.getFirst();
92 Vertex firstV = firsthe.origin;
93 Zlist.add(pdcel.getVertCenter(firsthe));
94 Zlist.add(pdcel.getVertCenter(firsthe.next));
95
96 // call routine to layout the chain of TriAspects
97 int j = aspectlist.get(0).vertIndex(firstV.vertIndx);
98 layTriAspects(aspectlist, j);
99
100 // look for successive ends of hlink edges in aspectlist
101 his = hlink.iterator();
102 his.next(); // toss first edge
103// Iterator<HalfEdge> llits=leftlink.iterator();
104 Iterator<TriAspect> tais = aspectlist.iterator();
105 while (his.hasNext()) {
106 HalfEdge he=his.next();
107 int V = he.next.origin.vertIndx; // next vert to find
108 boolean hit = false;
109 while (tais.hasNext() && !hit) {
110 TriAspect triaspect = tais.next();
111 int k = -1;
112 if ((k = triaspect.vertIndex(V)) >= 0) {
113 hit = true;
114 Zlist.add(triaspect.center[k]);
115 }
116 }
117 if (!hit) {
118 throw new CombException("didn't find center for V = "+V+" in" +
119 "half edge "+he+" among TriAspects");
120 }
121 }
122
123 return Zlist;
124 }

135 public static void layTriAspects(ArrayList<TriAspect> aspects, int j) {
136 Iterator<TriAspect> tais = aspects.iterator();
137 TriAspect nextasp = tais.next();
138 TriAspect lastasp = nextasp;
139
140 // set only the third center of first TriAspect
141 CircleSimple cs = lastasp.compOppCircle(j,false);
142 lastasp.center[(j + 2) % 3] = cs.center;
143
144 // now successively adjust the next TriAspect's centers
145 while (tais.hasNext()) {
146 TriAspect asp = tais.next(); // compute centers for this
147 int J = asp.nghb_Tri(lastasp);
148 if (J < 0)
149 throw new CombException("somehow, chain of aspects is broken," + " face" + asp.faceIndx);
150 int v = asp.vert[J];
151 int w = asp.vert[(J + 1) % 3];
152 int oldv = lastasp.vertIndex(v);
153 int oldw = lastasp.vertIndex(w);
154 asp.center[J] = new Complex(lastasp.center[oldv]); // reset shared centers
155 asp.center[(J + 1) % 3] = new Complex(lastasp.center[oldw]);
156 cs = asp.compOppCircle(J,false);
157 asp.center[(J + 2) % 3] = cs.center;
158 lastasp = asp;
159 }
160 }

173 public static int layoutFaceList(PackDCEL pdcel,FaceLink facelist,
174 int hes,boolean useSchw) {
175 if (facelist == null || facelist.size() == 0)
176 return 0;
177 Iterator<Integer> fis = facelist.iterator();
178 DcelFace last_face = pdcel.faces[fis.next()];
179 DcelFace next_face =last_face;
180 while (fis.hasNext()) {
181 last_face=next_face;
182 int g = fis.next(); // next face
183 next_face = pdcel.faces[g];
184 HalfEdge he = last_face.faceNghb(next_face).twin;
185
186 // shuck non-neighbors
187 while (he == null && fis.hasNext()) {
188 next_face = pdcel.faces[fis.next()];
189 he = next_face.faceNghb(last_face);
190 }
191
192 // note order: we use centers for 'he' as an edge
193 // of 'last_face', so 'he.twin' is the base in
194 // the next face.
195 TriAspect ftri=null;
196 TriAspect gtri=null;
197 int prev_g=-1;
198 if (he != null && next_face.faceIndx>0) {
199 if (last_face.faceIndx==prev_g)
200 ftri=gtri;
201 else
202 ftri=new TriAspect(pdcel,last_face);
203 gtri=new TriAspect(pdcel,next_face);
204 int ans=-1;
205 if (useSchw) {
206 double s=he.getSchwarzian();
207 try {
208 ans=workshops.LayoutShop.schwPropogate
209 (ftri, gtri,he,s,1);
210 } catch (Exception ex) {
211 throw new DataException(ex.getMessage());
212 }
213 }
214 else {
215 double rad=pdcel.getVertRadius(he.twin.prev);
216 ans=workshops.LayoutShop.radPropogate(ftri, gtri,
217 he,rad,1);
218 }
219
220 if (ans>0) {
221 int j=gtri.vertIndex(he.prev.origin.vertIndx);
222 pdcel.setCent4Edge(he.next.next, gtri.center[j]);
223 pdcel.setRad4Edge(he.next.next, gtri.radii[j]);
224 }
225 }
226 } // end of while
227
228 return last_face.faceIndx;
229 }

254 public static int radPropogate(TriAspect ftri, TriAspect gtri,
255 HalfEdge fedge,double rad,int mode) {
256 if (fedge.isBdry())
257 return -1;
258 int fv = ftri.edgeIndex(fedge);
259 if (fv < 0)

262 int fw = (fv+1)%3;
263 int gw=gtri.edgeIndex(fedge.twin);

313 public static int schwPropogate(TriAspect ftri,TriAspect gtri,
314 HalfEdge fedge,double s, int mode) {
315 if (fedge.isBdry())
316 return -1;
317 int fv = ftri.edgeIndex(fedge);
318 if (fv < 0)
319 throw new ParserException("TriAspect does not contain given "
320 + "HalfEdge " + fedge);
321 int fw = (fv+1)%3;
322 int gw=gtri.edgeIndex(fedge.twin);
323 int gv=(gw+1)%3;
324 int gopp=(gw+2)%3;
325
326 gtri.setCenter(ftri.getCenter(fv),gv);
327 gtri.setCenter(ftri.getCenter(fw),gw);
328 if (mode == 2) { // copy over shared labels:
329 gtri.setLabel(ftri.getLabel(fv),gv);
330 gtri.setLabel(ftri.getLabel(fw),gw);
331 } else { // default, shared radii
332 gtri.setRadius(ftri.getRadius(fv),gv);
333 gtri.setRadius(ftri.getRadius(fw),gw );
334 }
335
336 // update tang pts, compute map FROM "base equilateral"
337 ftri.setTanPts();
338 Mobius bm_f=new Mobius();
339 try {
340 bm_f = Mobius.mob_xyzXYZ(
341 CPBase.omega3[0],CPBase.omega3[1],CPBase.omega3[2],
342 ftri.tanPts[0],ftri.tanPts[1],ftri.tanPts[2],
343 0, ftri.hes);
344 } catch (MobException mex) {
345 throw new MobException(mex.getMessage());
346 }
347
348 // compute the target circle
349 CircleSimple sC = Schwarzian.getThirdCircle(s, fv, bm_f, ftri.hes);
350 gtri.setCenter(sC.center,gopp);
351 if (mode == 2)
352 gtri.setLabel(sC.rad,gopp);
353 else
354 gtri.setRadius(sC.rad,gopp);
355
356 return 1;
357 }
