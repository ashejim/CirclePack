72 public static PackData tri_to_Complex(Triangulation Tri,int hes) {

76 double min0=10000.0;

82 if (Tri.nodes!=null) {

137 int j=p.packDCEL.vertices[i].mark;
138 if (hes<=0) {
139 p.setCenter(i,new Complex(Tri.nodes[j].x,Tri.nodes[j].y));
140 }
141 else
142 p.setCenter(i,new Complex(Tri.nodes[p.getVertMark(i)])); // (theta,phi) form
143 }

154 p.fileName=new String("randKomplex");

166 int[] ans = new int[2];
167 ans[1] = 0; // default is not to reorient
168 for (int j = 0; j < 3; j++) {
169 if (fvert[j] == t) {

171 if (!fb)
172 ans[1] = 1; // signal to reorient
173 ans[0] = fvert[(j + 2) % 3];
174 return ans;
175 }
176 if (fvert[(j + 2) % 3] == v) {
177 if (fb)
178 ans[1] = 1; // signal to reorient
179 ans[0] = fvert[(j + 1) % 3];
180 return ans;
181 }
182 ans[0] = -1; // edge not found
183 return ans;
184 }
185 }
186 ans[0] = -1; // edge not found
187 return ans;

200 if (flower_c[hit-1]==w) return -1;
201 if (flower_c[hit+1]==w) return 1;
202 return 0;
203 }
204 if (hit==0) {
205 if (flower_c[1]==w) return 1;
206 if (flower_c[num]!=v || flower_c[num-1]!=w) return 0;
207 return -1;
208 }
209 if (hit==num) {
210 if (flower_c[num-1]==w) return -1;
211 if (flower_c[0]!=v || flower_c[1]!=w) return 0;
212 return 1;
213 }
214 return 0;

346 // Here we build flowers for the vertices. Begin with first vertex
347 // of 'start' face. For each petal w added to its flower, if w is not already processed, its utilFlag shows the

349 // adjusted, if necessary, to be positively oriented vis-a-vis the

377 int[] preflower = new int[2 * count[vtarget] + 4];

379 // put verts of first face in middle of preflower space
380 int front = 0;
381 int back = 0;
382 for (int j = 0; j < 3; j++) {
383 if (fvert[j] == vtarget) {
384 back = count[vtarget];
385 front = back + 1;
386 // Note: the very first face put down here determines
387 // orientation of the whole complex
388 preflower[back] = fvert[(j + 1) % 3];
389 preflower[front] = fvert[(j + 2) % 3];

393 pK[preflower[back]].utilFlag = first_face;

434 }

436 }
437 }
438 }
439 } // done forward
440
441 // flower still open? add petals backward
442 if (preflower[front] != preflower[back]) {
443 hit = 1;
444 while (hit != 0 && preflower[front] != preflower[back]) {
445 hit = 0;
446 for (int i = 0; (preflower[front] != preflower[back])
447 && i < count[vtarget]; i++) {
448 if ((next_face = nodefaces[vtarget][i]) > 0
449 && w != preflower[back + 1]) {

453 w = tmp[0];
454 swtch = tmp[1];
455 if (w >= 0) {
456 back--;
457 preflower[back] = w;
458 pK[vtarget].num++;
459 if (pK[w].utilFlag == 0)
460 pK[w].utilFlag = next_face;
461 hit = 1;
462 if (swtch != 0) { // face needs reorientation
463 // for possible later use
464 int temp = T.faces[next_face].vert[0];
465 T.faces[next_face].vert[0] = T.faces[next_face].vert[1];
466 T.faces[next_face].vert[1] = temp;
467 }
468 nodefaces[vtarget][i] = 0; // this face used for this vert
469 }
470 }
471 }
472 } // while
473 }
474 else if (pK[vtarget].num < 3) // interior vert needs at least 3 faces
475 throw new DataException("Interior vert is in less than 3 faces");
476
477 // Note: could do check here that we've used all the faces,
478 // but for now, we just forget this and use what faces we get.
479
480 // fix up the real flower
481 pK[vtarget].bdryFlag = 0;
482 if (preflower[front] != preflower[back])
483 pK[vtarget].bdryFlag = 1;
484 pK[vtarget].flower = new int[pK[vtarget].num + 1];
485 for (int nn = 0; nn <= pK[vtarget].num; nn++)
486 pK[vtarget].flower[nn] = preflower[back + nn];
487
488 // finished with this vertex, set utilFlag to -1
489 pK[vtarget].utilFlag = -1;
490
491 // look for next target vertex; this procedure should ensure
492 // that result is connected, but we may not reach every face.
493 vtarget = 0;
494 for (int nn = 1; nn <= top; nn++) {
495 if (pK[nn].utilFlag > 0) {
496 vtarget = nn;
497 nn = top + 1; // just to stop looping
498 }
499 }
500 } // end of while(vtarget)
501
502 // Now have to reprocess to make sure nodes are indexed from 1
503 // with no gaps; order is preserved.
504
505 int[] re_indx = new int[top + 1];
506 int click = 0;
507 for (int i = bottom; i <= top; i++) {
508 // error? vert is in some flower but has not itself been processed
509 if (pK[i].utilFlag > 0)
510 throw new DataException();
511 if (pK[i].utilFlag == -1) {
512 re_indx[i] = ++click;
513 }
514 }
515 if (click < 3) // too few vertices for a packing
516 throw new DataException("Too few vertices for a packing.");
517
518 // debug=true;
519 if (debug) {
520 int tb1=top-bottom+1;
521 if (click!=tb1) {
522 System.err.println("Debugging in 'parse_triangles.\n");
523 StringBuilder strb=new StringBuilder("These indices were not encountered: \n");
524 for (int i = bottom; i <= top; i++) {
525 if (pK[i].utilFlag ==0)
526 strb.append(" "+i);
527 }
528 strb.append("\n");
529 System.err.println(strb.toString());
530 strb=new StringBuilder("Flowers of these indices were not finished: \n");
531 for (int i = bottom; i <= top; i++) {
532 if (pK[i].utilFlag >0)
533 strb.append(" "+i);
534 }
535 strb.append("\n");
536 System.err.println(strb.toString());
537 }
538 }
539
540 ans[0] = click;
541
542 // create the final K_data
543 TmpVert[] tmpVert = new TmpVert[ans[0] + 1];

549 tmpVert[n].mark = i;

551 tmpVert[n].flower[j] = re_indx[tmpVert[n].flower[j]];
552 }
553 }
554 return tmpVert;
555 }

645 String errMsg="";

651
652 // catch all exceptions at end and use 'errMsg' as error.
653 try {
654
655 // check for 'VTK' first because we have to check a comment line
656 String line=StringUtil.ourNextLine(fp);
657 if (line==null) {
658 errMsg="Empty file error";

660 }

676 String str=tok.nextToken();

679 type=1;
680 else if (str.startsWith("TRIANGULATION:"))
681 type=2;
682 else if (str.startsWith("OFF") || str.startsWith("off"))
683 type=3;

685 tok=new StringTokenizer(line); // retokenize
686

709 faceHit=true;

712 }
713 else if (lt==3 && !vertHit) { // starts with a double, should be xyz data
714 vertHit=true;

716 V=tmpVerts.size();
717 }
718
719 } // end search for data
720 } // done with generic and OFF format
721
722 else if (type==1) { // CHECKCOUNT
723 // search for face/xyz data
724 while (line!=null && (!faceHit || !vertHit)) {
725 // flush lines until we get one of the key words
726 while ((line=StringUtil.ourNextLine(fp,true))!=null &&
727 !line.startsWith("FACE_TRIPLES:") &&
728 !line.startsWith("POINTS:"));

742 } // done with CHECKPOINT case
743

760 while ((line=StringUtil.ourNextLine(fp,true))!=null &&
761 !line.startsWith("FACE_TRIPLES:") &&
762 !line.startsWith("POINTS:"));
763 if (line!=null && line.startsWith("FACE_TRIPLES:")) {
764 faceHit=true;

781 // flush lines until we get one of the key words
782 while (!line.startsWith("POINT") &&
783 !line.startsWith("POLYGON") && !line.startsWith("CELL_DATA"))
784 line=StringUtil.ourNextLine(fp,true);

786 faceHit=true;
787 line=StringUtil.ourNextLine(fp,true);

790 }

792 vertHit=true;
793 line=StringUtil.ourNextLine(fp,true);

795 V=tmpVerts.size();
796 }

800 // ========== now to build 'tri' ==================
801
802 // first, reconfigure the data we gathered
803 Face[] holdFaces=null;

890 for (int f=1;f<=F;f++) {

893

946 vtick++;

957 ymean+=pt.y;
958 zmean+=pt.z;

962 if (tri.vertColors!=null)
963 tri.vertColors[vtick]=new Color(255,255,255); // white
964 }

967 newFaces[f]=new Face(3);
968 newFaces[f].vert=new int[3];
969 newFaces[f].vert[0]=vert[0];

971 newFaces[f].vert[2]=vtick;

975 if (tri.faceColors!=null)
976 tmpCol=(Color)tri.faceColors[f];

978 for (int j=1;j<num;j++) {
979 newFaces[++ftick]=new Face(3);
980 newFaces[ftick].vert=new int[3];
981 newFaces[ftick].vert[0]=vert[j];
982 newFaces[ftick].vert[1]=vert[(j+1)%num];
983 newFaces[ftick].vert[2]=vtick;

988 } // done breaking up this face
989 } // done going through all the face
990 tri.faces=newFaces;

992

1312 public static Triangulation zigzag_cutter(Triangulation Tri,
1313 Path2D.Double Gamma) {
1314
1315 // store vertex centers
1316 Complex[] Z = new Complex[Tri.nodeCount+1];
1317 for (int n = 1; n <= Tri.nodeCount; n++)
1318 Z[n] = new Complex(Tri.nodes[n].x, Tri.nodes[n].y);
1319
1320 // Now the processing to prune this; create and work with DCEL data
1321 int[] ans=new int[2];
1322 TmpVert[] tmpVert=Triangulation.parse_triangles(Tri,0,ans);
1323 int nodecount=ans[0];
1324 if (nodecount<=2)
1325 throw new DataException("DCEL: parse_triangles came up short");
1326
1327 // get the bouquet of flowers
1328 int [][]bouquet=new int[nodecount+1][];
1329 for (int v=1;v<=nodecount;v++)
1330 bouquet[v]=tmpVert[v].flower;
1331
1332 PackDCEL pdc=CombDCEL.getRawDCEL(bouquet);
1333 PackDCEL myDCEL=CombDCEL.extractDCEL(pdc,null,pdc.alpha);
1334 int facecount = myDCEL.intFaceCount;
1335
1336 // find the centroids
1337 Complex[] faceC = new Complex[facecount+1]; // centroids
1338 int tick = 0;
1339 for (int f=1;f<=myDCEL.faceCount;f++) {
1340 HalfLink edges=myDCEL.faces[f].getEdges();
1341 Iterator<HalfEdge> eit=edges.iterator();
1342 Complex accum=new Complex(0.0);
1343 while (eit.hasNext()) {
1344 HalfEdge he = eit.next();
1345 accum = accum.add(Z[he.origin.vertIndx]);
1346 }
1347 // average of vertex locations
1348 faceC[++tick] = new Complex(accum.times(1.0/(double)edges.size()));
1349 }
1350
1351 // determine which faces have centroids in Gamma; these are included
1352 Complex firstC = null;
1353 combinatorics.komplex.DcelFace firstFace=null;
1354 int[] facestat = new int[facecount + 1]; // 1=included
1355
1356 for (int j = 1; j <= facecount; j++) {
1357 if (Gamma.contains(faceC[j].x, faceC[j].y)) {
1358 facestat[j] = 1;
1359 if (firstC == null) { // mark the first included centroid and its face
1360 firstC = new Complex(faceC[j]);
1361 firstFace=myDCEL.faces[j];
1362 }
1363 }
1364 }
1365
1366 // There may be other faces included, we may have to fill in holes.
1367 // Idea is to find simple closed path from included, and
1368 // add those inside this path.
1369
1370 // Find all 'HalfEdge's of included faces with non-included (or
1371 // ideal) face on the other side.
1372 Vector<HalfEdge> putbdry = new Vector<HalfEdge>(0);
1373 for (int f = 1; f <= Tri.faceCount; f++) {
1374 if (facestat[f] > 0) {
1375 HalfLink edges=myDCEL.faces[f].getEdges();
1376 Iterator<HalfEdge> eit=edges.iterator();
1377 while (eit.hasNext()) {
1378 HalfEdge he=eit.next();
1379 combinatorics.komplex.DcelFace oppface = he.twin.face;
1380 if (Math.abs(oppface.faceIndx) > facecount || facestat[Math.abs(oppface.faceIndx)] == 0)
1381 putbdry.add(he);
1382 }
1383 }
1384 }
1385
1386 // sift through 'putbdry', removing cclw closed chains until
1387 // we get one which wraps positively about 'firstC'. This
1388 // should be "outer" bdry.
1389
1390 Path2D.Double outerPath = null;
1391
1392 while (outerPath == null && putbdry.size() > 0) {
1393 // debug: 'disp -ffc120 flist -tfc195t2 z zlist;'
1394
1395 // start a new closed edge path
1396 Vector<HalfEdge> bpath = new Vector<HalfEdge>(0);
1397 HalfEdge start = putbdry.remove(0);
1398 bpath.add(start);
1399
1400 HalfEdge nextedge = start.twin.prev.twin;
1401 while (!putbdry.contains(nextedge) && nextedge != start) {
1402 nextedge = nextedge.prev.twin;
1403 }
1404 if (nextedge == start)
1405 throw new CombException("comb problem in dcel");
1406 HalfEdge curredge = start;
1407 while (putbdry.size()>0 && nextedge != start) {
1408 curredge = nextedge;
1409 putbdry.remove(curredge);
1410 bpath.add(curredge);
1411
1412 nextedge = curredge.twin.prev.twin;
1413 while (!putbdry.contains(nextedge) && nextedge!=start && nextedge != curredge) {
1414 nextedge = nextedge.prev.twin;
1415 } // now have 'nextedge'
1416 if (nextedge == curredge)
1417 throw new CombException("combinatoric problem in dcel");
1418 } // end of while to close path
1419
1420 // create cclw Path2D.Double using ends of 'bpath'
1421 Path2D.Double putPath = new Path2D.Double();
1422 Complex pt = new Complex(Z[bpath.remove(0).twin.origin.vertIndx]);
1423 putPath.moveTo(pt.x, pt.y);
1424 Iterator<HalfEdge> bpi = bpath.iterator();
1425 while (bpi.hasNext()) {
1426 pt = new Complex(Z[bpi.next().twin.origin.vertIndx]);
1427 putPath.lineTo(pt.x, pt.y);
1428 }
1429 putPath.closePath();
1430
1431 // is 'putPath' the outer path? Yes, if it contains 'firstC'
1432 if (putPath.contains(firstC.x, firstC.y))
1433 outerPath = putPath;
1434 } // should have outer path
1435
1436 if (outerPath == null)
1437 throw new CombException("couldn't build 'outerPath'");
1438
1439 // convert to triangulation
1440 Triangulation zzTri = new Triangulation();
1441
1442 // Two steps: (1) find all faces inside 'outerPath';
1443 // (2) build out simply connected patch
1444 int startindx=-1;
1445 int []inouter=new int[myDCEL.faceCount+1];
1446 for (int j = 1; j <= facecount; j++) {
1447 if (outerPath.contains(faceC[j].x, faceC[j].y)) {
1448 inouter[j]=1;
1449 if (startindx==-1)
1450 startindx=j;
1451 }
1452 }
1453
1454 // if 'firstFace' is no longer included, use new starting place
1455 if (inouter[Math.abs(firstFace.faceIndx)]==0)
1456 firstFace=myDCEL.faces[startindx];
1457
1458 // cycle through adding faces to 'firstFace'
1459 Vector<combinatorics.komplex.DcelFace> nextF=new Vector<combinatorics.komplex.DcelFace>();
1460 Vector<combinatorics.komplex.DcelFace> currF=nextF;
1461 nextF.add(firstFace);
1462 int[] newfaces = new int[myDCEL.faceCount+1];
1463 newfaces[Math.abs(firstFace.faceIndx)]=1;
1464 int infacecount = 1;
1465 tick=0;
1466 while(nextF.size()>0) {
1467 currF=nextF;
1468 nextF=new Vector<combinatorics.komplex.DcelFace>();
1469 while (currF.size()>0) {
1470 combinatorics.komplex.DcelFace currface=currF.remove(0);
1471 HalfLink edges=currface.getEdges();
1472 Iterator<HalfEdge> eit=edges.iterator();
1473 while (eit.hasNext()) {
1474 HalfEdge he=eit.next();
1475 int findx=Math.abs(he.twin.face.faceIndx);
1476 if (findx<=facecount && inouter[findx]>0 && newfaces[findx]==0) {
1477 newfaces[findx]=1;
1478 infacecount++;
1479 nextF.add(he.twin.face);
1480 }
1481 }
1482 } // end of while on 'currF'
1483 } // end of while on 'nextF'
1484
1485 // reindex the needed vertices
1486 int[] old2new = new int[myDCEL.vertCount + 1];
1487 tick = 0;
1488 for (int j = 1; j <= facecount; j++) {
1489 if (newfaces[j]>0) {
1490
1491 // debug
1492 int []fv= myDCEL.faces[j].getVerts();
1493 StringBuilder ddstr=new StringBuilder("face: ");
1494 for (int nm=0;nm<fv.length;nm++)
1495 ddstr.append(" "+fv[nm]);
1496 System.err.println(ddstr.toString());
1497
1498 int []verts=myDCEL.faces[j].getVerts();
1499 for (int m=0;m<verts.length;m++) {
1500 int v=verts[m];
1501 if (old2new[v] == 0)
1502 old2new[v] = tick++; // new index;
1503 }
1504 }
1505 } // have new indices
1506
1507
1508 // fill triangulation data
1509 zzTri.nodeCount = tick;
1510 zzTri.maxIndex = tick;
1511 zzTri.nodes = new Point3D[tick + 1];
