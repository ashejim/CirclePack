120 public void hauptProcess() throws ParserException {
121 // break into left/right at ':'
122 String left=null;
123 String right=null;
124 int j;
125 try {
126 left=fullDeal.substring(0,(j=fullDeal.indexOf(':'))).trim();
127 right=fullDeal.substring(j+1,fullDeal.length()).trim();
128 if (right.length()==0) { // only need 'v'/'c' on left if '-p' flag
129 throw new ParserException();
130 }
131 } catch (Exception ex) {
132 errMsg=new String("No separator ':' or 'left'/'right' missing");
133 throw new ParserException();
134 }
135
136 // circles? faces? tiles?
137 if (left.length()>0) {
138 if (left.startsWith("f")) {
139 object='f';
140 }
141 else if (left.startsWith("c") || left.startsWith("v")) {
142 object='c';
143 }
144 else if (left.startsWith("t")) {
145 object='t';
146 }
147 // TODO: edges?
148 }
149
150 // Check left for '-p{p}' flag
151 if (left.length()>0 && (j=left.indexOf("-p"))>0) {
152 int pnum=-1;
153 try {
154 // expect integer
155 pnum=Integer.parseInt(left.substring(j+2,j+3));
156 if (pnum<0 || pnum>=CPBase.NUM_PACKS
157 || !PackControl.cpDrawing[pnum].getPackData().status)
158 throw new ParserException();
159 } catch (Exception ex) {
160 errMsg=new String("Malformed '-p{p}' pack or pack is empty");
161 throw new ParserException();
162 }
163 // change to this packing
164 packData=PackControl.cpDrawing[pnum].getPackData();
165 }

167 // Done with left; now work on 'right' string
168
169 /* ---- find/replace new-format conditions by traditional ones:
170 .eq.,.ne.,.lt.,.le.,.gt.,.ge.,.and.,.or. --> =,==,!=,
171 <,<=,>,>=,&&,|| (resp.) */
172 right=right.replace(".eq."," == ");
173 right=right.replace(".ne."," != ");
174 right=right.replace(".neq."," != "); // alternate
175 right=right.replace(".lt."," < ");
176 right=right.replace(".le."," <= ");
177 right=right.replace(".gt."," > ");
178 right=right.replace(".ge."," >= ");
179 right=right.replace(".and."," && ");
180 right=right.replace(".or."," || ");
181 right=right.replace(".not."," ! ");
182 right=right.replace("=<"," <= ");
183 right=right.replace("=>"," >= ");
184 // TODO: convert single '=' to ' == ' without infinite looping.

187 fullDeal=right;
188 specHit=false;

190 // break into segments by splitting at '&&' or '||',
191 Vector<String> segments=StringUtil.setB_segments(fullDeal);

220 public int parseSeg(String str) {
221 int j=-1;
222 int k=-1;
223 String lstr=null;
224 String rstr=null;
225 SelectSpec ent=new SelectSpec(object);
226
227 // check for leading connective first
228 if (str.startsWith("&&") || str.startsWith("||")) {
229 // must have preceeding specification
230 if (specs.size()==0 || specs.get(specs.size()-1).isConnective) return 0;
231 SelectSpec nspec=new SelectSpec(object);
232 nspec.isConnective=true;
233 nspec.isAnd=true;
234 if (str.startsWith("||")) nspec.isAnd=false;
235 specs.add(nspec); // add to 'specs'
236 str=str.substring(2).trim(); // get rid
237 }
238
239 ent.isConnective=false;
240 ent.negation=false;
241
242 // check for negation
243 if (str.startsWith("!")) {
244 ent.negation=true;
245 str=str.substring(1);
246 }
247
248 // check for parens --- should be first and last characters
249 j=str.indexOf('(');
250 k=str.indexOf(')',j+1);
251 if ((j<0 && k>=0) || (j>=0 && k<0) || k<j) return 0; // malformed
252 if (k>=0 && j>=0) {
253 str=str.substring(j+1,k).trim();
254 if (str.length()==0) return 0; // nothing left
255 }
256
257 // should have specification only now, look for
258 // typcial situations first
259 j=k=-1;
260 if ((j=str.indexOf("=="))>=0 ||
261 (j=str.indexOf(">="))>=0 ||
262 (j=str.indexOf("<="))>=0 ||
263 (j=str.indexOf("!="))>=0) { // 2-character condition?
264 k=j+2;
265 ent.setCondition(str.substring(j,k));
266 }
267 else if ((j=str.indexOf("="))>=0 || (j=str.indexOf(">"))>=0
268 || (j=str.indexOf("<"))>=0) { // 1-character condition?
269 k=j+1;
270 ent.setCondition(str.substring(j,k));
271 }
272 else { // can only be unary (TODO: should check?)
273 ent.unary=true;
274 ent.left_str=new String(str).trim();
275 ent.right_str=null;
276 specs.add(ent);
277 return 1;
278 }
279
280 // if not unary, should have j and k;
281 if (j>=0 && k>=0) {
282 if (k>=str.length()-1 || j<=0) return 0; // error: lstr/rstr missing
283 lstr=str.substring(0,j).trim();
284 rstr=str.substring(k).trim();
285
286 // check for explicit double(s)
287 try {
288 ent.left_value=Double.parseDouble(lstr);
289 } catch (Exception ex) {
290 ent.left_value=null;
291 ent.left_str=lstr;
292 }
293 try {
294 ent.right_value=Double.valueOf(rstr);
295 ent.right_str=rstr;
296 } catch (Exception ex) {
297 ent.right_value=null;
298 ent.right_str=rstr;
299 }
300 }
301
302 specs.add(ent);
303 return 1;
304 }
