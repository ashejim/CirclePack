67 public void parseDispStr(String str) {
68
69 if (str != null && str.length() > 0) {
70
71 if (str.startsWith("-"))
72 str=str.substring(1);
73
74 // need buffer to manipulate
75 StringBuilder strbuf = new StringBuilder(str.trim());
76
77 // truncate anything after first whitespace
78 int k = 0;
79 while (k < strbuf.length() && !Character.isWhitespace(strbuf.charAt(k))) {
80 k++;
81 }
82 if (k < (strbuf.length()-1))
83 strbuf.delete(k, strbuf.length());
84
85 // =========== searches ================
86
87 // foreground fill?
88 if ((k = strbuf.indexOf("fg")) >= 0) {
89 color = ColorUtil.getFGColor();
90 colorIsSet = true;
91 strbuf.delete(k, k + 2);
92 }
93 // else background fill?
94 else if ((k = strbuf.indexOf("bg")) >= 0) {
95 color = ColorUtil.getBGColor();
96 colorIsSet = true;
97 strbuf.delete(k, k + 2);
98 }
99
100 // fill object? ('fg' would have been removed)
101 if ((k = strbuf.indexOf("f")) >= 0)
102 fill = true;
103
104 // label? remove the 'n' so it doesn't interfere
105 if ((k = strbuf.indexOf("n")) >= 0) {
106 label = true;
107 strbuf.delete(k, k + 1);
108 }
109
110 // 'c' can mean draw object in color; but if
111 // followed by digit, must be color code
112 if ((k = strbuf.indexOf("c")) >= 0
113 && (strbuf.length() == (k + 1) ||
114 !Character.isDigit(strbuf.charAt(k + 1)))) {
115 colBorder = true;
116 strbuf.delete(k, k + 1);
117 }
118
119 // Can get at most one color code:
120 // might be 'fc' or 'cf' for fill color or just 'c'
121 // for coloring the object itself.
122 String digits = null;
123 if ((k = strbuf.indexOf("fc")) >= 0 || (k = strbuf.indexOf("cf")) >= 0)
124 digits = StringUtil.getDigitStr(strbuf, k + 2);
125 // 'c' with digits would have been caught earlier
126 else if ((k = strbuf.indexOf("c")) >= 0) {
127 colBorder = true;
128 digits = StringUtil.getDigitStr(strbuf, k + 1);
129 }
130 int K = 0;
131 if (digits != null) {
132 K = digits.length();
133 color = ColorUtil.coLor(Integer.parseInt(digits));
134 colorIsSet = true;
135 strbuf.delete(k, k + K + 1);
136 }
137
138 // 't' for thickness?
139 if ((k = strbuf.indexOf("t")) >= 0) {
140 digits = StringUtil.getDigitStr(strbuf, k + 1);
141 K = 0;
142 if (digits != null) {
143 K = digits.length();
144 thickness = Integer.parseInt(digits);
145 if (thickness < 0)
146 thickness = Integer.valueOf(0);
147 if (thickness > 15)
148 thickness = Integer.valueOf(15);
149 }

151 }

226 bg_set=true;

234 // typical

242 stb.append("bg");
243 }

246 stb.append('c');

248 if (color!=null)

253 if (thickness!=1)
