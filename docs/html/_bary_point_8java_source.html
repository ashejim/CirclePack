242 public static BaryPoint nearestBary(BaryPoint inpt,boolean cutoff) {
243 BaryPoint outpt=null;
244 double x=inpt.b1;
245 double y=inpt.b2;
246
247 // inpt is already inside
248 if (inpt.b0>=0 && x>=0 && y>=0) {
249 outpt=new BaryPoint(inpt);
250 outpt.face=inpt.face;
251 return outpt;
252 }
253
254 // 'cutoff' set and too far out of tolerance?
255 if (cutoff) {
256 if ((x+y)<1.0+THRESHOLD && x>-1.0*THRESHOLD && y>-1.0*THRESHOLD) {
257 outpt=new BaryPoint(1-(x+y),x); // this call fixes coords
258 outpt.face=inpt.face;
259 }
260 return outpt;
261 }
262
263 // project to corner or to edge (along line to opposite corner)
264 outpt=new BaryPoint();
265 outpt.face=inpt.face;
266 if (x<0) {
267 if (y<0) {
268 outpt.b1=outpt.b2=0;
269 outpt.b0=1.0;
270 }
271 else if ((x+y)>=1) {
272 outpt.b0=outpt.b1=0;
273 outpt.b2=1.0;
274 }
275 else {
276 outpt.b1=0.0;
277 outpt.b2=y*(1.0+x/(1.0-x));
278 outpt.b0=1.0-outpt.b2;
279 }
280 }
281 else if (y<0) {
282 if ((x+y)>=1) {
283 outpt.b0=outpt.b2=0.0;
284 outpt.b1=1.0;
285 }
286 else {
287 outpt.b2=0;
288 outpt.b1=x*(1.0+y/(1-y));
289 outpt.b0=1.0-outpt.b1;
290 }
291 }
292 else if ((x+y)>1.0) {
293 outpt.b0=0;
294 outpt.b1=x/(x+y);
295 outpt.b2=y/(x+y);
296 }
297 return outpt;
298 }

312 public static Complex vec2simplex(Complex vec,Complex z0,Complex z1,Complex z2) {
313 Complex a=z1.minus(z0);
314 Complex b=z2.minus(z0);
315
316 // are these too small or close to parallel?
317 if (Point3D.CrossProduct(a, b).norm()<PackData.OKERR)
318 return null;

320 // transformation is inverse of matrix (a.x b.x;a.y b.y)
321 double det=a.x*b.y-b.x*a.y;
322 return new Complex((b.y*vec.x -b.x*vec.y)/det,(-a.y*vec.x+a.x*vec.y)/det);

343 double x=b1;
344 if (x<PackData.OKERR) x=0.0;
345 double y=1-(b0+x);
346 if (y<PackData.OKERR) y=0.0;
347 if (y>1.0) y=1.0;
348 grad=grad.divide(grad.abs()); // unit vector
349 double a=grad.x;
350 double b=grad.y;
351 double m=b/a;

357
358 // going up
