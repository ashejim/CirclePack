112 public int loadSidePairs() {
113 int count=0;
114 PackData p=CirclePack.cpb.getActivePackData();
115 PackDCEL pdcel=p.packDCEL;
116 sidePairHandler.wipeoutTools();
117 sidePairHandler.clearToolBar();
118
119 if (pdcel.redChain==null || pdcel.pairLink==null ||
120 pdcel.pairLink.size()<2)
121 return 0;
122 Iterator<SideData> sides=pdcel.pairLink.iterator();
123 SideData sd=sides.next(); // first spot empty
124 while (sides.hasNext()) {
125 sd=(SideData)sides.next();
126 // yes, this is part of a side-pairing, add its tool
127 if (sd.label!=null && sd.mateIndex>0) {
128 String letterIcon=new String("mobius/mob_"+sd.label+".png");
129 String mobCmd=new String("appMob "+sd.mob.a.x+" "+
130 sd.mob.a.y+" "+sd.mob.b.x+" "+sd.mob.b.y+" "+
131 sd.mob.c.x+" "+sd.mob.c.y+" "+sd.mob.d.x+" "+
132 sd.mob.d.y);
133 MyTool but=new MyTool(new CPIcon(letterIcon),mobCmd,null,null,
134 new String("Drop: Mobius transformation for side-pairing '"+sd.label+"'"),
135 "MOBIUS:",true,this,(PopupBuilder)null);
136 sidePairHandler.addTool(but);
137 count++;
138 }
139 }
140
141 sidePairHandler.repopulateTools();
142 return count;
143 }
