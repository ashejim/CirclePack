423 public static PackData seed(ArrayList<Double> schvec,
424 CircleSimple homecs,Complex layoutErr,int order) {
425 int n=schvec.size();
426 PackDCEL pdcel=RawManip.seed_raw(n);
427 PackData p=new PackData(null);
428 pdcel.fixDCEL(p);
429
430 // set up structure, center of max index, spokes,
431 // record schwarzians, etc.
432 p.swap_nodes(1,n+1);
433 for (int j=1;j<n;j++)
434 p.swap_nodes(j, j+1);
435 HalfEdge nhe=p.packDCEL.findHalfEdge(new EdgeSimple(n+1,1));
436 // set up 'spokes', indexed from 1
437 HalfEdge[] spokes=new HalfEdge[n+1]; // indexed from 1
438 spokes[1]=nhe;
439 HalfEdge he=spokes[2]=nhe.prev.twin;
440 int tick=2;
441 while (he!=nhe) {
442 spokes[tick++]=he;
443 he=he.prev.twin;
444 }
445 p.packDCEL.vertices[n+1].halfedge=nhe;
446 p.packDCEL.setAlpha(n+1, null,false);
447 p.packDCEL.setGamma(1);
448
449 // get schwarzian closest to uniform n-flower
450 double aim=(order+1)*2.0*Math.PI;
451 double unifS=SchFlowerData.uniformS(n,aim); // uniform schwarzian
452 // find edge with schwarzian closest to uniform
453 tick=0;
454 double bestdiff=Math.abs(schvec.get(0)-unifS);
455 for (int j=1;j<n;j++) {
456 double diff=Math.abs(schvec.get(j)-unifS);
457 if (diff<bestdiff) {
458 bestdiff=diff;
459 tick=j;
460 }
461 }
462 int B=tick;
463 if (B==0) // need room for petal clw from B
464 B=1;
465
466 // We're indexing everything from 0 during
467 // layout; will adjust for vert indices
468 // as needed.
469 // start layout with B edge centered on
470 // positive axis, B-1 edge clw next to it,
471 // both with radius of uniform n-flower.
472 // Then layout clw until 0 edge is in place,
473 // then cclw until new 0 edge is computed.
474
475 // initialize 2 TriAspect's
476 TriAspect ftri=new TriAspect();
477 TriAspect gtri=new TriAspect();
478 ftri.hes=gtri.hes=0;
479 ftri.allocCenters();
480 gtri.allocCenters();
481 double a=aim/(2.0*(double)n);
482 double b=Math.sin(a);
483 double r=b/(1.0-b); // uniform radius
484 double len=(1+r);
485
486 // these won't change
487 ftri.center[0]=new Complex(0.0);
488 gtri.center[0]=new Complex(0.0);
489 ftri.radii[0]=1.0;
490 gtri.radii[0]=1.0;
491
492 // these will be repeatedly updated
493 ftri.center[1]=new Complex(len*Math.cos(2.0*a),len*Math.sin(-2.0*a));
494 gtri.center[1]=new Complex(len*Math.cos(2.0*a),len*Math.sin(-2.0*a));
495 ftri.center[2]=new Complex(len); // on x-axis
496 gtri.center[2]=new Complex(len); // on x-axis
497 ftri.radii[1]=ftri.radii[2]=r;
498 gtri.radii[2]=gtri.radii[2]=r;
499 ftri.setTanPts();
500 gtri.setTanPts();
501 ftri.setBaseMobius();
502 gtri.setBaseMobius();
503
504 // store center, B, and B-1 data in p
505 p.packDCEL.setVertCenter(n+1,new Complex(0.0));
506 p.packDCEL.setVertRadii(n+1,1.0);
507 p.packDCEL.setVertCenter(B+1,new Complex(ftri.center[2]));
508 p.packDCEL.setVertRadii(B+1,ftri.radii[2]);
509 p.packDCEL.setVertCenter(B,new Complex(ftri.center[1]));
510 p.packDCEL.setVertRadii(B,ftri.radii[1]);
511
512 CircleSimple cs=null;
513 double anglesum=2.0*a; // for accumulating total angle
514 int clwMoves=B-1;
515
516 // cross clwMoves edges clockwise
517 if (B>1) {
518 for (int j=1;j<=clwMoves;j++) {
519 double s=schvec.get(B-j);
520 // get next face angle
521 cs=Schwarzian.getThirdCircle(s,0,gtri.getBaseMobius(),0);
522 int v=B-j;
523 p.packDCEL.setVertCenter(v,cs.center);
524 p.packDCEL.setVertRadii(v,cs.rad);
525 double faceangle=gtri.center[1].divide(cs.center).arg();
526 anglesum +=faceangle;
527
528 // shift data preparing for next edge
529 gtri.radii[2]=gtri.radii[1];
530 gtri.center[2]=gtri.center[1];
531 gtri.radii[1]=cs.rad;
532 gtri.center[1]=cs.center;
533 gtri.setTanPts();
534 gtri.setBaseMobius();
535 }
536 }
537
538 // next cross edge 2 of ftri until done
539 // with edge n-2.
540 if (B<(n-1)) {
541 for (int j=B;j<n-1;j++) {
542 double s=schvec.get(j);
543 // get next face angle
544 cs=Schwarzian.getThirdCircle(s,2,ftri.getBaseMobius(),0);
545 int v=j+2;
546 p.packDCEL.setVertCenter(v,cs.center);
547 p.packDCEL.setVertRadii(v,cs.rad);
548 double faceangle=cs.center.divide(ftri.center[2]).arg();
549 anglesum +=faceangle;
550
551 // shift data preparing for next edge
552 ftri.radii[1]=ftri.radii[2];
553 ftri.center[1]=ftri.center[2];
554 ftri.radii[2]=cs.rad;
555 ftri.center[2]=cs.center;
556 ftri.setTanPts();
557 ftri.setBaseMobius();
558 }
559 }
560 // layout c_1 again, based on c_{n-2} and c_{n-1},
561 // but do not record in packdata
562 double s=schvec.get(n-1); // cs.center.arg() ftri.center[2].arg();
563 CircleSimple tmpcs=Schwarzian.getThirdCircle(s,2,ftri.getBaseMobius(),0);
564 double lastangle=tmpcs.center.divide(ftri.center[2]).arg();
565 anglesum +=lastangle;
566
567 // the 'error' is complex:
568 // * x = relative change in radius
569 // * y= relative change angle sum
570 layoutErr.x=(tmpcs.rad-r)/r;
571 layoutErr.y=(anglesum-aim)/aim;
572
573 // rotate by pi/2 to put c_1 on y-axis, center
574 // on unit circle
575 double rot=p.packDCEL.vertices[1].center.arg();
576 r=p.packDCEL.vertices[1].rad;
577 CommandStrParser.jexecute(p,"rotate "+(0.5-rot/Math.PI));
578 CommandStrParser.jexecute(p,"scale "+1/(1+r));
579 tmpcs.center=tmpcs.center.times(new Complex(0.0,Math.PI/2.0-rot).exp());
580 homecs.center=tmpcs.center.times(1.0/(1.0+r));
581 homecs.rad=tmpcs.rad/(1.0+r);
582 return p;
583 }

690 // the first element is always the 'S' element

717 redVert=4;

721 redVert=6;

723 }

727 if (rand.nextBoolean())
728 blueVert=6;
729 else
730 blueVert=2;
731 }
732

734 // successively adding new faces
735 for (int f=2;f<=n;f++) {
736
737 // randomize next action with 2 coin flips
738 newCoin=rand.nextBoolean();
739 redCoin=rand.nextBoolean();
740
741 if (redCoin) { // adding to get new red end
742 if (newCoin) {
743 adjoinFace(myPacking,hexFace,1,2,2,bdryLink);
744 }
745 else {
746 int redChk=bdryLink.get(bdryLink.size()-1);

748 adjoinFace(myPacking,hexFace,1,2,2,bdryLink);

751 blueVert=redVert;
752 }

754 adjoinFace(myPacking,hexFace,1,2,4,bdryLink);
755 }
756 }
757 }
758 else { // adding to get new blue end
759 if (newCoin) {
760 adjoinFace(myPacking,hexFace,1,4,2,bdryLink);
761 }
762 else {
763 int blueChk=bdryLink.get(1);
764 if (blueChk==blueVert) {
765 adjoinFace(myPacking,hexFace,1,4,2,bdryLink);
766 blueVert=myPacking.vertexMap.findW(2); // new vert
767 if (mode==1) // keep these in sync
768 redVert=blueVert;
769 }
770 else {
771 adjoinFace(myPacking,hexFace,3,2,4,bdryLink);
772 }
773 }
774 }
775
776 // mark center 1, red; nodes 2, blue
777 myPacking.setVertMark(myPacking.vertexMap.findW(1),1);
778 myPacking.setCircleColor(myPacking.vertexMap.findW(1),ColorUtil.cloneMe(ColorUtil.coLor(190))); // red
779 myPacking.setVertMark(myPacking.vertexMap.findW(2),2);
780 myPacking.setCircleColor(myPacking.vertexMap.findW(2),ColorUtil.cloneMe(ColorUtil.coLor(10))); // blue
781 myPacking.setVertMark(myPacking.vertexMap.findW(4),2);
782 myPacking.setCircleColor(myPacking.vertexMap.findW(4),ColorUtil.cloneMe(ColorUtil.coLor(10))); // blue
783 myPacking.setVertMark(myPacking.vertexMap.findW(6),2);
784 myPacking.setCircleColor(myPacking.vertexMap.findW(6),ColorUtil.cloneMe(ColorUtil.coLor(10))); // blue
785 // others white
786 myPacking.setCircleColor(myPacking.vertexMap.findW(3),ColorUtil.cloneMe(ColorUtil.coLor(100)));
787 myPacking.setCircleColor(myPacking.vertexMap.findW(5),ColorUtil.cloneMe(ColorUtil.coLor(100)));
788 myPacking.setCircleColor(myPacking.vertexMap.findW(7),ColorUtil.cloneMe(ColorUtil.coLor(100)));

1418 public static PackData buildKagome(int n) {
1419 // prepare 'TileData', mode 1
1420 int numtiles = (2 * n + 1) * (2 * n + 1) * 3;
1421 int mode = 1;
1422 TileData tileData = new TileData((PackData) null, numtiles, mode);
1423
1424 // first task: set vertex indices for all locations
1425 // in a (4n+2)x(4n+2) array that will be used.
1426 // This is basically a square in the Gaussian
1427 // lattice, but we don't use (i,j) if i and j are
1428 // both even.
1429 int[][] indices = new int[3 + 4 * n][3 + 4 * n];
1430 int offset = 2 * n + 1; // origin at (2n+1,2n+1)
1431 int tick = 0;
1432
1433 // do first generation
1434 int J = -1 + offset; // bottom edge
1435 for (int i = -1; i <= 1; i++)
1436 indices[i + offset][J] = ++tick;
1437 int I = 1 + offset; // right edge
1438 indices[I][1] = ++tick;
1439 J = 1 + offset; // top edge
1440 for (int i = 1; i >= -1; i--)
1441 indices[i + offset][J] = ++tick;
1442 I = -1 + offset; // left edge
1443 indices[I][1] = ++tick;
1444
1445 // iterate: 2n further generations of paths about
1446 // the origin, each forming a cclw "square"
1447 for (int g = 1; g <= n; g++) {
1448
1449 // next generation, omit when i, j both even
1450 J = -2 * g + offset;
1451 for (int i = 1; i < 2 * g; i = i + 2)
1452 indices[-2 * g + i + offset][J] = ++tick;
1453 I = 2 * g + offset;
1454 for (int j = 1; j < 2 * g; j = j + 2)
1455 indices[I][-2 * g + j + offset] = ++tick;
1456 J = 2 * g + offset;
1457 for (int i = 1; i < 2 * g; i = i + 2)
1458 indices[2 * g - i + offset][J] = ++tick;
1459 I = -2 * g + offset;
1460 for (int j = 1; j < 2 * g; j = j + 2)
1461 indices[I][2 * g - j + offset] = ++tick;
1462
1463 // next generation, take all vertices
1464 J = -2 * g - 1 + offset;
1465 for (int i = 0; i <= 4 * g + 1; i++)
1466 indices[-2 * g - 1 + i + offset][J] = ++tick;
1467 I = 2 * g - 1 + offset;
1468 for (int j = 1; j < 4 * g + 1; j++)
1469 indices[I][-2 * g - 1 + j + offset] = ++tick;
1470 J = 2 * g + 1 + offset;
1471 for (int i = 0; i <= 4 * g + 1; i++)
1472 indices[2 * g + 1 - i + offset][J] = ++tick;
1473 I = -2 * g - 1 + offset;
1474 for (int j = 1; j < 4 * g + 1; j++)
1475 indices[I][2 * g - 1 - j + offset] = ++tick;
1476 }
1477
1478 // build 3 prototype tiles: a hex and two opposite
1479 // triangles form a combinatorial square, which
1480 // we replicate in an (2n+1)x(2n+1) pattern.
1481 int[][] hextile = new int[6][2];
1482 hextile[0][0] = -1;
1483 hextile[0][1] = -1;
1484 hextile[1][0] = 0;
1485 hextile[1][1] = -1;
1486 hextile[2][0] = 1;
1487 hextile[2][1] = 0;
1488 hextile[3][0] = 1;
1489 hextile[3][1] = 1;
1490 hextile[4][0] = 0;
1491 hextile[4][1] = 1;
1492 hextile[5][0] = -1;
1493 hextile[5][1] = 0;
1494
1495 int[][] lowtri = new int[3][2];
1496 lowtri[0][0] = 0;
1497 lowtri[0][1] = -1;
1498 lowtri[1][0] = 1;
