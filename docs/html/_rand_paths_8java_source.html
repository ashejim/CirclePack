43 * want random points chosen in balanced way, in essence
44 * replacing the path x-->y by x-->(x+y)/2. This should
45 * maintain monotonicity, but not favor places where slope
46 * of homeomorphism is small versus those where's its large.
47 * We simultaneously do the linear interpolation.
48 */
49 if (hX!=null && hY!=null && hX.length==hY.length) {

60 start=ticks[indx];
61 if (indx==ticks.length-1) // reached end of homeo
62 end=1.0;
63 else end=ticks[indx+1];
