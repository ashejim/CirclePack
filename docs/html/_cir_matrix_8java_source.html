95 public static CirMatrix applyTransform(Mobius M, CirMatrix C, boolean oriented) {
96 // get working Mobius
97 Mobius MM=M.cloneMe();
98 if (!oriented) // want M^{-1}?
99 MM=(Mobius)MM.inverse();
100 boolean outsideC=false; // disc is outside of incoming C
101 Complex pt=new Complex(0.0);
102
103 // get working CirMatrix CC as normal circle (or straight line)
104 // and find reference pt "inside"
105 CirMatrix CC=new CirMatrix();
106 CC.a=new Complex(C.a);
107 CC.b=new Complex(C.b);
108 CC.c=new Complex(C.c);
109 CC.d=new Complex(C.d);
110
111 if (C.a.x==-1) { // outside of original circle
112 outsideC=true;
113 CC.a=CC.a.times(-1.0);
114 CC.b=CC.b.times(-1.0);
115 CC.c=CC.c.times(-1.0);
116 CC.d=CC.d.times(-1.0);
117 }
118 if (C.a.x==0) { // original is a line
119 pt=C.c.times(Math.abs(C.d.x)*20); // move in direction of normal
120 }
121 else pt=CC.c.times(-1.0);
122
123 // define G=MM^(-1) * det(MM)
124 Mobius G = new Mobius();
125 G.a = new Complex(MM.d);
126 G.b = MM.b.times(-1.0);
127 G.c = MM.c.times(-1.0);
128 G.d = new Complex(MM.a);
129
130 // compute G^{t}*C: do NOT use usual matrix 'rmult/lmult':
131 // then normalize
132 CirMatrix tmpC=new CirMatrix();
133 tmpC.a=(G.a.times(CC.a)).plus(G.c.times(CC.c));
134 tmpC.b=(G.a.times(CC.b)).plus(G.c.times(CC.d));
135 tmpC.c=(G.b.times(CC.a)).plus(G.d.times(CC.c));
136 tmpC.d=(G.b.times(CC.b)).plus(G.d.times(CC.d));
137
138 // compute tmpC*conj(G)
139 G.conj();
140 CirMatrix outC = new CirMatrix();
141 outC.a=(tmpC.a.times(G.a)).plus(tmpC.b.times(G.c));
142 outC.b=(tmpC.a.times(G.b)).plus(tmpC.b.times(G.d));
143 outC.c=(tmpC.c.times(G.a)).plus(tmpC.d.times(G.c));
144 outC.d=(tmpC.c.times(G.b)).plus(tmpC.d.times(G.d));
145
146 // due to round off, have to identify lines, then normalize
147 if (outC.a.abs()<.0001) {
148 outC.a=new Complex(0.0);
149 double divisor=outC.c.abs();
150 outC.b=outC.b.divide(divisor);
151 outC.c=outC.c.divide(divisor);
152 outC.d=new Complex(outC.d.divide(divisor).x); // should be real
153 }
154
155 // else should be a regular circle; normalize for roundoff
156 // and to insure a = +1
157 else {Complex scalar=outC.a.reciprocal();
158 outC.a=new Complex(1.0);
159 outC.b=outC.b.times(scalar);
160 outC.c=outC.c.times(scalar);
161 outC.d=outC.d.times(scalar);
162 }
163
164 // See if reference point for CC is mapped inside outC
165 // and adjust accordingly. If it's inside outC but
166 // but 'outsideC' is set, of if outC is a line and
167 // pt is not inside, then multiply by -1.
168 Complex MMpt=MM.apply(pt); // outC.getCenter();
169 int cent_inside=pt_inside(outC,MMpt); // outC.getRadius();
170 if ((cent_inside==1 && outsideC) ||
171 cent_inside==-1) { // && outC.a.x==0) { // put on same side of both
172 outC.a=outC.a.times(-1.0);
173 outC.b=outC.b.times(-1.0); // cent_inside=1.0;
174 outC.c=outC.c.times(-1.0);
175 outC.d=outC.d.times(-1.0);
176 }
177
178 return outC;
179 }

190 public static CirMatrix sph2CirMatrix(Complex center, double rad) {
191
192 CirMatrix C=new CirMatrix();
193 C.oriented =true;
194
195 // goes through S pole, so projects to straight line
196 if (Math.abs(center.y + rad - Math.PI) < CM_TOLER) {
197
198 C.a=new Complex(0.0);
199 // set b unit length so conj(b) is towards center
200 double theta=center.x;
201 C.b=new Complex(Math.cos(theta),-Math.sin(theta));
202 C.c=C.b.conj();
203
204 // contains N pole? line through origin
205 if (Math.abs(rad - Math.PI / 2.0) < CM_TOLER)
206 return C;
207
208 // straight line, but NOT through origin
209 double R = Math.sin(center.y - rad) / (1 + Math.cos(center.y - rad));
210 // R = signed eucl distance from origin to the line,
211 C.d=new Complex(R/(-2.0),0);
212 return C;
213 // when C.d.x>0, origin is inside the halfplane, else outside
214 }
215
216 // project to circle
217 CircleSimple sc = SphericalMath.s_to_e_data(center, rad);
218 Complex ez = sc.center;
219 double R = sc.rad;
220 if (sc.flag==-1)
221 R *=-1.0; // set negative
222
223 // this represents the circle
224 C.a=new Complex(1.0);
225 C.b = new Complex(-ez.x, ez.y); // B= - conj(z)
226 C.d = new Complex(ez.abs() * ez.abs() - R * R);
227 C.c=C.b.conj();
228
229 // if we want exterior, multiply through by -1
230 if (R<0) {
231 C.a=C.a.times(-1.0);
232 C.b=C.b.times(-1.0);
233 C.c=C.c.times(-1.0);
234 C.d=C.d.times(-1.0);
235 }
236 return C;
237 }

246 public static int pt_inside(CirMatrix C,Complex pt) {
247
248 if (Double.isNaN(pt.x) || Double.isNaN(pt.y))
249 pt.x=pt.y=1000000000.0;
250
251 if (C.a.abs()!=0 && Math.abs(C.a.abs()-1.0)>.000000000001)
252 throw new MiscException("'C' does not seem to be normalized.");
253
254 int ans=1;
255
256 // regular circle? C.a=+-1
257 if (C.a.abs()>CM_TOLER) {
258 double rad=Math.sqrt(C.a.x*(C.b.absSq()-C.d.x));
259 if (pt.abs()>100000000.0)
260 ans=-1;
261 else {
262 double dist=pt.add(C.c).abs(); // C.c= -center
263 if (dist>(rad+CM_TOLER)) // outside
264 ans=-1;
265 else if (dist>(rad-CM_TOLER)) // on
266 ans=0;
267 }
268 return (int)(C.a.x*ans); // swap if C bounds its outside
269 }
270
271 // line? C.c is unit normal into interior, so see if vector
272 // to point has dot product with it which is > +1
273 if (pt.abs()>10000000.0)
274 return 0; // infinity is on the line
275 double dot=pt.x*C.c.x+pt.y*C.c.y;
276 double transdist=-C.d.x/2.0;
277 if (dot<transdist-CM_TOLER) // outside
278 ans=-1;
279 else if (dot<transdist+CM_TOLER) // on the line
280 ans=0;
281 return ans;
282 }

293 public static CircleSimple cirMatrix_to_geom(CirMatrix CC,int hes) {
294 if (CC==null)
295 return null;
296 CircleSimple outCS=new CircleSimple();
297
298 // typical data
299 Complex ecent=CC.c.times(-1.0); // c entry is -center
300 double reald=CC.d.x; // throw out any extraneous imaginary part
301 if (CC.a.x<.5) { // a.x=-1 ==> all entries were multiplied by -1
302 ecent=ecent.times(-1.0);
303 reald *=-1.0;
304 }
305 double rsq=ecent.absSq()-reald; // rad^2
306 if (CC.a.x!=0 && rsq<=0) {
307 CirclePack.cpb.errMsg("error in a 'CirMatrix'");
308 return null;
309 }
310 // positive eucl radius
311 double erad=Math.sqrt(rsq);
312
313 // sph case, radius/center. See the conventions about inside/outside,
314 if (hes>0) {
315 // circle is a straight line (goes through south pole)
316 if (CC.a.abs() < CirMatrix.CM_TOLER) {
317 // through origin? Hence a hemisphere
318 if (CC.d.abs() < CirMatrix.CM_TOLER) {
319 outCS.center.y = outCS.rad = Math.PI / 2.0;
320 outCS.center.x=CC.b.conj().arg();
321 return outCS;
322 }
323
324 // straight line, but NOT through origin
325 double R=CC.d.abs(); // distance to origin
326 double theta=Math.atan2(-1.0*CC.b.y,CC.b.x);
327 double atn=Math.atan(R);
328 double rho=Math.PI/2.0-atn;
329 if (CC.d.x<0) // encloses origin (north pole)?
330 rho+=2.0*atn;
331 outCS.rad=rho;
332 outCS.center=new Complex(theta,Math.PI-rho);
333
334 return outCS;
335 } // end of 'straight line' cases
336
337 // else a circle
338 CircleSimple sc=SphericalMath.e_to_s_data(ecent, erad);
339 outCS.center=sc.center;
340 outCS.rad=sc.rad;
341 if (CC.a.x<0) { // want outside of euclidean circle
342 outCS.center=SphericalMath.getAntipodal(sc.center);
343 outCS.rad=Math.PI-sc.rad;
344 }
345 return outCS;
346 } // done with sph
347
348 // hyp case: return null if circle is not in unit disc
349 if (hes<0) {
350 if (CC.a.x<=0) { // straight line or outside
351 CirclePack.cpb.errMsg("Improper hyp conversion of 'CirMatrix'");
352 return null;
353 }
354 if (ecent.abs()+erad>1.0) // not in disc
355 return null;
356
357 return HyperbolicMath.e_to_h_data(ecent, erad);
358 }
359
360 // else eucl; watch for line
361 if (CC.a.x==0) { // yes, is a line
362 outCS.lineFlag=true;
363 Complex unitnormal=CC.c;
364 outCS.center=unitnormal.times(CPBase.FAUX_RAD); // unit normal toward interior
365
366 outCS.rad=CPBase.FAUX_RAD+(CC.d.x/2.0); // FAUX_RAD - signed distance from origin
367 return outCS;
368 }
369 outCS.center=ecent;
370 outCS.rad=erad*CC.a.x; // may be negative if a=-1
371 return outCS;
372 }
