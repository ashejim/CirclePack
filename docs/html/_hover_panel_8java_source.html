160 public Point setXY(MouseEvent me) {
161 // if the user mouses over the component
162 Point mousePoint=null;
163 if (me==null)
164 new Point(XLoc,YLoc);
165 else
166 mousePoint = new Point(me.getX(), me.getY());
167 SwingUtilities.convertPointToScreen(mousePoint, me.getComponent());
168 int x = (int) mousePoint.getX() + XOffset;
169 int y = (int) mousePoint.getY() + YOffset;
170
171 // if the hoverFrame would display off the screen
172 // move it to other side of mouse
173 if (x + myWidth > PackControl.displayDimension.width) x = x - myWidth - 10;
174 if (y + myHeight > PackControl.displayDimension.height) y = y - myHeight - 10;
175
176 return new Point(x,y);
177 }
