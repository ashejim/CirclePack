128 public void actionPerformed(ActionEvent aev) {
129 String cmd=aev.getActionCommand();
130 Component cmpt=VFETabbing.getSelectedComponent();
131 if (cmpt==null) return;
132 StringBuilder theCmd=new StringBuilder();
133 PackData p=CirclePack.cpb.getActivePackData();
134
135 if (cmd.equals("getLoc")) {
136 if (cmpt==(Component)vertTab) {
137 Iterator<Integer> vlist=p.vlist.iterator();
138 while (vlist.hasNext() && theCmd.length()<2000) {
139 theCmd.append(Integer.toString((Integer)vlist.next())+" ");
140 }
141 VlistArea.setText(theCmd.toString());
142 }
143 else if (cmpt==(Component)edgeTab) {
144 Iterator<EdgeSimple> elist=p.elist.iterator();
145 while (elist.hasNext() && theCmd.length()<2000) {
146 EdgeSimple edge=(EdgeSimple)elist.next();
147 theCmd.append(Integer.toString(edge.v)+" "+
148 Integer.toString(edge.w)+" ");
149 }
150 ElistArea.setText(theCmd.toString());
151 }
152 if (cmpt==(Component)faceTab) {
153 Iterator<Integer> flist=p.flist.iterator();
154 while (flist.hasNext() && theCmd.length()<2000) {
155 theCmd.append(Integer.toString((Integer)flist.next())+" ");
156 }
157 FlistArea.setText(theCmd.toString());
158 }
159 return;
160 }
161
162 boolean append=false;
163 if (cmd.startsWith("append"))
164 append=true;
165
166 if (cmpt==(Component)vertTab) {
167 theCmd.append("set_vlist ");
168 if (append) theCmd.append("vlist ");
169 theCmd.append(VlistArea.getText());
170 }
171 else if (cmpt==(Component)edgeTab) {
172 theCmd.append("set_elist ");
173 if (append) theCmd.append("elist ");
174 theCmd.append(ElistArea.getText());
175 }
176 else if (cmpt==(Component)faceTab) {
177 theCmd.append("set_flist ");
178 if (append) theCmd.append("flist ");
179 theCmd.append(FlistArea.getText());
180 }
181
182 try {
183 TrafficCenter.cmdGUI(CirclePack.cpb.getActivePackData(),theCmd.toString());
184 } catch (Exception ex) {
185 throw new ParserException("Error processing list tab");
186 }
187 }
