306 public int []findContig578() {
307 int []ans=new int[4];
308 int type=tntype;
309 if ((type!=CPTreeNode.COMMAND && type!=CPTreeNode.MODE &&
310 type!=CPTreeNode.FILE && type!=CPTreeNode.LINEUP) ||
311 stackBox.isOpen) {
312 ans[0]=-1;
313 return ans;
314 }
315 CPTreeNode parTN=(CPTreeNode)getParent();
316 int childCount=ans[3]=parTN.getChildCount();
317 int refspot=CPTreeNode.getMyIndex(parTN,this);
318 if (refspot<0) {
319 throw new MiscException("Error is finding node in tree");
320 }
321
322 ans[0]=ans[1]=ans[2]=refspot;
323 CPTreeNode aTN;
324 while (ans[0]>0 && (aTN=(CPTreeNode)parTN.getChild(ans[0]-1)).tntype==type &&
325 !aTN.stackBox.isOpen()) {
326 ans[0]--;
327 }
328
329 while (ans[2]<(childCount-1) && (aTN=(CPTreeNode)parTN.getChild(ans[2]+1)).tntype==type &&
330 !aTN.stackBox.isOpen()) {
331 ans[2]++;
332 }
333 return ans;

343 public int consolidateNodes() {
344
345 int count=0;
346 if ((tntype!=CPTreeNode.CPSCRIPT && tntype!=CPTreeNode.CPDATA &&
347 tntype!=CPTreeNode.SECTION) || !stackBox.isOpen())
348 return -1;
349 int spot;
350 boolean hit=true;
351 while (hit) {
352 hit=false;
353 for (spot=getChildCount()-1;spot>=0;spot--) {
354 CPTreeNode spotNode=getChild(spot);
355 int []contigData=spotNode.findContig578();
356 int n;
357 Vector<CPTreeNode> nodeVec=new Vector<CPTreeNode>();
358 CPTreeNode lineTN;
359
360 // first, do we have 'LineSbox's to consolidate?
361 if (spotNode.tntype==CPTreeNode.LINEUP && contigData[0]>=0) {
362 n=(contigData[1]-contigData[0])+1;
363 if (n>1) {
364 // gather nodes from the downstream 'LineSBox's
365 for (int j=n-1;j>0;j--) {
366 lineTN=getChild(contigData[0]+j);
367 int kidCount=lineTN.getChildCount();
368 for (int k=(kidCount-1);k>=0;k--)
369 nodeVec.add((CPTreeNode)lineTN.getChild(k));
370 remove(lineTN);
371 }
372 // put them into the first; careful with order
373 lineTN=getChild(contigData[0]);
374 for (int v=nodeVec.size()-1;v>=0;v--)
375 lineTN.add((MutableTreeNode)nodeVec.elementAt(v));
376 count++;
377 hit=true;
378 spot=contigData[0];
379 }
380 }
381
382 // do we have nodes to consolidate?
383 else if ( (((tntype==CPTreeNode.CPSCRIPT || tntype==CPTreeNode.SECTION) &&
384 (spotNode.tntype==CPTreeNode.COMMAND || spotNode.tntype==CPTreeNode.MODE)) ||
385 (tntype==CPTreeNode.CPDATA && spotNode.tntype==CPTreeNode.FILE)) &&
386 contigData[0]>=0) {
387 n=(contigData[1]-contigData[0])+1;
388 if (n>1) {
389 nodeVec=new Vector<CPTreeNode>(n+1);
390 for (int j=0;j<n;j++) { // have to get in right order
391 int m=contigData[0]+j;
392 nodeVec.add(j,(CPTreeNode)getChild(m));
393 }
394 for (int j=(n-1);j>=0;j--) {
395 int m=contigData[0]+j;
396 remove(m);
397 }
398
399 // create new 'LineSBox' node, move these nodes into it
400 CPTreeNode newTN=new CPTreeNode((String)null,CPTreeNode.LINEUP);
401 LineSBox lsb=(LineSBox)(newTN.stackBox);
402 lsb.fillEmIn(nodeVec);
403 insert((MutableTreeNode)lsb.tNode,contigData[0]);
404 count++;
405 lsb.redisplaySB(lsb.myWidth);
406 hit=true;
407 spot=contigData[0];
408 }
409 else { // only one node, closed and in DISPLAY mode
410
411 // absorb into following 'LineSBox'?
412 if (contigData[0]<(getChildCount()-1)) {
413 CPTreeNode postTN=getChild(contigData[0]+1);
414 if (postTN.tntype==CPTreeNode.LINEUP) {
415 postTN.insert(getChild(contigData[0]),0);
416 count++;
417 hit=true;
418 spot=contigData[0]-1;
419 }
420 }
421 // add to preceeding 'LineSBox'?
422 else if (contigData[0]>0) {
423 CPTreeNode postTN=getChild(contigData[0]-1);
424 if (postTN.tntype==CPTreeNode.LINEUP) {
425 postTN.add(getChild(contigData[0]));
426 count++;
427 hit=true;
428 spot=contigData[0]-1;
429 }
430 }
431
432 }
433 }
434 }
435 } // end of for loop on 'spot'
436 return count;
437 }
