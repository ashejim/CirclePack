41 public static boolean doesSegXTri(Complex p1,Complex p2,
42 Complex z1,Complex z2,Complex z3) {
43
44 // one point inside or on?
45 BaryPoint bp1=EuclMath.e_pt_to_bary(p1, z1, z2, z3);
46 if (BaryPoint.baryPtInside(bp1)>=0) return true;
47 BaryPoint bp2=EuclMath.e_pt_to_bary(p2, z1, z2, z3);
48 if (BaryPoint.baryPtInside(bp2)>=0) return true;
49
50 // thus, p1 or p2 must be on opposite side from some vert.
51
52 // both on opposite of any one of edges?
53 if ((bp1.b0<0 && bp2.b0<0) || (bp1.b1<0 && bp2.b1<0) ||
54 (bp1.b2<0 && bp2.b2<0)) return false;
55
56 // p1 must be on neg side of some edge, p2 across that edge.
57 double vx=p2.x-p1.x;
58 double vy=p2.y-p1.y;
59 double d1=(z1.x-p1.x)*vx+(z1.y-p1.y)*vy;
60 double d2=(z2.x-p1.x)*vx+(z2.y-p1.y)*vy;
61 double d3=(z3.x-p1.x)*vx+(z3.y-p1.y)*vy;
62 // if all dot products have same sign, false
63 if ((d1>=0 && d2>=0 && d3>=0) || (d1<=0 && d2<=0 && d3<=0))
64 return false;
65 return true;
66 }
