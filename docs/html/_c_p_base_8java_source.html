205 public static URL getResourceURL(String path) {
206 try {
207 ClassLoader cll = CPBase.sharedinstance.getClass().getClassLoader();
208
209 // null ClassLoader? look in user directory
210 if( cll==null || cll.getResource("Resources"+path)==null ) {

212 +File.separator+"Resources"+path);
213 return file.toURI().toURL();

216 // got ClassLoader? look there
217 else {
218 return cll.getResource("Resources"+path);

265 /* Extracts files in Resources/executables to the TempDirectory.
266 * E.g. qhull.exe and triangle.exe used for Delaunay triangulations.
267 * TODO: have to generate and load executables for other operating
268 * systems, e.g., mac.
269 * @param destDir destination directory.
270 * @param execName filename
271 * @return boolean
272 */
273 private boolean extractExeFiles(String destDir,String execName) {
274
275 InputStream ins=null;
276 try {
277 ins = getClass().getResourceAsStream("/Resources/executables/"+execName);
278 } catch (Exception ex) {
279 System.err.println("Didn't get InputStream for executables in 'CPBase'");
280 return false;
281 }
282
283 try {
284 java.io.File f = new java.io.File(destDir + java.io.File.separator + execName);
285 java.io.FileOutputStream fos = new java.io.FileOutputStream(f);
286 while (ins.available() > 0) { // write contents of 'ins' to 'fos'
287 fos.write(ins.read());
288 }
289 fos.close();
290 } catch (Exception iox) {
291 System.err.println("exception in writing an executable");
292 return false;
293 }
294
295 return true;
296 }
