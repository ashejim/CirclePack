171 Graphics2D g2d=bufImage.createGraphics();
172 if (colr==null)
173 colr=ColorUtil.getFGColor();
174 g2d.setColor(colr);
175 if (trink<0) trink=1;
176 else if (trink>=trinkets.size())
177 trink=trinkets.size()-1;
178 Path2D.Double gpath=new Path2D.Double(trinkets.get(trink));
179 gpath.closePath();
180 gpath.transform(new AffineTransform(factor,0,0,factor,0,0));
181// g2d.fill(gpath);
182 Color old_color=g2d.getColor();
183 if (colr==null)
184 g2d.setColor(Color.BLACK);
185 else
186 g2d.setColor(colr);
187 if (fill) {
188 g2d.fill(gpath);
189 }
190 g2d.draw(gpath);
191 g2d.setColor(old_color);
192 if (geometry>0) { // spherical, convert (theta,phi) to (y,z) of screen
193 pix_x=parent.toPixX(Math.sin(y)*Math.sin(x));
194 pix_y=parent.toPixY(Math.cos(y));
195 }
196 else {
197 pix_x=parent.toPixX(x);
198 pix_y=parent.toPixY(y);
199 }
200 // put in image buffer and check if it intersects the view
201 Graphics2D g2=parent.imageContextReal;
202 Rectangle imgrect=new Rectangle((int)(pix_x-side/2),(int)(pix_y-side/2),side+1,side+1);
203 if (imgrect.intersects(g2.getClipBounds())) // view check
204 g2.drawImage(bufImage,new AffineTransform(1,0,0,1,pix_x-side/2,pix_y-side/2),null);
205 }
