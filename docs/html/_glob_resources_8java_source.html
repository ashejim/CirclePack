86 public static String glob2regex(String g) {
87 // pattern that matches any number of characters that aren't '*' or
88 // '?' followed by one of those characters or the end of the string.
89 Pattern p = Pattern.compile("([^\\*\\?]*)([\\*\\?]|$)");
90 // matcher to look through matches to p in g
91 Matcher m = p.matcher(g);
92 // I'll build up my regular expression in this string
93 StringBuilder regex = new StringBuilder();
94 // glob pattern has to start matching at the beginning
95 regex.append("^");
96 String literal;
97 String special;
98 while (m.find()) {
99 literal = m.group(1); // literal string before special
100 special = m.group(2); // "*", "?", or "" (at end of string)
101 // quote literal in case it contains special characters
102 regex.append(Pattern.quote(literal));
103 // convert '*' or '?' to equivalent regular expression
104 if (special.equals("*")) {
105 regex.append(".*");
106 } else if (special.equals("?")){
107 regex.append(".");
108 } else {
109 // must be empty string, nothing to append
110 }
111 }
112 // glob pattern has to match all the way to the end
113 regex.append("$");
114 return regex.toString();
115 }

125 static Set<String> globJarURL(URL url, List<String> globs) {
126 try {
127 Set<String> results = new TreeSet<String>();
128 // convert glob patterns with path to regular expressions
129 List<String> patterns = glob2regexList(globs);
130 JarURLConnection conn = (JarURLConnection)url.openConnection();
131 // get the name of the directory from the URL
132 String dir = conn.getJarEntry().getName();
133 // make sure the directory ends with a '/'
134 dir = dir.replaceFirst("/?$", "/");
135 // find all entries from the jar file that match the patterns.
136 Enumeration<JarEntry> entries = conn.getJarFile().entries();
137 String path; // path within the jar file
138 while (entries.hasMoreElements()) {
139 path = entries.nextElement().getName();
140 if (path.startsWith(dir)) {
141 String name = path.substring(dir.length(), path.length());
142 for (String p : patterns) {
143 if (name.matches(p)) {
144 results.add(name);
145 break;
146 }
147 }
148 }
149 }
150 return results;
151 } catch (IOException e) {
152 e.printStackTrace();
153 }
154 return Collections.emptySet();
155 }
