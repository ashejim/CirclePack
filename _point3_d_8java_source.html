330 public Complex projToHorizon() {
331 double norm=Math.sqrt(y*y+z*z);
332 if (norm < .000000001) return new Complex(0,0);
333 double Y=y/norm;
334 double Z=z/norm;

401 public static double triArea(Point3D p0,Point3D p1,Point3D p2) {
402 double a=distance(p0,p1);
403 double b=distance(p1,p2);
404 double c=distance(p2,p0);
405 // Heron's formula in the numerically friendlier factored form
406 double ans=Math.sqrt((a+b+c)*(a+b-c)*(a+c-b)*(b+c-a))/4.0;
407 if (Double.isNaN(ans))
408 ans=0.0;
409 return ans;
410 }
