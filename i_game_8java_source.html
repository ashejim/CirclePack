83 public void set4Aspect() {
84 // get 'factor' for multiplying imaginary part
85 double wide=(extenderPD.getCenter(corner[1])).minus(extenderPD.getCenter(corner[0])).abs();
86 double high=(extenderPD.getCenter(corner[2])).minus(extenderPD.getCenter(corner[1])).abs();
87 double factor=wide/(high*GameAspect);
88
89 // find average of 4 corner locations
90 Complex c0=extenderPD.getCenter(corner[0]);
91 Complex c1=extenderPD.getCenter(corner[1]);
92 Complex c2=extenderPD.getCenter(corner[2]);
93 Complex c3=extenderPD.getCenter(corner[3]);
94 Complex mid=c0.add(c1).add(c2).add(c3);
95 mid=mid.times(.25);
96
97 // adjust all centers
98 for (int v=1;v<=extenderPD.nodeCount;v++) {
99 Complex z=extenderPD.getCenter(v).minus(mid);
100 z.y*=factor;
101 extenderPD.setCenter(v,z);
102 }

148 public int eatTarget(int playerNum,int targetNum) {
149 int p=player[playerNum];
150 int t=targets[targetNum];
151 int oldV=t;
152 if (p>t)
153 oldV=p;
154 if (extenderPD.nghb(p,t)>=0) {
155 HalfEdge edge=extenderPD.packDCEL.findHalfEdge(p,t);
156 int rslt=RawManip.meldEdge_raw(extenderPD.packDCEL,edge);
157 if (rslt<=0)
158 return -1;
159 extenderPD.packDCEL.fixDCEL(extenderPD);
160 extenderPD.fillcurves();
161
162 // which vertex survives?
163 int fused=t;
164 if (rslt==t)
165 fused=p;
166 player[playerNum]=fused;
167 for (int i=0;i<playerCount;i++) {
168 if (i!=playerNum && player[i]>oldV) player[i]--;
169 if (targets[i]>oldV) targets[i]--;
170 }
171 for (int i=0;i<4;i++)
172 if (corner[i]>oldV) corner[i]--;
173 Color col=playerCOlor[playerNum];
174 extenderPD.setCircleColor(player[playerNum],new Color(col.getRed(),col.getGreen(),col.getBlue()));
175 return fused;
176 }
177 return -1;
178 }

186 public int crunchEdge(int v,int u) {
187 HalfEdge edge=extenderPD.packDCEL.findHalfEdge(v,u);
188 if (edge==null ||
189 (extenderPD.isBdry(v) && extenderPD.isBdry(u)))
190 return 0;
191 int rslt=RawManip.meldEdge_raw(extenderPD.packDCEL,edge);
192 if (rslt!=0)
193 extenderPD.packDCEL.fixDCEL(extenderPD);
194 return rslt;

259 if (cmd.startsWith("rand")) {
260 int n=20;
261 try {
262 items=flagSegs.get(0);
263 n=Integer.parseInt((String)items.get(0));
264 } catch (Exception ex) { }

272 int v=0;

281 extenderPD.countFaces(v)!=3)
282 Oops("invalid choice");

285 }

288 if (cmd.startsWith("addT")) {

292 } catch (Exception ex) {}

303 } catch (Exception ex) {

307 Oops("invalid choice");

319 if (cmd.startsWith("set_game")) {
320 try {
321 items=flagSegs.get(0);
322 gameMode=Integer.parseInt((String)items.get(0));
323 } catch (Exception ex) {
324 gameMode=0;
325 Oops("didn't get valid game mode: "+ex.getMessage());
326 }

330 // ======== eat ===========
331 if (cmd.startsWith("eat")) {
332 try {
333 items=flagSegs.get(0);
334 EdgeSimple edge=EdgeLink.grab_one_edge(extenderPD,flagSegs);
335 if (edge==null)
336 throw new DataException("");
337 } catch (Exception ex) {
338 Oops("didn't get valid edge: "+ex.getMessage());
339 }
340 return eatTarget(0,0);
341 }
342
343 // ======== jump ==========
344 if (cmd.startsWith("jump")) {
345 EdgeSimple edge=null;
346 try {
347 items=flagSegs.get(0);
348 edge=EdgeLink.grab_one_edge(extenderPD,flagSegs);
349 if (edge==null)
350 throw new DataException("");
351 } catch (Exception ex) {
352 Oops("didn't get valid edge: "+ex.getMessage());
353 }
354 if (edge.v==player[currentPlayer])
355 player[currentPlayer]=edge.w;
356 else if (edge.w==player[currentPlayer])
357 player[currentPlayer]=edge.v;
358 Color col=playerCOlor[currentPlayer];
359 extenderPD.setCircleColor(player[currentPlayer],new Color(col.getRed(),col.getGreen(),col.getBlue()));
360 return 1;
361 }
362
363 // ======== set_player =======
364 if (cmd.startsWith("set_play")) {
365 try{
366 items=flagSegs.get(0);
367 int cp=Integer.parseInt((String)items.get(0));
368 if (cp<0 && cp>=playerCount) {
369 currentPlayer=cp;
370 return cp;
371 }
372 return -1;
373 } catch (Exception ex) {
374 Oops("illegal player number: "+ex.getMessage());
375 }
376 return -1;
377 }
378
379 // ======== players ========
380 if (cmd.startsWith("play")) {
381 for (int vv=1;vv<=extenderPD.nodeCount;vv++)
382 extenderPD.setVertMark(vv,-1);
383 try {
384 items=flagSegs.get(0);
385 playerCount=items.size();
386 if (playerCount>5) playerCount=5;
387 player=new int[playerCount];
388 playerCOlor=new Color[playerCount];
389 targets=new int[playerCount];
390 for (int j=0;j<playerCount;j++) {
391 player[j]=Integer.parseInt((String)items.get(j));
392 playerCOlor[j]=ColorUtil.spreadColor(j%16);
393 targets[j]=randNonPlayer(player[j]);
394 if (player[j]<=0 || player[j]>=extenderPD.nodeCount ||
395 extenderPD.isBdry(player[j])) {
396 player=null;
397 playerCOlor=null;
398 targets=null;
399 Oops("improper player specified");
400 }
401 extenderPD.setVertMark(player[j],j);
402 }
403 } catch (Exception ex) {
404 player=null;
405 playerCOlor=null;
406 targets=null;
407 Oops("problem in specifying 'player'");
408 }
409 return playerCount;
410 }
411
412 // ======== display =========
413 if (cmd.startsWith("disp")) {
414 return display(-1);
415 }
416
417 return 0;
418 }
