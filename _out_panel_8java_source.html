234 public static int outputter(BufferedWriter fp,PackData p,
235 String prefix,String datastr,String loopstr,String suffix) {
236
237 Vector<DataObj> dataObjects=null;
238 // process 'prefix' section
239 prefix=parsePrefix(p,prefix);
240 if (prefix!=null) {
241 try {
242 fp.write(prefix+" ");
243 } catch (Exception ex) {
244 return 0;
245 }
246 }
247
248 // First, check if loopstr exists: this guides the 'data' search
249 boolean loopstate=true;
250 if (loopstr==null || loopstr.trim().length()==0) loopstate=false;
251
252 // If 'non-looping', no lists, perhaps we're after data about packing
253 if (!loopstate) { // non-looping data:
254 boolean hit=true;
255 StringBuilder loopbuf=new StringBuilder(datastr);
256 String lit=null;
257 String thing=null;
258 while (hit && loopbuf.length()>0) {
259 hit=false;
260 // look for literal first
261 if ((lit=StringUtil.find_literal(loopbuf))!=null && lit.length()>0) {
262 try {
263 fp.write(lit);
264 } catch (Exception e) {}
265 hit=true;
266 }
267 else if ((thing=StringUtil.grabNext(loopbuf))!=null && thing.length()>0) {
268 switch(thing.charAt(0)) {
269 case '_': // variable indicator
270 {
271 String var_lit_str=StringUtil.varSub(thing);
272 try {
273 fp.write(var_lit_str+" "); // add trailing space
274 } catch (Exception e) {}
275 hit=true;
276 break;
277 }
278 case 'M':
279 {
280 if (thing.equals("Mob")) { // current mobius
281 Mobius mob=CPBase.Mob;
282 try {
283 fp.write(mob.a.x+" "+mob.a.y+"\n");
284 fp.write(mob.b.x+" "+mob.b.y+"\n");
285 fp.write(mob.c.x+" "+mob.c.y+"\n");
286 fp.write(mob.d.x+" "+mob.d.y+"\n\n");
287 hit=true;
288 } catch (Exception e) {}
289 }
290 break;
291 }
292 case 'P':
293 {
294 if (thing.equals("PATH")) { // current path
295 Vector<Vector<Complex>> path=PathUtil.gpPolygon(CPBase.ClosedPath);
296 Iterator<Vector<Complex>> ZP=path.iterator();
297 while (ZP.hasNext()) {
298 Vector<Complex> loc_path=(Vector<Complex>)ZP.next();
299 Iterator<Complex> zp=loc_path.iterator();
300 Complex z1=(Complex)zp.next();
301 Complex z=null;
302 try {
303 fp.write(+z1.x+" "+z1.y+" m\n");
304 while (zp.hasNext()) {
305 z=(Complex)zp.next();
306 fp.write(z.x+" "+z.y+" l\n");
307 }
308 // intended to be closed
309 if (z1.minus(z).abs()<.000001) fp.write("cp \n");
310 fp.write("s\n"); // stroke
311 } catch (IOException iox) {}
312 } // end of while
313 hit=true;
314 }
315 break;
316 }
317 case 't': // trace^2 of given mobius
318 {
319 if (thing.equals("tr")) {
320 Complex tr2=CPBase.Mob.getTraceSqr();
321 try {
322 fp.write(tr2.x+" "+tr2.y+"\n");
323 } catch(Exception e) {}
324 hit=true;
325 }
326 break;
327 }
328 case '#': {
329 if (thing.equals("#N")) { // packing nodecount
330 try {
331 fp.write(p.nodeCount);
332 } catch(Exception e) {}
333 hit=true;
334 }
335 else if (thing.equals("#F")) { // packing facecount
336 try {
337 fp.write(p.faceCount);
338 } catch(Exception e) {}
339 hit=true;
340 }
341 break;
342 }
343 case 'C': { // 'CPcallPacket' for passing computed data
344 if (thing.equals("CP")) { // print double value from 'CPcallPacket'
345 CallPacket cp=CPBase.CPcallPacket;
346 if (cp!=null && !cp.error) {
347 try {
348 fp.write(Double.toString(cp.double_vec.get(0)));
349 } catch (Exception e) {}
350 hit=true;
351 }
352 }
353 break;
354 }
355 } // end of switch
356 if (!hit) { // not found? treat as literal --- e.g. a numerical value
357 try {
358 fp.write(thing);
359 } catch (Exception e) {}
360 hit=true;
361 }
362 }
363 } // end of while
364 } // end of non-looping data
365
366 // looping data: process 'data' section
367 else {
368 StringBuilder dataBuffer=new StringBuilder(datastr);
369 dataObjects=parseData(p,dataBuffer);
370 if (dataObjects.size()==0) {
371 CirclePack.cpb.myErrorMsg("error: output: problems parsing data section");
372 return 0;
373 }
374 }
375
376 // process 'loop' section
377 int count=0;
378 if (loopstate) {
379 loopstr=loopstr.trim();
380 count = outputLoop(fp,p,loopstr,dataObjects);
381// if (count==0) {
382// CirclePack.cpb.myErrorMsg("error: output: error in looping description");
383// return 0;
384// }
385 }
386
387 // process 'suffix' section
388 suffix=suffix.trim();
389 suffix=parseSuffix(p,suffix);
390 if (suffix!=null && suffix.length()>0) {
391 try {
392 fp.write(suffix);
393 } catch (Exception ex) {
394 return 0;
395 }
396 }
397
398 // clean up and return
399 try {
400 fp.flush();
401 fp.close();
402 } catch (Exception ex) {
403 CirclePack.cpb.myErrorMsg("error: output: some error is closing the file");
404 return -1;
405 }
406 if (count==0)
407 return 1;
408 return count;
409 }

418 public static Vector<DataObj> parseData(PackData p,StringBuilder dataBuffer) {
419 Vector<DataObj> dataObjs=new Vector<DataObj>(5);
420 Vfe=0;
421 String nextStr=null;
422 while (dataBuffer.length()>0) { // cycle through data objects
423 int k=0;
424 while (k<dataBuffer.length() && dataBuffer.charAt(k)<=' ') k++;
425 if (k==dataBuffer.length()) break;
426 if (dataBuffer.charAt(k)=='\"' || dataBuffer.charAt(k)=='\'') { // literal
427 if ((nextStr=StringUtil.find_literal(dataBuffer))!=null) {
428 DataObj dObj=new DataObj();
429 dObj.spec=new String(nextStr);
430 dataObjs.add(dObj);
431 }
432 // didn't find literal?
433 else dataBuffer.deleteCharAt(k); // to prevent infinite loop
434 }
435 else { // NOTE: 'Vfe' update may disqualify previous 'dataObjs'
436 int kk=k;
437 while (kk<dataBuffer.length() && dataBuffer.charAt(kk)>' ')
438 kk++;
439 nextStr=dataBuffer.substring(k,kk);
440 dataBuffer.delete(k,kk);
441 DataObj dobj=data_obj_parse(nextStr);
442 if (dobj!=null && dobj.code!=dataCode.NULL) {
443 Vfe=dobj.vfe;
444 dataObjs.add(dobj);
445 }
446 }
447 } // end of while
448 return dataObjs;
449 }

451 public static int print_vert_obj(PackData p,BufferedWriter fp,
452 dataCode code,int v) {
453 int w;
454 Complex z=null;
455
456 try {
457 if (code==dataCode.VERT_INDEX) {
458 fp.write(Integer.toString(v)+" ");
459 return 1;
460 }
461 if (code==dataCode.VERT_FLOWER) {
462 StringBuilder strbld=new StringBuilder("");
463 int[] flower=p.getFlower(v);
464 for (int j=0;j<flower.length;j++)
465 strbld.append(flower[j]+" ");
466 fp.write(strbld.toString());
467 return 1;
468 }
469 if (code==dataCode.VERT_CENTER) {
470 z=p.getCenter(v);
471 fp.write(z.x+" "+z.y+" ");
472 return 1;
473 }
474 if (code==dataCode.VERT_CURV) {
475 fp.write(p.getCurv(v)+" ");
476 return 1;
477 }
478 if (code==dataCode.VERT_AIM) {
479 fp.write(p.getAim(v)+" ");
480 return 1;
481 }
482 if (code==dataCode.VERT_DEG) {
483 fp.write(Integer.toString((p.countFaces(v)+p.getBdryFlag(v)))+" ");
484 return 1;
485 }
486 if (code==dataCode.VERT_COLOR) {
487 Color col=p.getCircleColor(v);
488 fp.write(col.getRed()+" "+col.getGreen()+" "+col.getBlue()+" ");
489 return 1;
490 }
491 if (code==dataCode.VERTEX_MAP && p.vertexMap!=null && (w=p.vertexMap.findW(v))>0) { // vertex map image for v
492 fp.write(w+" ");
493 return 1;
494 }
495 if (code==dataCode.VERT_RADII) {
496 fp.write(p.getRadius(v)+" ");
497 return 1;
498 }
499 if (code==dataCode.VERT_XYZ) {
500 double []xyz;
501
502 // have xyz data stored?
503 if (p.xyzpoint!=null) {
504 Point3D pd=p.xyzpoint[v];
505 xyz=new double[3];
506 xyz[0]=pd.x;
507 xyz[1]=pd.y;
508 xyz[2]=pd.z;
509 }
510
511 else if (p.hes>0) {
512 xyz=SphericalMath.s_pt_to_vec(p.getCenter(v));
513 }
514 else { // eucl/hyp, just store flat data
515 xyz=new double[3];
516 xyz[0]=p.getCenter(v).x;
517 xyz[1]=p.getCenter(v).y;
518 xyz[2]=0.0;
519 }
520
521 fp.write(xyz[0]+" "+xyz[1]+" "+xyz[2]+" ");
522 return 1;
523 }
524 if (code==dataCode.PAVER_CORNERS) { // a 'paver' is polygon
525 // formed by union of faces containing a given vertex;
526 // we save the open list of polygon corners. For
527 // boundary v, include corner at v itself.
528 Complex []pts=p.corners_paver(v); // non-closed list
529 int num=pts.length;
530 fp.write(pts[0].x+" "+pts[0].y+"i ");
531 for (int j=1;j<num;j++) {
532 fp.write(pts[j].x+" "+pts[j].y+"i ");
533 }
534 fp.write("\n");
535 return 1;
536 }
537 if (code==dataCode.SHARP_PQ) {
538 double shp;
539 if (pData.status && qData.status && v<qData.nodeCount) {
540 shp=qData.getActualRadius(v)/pData.getActualRadius(v);
541 fp.write(shp+" ");
542 return 1;
543 }
544 }
545 } catch (Exception ex) {}
546
547 return 0;
548 }

646 public static int outputLoop(BufferedWriter fp,PackData p,
647 String loopstr,Vector<DataObj> dataobj) {
648 int count=0;
649
650 try {
651 // output calls for vertex indices? expect vert list
652 if (Vfe==1) {
653 int v;
654 NodeLink vertlist= new NodeLink(p,loopstr);
655
656 if (vertlist.size()==0)
657 throw new ParserException("error: output: no vertices specified");
658 Iterator<Integer> vlist=vertlist.iterator();
659 while (vlist.hasNext()) { // loop through vertices
660 v=(Integer)vlist.next();
661 for (int i=0;i<dataobj.size();i++) {
662 DataObj dtrace=(DataObj)dataobj.get(i);
663 // loop through objects, stopping at any improper one.
664 if (dtrace!=null && (dtrace.spec!=null || dtrace.vfe==Vfe)) {
665 if (dtrace.spec!=null) { // literal
666 fp.write(dtrace.spec);
667 count++;
668 }
669 else count+=print_vert_obj(p,fp,dtrace.code,v);
670 }
671 } // done going through objects
672 } // done going through vertices
673 return count;
674 }
675
676 // output calls for face indices? expect face list
677 if (Vfe==2) {
678 int f;
679 FaceLink facelist= new FaceLink(p,loopstr);
680
681 if (facelist.size()==0)
682 throw new ParserException("error: output: no faces specified");
683 Iterator<Integer> flist=facelist.iterator();
684 while (flist.hasNext()) { // loop through vertices
685 f=(Integer)flist.next();
686 for (int i=0;i<dataobj.size();i++) {
687 DataObj dtrace=(DataObj)dataobj.get(i);
688 // loop through objects, stopping at any improper one.
689 if (dtrace!=null && (dtrace.spec!=null|| dtrace.vfe==Vfe)) {
690 if (dtrace.spec!=null) { // literal
691 fp.write(dtrace.spec);
692 count++;
693 }
694 else {
695 count+=print_face_obj(p,fp,dtrace.code,f);
696 }
697 }
698 } // done going through objects
699 } // done going through faces
700 return count;
701 }
702
703 // output calls for edges? expect edge list
704 else if (Vfe==3) {
705 EdgeLink edgelist=new EdgeLink(p,loopstr);
706

715 // loop through objects, stopping at any improper one.
716 if (dtrace!=null && (dtrace.spec!=null|| dtrace.vfe==Vfe)) {
717 if (dtrace.spec!=null) { // literal
718 fp.write(dtrace.spec);

720 }
721 else count+=print_edge_obj(p,fp,dtrace.code,edge.v,edge.w);
722 }
723 } // done going through objects
724 } // done going through vertices
725 return count;
726 }

729 else if (Vfe==4) {

740 Mobius mb=p.namedSidePair(mlabels[k]);

742 fp.write(mb.mob2String().toString()+"\n\n");
743 }
744 }
745 }
746 } // end of while
747 }
748
749 // error
750 else {
751 CirclePack.cpb.myErrorMsg("output: didn't find proper list.");
752 return 0;
753 }
754 } catch (Exception ex) {}
755 return 1;

770 public static DataObj data_obj_parse(String datastr) {
771 if (datastr==null || datastr.length()==0) return null;
772 datastr=datastr.trim();
773 if (datastr.length()==0) return null;
774 char c=datastr.charAt(0);
775 DataObj dobj=new DataObj();
776 dobj.spec=null;
777 dobj.code=dataCode.NULL;
778 switch(c) {
779 case 'V':
780 {
781 dobj.vfe=1;
782 if (datastr.equals("VI"))
783 dobj.code=dataCode.VERT_INDEX;
784 else if (datastr.equals("VF"))
785 dobj.code=dataCode.VERT_FLOWER;
786 else if (datastr.equals("VZ"))
787 dobj.code=dataCode.VERT_CENTER;
788 else if (datastr.equals("VA"))
789 dobj.code=dataCode.VERT_CURV;
790 else if (datastr.equals("VT"))
791 dobj.code=dataCode.VERT_AIM;
792 else if (datastr.equals("VD"))
793 dobj.code=dataCode.VERT_DEG;
794 else if (datastr.equals("VC"))
795 dobj.code=dataCode.VERT_COLOR;
796 else if (datastr.equals("VR"))
797 dobj.code=dataCode.VERT_RADII;
798 else if (datastr.equals("VM"))
799 dobj.code=dataCode.VERTEX_MAP;
800 else if (datastr.equals("VXYZ"))
801 dobj.code=dataCode.VERT_XYZ;
802 else if (datastr.startsWith("VS")) { // 'sharp' function (ratios)
803 try {
804 int pnum=Integer.parseInt(datastr.substring(2,3));
805 int qnum=Integer.parseInt(datastr.substring(3,4));
806 pData=PackControl.cpDrawing[pnum].getPackData();
807 qData=PackControl.cpDrawing[qnum].getPackData();
808 } catch (Exception ex) {
809 throw new ParserException("error: output: bad 'VS' perscription");
810 }
811 dobj.code=dataCode.SHARP_PQ;
812 }
813 break;
814 }
815 case 'F':
816 {
817 dobj.vfe=2;
818 if (datastr.equals("FI"))
819 dobj.code=dataCode.FACE_INDEX;
820 else if (datastr.equals("FZ"))
821 dobj.code=dataCode.FACE_CORNERS;
822 else if (datastr.equals("FC"))
823 dobj.code=dataCode.FACE_COLOR;
824 else if (datastr.equals("FA"))
825 dobj.code=dataCode.FACE_AREA;
826 else if (datastr.equals("FV"))
827 dobj.code=dataCode.FACE_VERTICES;
828 else if (datastr.equals("FDR"))
829 dobj.code=dataCode.FACE_DUAL_RADII;
830 else if (datastr.equals("FDZ"))
831 dobj.code=dataCode.FACE_DUAL_CENTER;
832 break;
833 }
834 case 'E':
835 {
836 dobj.vfe=3;
837 if (datastr.equals("EC"))
838 dobj.code=dataCode.EDGE_COLOR;
839 else if (datastr.equals("ES"))
840 dobj.code=dataCode.EDGE_SCHW;
841 else if (datastr.equals("EL"))
842 dobj.code=dataCode.EDGE_LENGTH;
843 else if (datastr.equals("ER"))
844 dobj.code=dataCode.EDGE_INT_LENGTH;
845 else if (datastr.equals("EI"))
846 dobj.code=dataCode.EDGE_INDICES;
847 else if (datastr.equals("EDZ"))
848 dobj.code=dataCode.EDGE_DUAL_CENTERS;
849 else if (datastr.equals("EDI"))
850 dobj.code=dataCode.EDGE_DUAL_INDICES;
851 break;
852 }
853 case 'C': // 'CPcallPacket' is global for passing information
854 {
855 if (datastr.equals("CP"))
856 dobj.vfe=0;
857 dobj.code=dataCode.CALL_PACKET;
858 break;
859 }
860 case 'P':
861 {
862 if (datastr.equals("PZ"))
863 dobj.vfe=1; // based on vertices
864 dobj.code=dataCode.PAVER_CORNERS;
865 break;
866 }
867 case 'M':
868 {
869 if (datastr.equals("Mob"))
870 dobj.vfe=4;
871 dobj.code=dataCode.MOBIUS_LABELS;
872 break;
873 }
874 } /* end of switch */
875 return dobj; // no legal code
876 }
