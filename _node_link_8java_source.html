1616 public static NodeLink circleSpecs(PackData p,Vector<SelectSpec> specs) {
1617 if (specs==null || specs.size()==0)
1618 return null;
1619 SelectSpec sp=null;
1620 int count=0;
1621 // will store results in 'tmpUtil'
1622 int[] tmpUtil=new int[p.nodeCount+1];
1623 // loop through all the specifications: these should alternate
1624 // between 'specifications' and 'connectives', starting
1625 // with the former, although typically there will be just
1626 // one specification in the vector and no connective.
1627 UtilPacket uPx=null;
1628 UtilPacket uPy=null;
1629 boolean isAnd=false; // true for '&&' connective, false for '||'.
1630 for (int j=0;j<specs.size();j++) {
1631 sp=(SelectSpec)specs.get(j);
1632 if (sp.object!='c')
1633 throw new ParserException(); // spec must be for circles
1634 try {
1635 for (int v=1;v<=p.nodeCount;v++) {
1636
1637 // success?
1638 boolean outcome=false;
1639 uPx=sp.node_to_value(p,v,0);
1640 if (sp.unary) {
1641 if (uPx.rtnFlag!=0)
1642 outcome=sp.comparison(uPx.value,0);
1643 }
1644 else {
1645 uPy=sp.node_to_value(p,v,1);
1646 if (uPy.rtnFlag!=0)
1647 outcome=sp.comparison(uPx.value, uPy.value);
1648 }
1649 if (outcome) { // yes, this value satisfies condition
1650 if (!isAnd && tmpUtil[v]==0) { // 'or' situation
1651 tmpUtil[v]=1;
1652 count++;
1653 }
1654 }
1655 else { // no, fails this condition
1656 if (isAnd && tmpUtil[v]!=0) { // 'and' situation
