235 public static double getEnergy(PackData p) {
236 double erg=0.0;
237// double [][]conductance=ComplexAnalysis.setConductances(p);
238 double groundLength=2.0*(TorusEnergy.mycon/Math.sqrt(p.faceCount));
239 for (int v=1;v<=p.nodeCount;v++) {
240 int[] petals=p.packDCEL.vertices[v].getPetals();
241 for (int j=0;j<petals.length;j++)
242 if (petals[j]>v) {
243 double len=groundLength-(p.getRadius(v)+
244 p.getRadius(petals[j]));
245 erg += len*len;
246 }
247 }
248 return erg;
249 }

255 public int normalize() {
256 double area=0;
257 for (int f=1;f<=extenderPD.faceCount;f++) {
258 int []verts=extenderPD.packDCEL.faces[f].getVerts();
259
260 // assume tangency
261 double r0=extenderPD.getRadius(verts[0]);
262 double r1=extenderPD.getRadius(verts[1]);
263 double r2=extenderPD.getRadius(verts[2]);
264 area += EuclMath.eArea(r0,r1,r2,1.0,1.0,1.0);
265 }
266 double factor=Math.sqrt(area);
267 for (int v=1;v<=extenderPD.nodeCount;v++)
268 extenderPD.setRadius(v,extenderPD.getRadius(v)/factor);
269 return 1;
270 }
