472 public void resetDisplay(double fracMax) {
473
474 // what are actual display dimensions?
475 PackControl.displayDimension=Toolkit.getDefaultToolkit().getScreenSize();
476
477 // set active canvas size
478 Dimension dimMax=activeFrame.getMaxDim(PackControl.displayDimension);
479 int maxSize=activeFrame.getWouldBeSize(dimMax.height);
480 int curSize=MainFrame.getCanvasDim().height;
481 int newSize=curSize;
482 if (fracMax>0.0) {
483 if (fracMax>1.0)
484 fracMax=1.0;
485 newSize=(int)(fracMax*(double)maxSize);
486 }
487 else if (maxSize<curSize)
488 newSize=maxSize;
489 activeFrame.layMeOut(new Dimension(newSize,newSize));
490
491 // set pair canvas size
492 dimMax=mapPairFrame.getMaxDim(displayDimension);
493 maxSize=mapPairFrame.getWouldBeSize(dimMax.height);
494 newSize=curSize=PairedFrame.getCanvasDim().height;
495 if (fracMax>0.0) {
496 if (fracMax>1.0)
497 fracMax=1.0;
498 newSize=(int)(fracMax*(double)maxSize);
499 }
500 else if (maxSize<curSize)
501 newSize=maxSize;
502 mapPairFrame.layMeOut(new Dimension(newSize,newSize));
503
504 // set active location
505 int rightedge=frame.getLocation().x+frame.getWidth()+6;
506 int X=PackControl.displayDimension.width-activeFrame.getWidth()-4;
507 if (X>rightedge && rightedge>100) X=rightedge;
508 int Y=PackControl.displayDimension.height-activeFrame.getHeight()-HeightBuffer;
509 if (Y>0) Y=Y/2;
510 else Y=2;
511 activeFrame.setLocation(new Point(X,Y));
512
513 // set pair locatiion
514 X=PackControl.displayDimension.width-mapPairFrame.getWidth()-4;
515 if (X>rightedge && rightedge>100) X=rightedge;
516 Y=PackControl.displayDimension.height-mapPairFrame.getHeight()-HeightBuffer;
517 if (Y>0) Y=Y/2;
518 else Y=2;
519 mapPairFrame.setLocation(new Point(X,Y));
520 }

1148 public void queryUserForQuit() {
1149 if (scriptManager.hasChanged) {
1150 // Should the script be saved?
1151 Object[] options = {"Save", "Discard", "Cancel"};
1152 String message = "Script contents may have changed. Would you like to save before exiting?";
1153 int result = JOptionPane.showOptionDialog(null, message, "Save?", JOptionPane.YES_NO_CANCEL_OPTION,
1154 JOptionPane.QUESTION_MESSAGE, null, options, options[0]);
1155
1156 if (result == JOptionPane.YES_OPTION) {
1157 File scriptFile;
1158 if ((scriptFile = FileDialogs.saveDialog(FileDialogs.SCRIPT, true)) != null) {
1159 scriptManager.tnWriter.Write_from_TN(scriptFile);
1160 if (scriptFile.getName().startsWith("new_script.")) {
1161 CirclePack.cpb.errMsg("Not allowed to save as \"new_script.*\".");
1162 return;
1163 }
1164
1165 try {
1166 ScriptBundle.m_locator.add2List(scriptFile.getCanonicalPath(), true);
1167 } catch (Exception e) {}
1168 } else return;
1169
1170 scriptManager.hasChanged = false;
1171 scriptHover.scriptTitle(scriptManager.scriptName, false);
1172 exit();
1173 } else if (result == JOptionPane.NO_OPTION) exit();
1174 } else {
1175 // The script hasn't changed.
1176 int result = JOptionPane.showConfirmDialog(null, "Exit CirclePack?", "Exit?", JOptionPane.YES_NO_OPTION);
1177 if (result == JOptionPane.YES_OPTION) exit();
1178 }
1179 }
