728 public static double weakConError(PackData p,
729 TriAspect []aspects,int v) {
730 if (p.isBdry(v))
731 return 1.0;
732 double rtio=1.0;
733 int[] faceFlower=p.getFaceFlower(v);
734 for (int j=0;j<faceFlower.length;j++) {
735 int ff=faceFlower[j];
736 int k=aspects[ff].vertIndex(v);

738 rtio /= aspects[ff].sidelengths[k]; // right sidelength
739 }

751 public static double []getErrors(PackData p,TriAspect []aspects) {
752 double weak_err=0.0;
753 double weak_max=0.0;
754 double TLog_err=0.0;
755 double TLog_max=0.0;
756 double ang_err=0.0;
757 double ang_max=0.0;
758 double []ans=new double[6];
759
760 for (int v=1;v<=p.nodeCount;v++) {
761 // Strong: find sum[|Log(t.t')|]^2 for interior edges
762 HalfLink vspokes=p.packDCEL.vertices[v].getEdgeFlower();
763 Iterator<HalfEdge> sis=vspokes.iterator();
764 while (sis.hasNext()) {
765 HalfEdge he=sis.next();
766 int w=he.twin.origin.vertIndx;
767 if (w>v && !he.isBdry()) {
768 double prd=Math.abs(Math.log(Math.abs(
769 edgeRatioError(p,aspects,he))));
770 TLog_max=(prd>TLog_max) ? prd:TLog_max;
771 TLog_err += prd*prd;
772 }
773 }
774
775 // weak;
776 double werr=Math.abs(Math.log(
777 ProjStruct.weakConError(p,aspects,v)));
778 weak_err += werr*werr;
779 weak_max=(werr>weak_max) ? werr:weak_max;
780
781 // angle sum
782 double ang=Math.abs(angSumTri(p,v,1.0,aspects)[0]-p.getAim(v));

1268 public static double sideCalc(PackData p,int v,double aim,int N,
1269 TriAspect []asps) {
1270 double bestcurv,upcurv,lowcurv;
1271 double lower,upper;
1272 double limit=0.5;
1273 double best=1.0;
1274
1275 // starting curvature
1276 bestcurv=lowcurv=upcurv=ProjStruct.angSumSide(p,v,best,asps);
1277
1278 if (Math.abs(bestcurv-aim)<=OKERR)
1279 return 1.0;
1280
1281 // set upper/lower limits on possible factors due to triangle inequality
1282 double []bds=ProjStruct.sideBounds(p,v,asps);
1283 lower=1.0-(1-bds[0])*limit; // interpolate between bds[0] and 1
1284 upper=1.0+(bds[1]-1.0)*limit; // interpolate between 1 and bds[1]
1285
1286 // does lowest allowed factor undershoot?
1287 if (bestcurv<(aim-OKERR)) {
1288 lowcurv=ProjStruct.angSumSide(p,v,lower,asps);
1289 if (lowcurv<aim) { // still not enough, but return
1290 return lower;
1291 }
1292 }
1293
1294 // does largest allowed factor overshoot?
1295 else if (bestcurv>(aim+OKERR)) {
1296 upcurv=ProjStruct.angSumSide(p,v,upper,asps);
1297 if (upcurv>aim) { // still not enough, but return
1298 return upper;
1299 }
1300 }
1301
1302 // successive interpolation adjustments
1303 for (int n=1;n<=N;n++) {
1304 if (bestcurv>(aim+OKERR)) {
1305 lower=best;
1306 lowcurv=bestcurv;
1307 best += (aim-bestcurv)*(upper-best)/(upcurv-bestcurv);
1308 }
1309 else if (bestcurv<(aim-OKERR)) {
1310 upper=best;
1311 upcurv=bestcurv;
1312 best -= (bestcurv-aim)*(lower-best)/(lowcurv-bestcurv);
1313 }
1314 else {
1315 return best;
1316 }
1317
1318 // angle sum with current 'best' factor
1319 bestcurv=ProjStruct.angSumSide(p,v,best,asps);
1320 }
1321 return best;
1322 }
