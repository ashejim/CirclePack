151 public Complex sToZ(double s) {
152 if (hes<0) return null;
153
154 // get length and adjust s if necessary
155 double pathLength=domain.lastElement();
156 if (s<=0.0 || pathLength==0) return pathZ.firstElement();
157 if (pathLength<0.0) throw new DataException("path length is <= 0.0");
158 if (!closed && (s<0.0 || s>pathLength))
159 throw new DataException("Improper arclenth parameter");
160 while (s<0.0) s +=pathLength;
161 while (s>pathLength) s-=pathLength;
162
163 // find indices [i-1,i] for location of s in 'domain' and
164 // interpolating factor 'fac'
165 int i=1;
166 while (i<N && s>domain.get(i)) i++;
167 if (i==N) throw new DataException("arclength parameter greater than length");
168 double fac=(s-domain.get(i-1))/(domain.get(i)-domain.get(i-1));
169 Complex startZ=pathZ.get(i-1);
170 Complex endZ=pathZ.get(i);
171
172 if (hes==0) { // euclidean
173 return endZ.sub(startZ).times(fac).add(startZ);
174 }
175 double ang=fac*SphericalMath.s_dist(startZ,endZ);
176 return SphericalMath.s_shoot(startZ,endZ,ang);
177 }

197 public double jumpParam(double s,double minDist,double maxDist) {
198 Complex z=null;
199 if (s<0 || s>length || // bad parameter
200 minDist>length || // path too short to reach
201 (!closed && (s+minDist)>length) || // remaining path too short
202 (hes>0 && minDist>Math.PI) || // too far for sphere
203 (z=sToZ(s))==null) // failure to find point on path
204 return -1;
205
206 // jump ahead in 'maxDist' size steps until we:
207 // * find t,
208 // * exceed length for non-closed path
209 // * go fully around and pass s for clsoed path
210
211
212 double inc=maxDist;
213 double old_inc=s;
214 double t;
215 while (inc<length && (t=newParam(s,inc))>=0) {
216 Complex w=sToZ(t);
217 double dist=dist(z,w);
218 if (dist>=minDist && dist<=maxDist) return t;
219 while (dist>=minDist) { // must have gone too far
220 inc=(old_inc+inc)/2.0;
221 t=newParam(s,inc); // should be >=0
222 w=sToZ(t);
223 dist=dist(z,w);
224 if (dist<=maxDist) return t;
225 }
226 // take another step
227 old_inc=inc;
228 inc+=maxDist;
229 } // end of while
230 return -1.0; // failure
231 }
