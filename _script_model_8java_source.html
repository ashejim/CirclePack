187 protected void processIncludedFiles(BufferedReader reader)
188 throws IOException {
189 String line;
190 String orig_name="";
191 String newName=null;
192 File temp;
193 int safeID=new Random().nextInt(32000);
194
195 while ((line = reader.readLine()) != null) {
196
197 // shuck lines until we reach a tagged line
198 while (line!=null && line.indexOf("<")==-1)
199 line=reader.readLine();
200 if (line==null || (line.indexOf("</CPdata>")) != -1)
201 return;
202
203 // else, starting with the tagged line
204 PrintWriter tempWriter;
205
206 // Have to check deprecated name style: i.e., <name>..</name> tag
207 if (line.indexOf("<name>")!=-1) {
208 orig_name = line.substring(line.indexOf(">") + 1,
209 line.indexOf("</"));
210 orig_name = orig_name.trim();
211 if (orig_name.length()==0)
212 orig_name = new String("notnamed"+includedFiles.size());
213 // read through whitespace lines
214 while ((line=reader.readLine()) !=null && line.indexOf("<")==-1);
215 // didn't find any subsequent tag?
216 if (line==null)
217 return;
218 }
219
220 // reaching here, should have the initial tag line of tagged region.
221 // If it has 'name' attribute, that overrides any old style "name" info.
222 int indx=line.indexOf("<");
223 int newidx=line.indexOf("name=");
224 if (newidx != -1) { // found name attribute
225 newidx = newidx + 6; // move past the first "
226 int end = line.indexOf("\"", newidx);
227 if (end == -1 || (end - 1) == newidx) { // default name
228 orig_name = new String("notnamed"+includedFiles.size());
229 }
230 else // given name
231 orig_name = line.substring(newidx, end);
232 }
233 else if (orig_name.length()==0)
234 orig_name = new String("notnamed"+includedFiles.size());
235
236 // In any case, should now have name, be ready for data lines
237 newName=new String(safeID + orig_name);
238 temp = new File(System.getProperty("java.io.tmpdir"),newName);
239 temp.createNewFile();
240 temp.deleteOnExit();
241 tempWriter = new PrintWriter(new FileWriter(temp));
242
243 int datatype=IncludedFile.RAW; // default
244 int count=0;
245 if (line.length()>indx+14 && line.substring(indx+1,indx+14).equalsIgnoreCase("circlepacking")) {
246 datatype=IncludedFile.PACKING;
247 }
248 else if(line.length()>indx+5 && line.substring(indx+1,indx+5).equalsIgnoreCase("path")) {
249 datatype=IncludedFile.PATH;
250 }
251 else if(line.length()>indx+8 && line.substring(indx+1,indx+8).equalsIgnoreCase("xyzData")) {
252 datatype=IncludedFile.XYZ;
253 }
254 else if (line.length()>indx+9 && line.substring(indx+1,indx+9).equalsIgnoreCase("commands")) {
255 datatype=IncludedFile.CMDS;
256 }
257 else if (line.length()>indx+9 && line.substring(indx+1,indx+6).equalsIgnoreCase("image")) {
258 datatype=IncludedFile.IMAGE;
259 }
260 else if (line.length()>indx+14 && line.substring(indx+1,indx+11).equalsIgnoreCase("aboutimage")) {
261 datatype=IncludedFile.ABOUT_IMAGE;
262 }
263
264 // base64 encoded image files? read characters
265 if (datatype==IncludedFile.IMAGE || datatype==IncludedFile.ABOUT_IMAGE) {
266 char c;
267 while ((c=(char) reader.read())>0 && !(c=='<')) {
268 tempWriter.write(c);
269 }
270 count++;
271 }
272
273 // transfer lines until another tagged line is encountered;
274 // (last line should have 'END', but this isn't checked)
275 // TODO: don't know if '<' can occur in base64 encoded image file.
276 else {
277 while ((line=reader.readLine().trim())!=null && !line.startsWith("<")) {
278 tempWriter.println(line);
279 if (line.length()!=0) count++;
280 }
281 }
282
283 tempWriter.flush();
284 tempWriter.close();
285
286 // get anything? add to vector of included files
287 if (count>0) {
288
289 // IMAGE should be base64 encoded, so decode
290 if (datatype==IncludedFile.IMAGE || datatype==IncludedFile.ABOUT_IMAGE) {
291 temp=Base64InOut.fileOutof64(temp);
292 }
293
294 // add to 'includedFiles' list: duplicate names stored once, file contains last contents
295 boolean dup=false;
296 for (int i=0;i<includedFiles.size();i++)
297 if (includedFiles.get(i).origName.equals(orig_name))
298 dup=true;
299 if (!dup) {
300 includedFiles.add(new IncludedFile(datatype,orig_name,temp));
301 }
302 }
303
304 orig_name="";
305 } // end of main while loop
306 }

391 public URL getScriptURL(String namE) {
392 if (namE==null)
393 return null;
394 String name=namE.trim();
395 if (name.startsWith("file:")) {
396 if (name.length()==5)
397 return null;
398 name=name.substring(5);
399 }
400
401 if (name.startsWith("www."))
402 name=new String("http://"+name);
403
404 // If we are looking in the file system (versus web)
405 URL url=null;
406 if (!name.startsWith("htt") && !name.startsWith("ftp") && !name.startsWith("gopher")) {
407 File temp=null;
408 // TODO: other syntax for home?
409 if (name.startsWith("~")) {
410 name=name.substring(1);
411 temp=new File(CPFileManager.HomeDirectory,name);
412 }
413 else temp = new File(name);
414 if (temp==null || !temp.exists()) {
415 // next, try in 'ScriptDirectory':
416 temp=new File(CPFileManager.ScriptDirectory,name);
417 if (temp.exists() == false) {
418 String errmsg=new String("Requested script '"+namE+"' not found");
419 CirclePack.cpb.myErrorMsg(errmsg);
420 return null;
421 }
422 }
423 name=new String("file:"+temp.toString());
424 }
425 else { // if from web, need to pick off file name from end
426 int index=name.lastIndexOf(File.separatorChar);
427 if (index<0) index=0;
428 String nameonly=name.substring(index+1,name.length());
429 if (nameonly.length()==0)
430 return null;
431 }
432
433 // AF:
434 if ((url=FileUtil.tryURL(name))==null) {
435 String errmsg=new String("failed to create URL for "+name);
436 CirclePack.cpb.myErrorMsg(errmsg);
437 url=null;
438 }
439 return url;
440 }
