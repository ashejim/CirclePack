93 public void initiateEdgeData() {
94 NodeLink intv=new NodeLink(extenderPD,"a");
95 Iterator<Integer> ilst=intv.iterator();
96 while (ilst.hasNext()) {
97 Vertex vert=extenderPD.packDCEL.vertices[ilst.next()];
98 if (vert.isBdry()) {
99 continue;
100 }
101 HalfEdge htrace=vert.halfedge;
102 if (vert.getNum()==3) { // not subject to adjustment
103 htrace.setSchwarzian(S3);
104 htrace=htrace.prev.twin;
105 htrace.setSchwarzian(S3);
106 htrace=htrace.prev.twin;
107 htrace.setSchwarzian(S3);
108 continue;
109 }
110 do {
111 Vertex oppv=htrace.twin.origin;
112 if (oppv.getNum()!=3 || oppv.isBdry()) {
113 SchEdgeData sed=new SchEdgeData(htrace);
114 edgeData[htrace.edgeIndx]=sed;
115 }
116 htrace=htrace.prev.twin; // cclw
117 } while (htrace!=vert.halfedge);
118 } // done with all interior vertices
119 }

377 public double newSchwarzian(HalfEdge target,
378 double factor,boolean print) {
379 double currSch=target.getSchwarzian();
380 SchEdgeData sed1=edgeData[target.edgeIndx];
381 SchEdgeData sed2=edgeData[target.twin.edgeIndx];
382 if (sed1==null && sed2==null)
383 return currSch;
384
385 // scherr will be thw average of cclw and clw
386 // errors for one or both HalfEdges.
387 double scherr=0.0;
388 double denom=0.0;
389 if (sed1!=null) {
390 scherr=sed1.newSch_cclw[0]+sed1.newSch_clw[0];
391 denom=2.0;
392 }
393
394 if (sed2!=null) {
395 scherr=scherr+sed1.newSch_cclw[0]+sed1.newSch_clw[0];
396 if (denom>0.0)
397 denom=4.0;
398 else
399 denom=2.0;
400 }
401
402 double newSch=currSch+factor*scherr/denom;
403 StringBuilder strbld=null;
404
405//debugging
406if (print) {
407 strbld=new StringBuilder(" Next adjustment:");
408 strbld.append(" target="+target);
409 strbld.append(": original="+currSch+"; new="+newSch);
410}
411
412 // too small to change?
413 if (Math.abs(scherr)<.0000000001)
414 return currSch;
415
416 // else, return new value
417
418// debugging
419if (print)
420 System.out.println(strbld.toString());
421
422 return newSch;
423 }

472 public int[] worstErrors() {
473 int[] ans=new int[4];
474 double radplus=0.0;
475 double radminus=0.0;
476 double angplus=0.0;
477 double angminus=0.0;
478 for (int v=1;v<=extenderPD.nodeCount;v++) {
479 Complex err=vertErrors[v];
480 if (err==null)
481 continue;
482 double r=err.x;
483 double a=err.y;
484
485 if (r>radplus) {
486 ans[0]=v;
487 radplus=r;
488 }
489 if (r<radminus) {
490 ans[1]=v;
491 radminus=r;
492 }
493 if (a>angplus) {
494 ans[2]=v;
495 angplus=a;
496 }
497 if (a<angminus) {
498 ans[3]=v;
499 angminus=a;
500 }
501 }
502 return ans;
503 }

526 if (vert.isBdry())
527 return null;
528 int n=vert.getNum();
529 double[] uzians=new double[n];
530 HalfEdge he=vert.halfedge;
531 uzians[0]=1.0-he.getSchwarzian();

533 int tick=1;

536 he=he.prev.twin;
