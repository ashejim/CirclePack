96 public static ImageIcon embellishIcon(ImageIcon startImg,
97 String key,boolean xy_tear,boolean hot_arrow,boolean handy,
98 boolean menu_pop) {
99
100 if (startImg==null) return null;
101 if (key!=null && key.trim().length()>0)
102 key=key.trim().substring(0,1);
103 else key=null;
104
105 Image img=startImg.getImage();
106 int high=img.getHeight(null);
107 int wide=img.getWidth(null);
108 int voffset=0;
109 int hoffset=0;
110
111 // extra height for xy_tear, handy, key, and hotpoint
112 if (xy_tear || handy || hot_arrow)
113 high +=3;
114 if (xy_tear || handy)
115 wide +=6;
116 else if (hot_arrow) {
117 wide +=6;
118 hoffset+=5;
119 }
120 if (key!=null) {
121 high +=3;
122 voffset+=3;
123 }
124
125 BufferedImage bufImage=null;
126 try {
127 bufImage=new BufferedImage(wide,high,BufferedImage.TYPE_INT_ARGB);
128 } catch (Exception ex) {
129 if (wide<1 || high <1) {
130 wide=high=22;
131 bufImage=new BufferedImage(wide,high,
132 BufferedImage.TYPE_INT_ARGB);
133 }
134 else throw new ParserException("Problem embellishing: "+ex.getMessage());
135 }
136 Graphics2D g2d=bufImage.createGraphics();
137 g2d.drawImage(img,new AffineTransform(1,0,0,1,hoffset,voffset),null);
138
139 // in future add small hand if dragging allowed; 'handy.png' is 12x12
140 if (handy) {
141// g2d.drawImage(handyImg.getImage(),new AffineTransform(1,0,0,1,wide-12,high-12),null);
142 }
143
144 // add 'teardrop' if the command has an '#XY', 'teardrop.png' is 7x12
145 if (xy_tear && !handy) {
146 g2d.drawImage(teardropImg.getImage(),new AffineTransform(1,0,0,1,wide-7,high-12),null);
147 }
148
149 // add hotpoint cursor spot (lower left), 'cursorHotPt.png' is 10x12
150 if (hot_arrow) {
151 g2d.drawImage(hotPtImg.getImage(),new AffineTransform(1,0,0,1,0,high-10),null);
152 }
153
154 // add letter/number, upper right hand corner
155 if (key!=null && key.trim().length()>0) {
156 if (key.equals("*")) {
157 g2d.drawImage(astrixImg.getImage(),new AffineTransform(1,0,0,1,wide-15,0),null);
158 }
159 else {
160 Rectangle2D.Double rect=new Rectangle2D.Double(wide-11,0,10,12);
161 g2d.setColor(Color.WHITE);
162 g2d.fill(rect);
163 g2d.setColor(Color.BLACK);
164 g2d.draw(rect);
165 try {
166 g2d.setFont(new Font("truetype",Font.ROMAN_BASELINE,10));
167 g2d.drawString(key,wide-8,9);
168 } catch (Exception ex) {}
169 }
170 }
171
172 // Expand length for arrow for popup
173 if (menu_pop) {
174 // have to extend the basic image to make room
175 BufferedImage bufImage2=new BufferedImage(wide+18,high,BufferedImage.TYPE_INT_ARGB);
176 Graphics2D gd=bufImage2.createGraphics();
177 // load current image
178 gd.drawImage(bufImage,new AffineTransform(1,0,0,1,0,0),null);
179 gd.drawImage(popImg.getImage(),new AffineTransform(1,0,0,1,wide+1,2),null);
180 gd.dispose();
181 return new ImageIcon(bufImage2);
182 }
183
184 return new ImageIcon(bufImage); // Note: may have returned earlier
185 }
