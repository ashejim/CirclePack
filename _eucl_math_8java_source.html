72 double ivd0, double ivd1, double ivd2) {

341 Complex a,Complex b) {

462 public static CircleSimple e_compcenter(Complex z0,Complex z1,double r0,
463 double r1,double r2,double ivd0,double ivd1,double ivd2) {
464 // squared lengths of sides
465 double l0 = r0 * r0 + r1 * r1 + 2 * r0 * r1 * ivd0; // l0=a^2
466 double l1 = r1 * r1 + (r2) * (r2) + 2 * r1 * (r2) * ivd1; // l1=b^2
467 double l2 = r0 * r0 + (r2) * (r2) + 2 * r0 * (r2) * ivd2; // l2=c^2
468
469 double ld = 0.5 * (l2 + l0 - l1); // a c cos(theta)=(c^2+a^2-b^2)/2
470 if ((ld * ld) > (l2 * l0))
471 return new CircleSimple(false);
472
473 // drop perp to side a, get base, height G, H
474 double B = ld / Math.sqrt(l0); // v.x= c cos(theta)
475 double H = Math.sqrt(l2 - B*B);
476 Complex v = new Complex(B,H);
477
478 // if a were laid out along x-axis, v is vector to center
479 Complex w = z1.sub(z0);
480 double s = w.abs();
481 if (s < 1.0e-30)
482 return new CircleSimple(false);
483 w = w.times(1 / s);
484 // answer is z0+w*v
485 Complex z=new Complex((z0.x + w.x * v.x - w.y * v.y),
486 (z0.y + w.x * v.y + w.y * v.x));
487 return new CircleSimple(z, r2, 1);
488 }

536 public static double e_dilatation(double A,double B,double C,
537 double a,double b,double c) {
538 double AB2,ab2;
539 double okerr=OKERR;
540
541 if ((AB2=2.0*A*B)<okerr || (ab2=2.0*a*b)<okerr
542 || C<okerr || c<okerr
543 || (A>=(B+C+okerr) && B>=(A+C+okerr) && C>=(A+B+okerr))
544 || (a>=(b+c+okerr) && b>=(a+c+okerr) && c>=(a+b+okerr))

548 double cos2=(a*a+b*b-c*c)/ab2;
549 double sin1=Math.sqrt(1-cos1*cos1);
550 double sin2=Math.sqrt(1-cos2*cos2);
551 double x=B*cos1;
552 double y=1.0/(B*sin1);
553 double u=b*cos2;
554 double v=b*sin2;
555 return affine_dilatation(a/A, (u-(a/A)*x)*y,(double) 0.0, v*y);

607 public static Complex quad_cross_ratio(
608 double r1,double r2,double r3,double r4,
609 double ivd1,double ivd2,double ivd3,double ivd4,
610 double ivd_common) {
611
612 // compute edge lengths using radii/overlaps
613 double L5=r1*r1+r3*r3+2*r1*r3*ivd_common;
614 double l5=Math.sqrt(L5);
615 double L1=r1*r1+r2*r2+2*r1*r2*ivd1;
616 double L2=r2*r2+r3*r3+2*r2*r3*ivd2;
617 double L3=r3*r3+r4*r4+2*r3*r4*ivd3;
618 double L4=r4*r4+r1*r1+2*r1*r4*ivd4;
619
620 Complex z1=new Complex(0.0);
621 Complex z3=new Complex(l5);
622 double l1c=(L1+L5-L2)/(2.0*l5);
623 Complex z2=new Complex(l1c,-Math.sqrt(L1-l1c*l1c));
624 double l4c=(L4+L5-L3)/(2.0*l5);
625 Complex z4=new Complex(l4c,Math.sqrt(L4-l4c*l4c));
626 Complex num1=z1.minus(z2).divide(z1.minus(z4));
627 Complex num2=z3.minus(z4).divide(z3.minus(z2));
628 return num1.times(num2);

767 public static int effectiveRad(PackData p,double []radii) {
768 for (int v=1;v<=p.nodeCount;v++) {
769 // for each node find the area sum and angle sum
770 Complex z=p.getCenter(v);
771
772 double thsum = 0.0;
773 double asum = 0.0;
774 int num=p.countFaces(v);
775 int[] flower=p.packDCEL.vertices[v].getFlower(true);
776 for (int j=0;j<num;j++) {
777 Complex spoke0 =
778 z.minus(p.getCenter(flower[j]));
779 Complex spoke1 =
780 z.minus(p.getCenter(flower[j+1]));
781 Complex farside = (p.getCenter(flower[j])).
782 minus(p.getCenter(flower[j+1]));
783 double a, b, c;
784 a = spoke0.abs(); b = farside.abs(); c = spoke1.abs();
785 double facerad = (a+c-b)/2.0;
786 double th = Math.acos((a*a+c*c-b*b)/(2*a*c));
787 thsum += th;
788 asum += facerad*facerad*th/2.0;
789 }
790 if (radii==null)
791 p.setRadius(v,Math.sqrt(asum/(thsum/2.0)));
792 else radii[v]= Math.sqrt(asum/(thsum/2.0));
793 }
794 return 1;

906 Complex rz,Complex wz,Complex lz) {
907 double []angles=new double[4];
908 double edg=vz.minus(wz).abs();
909 double lv_edge=lz.minus(vz).abs();
910 double lw_edge=lz.minus(wz).abs();
911 double rv_edge=rz.minus(vz).abs();
912 double rw_edge=rz.minus(wz).abs();
913
914 // angles at v, left/right
915 double al=Math.acos((lv_edge*lv_edge+edg*edg-lw_edge*lw_edge)/(2.0*lv_edge*edg));
916 double ar=Math.acos((rv_edge*rv_edge+edg*edg-rw_edge*rw_edge)/(2.0*rv_edge*edg));
917 angles[0]=al+ar;
918
919 // angles at w, left/ right
920 al=Math.acos((lw_edge*lw_edge+edg*edg-lv_edge*lv_edge)/(2.0*lw_edge*edg));
921 ar=Math.acos((rw_edge*rw_edge+edg*edg-rv_edge*rv_edge)/(2.0*rw_edge*edg));
922 angles[2]=al+ar;
923
924 angles[1]=Math.acos((rw_edge*rw_edge+rv_edge*rv_edge-edg*edg)/(2.0*rw_edge*rv_edge));
925 angles[3]=Math.acos((lw_edge*lw_edge+lv_edge*lv_edge-edg*edg)/(2.0*lw_edge*lv_edge));
926
927 return angles;
928 }
