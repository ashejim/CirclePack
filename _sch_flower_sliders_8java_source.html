108 public void populate() {
109 // note: sliders indexed from 0
110 ActiveSlider[] tmpSliders = new ActiveSlider[sliderCount];
111 Iterator<HalfEdge> his=hedges.iterator();
112 int tick=0;
113 while (his.hasNext() && tick<sliderCount) {
114 HalfEdge edge=his.next();
115 // arrange so v < w.
116 int vv=edge.origin.vertIndx;
117 int ww=edge.twin.origin.vertIndx;
118 if (vv>ww) {
119 int hld=vv;
120 vv=ww;
121 ww=hld;
122 }
123 String str=new String(vv+" "+ww);
124 double sch=getParentValue(tick);
125 if (tick<sliderCount)
126 tmpSliders[tick]=
127 new ActiveSlider(this,tick,str,sch,true);
128 tick++;
129 }
130
131 mySliders = new ActiveSlider[sliderCount];
132 for (int j=0;j<sliderCount;j++) {
133 mySliders[j]=tmpSliders[j];
134 sliderPanel.add(mySliders[j]);
135 }
