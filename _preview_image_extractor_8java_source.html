57 public Image getImage() throws IOException {
58 /*
59 * Due to difficulties with the network coding involved with extracting preview
60 * images from remote resources, specifically in regards to connection failure
61 * and time delays, we'll only support local file resources for now. If the URL
62 * does not point to a local resource, return a null image.
63 *
64 * TODO: Add support for remote network resources. This will likely require
65 * custom code to connect to the remote host and iteratively download partial
66 * chunks of the resource, as well as a custom XMD parser to find the preview
67 * image tag. Local caching of remote resources would help tremendously.
68 * Consider checking the modification date of remote XMD scripts (most servers
69 * will provide this on request) to determine if a resource needs to be
70 * redownloaded and recached.
71 */
72 if (!xmlResource.getProtocol().equals("file")) return null;
73
74 // Get a SAX parser to interpret our XML resource.
75 SAXParser sp;
76 try {
77 sp = SAXParserFactory.newInstance().newSAXParser();
78 } catch (ParserConfigurationException e) {
79 // I'm not certain, but this exception should never occur.
80 // Print a stack trace for debugging, and return a null image.
81 e.printStackTrace();
82 return null;
83 } catch (SAXException e) {
84 // Same as above.
85 e.printStackTrace();
86 return null;
87 }
88
89 /*
90 * Get a stream to the XML resource.
91 *
92 * Set connection and read timeouts on the stream. By default no
93 * timeouts are set, allowing network sockets that have silently
94 * failed to block access to the resource indefinitely. This
95 * potentially floods the thread pool and causes deadlock.
96 */
97 final InputStream is;
98 try {
99 URLConnection connection = xmlResource.openConnection();
100 connection.setConnectTimeout(TIMEOUT);
101 connection.setReadTimeout(TIMEOUT);
102 is = connection.getInputStream();
103 } catch (SocketTimeoutException e) {
104 // Connection timed out. Return a null image.
105 return null;
106 }
107
108 /*
109 * Parse the XML resource to search for the preview image. If a preview image is
110 * found, catch the PreviewImageFoundException and get the preview image from it.
111 */
112 String base64Image = null;
113 try {
114 // Start parsing.
115 sp.parse(is, new PreviewImageSearchHandler());
116 } catch (PreviewImageFoundException e) {
117 // The base 64 string encoded image has been found. Grab it.
118 base64Image = e.getBase64Image();
119 } catch (SAXException e) {
120 /*
121 * This is likely caused by an invalid XML resource. This could mean anything
122 * from minor XML formatting errors to the resource not being XML at all.
123 * This shouldn't happen given that the caller is responsible for the validity
124 * of the XML resource, so print the stack trace for debugging and return a
125 * null image.
126 */
127 e.printStackTrace();
128 return null;
129 } catch (SocketTimeoutException e) {
130 // Connection timed out. Return a null image.
131 return null;
132 } finally {
133 // Whatever happens, close the stream. This shouldn't fail. If it does, print
134 // a stack trace for debugging and continue.
135 try {is.close();}
136 catch (IOException e) {e.printStackTrace();}
137 }
138
139 // If we haven't found an image, return null. Otherwise, decode and construct the
140 // image and return the result.
141 if (base64Image == null) return null;
142 // TODO: used to be 'decodeBase64' instead of 'decode', but it seems to be
143 // absent; maybe I have the wrong "codec"??
144 else return ImageIO.read(new ByteArrayInputStream(Base64.decodeBase64(base64Image)));
145 }
