239 String name=(String)nameField.getText();
240 String cmd=formulateCmd();
241 String tip=tooltipField.getText();
242 if (tip!=null && tip.length()==0) tip=null;
243 String mnemonic=null;
244 char c=' ';
245 if (name!=null && name.length()>0 && (((c=name.charAt(0))>='A' && c<='z') ||
246 (c>='0' && c<='9') || c=='*')) mnemonic=String.valueOf(c);
247 CPBase.hashedTools.remove(name);
248 // now createTool will put it back in
249 CPIcon cpIcon=(CPIcon)iconCombo.iconBox.getSelectedItem();
250 parentHandler.createTool(cpIcon,cmd,name,mnemonic,tip,
251 isDropable(),(PopupBuilder)null);
