59 public static Complex[] rand_bdry_pts(Path2D.Double gpath,
60 int M,boolean sS)
61 throws DataException {
62 setSeed=sS;
63 // need length and polynomial approx of Gamma (should be
64 // just one component)
65 double flatness = PathUtil.gpExtent(gpath) * PathUtil.FLAT_FACTOR;
66 double length = PathUtil.gpLength(gpath, flatness);
67 Vector<Complex> poly;
68 Vector<Vector<Complex>> polyGamma = PathUtil.gpPolygon(gpath,
69 flatness);
70 try {
71 poly = (Vector<Complex>) polyGamma.get(0);
72 } catch (Exception ex) {
73 throw new DataException("problem with specified path");
74 }
75 if (gpath == null || M < 2)
76 throw new DataException("Problem: path empty or too short");
77
78 // find M random ordered arclength param spots in [0,length]
79 Random CPrand = new Random();
80 if (setSeed) // for debugging
81 CPrand.setSeed((long)1);
82 double[] arc_spots = new double[M];
83 for (int j = 0; j < M; j++) {
84 arc_spots[j] = CPrand.nextDouble()*length;
85 }
86 Arrays.sort(arc_spots); // put in increasing order
87
88 // convert by interpolation to Complex points on polygonal 'path'
89 double last_length = 0.0;
90 double spot;
91 Complex[] bdry_pts = new Complex[M];
92
93 Iterator<Complex> gamma = poly.iterator();
94 Complex initPt = new Complex((Complex) gamma.next());
95 Complex nextPt = new Complex((Complex) gamma.next());
96 double next_length = nextPt.minus(initPt).abs();
97 int indx = 0;
98 while (indx < M) {
99 spot = arc_spots[indx];
100 while (gamma.hasNext() && spot > next_length) {
101 initPt = nextPt;
102 nextPt = (Complex) gamma.next();
103 last_length = next_length;
104 next_length += nextPt.minus(initPt).abs();
105 }
106 if (spot <= next_length) { // interpolate x,y
107 double t = (spot - last_length) / (next_length - last_length);
108 bdry_pts[indx] = nextPt.times(t).add(initPt.times(1 - t));
109 indx++;
110 } else {
111 throw new DataException();
112 }
113 }
114 return bdry_pts;
115 }

237 c1=c1.divide(maxdist);
238 c2=c2.divide(maxdist);

253 // later map random points of unit disc to random points
254 // in disc centered at bounding box of Gamma containing
255 // Gamma.

331 // translated points have indices differing by 0(mod N).
332 if (torus_flag) {

367 Complex z=zb.next();
368 Z_list.add(z);
369 elink.add(new EdgeSimple(tickv,++tickv));
370 }

372 elink.add(new EdgeSimple(tickv,iCount+1));
373 }

377 dData=new DelaunayData(hes,Z_list,elink);

381 "'ProcessDelaunay' exception in planeDelaunay: "+ex.getMessage());
382 return null;
383 }

386
387 // For torus, remove triangles involving only added vertices
388 // (i.e., all indices > N). Leave other faces, even though
389 // this may leave some boundary vertices with no interior face.
390 int tick=0;
391 if (torus_flag) {
392 for (int i=1;i<=Tri.faceCount;i++) {
393 if (Tri.faces[i].vert[0]>N // all added vertices
394 && Tri.faces[i].vert[1]>N
395 && Tri.faces[i].vert[2]>N)
396 tick++;
397 }
398 // found some? create new faces without them.
399 if (tick>0) {
400 tick=0;
401 Face []newfaces=new Face[Tri.faceCount+1];
402 for (int i=1;i<=Tri.faceCount;i++) {
403 if (Tri.faces[i].vert[0]>N && Tri.faces[i].vert[1]>N
404 && Tri.faces[i].vert[2]>N)
405 tick++;
406 else
407 newfaces[i-tick]=Tri.faces[i];
408 }
409

411 int facecount=Tri.faceCount;
412 Tri=new Triangulation();
413 Tri.faceCount=facecount-tick;
414 Tri.faces=newfaces;

416 // they should be ignored by 'tri_to_pack'
417 int v;
418 for (int i=1;i<=Tri.faceCount;i++) {
419 for (int j=0;j<3;j++)
420 if ((v=Tri.faces[i].vert[j])>N)
421 Tri.faces[i].vert[j]=1+(v-1)%N;
422 }
423
424 // throw away redundant point locations
425 Point3D []newNodes=new Point3D[N+1];
426 for (int i=1;i<=N;i++) {
427 newNodes[i]=new Point3D(old_nodes[i]);
428 }
429 Tri.nodes=newNodes;
430 }
431 } // end of torus adjustment
432
433 return Tri;

528 public static PackData randomHypKomplex(int N,boolean sS) {
529 setSeed=sS;
530
531 if (N<4)
532 return null;
533
534 // set polygonal path enclosing disc for call
535 int bdryN=(int)(Math.PI*Math.sqrt((double)N));
536 N=N+(int)(.25*bdryN); // extra to cover pruning losses
537 Path2D.Double cirGamma=new Path2D.Double();
538 double inc=2.0*Math.PI/(double)bdryN;
539 cirGamma.moveTo(1.0,0.0);
540 double ta;
541 for (int i=1;i<bdryN;i++) {
542 ta=i*inc;
543 cirGamma.lineTo(Math.cos(ta),Math.sin(ta));
544 }
545 cirGamma.closePath();
546
547 Triangulation Tri=null;
548 try {
549 Tri=random_Triangulation(N,sS,-1,0.0,cirGamma,null); // call with circle as bdry
550 if (Tri==null) {
551 CirclePack.cpb.errMsg("random_Triangulation failed");
552 return null;
553 }
554 } catch (Exception ex) {
555 throw new CombException(
556 "random_Triangulation error: "+ex.getMessage());
557 }
558
559 // eliminate faces containing points of cirGamma (indices > N)
560 Triangulation newTri=new Triangulation();
561 newTri.faces=new Face[Tri.faceCount+1];
562 int count=0;
563 int v,u,w;
564 try {
565 for (int f=1;f<=Tri.faceCount;f++) {
566 Face face=Tri.faces[f];
567 v=face.vert[0];
568 u=face.vert[1];
569 w=face.vert[2];

573 newTri.faces[count].vert[0]=v;
574 newTri.faces[count].vert[1]=u;
575 newTri.faces[count].vert[2]=w;
576 }
577 }
578 } catch (Exception ex){
579 ex.printStackTrace();
580 }
581 newTri.faceCount=count;
582 newTri.nodeCount=Tri.nodeCount;
583 newTri.nodes=Tri.nodes;
