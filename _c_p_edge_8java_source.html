147 public static void sphCreateEdge(Path2D.Double gpath,
148 SphGeodesic sg,CPdrawing cpS) {
149 if (!sg.isVisible()) return; // both are on back
150 if (sg.lineFlag) { // handle a straight line
151 Complex a=SphericalMath.sphToVisualPlane(sg.z1.x,sg.z1.y);
152 Complex b=SphericalMath.sphToVisualPlane(sg.z2.x,sg.z2.y);
153 gpath.moveTo(cpS.toPixX(a.x),cpS.toPixY(a.y));
154 gpath.lineTo(cpS.toPixX(b.x),cpS.toPixY(b.y));
155 return;
156 }
157 if (sg.followHorizon()!=0) { // handle geodesic along the horizon
158 double ang1=rad2deg*Math.atan2(sg.z1.y,sg.z1.x);
159 double extent=rad2deg*sg.z2.divide(sg.z1).arg();
160 gpath.append((Shape)(new Arc2D.Double(cpS.toPixX(-1.0),
161 cpS.toPixY(1.0),2.0*cpS.pixFactor,
162 2.0*cpS.pixFactor,ang1,extent,Arc2D.OPEN)),true);
163 return;
164 }
165
166 // part of great circle
167 Point3D C=new Point3D(sg.center.x,sg.center.y);
168 double plen=Math.sqrt(C.y*C.y+C.z*C.z);
169 Point3D A=new Point3D(0,-C.z/plen,C.y/plen); // unit orthog to yz-proj of C
170
171 // Creation of 'sg' should have made both ends visible
172 Point3D end1=new Point3D(sg.z1.x,sg.z1.y);
173 Point3D end2=new Point3D(sg.z2.x,sg.z2.y);
174
175 double dp=Point3D.DotProduct(A,end1);
176 double a1=Math.acos(dp);
177 if (dp>=1.0) a1=0.0;
178 else if (dp<=-1.0) a1=Math.PI;
179 dp=Point3D.DotProduct(A,end2);
180 double a2=Math.acos(dp);
181 if (dp>=1.0) a2=0.0;
182 else if (dp<=-1.0) a2=Math.PI;
183 double extent=MathComplex.radAngDiff(a1,a2);
184 if (C.x <= 0) { // we're seeing the ellipse from the back
185 a1 *=-1.0;
186 extent *=-1.0;
187 }
188 a1+=Math.PI/2.0;
189 double origx=cpS.toPixX(0.0);
190 double origy=cpS.toPixY(0.0);
191 double wide=Math.abs(C.x)*cpS.pixFactor;
192 Arc2D.Double tmpArc= new Arc2D.Double(origx-wide,origy-cpS.pixFactor,
193 2.0*wide,2.0*cpS.pixFactor,a1*rad2deg,extent*rad2deg,Arc2D.OPEN);
194
195 // Rotate to true orientation
196 AffineTransform rot=new AffineTransform();
197 rot.translate(origx,origy);
198 rot.rotate(-Math.atan2(C.z,C.y));
199 rot.translate(-origx,-origy);
200 Path2D.Double tmpPath=new Path2D.Double((Shape)tmpArc,rot);
201 gpath.append(tmpPath,true);
202 return;
203 }
