128 appearSleeper.interrupt();

135 /*
136 * Calculate where to put the locked frame, accounting for the size of the
137 * frame. We want the frame to visually pop into place around the hover
138 * panel without it seeming to move, so we subtract the height of the title
139 * bar from the x location and the width of the left frame border from the
140 * y location.
141 */
142 int x = hoverFrame.getLocation().x;
143 int y = hoverFrame.getLocation().y;
144 x -= lockedFrame.getInsets().left;
145 y -= lockedFrame.getInsets().top;
146

148 // This happens if the hover frame hasn't displayed before the instance is locked.
149 if (x < 0 || y < 0) {

151 x = mousePoint.x;
152 y = mousePoint.y;
153 }
154

159 lockedFrame.setVisible(true);
160 } else {
161 // If we are unlocking, just mark it and hide the locked frame.

196 protected class HoverAdapter extends MouseAdapter {
197 @Override
198 public void mouseEntered(MouseEvent me) {
199 // The user has moused over the parent component.
200 // Create the thread to show the hover frame after a delay.
201 appearSleeper = new Thread() {
202 @Override
203 public void run() {
204 try {
205 if (locked) Thread.sleep(2 * APPEAR_DELAY); // Sleep for double time if the frame is locked.
206 else Thread.sleep(APPEAR_DELAY);
207 } catch (InterruptedException e) {return;} // If we get interrupted, just return.
208
209 // If we haven't been interrupted, pop up the appropriate frame.
210 if (locked) {
211 // The frame is currently locked, so restore and focus on the locked frame.
212 // We are in a separate thread, and all GUI changes must be made from the EventQueue.
213 EventQueue.invokeLater(new Runnable() {
214 @Override
215 public void run() {
216 lockedFrame.setExtendedState(Frame.NORMAL);
217 lockedFrame.requestFocus();
218 }
219 });
220 } else {
221 // The frame is currently unlocked, so show the hover frame.
222 // We are in a separate thread, and all GUI changes must be made from the EventQueue.
223 EventQueue.invokeLater(new Runnable() {
224 @Override
225 public void run() {
226 // Get the absolute mouse location.
227 Point mousePoint = MouseInfo.getPointerInfo().getLocation();
228
229 // Calculate the location for the hover frame to appear, accounting for offsets.
230 int x = mousePoint.x + X_OFFSET;
231 int y = mousePoint.y + Y_OFFSET;
232
233 // Move the frame if it would appear off the screen.
234 Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
235 if (x + hoverFrame.getWidth() > screenSize.width) x -= hoverFrame.getWidth() + X_OFFSET * 2;
236 if (y + hoverFrame.getHeight() > screenSize.height) y -= hoverFrame.getHeight() + Y_OFFSET * 2;
237
238 // Prepare and show the hover frame.
239 Point hoverLocation = new Point(x, y);
240 hoverFrame.setLocation(hoverLocation);
241 hoverFrame.add(currentInstance);
242 hoverFrame.pack();
243 hoverFrame.setVisible(true);
244 }
245 });
246 }
247 }
248 };
249 appearSleeper.start();
250 }

252 @Override
253 public void mouseExited(MouseEvent me) {
254 // If the mouse leaves the parent component, stop the hover frame appearance process.
255 appearSleeper.interrupt();
256
257 // If the panel isn't locked and the hover frame is visible, count down to make it disappear.
258 if (!locked && hoverFrame.isVisible()) {
259 disappearSleeper = new Thread() {
260 @Override
261 public void run() {
262 try {Thread.sleep(DISAPPEAR_DELAY);}
263 catch (InterruptedException e) {return;} // If we get interrupted, just return.
264
265 /*
266 * Manually check to see if the mouse is in the hover frame. If it is, lock this instance.
267 * We do this because sometimes components displayed in this panel will consume the mouse
268 * entered event that normally locks this instance.
269 */
270 Point mousePoint = MouseInfo.getPointerInfo().getLocation();
271 Point hoverLocation = hoverFrame.getLocation();
272 Dimension hoverSize = hoverFrame.getSize();
273
274 if (mousePoint.x >= hoverLocation.x && mousePoint.y >= hoverLocation.y
275 && mousePoint.x <= hoverLocation.x + hoverSize.width && mousePoint.y <= hoverLocation.y + hoverSize.width) {
276 // The mouse is in the hover frame. Lock this instance.
277 // We are in a separate thread, and all GUI changes must be made from the EventQueue.
278 EventQueue.invokeLater(new Runnable() {
279 @Override
280 public void run() {
281 currentInstance.setLocked(true);
282 }
283 });
284 } else {
285 // The mouse is not in the hover frame. Hide the hover frame.
286 // We are in a separate thread, and all GUI changes must be made from the EventQueue.
287 EventQueue.invokeLater(new Runnable() {
288 @Override
289 public void run() {
290 hoverFrame.setVisible(false);
291 }
292 });

294 }
295 };
