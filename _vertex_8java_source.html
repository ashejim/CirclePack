108 public static HalfEdge oppSpoke(HalfEdge spoke,boolean hexflag) {
109 Vertex vert=spoke.origin;
110 int safety=2000;
111 HalfEdge cclw=spoke;
112 HalfEdge clw=spoke;
113 int N=vert.getNum();
114
115 // interior case
116 if (vert.bdryFlag==0) {
117 if (hexflag && N!=6)
118 return null;
119 do {
120 safety--;
121 cclw=cclw.prev.twin;
122 clw=clw.twin.next;
123 if (cclw==clw && cclw!=spoke)
124 return cclw;
125 } while (cclw!=spoke && clw!=spoke && safety>0);
126 if (safety==0)
127 throw new CombException("safety value blew in looking for opposite to "+spoke);
128 return null;
129 }
130
131 if (hexflag && (N!=3 || !spoke.isBdry()))
132 return null;
133
134 // else 'vert' is bdry: first check if 'spoke' is bdry
135 if (spoke==vert.halfedge) // downstream bdry edge?
136 return spoke.twin.next.twin; // upstream bdry
137 if (spoke.twin.isBdry()) // upstream bdry edge?
138 return vert.halfedge; // downstream bdry
139
140 // else spoke points to interior
141 do {
142 safety--;
143 cclw=cclw.prev.twin;
144 clw=clw.twin.next;
145 if (cclw.origin.bdryFlag>0)
146 return vert.halfedge;
147 else if (clw.origin.bdryFlag>0)
148 return vert.halfedge.twin.next.twin;
149 } while (safety>0);
150 if (safety==0)
151 throw new CombException("inconsistency looking for bdry oppSpoke");
152 return null;
153 }
