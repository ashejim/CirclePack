247 public SolverData initSolverData(PackLite pL) {
248
249 int ijCount=0;
250 for (int v=1;v<=pL.intVertCount;v++)
251 ijCount +=pL.vNum[v]+1; // larger than needed, but will be adjusted later
252
253 // create Ap and Ai data for compressed column format
254 // Ap has 1+number of column (=intVertCount+1) entries, with Ap[0]=0.
255 SolverData sdata=new SolverData();
256
257 VWJ []tmp_vwj=new VWJ[ijCount];
258 VWJ []tmp_rhs=new VWJ[ijCount];
259 sdata.nz_entries=0;
260 rhsIJcount=0; // separate pointer to non-zero entries
261
262 // first, find and list (possibly) non-zero entry spots
263 // in the square matrix, interior-to-interior
264 for (int v=1;v<=pL.intVertCount;v++) {
265 tmp_vwj[sdata.nz_entries++]=new VWJ(v,v,-1);
266 int []flower=pL.flowers[v];
267 int num=pL.vNum[v];
268 if (flower[0]!=flower[num]) // this shouldn't happen
269 num++;
270 for (int j=0;j<num;j++) {
271 int w=flower[j];
272 if (w<=pL.intVertCount) // w interior
273 tmp_vwj[sdata.nz_entries++]=new VWJ(v,w,j);
274 else
275 tmp_rhs[rhsIJcount++]=new VWJ(v,w,j);
276 }
277 }
278
279 // non-zero spots are in vwjA; sort these and store in 'Ai',
280 // put counts in 'Ap'
281 vwjA=new VWJ[sdata.nz_entries];
282 for (int i=0;i<sdata.nz_entries;i++)
283 vwjA[i]=tmp_vwj[i];
284 Arrays.sort(vwjA);
285 sdata.Ai=new int[sdata.nz_entries];
286 sdata.Ap=new int[pL.intVertCount+1];
287 for (int i=0;i<sdata.nz_entries;i++) {
288 sdata.Ap[vwjA[i].w]++;
289 sdata.Ai[i]=vwjA[i].v-1;
290 }
291 for (int w=2;w<=pL.intVertCount;w++) // Ap entries are cummulative
292 sdata.Ap[w] += sdata.Ap[w-1];
293
294 vwjRHS=new VWJ[rhsIJcount];
295 for (int j=0;j<rhsIJcount;j++)
296 vwjRHS[j]=tmp_rhs[j];
297
298 sdata.intNum=pL.intVertCount;
299 sdata.bdryNum=pL.bdryCount;
300
301 return sdata;
302 }

357 // first time through?

389 localError= l2quality(.001);

392 if (myPLiteError==null)
393 myPLiteError=new Vector<Double>(1);
394 myPLiteError.addElement(Double.valueOf(localError));
395 status=RePacker.RIFFLE; // ready to continue riffling
396 localPasses += pass;
397 totalPasses += pass;
398 reapResults();
399 return pass;// kick out
400 }
401
402 } // end of outer while
403 // ********************************************************
404
405 // store results in parent packing
406 reapResults();
407
408 status=RePacker.RIFFLE;
409 return pass;

512 public int updateMatrices() {
513
514 // update
515 sectorTans=new double[myPLite.intVertCount+1][];
516 for (int v=1;v<=myPLite.intVertCount;v++) {
517// double ckangsum=0.0; // for debugging
518 int []flower=myPLite.flowers[v];
519 double []data=new double[myPLite.vNum[v]];
520 for (int j=0;j<myPLite.vNum[v];j++) {
521 int w=flower[j];
522 int u=flower[j+1];
523 double cang=EuclMath.e_cos_overlap(myPLite.radii[v],myPLite.radii[w],myPLite.radii[u]);
524// ckangsum +=Math.acos(cang); // debugging: check angle sum
525 data[j]=Math.sqrt((1.0-cang)/(1.0+cang)); // half angle formula
526 }
527 sectorTans[v]=data;
528 }
529
530 // store total conductances for variable vertices
531 conductances=new double[myPLite.intVertCount+1];
532 for (int v=1;v<=myPLite.intVertCount;v++) {
533 double []st=sectorTans[v];
534 int num=myPLite.vNum[v];
535 int []flower=myPLite.flowers[v];
536 if (flower[0]==flower[num]) { // interior vertex
537 int w=flower[0];
538 conductances[v]=(st[num-1]+st[0])/(myPLite.radii[v]+myPLite.radii[w]);
539 for (int j=1;j<num;j++) {
540 double t1=st[j-1];
541 double t2=st[j];
542 w=flower[j];
543 conductances[v] +=(t1+t2)/(myPLite.radii[v]+myPLite.radii[w]);
544 }
545 }
546 else { // bdry vertex
547 int w=flower[0];
548 conductances[v]=st[0]/(myPLite.radii[v]+myPLite.radii[w]);
549 for (int j=1;j<num;j++) {
550 double t1=st[j-1];
551 double t2=st[j];
552 w=flower[j];
553 conductances[v] +=(t1+t2)/(myPLite.radii[v]+myPLite.radii[w]);
554 }
555 w=flower[num];
556 conductances[v] +=st[num-1]/(myPLite.radii[v]+myPLite.radii[w]);
557 }
558 }
559
560 // set 'Aentries'
561 int n=sData.nz_entries;
562 sData.Aentries=new double[n];
563 for (int k=0;k<n;k++) {
564 VWJ vwj=vwjA[k];
565 int v=vwj.v;
566 int w=vwj.w;
567 int j=vwj.j;
568 int num=myPLite.vNum[v];
569 double []st=sectorTans[v];
570 int []flower=myPLite.flowers[v];
571 if (j<0) { // edge to self if j=-1
572 sData.Aentries[k]=-1.0;
573 }
574 else if (flower[0]==flower[num]) { // interior vertex
575 double t1;
576 if (j==0)
577 t1=st[num-1];
578 else
579 t1=st[j-1];
580 double t2=st[j];
581
582 // this is transition probability from v to w. Conductance on an edge is ratio
583 // of dual circle center distance divided by circle center distance. rad[v]
584 // doesn't appear in t1, t2 because it cancels with rad[v] that would appear
585 // in conductance.
586 sData.Aentries[k]=((t1+t2)/(myPLite.radii[v]+myPLite.radii[w]))/conductances[v];
587 }
588 else { // bdry vertex
589 double t1;
590 if (j==0 || j==num)
591 t1=0.0;
592 else
593 t1=st[j-1];
594 double t2=st[j];
595 sData.Aentries[k]=((t1+t2)/(myPLite.radii[v]+myPLite.radii[w]))/conductances[v];
596 }
597 }
598
599 // set rhs stuff, which is -B*zb, with B the m x n matrix
600 // of transition, interior-to-bdry, m=intVertCount, n=bdryCount, and
601 // zb is the vertor of bdry centers.
602 sData.rhsX=new double[myPLite.intVertCount];
603 sData.rhsY=new double[myPLite.intVertCount];
604 for (int k=0;k<rhsIJcount;k++) {
605 VWJ vwj=vwjRHS[k];
606 int v=vwj.v;
607 int w=vwj.w;
608 int j=vwj.j;
609 int num=myPLite.vNum[v];
610 double []st=sectorTans[v];
611 int []flower=myPLite.flowers[v];
612 double tranprob=0.0;
613 boolean bdry=false;
614 if (flower[0]!=flower[num]) { // v is bdry
615 bdry=true;
616 num++;
617 }
618 double t=0.0;
619 if (bdry) {
620 if (j==0)
621 t=st[0];
622 if (j==num)
623 t=st[j-1];
624 else
625 t=st[j-1]+st[j];
626 }
627 else {
628 t=st[j];
629 if (j==0)
630 t +=st[num-1];
631 else
632 t+=st[j-1];
633 }
634 tranprob=1.0*(t/(myPLite.radii[v]+myPLite.radii[w]))/conductances[v];
635
636 // have to introduce negative sign
637 sData.rhsX[v-1] -= tranprob*myPLite.centers[w].x;
638 sData.rhsY[v-1] -= tranprob*myPLite.centers[w].y;
639 }
640
641 return 1;
642 }

668 public int setEffective() {
669
670 // handle interior first: their anglesum and aim are 2*pi
671 for (int v = 1; v <= myPLite.intVertCount; v++) {
672 double area = 0;
673 int num = myPLite.vNum[v];
674 Complex z = new Complex(myPLite.centers[v]);
675 int[] flower = myPLite.flowers[v];
676 for (int j = 0; j < num; j++) {
677 int jr = flower[j];
678 int jl = flower[j + 1];
679 Complex zr = new Complex(myPLite.centers[jr]);
680 Complex zl = new Complex(myPLite.centers[jl]);
681 // radius of sector at v
682 double r = 0.5 * (zr.minus(z).abs() + zl.minus(z).abs() - zr.minus(zl).abs());
683 double cC = EuclMath.e_cos_corners(myPLite.centers[v],
684 myPLite.centers[jr], myPLite.centers[jl]);
685 if (cC < -1.0)
686 cC = -1.0;
687 if (cC > 1.0)
688 cC = 1.0;
689 double ang = Math.acos(cC);
690 area += 0.5 * r * r * ang; // add area of sector
691 }
692
693 // effective radii at interior if it is adjustable
694 if (myPLite.aims[v]>0.0)
695 myPLite.radii[v] = Math.sqrt(area / Math.PI);
696 } // finished with interiors
697
698 if (mode==FIXED_BDRY)
699 return 1;
700
701 // now do boundary, where, if there's not specified angle sum,
702 // we try to maintain the current angle sum;
703 for (int v = myPLite.intVertCount+1; v <= myPLite.vertCount; v++) {
704 double area = 0;
705 double angsum=0;
706 Complex z = new Complex(myPLite.centers[v]);
707 int[] flower = myPLite.flowers[v];
708 int num = myPLite.vNum[v]-1; // num faces
709 for (int j = 0; j <= num; j++) {
710 int jr = flower[j];
711 int jl = flower[j + 1];
712 Complex zr = new Complex(myPLite.centers[jr]);
713 Complex zl = new Complex(myPLite.centers[jl]);
714 // radius of sector at v
715 double r = 0.5 * (zr.minus(z).abs() + zl.minus(z).abs() - zr.minus(zl).abs());
716 double cC = EuclMath.e_cos_corners(myPLite.centers[v],
717 myPLite.centers[jr], myPLite.centers[jl]);
718 if (cC < -1.0)
719 cC = -1.0;
720 if (cC > 1.0)
721 cC = 1.0;
722 double ang = Math.acos(cC);
723 angsum += ang;
724 area += 0.5 * r * r * ang; // add area of sector
725 }
726
727 // for positive aim, want effective radii to give same area as aim would
728 if (myPLite.aims[v]>0.001) { //
729 angsum=myPLite.aims[v];
730 myPLite.radii[v] = (Math.sqrt(2.0*area/angsum)); // + myPLite.radii[v]) / 2.0;
731 }
732 // for negative aim, want area as current angle sum would give, except we
733 // average radius with current value to prevent oscillations (can test if
734 // this is helpful later
735 else if (myPLite.aims[v]<-.001) { // or same
736 myPLite.radii[v] = (Math.sqrt(2.0*area/angsum) + myPLite.radii[v]) / 2.0;
737 }
738
739 // Note: if 'targetArea' is between -.001 and .001, then the
740 // local radius does not change. (E.g., with 3 bdry circles
741 // in the spherical case.)
742
743 } // finished with boundary
744
745 return 1;
746 }

761 public double []packQuality(double crit) {
762 double []ans=new double[2];
763 double sas=0.0;
764 for (int k=1;k<=myPLite.intVertCount;k++) {
765 double r=myPLite.radii[k];
766 int num=myPLite.vNum[k];
767 int []flower=myPLite.flowers[k];
768 sas=-myPLite.aims[k];
769 for (int j=0;j<num;j++) {
770 sas += Math.acos(EuclMath.e_cos_overlap(r,
771 myPLite.radii[flower[j]],myPLite.radii[flower[j+1]]));
772 }
773 sas=Math.abs(sas);
774 if (sas>ans[0]) ans[0]=sas;
775 ans[1] += sas*sas;
776 }
777 return ans;

833 public double setHoroCenters() {
834 double fvalue=0.0;
835 double fprime=0.0;
836
837 // initial guess: (sum of bdry radii)/pi
838 double R=0.0;
839 double minrad=0.0;
840 double []r=new double[myPLite.bdryCount+1]; // closed list of radii
841 for (int j=0;j<myPLite.bdryCount;j++) {
842 R += r[j]=myPLite.radii[j+myPLite.intVertCount+1];
843 minrad= (r[j]>minrad) ? r[j] : minrad;
844 if (java.lang.Double.isNaN(R))
845 throw new DataException("R is NaN ???");
846 }
847 r[myPLite.bdryCount]=r[0]; // to close up
848 R /= Math.PI;
849 if (R<2.0*minrad)

915 public double setOrthoCenters() {
916 double fvalue=0.0;
917 double fprime=0.0;
918
919 // initial guess: (sum of bdry radii)/pi
920 double R=0.0;
921 double minrad=0.0;
922 double []r=new double[myPLite.bdryCount+1]; // closed list of radii
923 for (int j=0;j<myPLite.bdryCount;j++) {
924 R += r[j]=myPLite.radii[j+myPLite.intVertCount+1];
925 minrad= (r[j]>minrad) ? r[j] : minrad;

928 }
929 r[myPLite.bdryCount]=r[0]; // to close up
930 R /= 2.0*Math.PI;
931 if (R<minrad)
932 R=2.0*minrad;

934 // Newton iteration to find R
935 boolean keepon=true;
936 while (keepon) {
937
938 double RR=R*R;
939 fvalue=-2.0*Math.PI;
940 fprime=0.0;
941 for (int j=0;j<myPLite.bdryCount;j++) {
942 fvalue += 2.0*Math.atan(r[j]/R); // angle each boundary circle intercepts at origin
943 if (java.lang.Double.isNaN(fvalue))
944 throw new DataException("acos failed: R="+R+"; r[j]="+r[j]);
945 fprime += -2.0*r[j]/(RR+r[j]*r[j]);
946 }
947
948 // is this working?
949 double newR=R-fvalue/fprime;
950 if (newR<R/2.0 || newR>2.0*R) {
951// System.err.println("big change in R="+R+"; newR="+newR);
952 if (newR<R/2.0) {
953 newR=R/2;
954 if (newR<minrad)
955 newR=(minrad+R)/2.0;
956 }
957 if (newR>2.0*R)
958 newR=2.0*R;
959 }
960
961 if (Math.abs(newR-R)<.00001)
962 keepon=false;
963 R=newR;
964 } // end of while
965
966 // scale all radii by 1/R
967 for (int k=1;k<=myPLite.vertCount;k++)
968 myPLite.radii[k] /= R;
969
970 // set boundary centers, first on y-axis; center is distance sqrt(r^2+1^2) from origin
971
972 double rad=myPLite.radii[myPLite.intVertCount+1];
973 myPLite.centers[myPLite.intVertCount+1]=new Complex(0.0,Math.sqrt(rad*rad+1.0));
974 double accumArg=Math.atan(rad)+Math.PI/2.0; // pi/2 + half of angle subtended by first bdry circle
975 for (int k=myPLite.intVertCount+2;k<=myPLite.vertCount;k++) {
976 rad=myPLite.radii[k];
977 double delta=Math.atan(rad);
978 accumArg +=delta; // increment by half the angle subtended
979 double d=Math.sqrt(rad*rad+1.0);

1168 int []kolst=new int[n];
1169 for (int i=1;i<n;i++)
1170 kolst[i]=crnrs[i];

1177 neworder[tick]=indx;
1178 kolst[j]=0;
1179 tick++;
1180 }
1181 }
1182 }
1183 if (tick<n) {
1184 throw new ParserException("didn't get all the corners in order");
1185 }

1189
1190 for (int i=0;i<n;i++) {
1191 NodeLink nlk=new NodeLink();

1194 int k=(current+j-myPLite.intVertCount-1)%myPLite.bdryCount+myPLite.intVertCount+1;
1195 if (k==neworder[i+1]) {
1196 nlk.add(k);
1197 current=k;

1199 }

1201 nlk.add(k);

1204 }
