63 public int cmdParser(String cmd,Vector<Vector<String>> flagSegs) {
64 Vector<String> items=null;
65
66 if (cmd.startsWith("hist")) {
67 BufferedReader fp=null;
68 try {
69 items=flagSegs.get(0);
70 String str=items.get(0);
71 boolean script_flag=false;
72 if (str.equals("-s")) {
73 script_flag=true;
74 items.remove(0);
75 }
76 histFile=StringUtil.reconItem(items);
77 fp=CPFileManager.openReadFP(histFile,script_flag);
78 } catch (Exception ex) {
79 errorMsg("Failed to open 'history' file");
80 return 0;
81 }
82
83 tcState = INITIAL; // starting over
84 String line;
85
86 // Must find 'HISTORIES:'
87 int count=0;
88 try {
89 while(tcState==INITIAL && (line=StringUtil.ourNextLine(fp))!=null) {
90 StringTokenizer tok = new StringTokenizer(line);
91 while(tok.hasMoreTokens()) {
92 String mainTok = tok.nextToken();
93 if(mainTok.equals("HISTORIES:")) {
94 tcState=READING;
95 }
96 }
97 }
98 tiles=new Vector<TileInfo>(10);
99 while(tcState==READING && (line=StringUtil.ourNextLine(fp))!=null) {
100 StringTokenizer tok = new StringTokenizer(line);
101 TileInfo newTile=new TileInfo();
102 newTile.vert=Integer.parseInt(tok.nextToken());
103 if (newTile.vert<1 || newTile.vert>packData.nodeCount) {
104 throw new DataException("circle index out of range");
105 }
106
107 // set type
108 newTile.type=packData.countFaces(newTile.vert)-2;
109
110 String lineage=tok.nextToken();
111 depth=lineage.length();
112 newTile.history=new int[depth];
113 for (int j=0;j<depth;j++) {
114 String nextchar=lineage.substring(j,j+1);
115 newTile.history[j]=Integer.parseInt(nextchar);
116 }
117 tiles.add(newTile);
118 count++;
119 }
120 } catch (Exception ex) {
121 errorMsg("Exception in reading '"+histFile+"': "+ex.getMessage());
122 tcState=INITIAL; // ready to restart
123 return 0;
124 }
125 tcState=HAVE_HISTORY;
126 return count;
127 }
128 if (cmd.startsWith("set_col")) {
129 if (tcState < HAVE_HISTORY) {
130 errorMsg("can't set colors, TileColoring state is only "+stateStr[tcState]);
131 return 0;
132 }
133
134 // cycle through the tiles and set colors
135 int Red,Green,Blue;
136 for (int n=0;n<tiles.size();n++) {
137 Red=Green=Blue=0;
138 TileInfo tc=tiles.get(n);
139
140 // ======= here's where 'mode' chooses strategy
141 if (color_mode==0) { // equal weight
142 for (int i=1;i<depth;i++) {
143 // equal weight (could change weight depending on color)
144 if (tc.history[i]==TRIANGLE)
145 Red++;
146 else if (tc.history[i]==QUADRANGLE)
147 Green++;
148 else if (tc.history[i]==PENTAGON)
149 Blue++;
150 }
151 tc.red=(Red*colorHit)%256;
152 tc.green=(Green*colorHit)%256;
153 tc.blue=(Blue*colorHit)%256;
154 }
155 } // end of for loop
156 tcState=COLORS_SET;
157 }
158 if (cmd.startsWith("postfile")) { // get PostScript filename and open
159 if (tcState<COLORS_SET) {
160 errorMsg("can't open post, TileColoring state is only "+stateStr[tcState]);
161 return 0;
162 }
163 try {
164 items=flagSegs.get(0);
165 postFile=StringUtil.reconItem(items);
166 if (!postFile.endsWith(".ps"))
167 postFile=new String(postFile+".ps");
168 } catch (Exception ex) {
169 errorMsg("error in postscript file name"+postFile);
170 }
171 }
172 if (cmd.startsWith("set_hit")) {
173 try {
174 items=flagSegs.get(0);
175 int hit=Integer.parseInt((String)items.get(0));
176 if (hit<0 || hit>200) {
177 errorMsg("TileColoring increment must be between 0 and 200");
178 return 0;
179 }
180 colorHit=hit;
181 } catch (Exception ex) {
182 errorMsg("error in 'set_hit'");
183 return 0;
184 }
185 }
186 if (cmd.startsWith("draw")) {
187 if (tcState<COLORS_SET) {
188 errorMsg("colors not set yet, TileColoring state is only "+stateStr[tcState]);
189 return 0;
190 }
191 try {
192 cpCommand("set_screen -a");

194 for (int i=0;i<tiles.size();i++) {
195 TileInfo td=tiles.get(i);
196 int v=td.vert;
197 int num=packData.countFaces(v);
198 int[] flower=packData.getFlower(v);
199 int N=num+2*packData.getBdryFlag(v);
200 double []crnr=new double[2*N];
201 for (int j=0;j<num;j++) {
202 Complex Z=packData.getCenter(flower[j]);
203 if (packData.hes>0)
204 Z=cpDrawing.sphView.toApparentSph(Z);
205 crnr[2*j]=Z.x;
206 crnr[2*j+1]=Z.y;
207 }
208 if (packData.isBdry(v)) { // need extra points
209 Complex Z=packData.getCenter(flower[num]);
210 if (packData.hes>0)

245 int N=num+2*packData.getBdryFlag(v);
