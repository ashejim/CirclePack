118 public int update() {
119 NodeLink bdry=new NodeLink(extenderPD,"b");
120 int N=0;
121 if (bdry==null || (N=bdry.size())<3)
122 return 0;
123 bdryVerts=new int[N];
124 bdryRads=new double[N];
125 bdryAngSums=new double[N];
126
127 extenderPD.fillcurves();
128 for (int j=0;j<N;j++) {
129 int v=bdry.get(j);
130 bdryVerts[j]=v;
131 bdryRads[j]=extenderPD.getRadius(v);
132 bdryAngSums[j]=extenderPD.getCurv(v);
133 }
134
135 return N;

270 static double[] flatCurv(PackData p,int v,double r) {
271 if (!p.isBdry(v) || p.hes!=0)
272 return null;
273 double C=Math.PI-p.getCurv(v);
274
275 // compute the derivative of theta w.r.t. r
276 int[] petals=p.getPetals(v);
277 double dtdr=0.0;
278 for (int j=0;j<petals.length;j++) {
279 int k=petals[j];
280 double y=p.getRadius(k); // TODO: ?????
281 double z=p.getRadius(k);
282 dtdr +=EuclMath.Fx(r,y,z);
283 }
284
285 double[] results=new double[2];
286 if (Math.abs(C)<.0000001)
287 return results; // both zero
288 results[0]=C;
289 results[1]=-1.0*dtdr;

291 return results;

312 static double[] kissingCurv(PackData p,int v,double r) {
313 if (!p.isBdry(v) || p.hes!=0)
314 return null;
315 double t2=p.getCurv(v)/2.0;
316 double tant2=Math.tan(t2);
317 double cost2=Math.cos(t2);
318 double sect22=1.0/(cost2*cost2);
319
320 double R=tant2*r;
321 double dtdr=0.0;
322
323 // compute the derivative of theta w.r.t. r
324 int[] petals=p.getPetals(v);
325 double r2=p.getRadius(petals[0]);
326 for (int j=1;j<petals.length;j++) {
327 double r1=r2;
328 int k=petals[j];
329 r2=p.getRadius(k);
330 dtdr +=EuclMath.Fx(r,r1,r2);
331 }
332
333 double []results=new double[2];
334
335 // curvature zero?
336 if (Math.abs(t2-CPBase.piby2)<.0001) {
337 results[0]=0.0;
338 results[1]=0.0;
339 }
340 else {
341 results[0]=1.0/R-1.0;
342 results[1]=-dtdr;
343 }
344
345 return results;
