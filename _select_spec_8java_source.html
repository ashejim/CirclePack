538 if (unary) { // for unary (eg., 'b'), x>0 means true.
539 if (x > 0.0) {
540 if (negation)
541 return false;
542 return true;
543 }
544 if (negation)
545 return true;
546 return false;
547 }
548 if (condition == Condition.NULL)
549 return false;
550 else if (condition == Condition.EQ) {
551 if (Math.abs(x - y) > TOLER) {
552 if (negation)
553 return true;
554 return false;
555 }
556 if (negation)
557 return false;
558 return true;
559 } else if (condition == Condition.NE) {
560 if (Math.abs(x - y) < TOLER) {
561 if (negation)
562 return true;
563 return false;
564 }
565 if (negation)
566 return false;
567 } else if (condition == Condition.GT) {
568 if (x <= y) {
569 if (negation)
570 return true;
571 return false;
572 }
573 if (negation)
574 return false;
575 } else if (condition == Condition.GE) {
576 if (x < y) {
577 if (negation)
578 return true;
579 return false;
580 }
581 if (negation)
582 return false;
583 } else if (condition == Condition.LT) {
584 if (x >= y) {
585 if (negation)
586 return true;

589 if (negation)
590 return false;
